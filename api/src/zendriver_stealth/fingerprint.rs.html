13/// Sent to CDP as `Emulation.setUserAgentOverride.userAgentMetadata`.
14/// Mirrors the [W3C UA-CH spec](https://wicg.github.io/ua-client-hints/).
15#[derive(Debug, Clone, Serialize)]
16pub struct UserAgentMetadata {
17    pub brands: Vec<Brand>,
18    #[serde(rename = "fullVersionList")]
19    pub full_version_list: Vec<Brand>,
20    pub platform: String,
21    #[serde(rename = "platformVersion")]
22    pub platform_version: String,
23    pub architecture: String,
24    pub bitness: String,
25    pub wow64: bool,
26    pub mobile: bool,
27    pub model: String,
28}

30impl UserAgentMetadata {
31    /// Build a realistic UAM for the given platform + Chrome major version.
32    /// Uses the Chrome convention of three brands: "Not_A Brand;v=8",
33    /// "Chromium;v=N", "Google Chrome;v=N".
34    pub fn realistic(platform: Platform, chrome_major: u32, chrome_full: &str) -> Self {
35        let brands = vec![
36            Brand {
37                brand: "Not_A Brand".into(),
38                version: "8".into(),
39            },
40            Brand {
41                brand: "Chromium".into(),
42                version: chrome_major.to_string(),
43            },
44            Brand {
45                brand: "Google Chrome".into(),
46                version: chrome_major.to_string(),
47            },
48        ];
49        let full_version_list = vec![
50            Brand {
51                brand: "Not_A Brand".into(),
52                version: "8.0.0.0".into(),
53            },
54            Brand {
55                brand: "Chromium".into(),
56                version: chrome_full.to_string(),
57            },
58            Brand {
59                brand: "Google Chrome".into(),
60                version: chrome_full.to_string(),
61            },
62        ];
63        let (platform_version, architecture, bitness) = match platform {
64            Platform::Win32 => ("15.0.0", "x86", "64"),
65            Platform::MacIntel => ("10.15.7", "x86", "64"),
66            Platform::LinuxX86_64 => ("5.15.0", "x86", "64"),
67        };
68        Self {
69            brands,
70            full_version_list,
71            platform: platform.ch_platform().to_string(),
72            platform_version: platform_version.to_string(),
73            architecture: architecture.to_string(),
74            bitness: bitness.to_string(),
75            wow64: false,
76            mobile: false,
77            model: String::new(),
78        }
79    }
80}

87/// Default Chrome version used when `chrome --version` probe fails.
88/// Bump on each release of zendriver-rs.
89const FALLBACK_CHROME_FULL: &str = "120.0.6099.234";
90const FALLBACK_CHROME_MAJOR: u32 = 120;
91
92/// Probed system + Chrome facts used to compose stealth values.
93#[derive(Debug, Clone, Serialize)]
94pub struct Fingerprint {
95    pub platform: Platform,
96    pub chrome_major: u32,
97    pub chrome_full: String,
98    pub cpu_count: u32,
99    pub memory_gb: u32,
100    pub ua_string: String,
101    pub ua_metadata: UserAgentMetadata,
102    pub timezone: Option<String>,
103    pub locale: Option<String>,
104}
105
106impl Fingerprint {
107    /// Probe host system + installed Chrome to build a realistic fingerprint.
108    // `StealthError` is large because `PatchFailed` wraps `CallError` (~152B).
109    // Boxing it would cross the Task 5 file scope; bypass per-fn instead.
110    #[allow(clippy::result_large_err)]
111    pub fn auto_detect(chrome_executable: &Path) -> Result<Self, StealthError> {
112        let platform = detect_platform();
113        let (chrome_major, chrome_full) =
114            probe_chrome_version(chrome_executable).unwrap_or_else(|e| {
115                tracing::warn!("chrome version probe failed: {e}; using fallback");
116                (FALLBACK_CHROME_MAJOR, FALLBACK_CHROME_FULL.to_string())
117            });
118        let cpu_count = clamp_cpu_count(num_cpus::get() as u32);
119        let memory_gb = detect_memory_gb()?;
120        let ua_string = crate::ua::compose_ua_string(platform, &chrome_full);
121        let ua_metadata = UserAgentMetadata::realistic(platform, chrome_major, &chrome_full);
122        Ok(Self {
123            platform,
124            chrome_major,
125            chrome_full,
126            cpu_count,
127            memory_gb,
128            ua_string,
129            ua_metadata,
130            timezone: None,
131            locale: None,
132        })
133    }
134
135    /// Recompose UA string + UAM after platform/version overrides.
136    pub fn recompose(&mut self) {
137        self.ua_string = crate::ua::compose_ua_string(self.platform, &self.chrome_full);
138        self.ua_metadata =
139            UserAgentMetadata::realistic(self.platform, self.chrome_major, &self.chrome_full);
140    }
141}

143pub(crate) fn detect_platform() -> Platform {
144    #[cfg(target_os = "windows")]
145    {
146        Platform::Win32
147    }
148    #[cfg(target_os = "macos")]
149    {
150        Platform::MacIntel
151    }
152    #[cfg(any(target_os = "linux", target_os = "freebsd", target_os = "openbsd"))]
153    {
154        Platform::LinuxX86_64
155    }
156    #[cfg(not(any(
157        target_os = "windows",
158        target_os = "macos",
159        target_os = "linux",
160        target_os = "freebsd",
161        target_os = "openbsd"
162    )))]
163    {
164        Platform::LinuxX86_64 // unknown unix-likes -> linux is the safest plausibility
165    }
166}

168#[allow(clippy::result_large_err)]
169pub(crate) fn probe_chrome_version(exe: &Path) -> Result<(u32, String), StealthError> {
170    let output = Command::new(exe)
171        .arg("--version")
172        .output()
173        .map_err(|e| StealthError::ChromeVersionDetect(format!("spawn failed: {e}")))?;
174    if !output.status.success() {
175        return Err(StealthError::ChromeVersionDetect(format!(
176            "exit {:?}",
177            output.status.code()
178        )));
179    }
180    let stdout = String::from_utf8_lossy(&output.stdout);
181    // Format: "Google Chrome 120.0.6099.234" (sometimes "Chromium 120.0.6099.0")
182    let full = stdout
183        .split_whitespace()
184        .find(|tok| tok.chars().next().is_some_and(|c| c.is_ascii_digit()))
185        .ok_or_else(|| StealthError::ChromeVersionDetect(format!("no version token in: {stdout}")))?
186        .to_string();
187    let major: u32 = full
188        .split('.')
189        .next()
190        .and_then(|s| s.parse().ok())
191        .ok_or_else(|| StealthError::ChromeVersionDetect(format!("bad major in: {full}")))?;
192    Ok((major, full))
193}

195pub(crate) fn clamp_cpu_count(n: u32) -> u32 {
196    n.clamp(2, 32)
197}

199/// Detect total RAM in GB, clamped to the spec-compliant values
200/// for `navigator.deviceMemory` (capped at 8 per W3C; floor at 4 for plausibility).
201#[allow(clippy::result_large_err)]
202pub(crate) fn detect_memory_gb() -> Result<u32, StealthError> {
203    let mut sys = sysinfo::System::new();
204    sys.refresh_memory();
205    // sysinfo 0.32: total_memory() returns BYTES, not KiB. Verified against
206    // sysinfo-0.32.1/src/common/system.rs::total_memory doc comment
207    // ("Returns the RAM size in bytes.").
208    let total_bytes = sys.total_memory();
209    if total_bytes == 0 {
210        return Err(StealthError::SystemInfo("total_memory returned 0".into()));
211    }
212    let total_gb = (total_bytes / 1_073_741_824) as u32;
213    Ok(round_to_navigator_memory(total_gb))
214}
215
216fn round_to_navigator_memory(gb: u32) -> u32 {
217    // navigator.deviceMemory spec values: 0.25, 0.5, 1, 2, 4, 8. Cap at 8.
218    // We floor at 4 for plausibility (sub-4GB consumer desktops are extinct).
219    if gb >= 8 { 8 } else { 4 }
220}
