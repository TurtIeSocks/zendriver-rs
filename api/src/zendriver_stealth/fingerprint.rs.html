93/// Probed system + Chrome facts used to compose stealth values.
94#[derive(Debug, Clone, Serialize)]
95pub struct Fingerprint {
96    pub platform: Platform,
97    pub chrome_major: u32,
98    pub chrome_full: String,
99    pub cpu_count: u32,
100    pub memory_gb: u32,
101    pub ua_string: String,
102    pub ua_metadata: UserAgentMetadata,
103    pub timezone: Option<String>,
104    pub locale: Option<String>,
105    pub languages: Option<Vec<String>>,
106}

108impl Fingerprint {
109    /// Probe host system + installed Chrome to build a realistic fingerprint.
110    // `StealthError` is large because `PatchFailed` wraps `CallError` (~152B).
111    // Boxing it would cross the Task 5 file scope; bypass per-fn instead.
112    #[allow(clippy::result_large_err)]
113    pub fn auto_detect(chrome_executable: &Path) -> Result<Self, StealthError> {
114        let platform = detect_platform();
115        let (chrome_major, chrome_full) =
116            probe_chrome_version(chrome_executable).unwrap_or_else(|e| {
117                tracing::warn!("chrome version probe failed: {e}; using fallback");
118                (FALLBACK_CHROME_MAJOR, FALLBACK_CHROME_FULL.to_string())
119            });
120        let cpu_count = clamp_cpu_count(num_cpus::get() as u32);
121        let memory_gb = detect_memory_gb()?;
122        let ua_string = crate::ua::compose_ua_string(platform, &chrome_full);
123        let ua_metadata = UserAgentMetadata::realistic(platform, chrome_major, &chrome_full);
124        Ok(Self {
125            platform,
126            chrome_major,
127            chrome_full,
128            cpu_count,
129            memory_gb,
130            ua_string,
131            ua_metadata,
132            timezone: None,
133            locale: None,
134            languages: None,
135        })
136    }
137
138    /// Recompose UA string + UAM after platform/version overrides.
139    pub fn recompose(&mut self) {
140        self.ua_string = crate::ua::compose_ua_string(self.platform, &self.chrome_full);
141        self.ua_metadata =
142            UserAgentMetadata::realistic(self.platform, self.chrome_major, &self.chrome_full);
143    }
144}

171/// Parse `Google Chrome 120.0.6099.234` / `Chromium 120.0.6099.0` into
172/// `(major, full)`. Shared by the Unix probe and its tests.
173#[cfg(not(windows))]
174#[allow(clippy::result_large_err)]
175fn parse_version_banner(stdout: &str) -> Result<(u32, String), StealthError> {
176    let full = stdout
177        .split_whitespace()
178        .find(|tok| tok.chars().next().is_some_and(|c| c.is_ascii_digit()))
179        .ok_or_else(|| StealthError::ChromeVersionDetect(format!("no version token in: {stdout}")))?
180        .to_string();
181    let major: u32 = full
182        .split('.')
183        .next()
184        .and_then(|s| s.parse().ok())
185        .ok_or_else(|| StealthError::ChromeVersionDetect(format!("bad major in: {full}")))?;
186    Ok((major, full))
187}

189/// Probe the Chrome version by running `chrome --version`.
190///
191/// Unix-only *by design*. Chrome on Windows does not implement `--version` as a
192/// print-and-exit: with no existing browser session it starts a whole browser
193/// and never exits, so reading its output blocks forever. See the `cfg(windows)`
194/// twin below for what Windows does instead, and why it must not exec.
195///
196/// Bounded even here. `--version` returns in milliseconds on a healthy Unix
197/// Chrome, but this runs *synchronously inside* `Browser::launch()`'s future:
198/// an unbounded wait would wedge the poll, and a future that never yields
199/// starves the tokio timer driver — which silently disables every
200/// `tokio::time::timeout` in the process, including the ones meant to catch
201/// exactly this. A blocking call on the async path must carry its own deadline,
202/// because no outer timeout can rescue it.
203#[cfg(not(windows))]
204#[allow(clippy::result_large_err)]
205pub(crate) fn probe_chrome_version(exe: &Path) -> Result<(u32, String), StealthError> {
206    use std::io::Read;
207    use std::process::Stdio;
208    use std::time::{Duration, Instant};
209
210    /// Generous: a healthy `chrome --version` answers in ~50ms. This is a
211    /// backstop against a wedged binary, not a performance budget.
212    const PROBE_TIMEOUT: Duration = Duration::from_secs(10);
213    const POLL_INTERVAL: Duration = Duration::from_millis(20);
214
215    let mut child = Command::new(exe)
216        .arg("--version")
217        .stdin(Stdio::null())
218        .stdout(Stdio::piped())
219        .stderr(Stdio::null())
220        .spawn()
221        .map_err(|e| StealthError::ChromeVersionDetect(format!("spawn failed: {e}")))?;
222
223    let deadline = Instant::now() + PROBE_TIMEOUT;
224    let status = loop {
225        match child.try_wait() {
226            Ok(Some(status)) => break status,
227            Ok(None) => {
228                if Instant::now() >= deadline {
229                    // Reap rather than leak: kill_on_drop is not in play here.
230                    let _ = child.kill();
231                    let _ = child.wait();
232                    return Err(StealthError::ChromeVersionDetect(format!(
233                        "`--version` did not exit within {PROBE_TIMEOUT:?}"
234                    )));
235                }
236                std::thread::sleep(POLL_INTERVAL);
237            }
238            Err(e) => {
239                return Err(StealthError::ChromeVersionDetect(format!(
240                    "wait failed: {e}"
241                )));
242            }
243        }
244    };
245    if !status.success() {
246        return Err(StealthError::ChromeVersionDetect(format!(
247            "exit {:?}",
248            status.code()
249        )));
250    }
251
252    let mut stdout = String::new();
253    child
254        .stdout
255        .take()
256        .ok_or_else(|| StealthError::ChromeVersionDetect("no stdout pipe".to_string()))?
257        .read_to_string(&mut stdout)
258        .map_err(|e| StealthError::ChromeVersionDetect(format!("read stdout: {e}")))?;
259    parse_version_banner(&stdout)
260}

262/// Probe the Chrome version by reading the binary's own PE version resource.
263///
264/// **Never exec Chrome to ask its version on Windows.** `chrome.exe --version`
265/// is not the Unix print-and-exit: with no existing browser session to hand off
266/// to, it launches a full browser (GPU + network service + crashpad children)
267/// and never exits. `Command::output()` waits for the child to exit *and* for
268/// its stdout/stderr to hit EOF, so it blocks forever.
269///
270/// That block is what made this pathological rather than merely slow. The call
271/// sits synchronously inside `Browser::launch()`'s future, so the task never
272/// yields; `tokio::time::timeout` can only cancel at an await point, leaving
273/// every guard in the launch path — `WS_ENDPOINT_TIMEOUT` (15s),
274/// `HANDSHAKE_TIMEOUT` (30s) — structurally unable to fire. The result was an
275/// infinite, *silent* hang: on windows-latest CI all three
276/// `prerelease_verification` tests timed out at 360s having emitted nothing.
277///
278/// It hides on dev machines because a developer usually has Chrome already
279/// open, and `--version` then hands off to the running instance
280/// ("Opening in existing browser session.") and exits in milliseconds. It only
281/// bites where no session exists — i.e. exactly on CI.
282///
283/// A version is a static property of the file, so read it from the file. This
284/// parses the documented `VS_FIXEDFILEINFO` block, which keeps the crate's
285/// `unsafe_code = "deny"` intact and adds no dependency.
286#[cfg(windows)]
287#[allow(clippy::result_large_err)]
288pub(crate) fn probe_chrome_version(exe: &Path) -> Result<(u32, String), StealthError> {
289    let bytes = std::fs::read(exe)
290        .map_err(|e| StealthError::ChromeVersionDetect(format!("read {}: {e}", exe.display())))?;
291    parse_pe_file_version(&bytes).ok_or_else(|| {
292        StealthError::ChromeVersionDetect(format!(
293            "no VS_FIXEDFILEINFO version resource in {}",
294            exe.display()
295        ))
296    })
297}

299/// Find the first occurrence of `needle` in `haystack`.
300#[cfg(windows)]
301fn find_bytes(haystack: &[u8], needle: &[u8]) -> Option<usize> {
302    haystack.windows(needle.len()).position(|w| w == needle)
303}

305/// Extract `(major, "a.b.c.d")` from a PE image's `VS_FIXEDFILEINFO`.
306///
307/// Anchors on the UTF-16LE `VS_VERSION_INFO` key, then reads the `0xFEEF04BD`
308/// signature that opens the fixed-info block a few bytes later. Layout
309/// (little-endian): `+0` signature, `+8` dwFileVersionMS, `+12` dwFileVersionLS,
310/// each packing two 16-bit fields as `high.low`.
311///
312/// Both halves of that search are load-bearing, and real `chrome.exe` punishes
313/// getting either wrong:
314///
315/// - **Anchor on the key, not the signature.** The signature is only four bytes
316///   and does occur by chance in a multi-megabyte image: Chrome 150 carries one
317///   at file offset ~2.79M, ~475KB *before* the resource, which decodes to
318///   nonsense (`9340.36168.8075.18720`).
319/// - **Try every key match, not just the first.** Chrome's `.rdata` contains the
320///   literal text `VS_VERSION_INFO` — error strings from Chrome's *own* resource
321///   parser ("unexpected VS_VERSIONINFO in ") — ~645KB before the genuine
322///   resource. The first match is therefore a decoy with no signature after it;
323///   only a later one is real. Scanning all matches and keeping the first that
324///   is actually followed by a signature tolerates any number of such decoys.
325#[cfg(windows)]
326fn parse_pe_file_version(bytes: &[u8]) -> Option<(u32, String)> {
327    const SIGNATURE: [u8; 4] = 0xFEEF_04BDu32.to_le_bytes();
328    /// The signature sits just past the key + NUL + 32-bit alignment padding
329    /// (34 bytes in practice). Bounding the search is what makes a decoy match
330    /// fail fast and fall through to the next candidate.
331    const SIG_SEARCH_WINDOW: usize = 64;
332
333    let key: Vec<u8> = "VS_VERSION_INFO"
334        .encode_utf16()
335        .flat_map(u16::to_le_bytes)
336        .collect();
337
338    let mut search_from = 0usize;
339    while let Some(rel) = find_bytes(bytes.get(search_from..)?, &key) {
340        let anchor = search_from + rel + key.len();
341        let window_end = anchor.saturating_add(SIG_SEARCH_WINDOW).min(bytes.len());
342
343        if let Some(sig) = bytes
344            .get(anchor..window_end)
345            .and_then(|w| find_bytes(w, &SIGNATURE))
346            .map(|sig_rel| anchor + sig_rel)
347        {
348            let word = |off: usize| -> Option<u32> {
349                Some(u32::from_le_bytes(
350                    bytes.get(sig + off..sig + off + 4)?.try_into().ok()?,
351                ))
352            };
353            let ms = word(8)?;
354            let ls = word(12)?;
355            let major = ms >> 16;
356            // A zero major means we locked onto something that is not a version
357            // block; keep looking rather than reporting `0.x.y.z`.
358            if major != 0 {
359                let full = format!("{}.{}.{}.{}", major, ms & 0xFFFF, ls >> 16, ls & 0xFFFF);
360                return Some((major, full));
361            }
362        }
363        search_from += rel + 1;
364    }
365    None
366}

372/// Detect total RAM in GB, clamped to the spec-compliant values
373/// for `navigator.deviceMemory` (capped at 8 per W3C; floor at 4 for plausibility).
374#[allow(clippy::result_large_err)]
375pub(crate) fn detect_memory_gb() -> Result<u32, StealthError> {
376    let mut sys = sysinfo::System::new();
377    sys.refresh_memory();
378    // sysinfo 0.32: total_memory() returns BYTES, not KiB. Verified against
379    // sysinfo-0.32.1/src/common/system.rs::total_memory doc comment
380    // ("Returns the RAM size in bytes.").
381    let total_bytes = sys.total_memory();
382    if total_bytes == 0 {
383        return Err(StealthError::SystemInfo("total_memory returned 0".into()));
384    }
385    let total_gb = (total_bytes / 1_073_741_824) as u32;
386    Ok(round_to_navigator_memory(total_gb))
387}
388
389fn round_to_navigator_memory(gb: u32) -> u32 {
390    // navigator.deviceMemory spec values: 0.25, 0.5, 1, 2, 4, 8. Cap at 8.
391    // We floor at 4 for plausibility (sub-4GB consumer desktops are extinct).
392    if gb >= 8 { 8 } else { 4 }
393}
