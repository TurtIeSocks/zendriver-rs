17/// UA string + UA-CH metadata override.
18#[derive(Debug, Clone, PartialEq, Default, Serialize, Deserialize)]
19pub struct UaSpec {
20    pub ua_string: Option<String>,
21    /// Free-form UA-CH overrides; merged onto realistic() output at resolve.
22    pub platform: Option<String>,
23    /// Full UA-CH (`Emulation.setUserAgentOverride.userAgentMetadata`)
24    /// override, field-wise. Any sub-field left `None` falls back to the
25    /// fingerprint-derived value at resolve (see [`UaMetadata::resolve`]).
26    /// JSON key is `userAgentMetadata` to match the wire shape callers
27    /// harvest it from (e.g. `navigator.userAgentData.getHighEntropyValues`).
28    #[serde(rename = "userAgentMetadata")]
29    pub ua_metadata: Option<UaMetadata>,
30}
31
32impl UaSpec {
33    /// Field-wise merge: `Some` in `over` wins, `None` inherits from `self`.
34    /// `ua_metadata` recurses into [`UaMetadata::overlay`] when both sides
35    /// carry one, so layering two personas composes UA-CH sub-fields instead
36    /// of one side wholesale-replacing the other.
37    #[must_use]
38    pub fn overlay(self, over: UaSpec) -> UaSpec {
39        UaSpec {
40            ua_string: over.ua_string.or(self.ua_string),
41            platform: over.platform.or(self.platform),
42            ua_metadata: match (self.ua_metadata, over.ua_metadata) {
43                (Some(base), Some(add)) => Some(base.overlay(add)),
44                (base, add) => add.or(base),
45            },
46        }
47    }
48}

50/// Field-wise `userAgentMetadata` (UA-CH) override. Mirrors
51/// [`UserAgentMetadata`] but every field is optional: unset fields fall back
52/// to the fingerprint-derived value at resolve (see [`Self::resolve`]).
53/// `wow64` is intentionally absent — it has no independent override surface
54/// here and always comes from the derived base.
55#[derive(Debug, Clone, PartialEq, Default, Serialize, Deserialize)]
56pub struct UaMetadata {
57    pub brands: Option<Vec<Brand>>,
58    #[serde(rename = "fullVersionList")]
59    pub full_version_list: Option<Vec<Brand>>,
60    pub platform: Option<String>,
61    #[serde(rename = "platformVersion")]
62    pub platform_version: Option<String>,
63    pub architecture: Option<String>,
64    pub bitness: Option<String>,
65    pub mobile: Option<bool>,
66    pub model: Option<String>,
67}
68
69impl UaMetadata {
70    /// Field-wise merge: `Some` in `over` wins, `None` inherits from `self`.
71    #[must_use]
72    pub fn overlay(self, over: UaMetadata) -> UaMetadata {
73        UaMetadata {
74            brands: over.brands.or(self.brands),
75            full_version_list: over.full_version_list.or(self.full_version_list),
76            platform: over.platform.or(self.platform),
77            platform_version: over.platform_version.or(self.platform_version),
78            architecture: over.architecture.or(self.architecture),
79            bitness: over.bitness.or(self.bitness),
80            mobile: over.mobile.or(self.mobile),
81            model: over.model.or(self.model),
82        }
83    }
84
85    /// Resolve into a complete [`UserAgentMetadata`], filling any unset
86    /// sub-field from `base` (the fingerprint-derived value). `wow64` always
87    /// comes from `base` — it has no override surface on [`UaMetadata`].
88    #[must_use]
89    pub fn resolve(&self, base: &UserAgentMetadata) -> UserAgentMetadata {
90        UserAgentMetadata {
91            brands: self.brands.clone().unwrap_or_else(|| base.brands.clone()),
92            full_version_list: self
93                .full_version_list
94                .clone()
95                .unwrap_or_else(|| base.full_version_list.clone()),
96            platform: self
97                .platform
98                .clone()
99                .unwrap_or_else(|| base.platform.clone()),
100            platform_version: self
101                .platform_version
102                .clone()
103                .unwrap_or_else(|| base.platform_version.clone()),
104            architecture: self
105                .architecture
106                .clone()
107                .unwrap_or_else(|| base.architecture.clone()),
108            bitness: self.bitness.clone().unwrap_or_else(|| base.bitness.clone()),
109            wow64: base.wow64,
110            mobile: self.mobile.unwrap_or(base.mobile),
111            model: self.model.clone().unwrap_or_else(|| base.model.clone()),
112        }
113    }
114}

136/// WebGPU adapter value substitution + opt-in synthetic-adapter fabrication.
137///
138/// `None`-everywhere ([`WebgpuSpec::default`]) is the regression guard: it
139/// behaves **byte-for-byte** like the pre-`WebgpuSpec` `SurfaceCfg` did — a
140/// real `navigator.gpu` adapter's `.info` is decorated with a vendor /
141/// architecture DERIVED from the [`WebglSpec`] renderer (never fabricated),
142/// `device` / `description` are emitted empty (Chrome masks them), `.limits`
143/// / `.features` are left untouched, and a GPU-less host is left alone —
144/// `navigator.gpu.requestAdapter()` still resolves `null`, same as native
145/// Chrome.
146///
147/// # You own value accuracy
148///
149/// Every field below is **caller-supplied** — nothing here is probed or
150/// invented from a real GPU. A `vendor` / `limits` / `features` combination
151/// that does not correspond to any real device is **more detectable than
152/// leaving the field `None`**: fingerprinting scripts cross-check
153/// `GPUAdapterInfo` against `GPUSupportedLimits` / `GPUSupportedFeatures` and
154/// against the WebGL renderer string, so an incoherent combination reads as a
155/// bot faster than honest absence does. Only set these fields to values
156/// you've verified against a real device (e.g. probed live from
157/// `navigator.gpu` on that device) — never invent plausible-looking numbers.
158///
159/// # What fabrication does
160///
161/// [`fabricate_when_absent`](Self::fabricate_when_absent) covers **both**
162/// GPU-less shapes:
163/// - **`navigator.gpu` entirely absent** (`'gpu' in navigator === false` —
164///   the common case, e.g. Chrome launched with `--disable-gpu`, which is
165///   zendriver's default under headless): a synthetic `navigator.gpu` is
166///   defined on `Navigator.prototype` whose `requestAdapter()` resolves the
167///   synthetic adapter. This flips `'gpu' in navigator` to **true**, which is
168///   coherent for a modern-Chrome persona — real modern Chrome always exposes
169///   `navigator.gpu` even with no usable GPU (there `requestAdapter()` merely
170///   resolves `null`). Restoring that presence is your explicit opt-in.
171/// - **`navigator.gpu` present but `requestAdapter()` resolves `null`**: the
172///   real `requestAdapter` is wrapped so a `null`/rejected result falls back
173///   to the synthetic adapter (a real adapter passes through untouched).
174///
175/// # v1 limitations
176///
177/// - `navigator.gpu`, the fabricated adapter, and its `.info` inherit the real
178///   `GPU` / `GPUAdapter` / `GPUAdapterInfo` prototypes (or a synthesized
179///   same-named constructor when the WebGPU IDL is absent), so `instanceof`
180///   holds for all three. Their `.limits` / `.features` are still a plain
181///   object / `Set`, not real `GPUSupportedLimits` / `GPUSupportedFeatures`
182///   instances — an `instanceof` check on those two still tells.
183/// - [`fabricate_when_absent`](Self::fabricate_when_absent)'s synthetic
184///   adapter's `requestDevice()` **always rejects**. Faking a working
185///   `GPUDevice` needs a real GPU behind it, which this patch cannot
186///   provide — it only makes `navigator.gpu.requestAdapter()` resolve a
187///   coherent adapter for detection scripts that stop at the adapter, it
188///   does not unlock actual WebGPU rendering on a GPU-less host.
189///
190/// ```no_run
191/// use std::collections::BTreeMap;
192/// use zendriver_stealth::{Persona, WebgpuSpec};
193///
194/// // Decorate a REAL adapter with values probed from an actual device.
195/// let persona = Persona {
196///     webgpu: Some(WebgpuSpec {
197///         vendor: Some("apple".into()),
198///         architecture: Some("metal-3".into()),
199///         ..Default::default()
200///     }),
201///     ..Persona::default()
202/// };
203///
204/// // Opt-in fabrication on a GPU-less host: requires an explicit vendor AND
205/// // limits (see `fabricate_when_absent` below) — anything less is refused.
206/// let mut limits = BTreeMap::new();
207/// limits.insert("maxTextureDimension2D".to_string(), 16384);
208/// let fabricated = WebgpuSpec {
209///     vendor: Some("apple".into()),
210///     architecture: Some("metal-3".into()),
211///     limits: Some(limits),
212///     features: Some(vec!["texture-compression-bc".into()]),
213///     fabricate_when_absent: Some(true),
214///     ..Default::default()
215/// };
216/// let _ = fabricated;
217/// ```
218#[derive(Debug, Clone, PartialEq, Default, Serialize, Deserialize)]
219pub struct WebgpuSpec {
220    /// Render strategy for this surface. `None` → kind default (`Value`).
221    pub strategy: Option<Strategy>,
222    /// `GPUAdapterInfo.vendor`. `None` → derived from the [`WebglSpec`]
223    /// renderer (today's behavior, via the internal `adapter_for_renderer`
224    /// dataset mapping).
225    pub vendor: Option<String>,
226    /// `GPUAdapterInfo.architecture`. `None` → derived from the WebGL
227    /// renderer, same as [`vendor`](Self::vendor).
228    pub architecture: Option<String>,
229    /// `GPUAdapterInfo.device`. `None` → `""` (real Chrome masks this field).
230    pub device: Option<String>,
231    /// `GPUAdapterInfo.description`. `None` → `""` (real Chrome masks this
232    /// field too).
233    pub description: Option<String>,
234    /// Caller-supplied `GPUSupportedLimits` caps (e.g.
235    /// `"maxTextureDimension2D"`). Applied to a real (or fabricated) adapter's
236    /// `.limits` getter. `None` → the adapter's own limits are left
237    /// untouched.
238    pub limits: Option<BTreeMap<String, u64>>,
239    /// Caller-supplied `GPUSupportedFeatures` strings (e.g.
240    /// `"texture-compression-bc"`). `None` → the adapter's own features are
241    /// left untouched.
242    pub features: Option<Vec<String>>,
243    /// Explicit opt-in: synthesize a `navigator.gpu` adapter on a host with no
244    /// real one, instead of leaving `requestAdapter()` at `null` (or
245    /// `navigator.gpu` entirely absent). Covers both GPU-less shapes — a
246    /// missing `navigator.gpu` is defined fresh (flipping `'gpu' in navigator`
247    /// to true), and a present-but-null `requestAdapter` is wrapped; see the
248    /// "What fabrication does" section above. Requires [`vendor`](Self::vendor)
249    /// AND [`limits`](Self::limits) to BOTH be explicitly set — a bare `true`
250    /// with nothing else is refused (silently, no-op) because there is nothing
251    /// coherent to fabricate; this project never auto-invents fingerprint
252    /// values. See the v1 limitations above for what fabrication does NOT
253    /// cover (no working `GPUDevice`; `limits`/`features` `instanceof`).
254    pub fabricate_when_absent: Option<bool>,
255}
