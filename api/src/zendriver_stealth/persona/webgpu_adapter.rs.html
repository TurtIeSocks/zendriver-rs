1//! Derive a coherent WebGPU adapter (vendor + architecture) from the spoofed
2//! WebGL renderer string. Dataset-mapped, deterministic — NEVER randomized
3//! (WAFs hash the WebGPU fingerprint; a random/unknown value reads as a bot).
4//!
5//! Architecture tokens come from Dawn's `gpu_info.json`, normalized
6//! (lowercase, spaces→hyphens) — the scheme Chrome's WebGPU backend uses for
7//! `GPUAdapterInfo.architecture`. Validated: Apple M4 Pro → "metal-3";
8//! NVIDIA Turing → "turing" (MDN). Only confident model→µarch mappings emit a
9//! token; unrecognized models get "" — Chrome legitimately returns "" for
10//! unclassified GPUs, so empty is coherent and safe. A WRONG token reads as an
11//! unknown device to a fingerprinting WAF.
12
13/// vendor + architecture for the spoofed WebGPU adapter. `device` and
14/// `description` are always emitted empty by the patch (Chrome masks them).
15#[derive(Debug, Clone, PartialEq, Eq)]
16pub(crate) struct GpuAdapterInfo {
17    pub vendor: String,
18    pub architecture: String,
19}
20
21/// Map a WebGL `UNMASKED_RENDERER` string to a coherent WebGPU adapter
22/// (vendor + architecture). Architecture tokens are Dawn's `gpu_info.json`
23/// names normalized (lowercase, spaces→hyphens) — the scheme Chrome's WebGPU
24/// backend uses for `GPUAdapterInfo.architecture` (validated: Apple M4 Pro →
25/// "metal-3"; NVIDIA Turing → "turing" per MDN). Only confident model→µarch
26/// mappings emit a token; unrecognized models get "" (Chrome legitimately
27/// returns "" for unclassified GPUs, so empty is coherent and safe — a WRONG
28/// token reads as an unknown device to a fingerprinting WAF). Add families
29/// here as they're confirmed.
30pub(crate) fn adapter_for_renderer(renderer: &str) -> GpuAdapterInfo {
31    let r = renderer.to_ascii_lowercase();
32
33    if r.contains("nvidia") || r.contains("geforce") || r.contains("rtx") || r.contains("gtx") {
34        let arch = if r.contains("rtx 50") || r.contains("rtx50") {
35            "blackwell"
36        } else if r.contains("rtx 40") || r.contains("rtx40") {
37            "lovelace"
38        } else if r.contains("rtx 30") || r.contains("rtx30") {
39            "ampere"
40        } else if r.contains("rtx 20")
41            || r.contains("rtx20")
42            || r.contains("rtx 16")
43            || r.contains("gtx 16")
44            || r.contains("gtx16")
45        {
46            "turing"
47        } else if r.contains("gtx 10") || r.contains("gtx10") {
48            "pascal"
49        } else if r.contains("titan v") {
50            "volta"
51        } else {
52            ""
53        };
54        return GpuAdapterInfo {
55            vendor: "nvidia".into(),
56            architecture: arch.into(),
57        };
58    }
59
60    if r.contains("amd") || r.contains("radeon") {
61        let arch = if r.contains("rx 9") || r.contains("rx9") {
62            "rdna-4"
63        } else if r.contains("rx 7") || r.contains("rx7") {
64            "rdna-3"
65        } else if r.contains("rx 6") || r.contains("rx6") {
66            "rdna-2"
67        } else if r.contains("rx 5700") || r.contains("rx 5600") || r.contains("rx 5500") {
68            "rdna-1"
69        } else {
70            ""
71        };
72        return GpuAdapterInfo {
73            vendor: "amd".into(),
74            architecture: arch.into(),
75        };
76    }
77
78    if r.contains("apple") {
79        return GpuAdapterInfo {
80            vendor: "apple".into(),
81            architecture: "metal-3".into(),
82        };
83    }
84
85    // Intel + everything unrecognized → Intel (the common integrated default).
86    let arch = if r.contains("iris xe") || r.contains("xe graphics") {
87        "gen-12-lp"
88    } else if r.contains("uhd graphics 6")
89        || r.contains("hd graphics 6")
90        || r.contains("uhd graphics 5")
91        || r.contains("hd graphics 5")
92    {
93        "gen-9"
94    } else {
95        ""
96    };
97    GpuAdapterInfo {
98        vendor: "intel".into(),
99        architecture: arch.into(),
100    }
101}

103#[cfg(test)]
104mod tests {
105    use super::*;
106
107    #[test]
108    fn nvidia_rtx_generations() {
109        assert_eq!(
110            adapter_for_renderer(
111                "ANGLE (NVIDIA, NVIDIA GeForce RTX 4090 Direct3D11 vs_5_0 ps_5_0, D3D11)"
112            )
113            .architecture,
114            "lovelace"
115        );
116        assert_eq!(
117            adapter_for_renderer("NVIDIA GeForce RTX 3080").architecture,
118            "ampere"
119        );
120        assert_eq!(
121            adapter_for_renderer("NVIDIA GeForce RTX 2070 SUPER").architecture,
122            "turing"
123        );
124        assert_eq!(
125            adapter_for_renderer("NVIDIA GeForce GTX 1080 Ti").architecture,
126            "pascal"
127        );
128        let a = adapter_for_renderer("NVIDIA GeForce RTX 4090");
129        assert_eq!(a.vendor, "nvidia");
130    }
131
132    #[test]
133    fn nvidia_unknown_generation_is_empty() {
134        // Vendor still nvidia, but an unrecognized model → empty arch (safe).
135        assert_eq!(adapter_for_renderer("NVIDIA Quadro K2200").architecture, "");
136    }
137
138    #[test]
139    fn amd_rx_generations() {
140        assert_eq!(
141            adapter_for_renderer("ANGLE (AMD, AMD Radeon RX 7900 XT, D3D11)").architecture,
142            "rdna-3"
143        );
144        assert_eq!(
145            adapter_for_renderer("AMD Radeon RX 6800 XT").architecture,
146            "rdna-2"
147        );
148        assert_eq!(
149            adapter_for_renderer("AMD Radeon RX 5700 XT").architecture,
150            "rdna-1"
151        );
152        assert_eq!(adapter_for_renderer("AMD Radeon RX 7900 XT").vendor, "amd");
153    }
154
155    #[test]
156    fn apple_is_metal3() {
157        let a = adapter_for_renderer(
158            "ANGLE (Apple, ANGLE Metal Renderer: Apple M4 Pro, Unspecified Version)",
159        );
160        assert_eq!(a.vendor, "apple");
161        assert_eq!(a.architecture, "metal-3");
162    }
163
164    #[test]
165    fn intel_integrated_generations() {
166        assert_eq!(
167            adapter_for_renderer(
168                "ANGLE (Intel, Intel(R) UHD Graphics 630 Direct3D11 vs_5_0 ps_5_0, D3D11)"
169            )
170            .architecture,
171            "gen-9"
172        );
173        assert_eq!(
174            adapter_for_renderer("ANGLE (Intel, Intel(R) Iris(R) Xe Graphics, D3D11)").architecture,
175            "gen-12-lp"
176        );
177        assert_eq!(
178            adapter_for_renderer("Intel(R) UHD Graphics 630").vendor,
179            "intel"
180        );
181    }
182
183    #[test]
184    fn unknown_renderer_falls_back_to_intel_empty_arch() {
185        let a = adapter_for_renderer("Mesa OffScreen");
186        assert_eq!(a.vendor, "intel");
187        assert_eq!(a.architecture, "");
188    }
