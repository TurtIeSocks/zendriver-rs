9/// Stealth modes shipped by the library.
10#[derive(Debug, Clone, Copy, PartialEq, Eq)]
11pub enum ProfileKind {
12    /// No stealth applied. Browser is launched stock; no JS patches, no UA scrub.
13    Off,
14    /// Launch flags + UA scrub (HeadlessChrome → Chrome). No JS bootstrap.
15    /// Safe against `Function.prototype.toString` detection. Default.
16    Native,
17    /// Native + Navigator-prototype JS patches. Passes sannysoft. Detectable
18    /// by sophisticated bots that probe `toString` on Navigator getters.
19    Spoofed,
20}

22/// JS `navigator.platform` value.
23#[derive(Debug, Clone, Copy, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
24pub enum Platform {
25    Win32,
26    MacIntel,
27    LinuxX86_64,
28}
29
30impl Platform {
31    /// Map to the `navigator.platform` string Chrome reports for that OS.
32    #[must_use]
33    pub fn js_string(self) -> &'static str {
34        match self {
35            Platform::Win32 => "Win32",
36            Platform::MacIntel => "MacIntel",
37            Platform::LinuxX86_64 => "Linux x86_64",
38        }
39    }
40
41    /// CDP `userAgentMetadata.platform` value (no version).
42    #[must_use]
43    pub fn ch_platform(self) -> &'static str {
44        match self {
45            Platform::Win32 => "Windows",
46            Platform::MacIntel => "macOS",
47            Platform::LinuxX86_64 => "Linux",
48        }
49    }
50
51    /// UA-string OS token (the bit inside parentheses).
52    #[must_use]
53    pub fn ua_token(self) -> &'static str {
54        match self {
55            Platform::Win32 => "Windows NT 10.0; Win64; x64",
56            Platform::MacIntel => "Macintosh; Intel Mac OS X 10_15_7",
57            Platform::LinuxX86_64 => "X11; Linux x86_64",
58        }
59    }
60}

75/// Stealth configuration passed to `BrowserBuilder::stealth(...)`.
76#[derive(Debug, Clone)]
77pub struct StealthProfile {
78    pub(crate) kind: ProfileKind,
79    pub(crate) extra_flags: Vec<String>,
80    pub(crate) fingerprint_override: Option<Fingerprint>,
81    pub(crate) per_field: PerFieldOverride,
82    pub(crate) bypass_csp: bool,
83    // Wired by `BrowserBuilder::stealth` in Task 17.
84    #[allow(dead_code)]
85    pub(crate) user_data_dir: Option<PathBuf>,

88impl StealthProfile {
89    /// No stealth: stock browser launch.
90    ///
91    /// Use this when you want a bare-bones Chrome with none of the launch
92    /// flags or CDP overrides applied — e.g. when verifying that a problem
93    /// reproduces in vanilla Chrome.
94    ///
95    /// ```
96    /// use zendriver_stealth::StealthProfile;
97    /// let p = StealthProfile::off();
98    /// assert!(p.build_flags().is_empty());
99    /// ```
100    #[must_use]
101    pub fn off() -> Self {
102        Self {
103            kind: ProfileKind::Off,
104            extra_flags: Vec::new(),
105            fingerprint_override: None,
106            per_field: PerFieldOverride::default(),
107            bypass_csp: false,
108            user_data_dir: None,
109        }
110    }
111
112    /// Launch flags + UA scrub + Emulation overrides. No JS bootstrap.
113    ///
114    /// Safe against `Function.prototype.toString` detection (it doesn't
115    /// patch any prototype getter). The default when stealth is requested.
116    ///
117    /// ```
118    /// use zendriver_stealth::StealthProfile;
119    /// let p = StealthProfile::native();
120    /// assert!(!p.build_flags().is_empty());
121    /// ```
122    #[must_use]
123    pub fn native() -> Self {
124        Self {
125            kind: ProfileKind::Native,
126            extra_flags: Vec::new(),
127            fingerprint_override: None,
128            per_field: PerFieldOverride::default(),
129            bypass_csp: false,
130            user_data_dir: None,
131        }
132    }
133
134    /// `native` + Navigator-prototype JS patches. Passes the sannysoft
135    /// detection battery.
136    ///
137    /// Sets [`bypass_csp`](Self::bypass_csp) on by default so the bootstrap
138    /// script can install on pages with strict CSP headers.
139    ///
140    /// ```
141    /// use zendriver_stealth::StealthProfile;
142    /// let p = StealthProfile::spoofed();
143    /// assert!(p.bypass_csp_enabled());
144    /// ```
145    #[must_use]
146    pub fn spoofed() -> Self {
147        Self {
148            kind: ProfileKind::Spoofed,
149            extra_flags: Vec::new(),
150            fingerprint_override: None,
151            per_field: PerFieldOverride::default(),
152            bypass_csp: true, // default ON for spoofed; see spec assumption #2
153            user_data_dir: None,
154        }
155    }
156
157    /// Override the auto-detected [`Fingerprint`] wholesale.
158    ///
159    /// Use this when you need to pin a specific Chrome version / platform /
160    /// hardware combination across runs (e.g. to keep request fingerprints
161    /// stable across CI invocations).
162    #[must_use]
163    pub fn fingerprint(mut self, f: Fingerprint) -> Self {
164        self.fingerprint_override = Some(f);
165        self
166    }
167    /// Override the reported `navigator.deviceMemory` (in GB).
168    ///
169    /// Per the [HTML Device Memory spec][spec] the value Chrome reports
170    /// is one of `{1, 2, 4, 8}` (Chrome does not expose fractional values
171    /// to JS). The resolver snaps `gb` to the nearest valid integer at
172    /// or below it:
173    ///
174    /// | Input `gb` | Reported |
175    /// |-----------:|---------:|
176    /// | `0` or `1` | `1`      |
177    /// | `2` or `3` | `2`      |
178    /// | `4`–`7`    | `4`      |
179    /// | `8`+       | `8`      |
180    ///
181    /// Snap-down (not nearest-round) matches Chrome's own behavior and
182    /// avoids inflating the reported value above what the host plausibly
183    /// has. Anything outside `{1, 2, 4, 8}` is itself a stealth tell
184    /// because real browsers never report it.
185    ///
186    /// [spec]: https://www.w3.org/TR/device-memory/
187    #[must_use]
188    pub fn memory_gb(mut self, gb: u32) -> Self {
189        self.per_field.memory_gb = Some(gb);
190        self
191    }
192    /// Override the reported `navigator.hardwareConcurrency` (CPU count).
193    ///
194    /// Clamped to `2..=32` at resolve time — values outside that range are
195    /// implausibly low/high and trip simple heuristics.
196    #[must_use]
197    pub fn cpu_count(mut self, n: u32) -> Self {
198        self.per_field.cpu_count = Some(n);
199        self
200    }
201    /// Override the reported Chrome major version (e.g. `125`).
202    #[must_use]
203    pub fn chrome_version(mut self, major: u32) -> Self {
204        self.per_field.chrome_major = Some(major);
205        self
206    }
207    /// Override the reported [`Platform`] (`navigator.platform` + UA OS
208    /// token + UAM `platform`).
209    #[must_use]
210    pub fn platform(mut self, p: Platform) -> Self {
211        self.per_field.platform = Some(p);
212        self
213    }
214    /// Override the reported locale (e.g. `"en-US"`, `"fr-FR"`).
215    ///
216    /// Sends `Emulation.setLocaleOverride` and adds `--lang=...` to the
217    /// launch flags.
218    #[must_use]
219    pub fn locale(mut self, l: impl Into<String>) -> Self {
220        self.per_field.locale = Some(l.into());
221        self
222    }
223    /// Override the reported language list (drives `navigator.languages` +
224    /// q-weighted `Accept-Language`). When unset, derived from
225    /// [`locale`](Self::locale).
226    #[must_use]
227    pub fn languages(mut self, langs: impl IntoIterator<Item = String>) -> Self {
228        self.per_field.languages = Some(langs.into_iter().collect());
229        self
230    }
231    /// Override the reported timezone via `Emulation.setTimezoneOverride`
232    /// (IANA name, e.g. `"America/Los_Angeles"`).
233    #[must_use]
234    pub fn timezone(mut self, tz: impl Into<String>) -> Self {
235        self.per_field.timezone = Some(tz.into());
236        self
237    }
238    /// Override the reported User-Agent string verbatim.
239    ///
240    /// Skips the composed-from-fingerprint step — prefer
241    /// [`platform`](Self::platform) + [`chrome_version`](Self::chrome_version)
242    /// unless you need an exact UA string.
243    #[must_use]
244    pub fn user_agent(mut self, ua: impl Into<String>) -> Self {
245        self.per_field.ua_string = Some(ua.into());
246        self
247    }
248    /// Toggle `Page.setBypassCSP`. Default `true` for [`spoofed`](Self::spoofed),
249    /// `false` for [`native`](Self::native) / [`off`](Self::off).
250    #[must_use]
251    pub fn bypass_csp(mut self, on: bool) -> Self {
252        self.bypass_csp = on;
253        self
254    }
255    /// Add a single extra Chrome launch flag (e.g. `"--proxy-server=..."`).
256    #[must_use]
257    pub fn arg(mut self, flag: impl Into<String>) -> Self {
258        self.extra_flags.push(flag.into());
259        self
260    }
261    /// Add a batch of Chrome launch flags.
262    #[must_use]
263    pub fn args(mut self, flags: impl IntoIterator<Item = String>) -> Self {
264        self.extra_flags.extend(flags);
265        self
266    }

274    /// Resolve the final [`Fingerprint`]: either the explicit override or an
275    /// auto-detected baseline, with per-field tweaks (`platform`, `locale`,
276    /// `memory_gb`, …) applied on top.
277    ///
278    /// `chrome_exe` is probed for its Chrome major: by reading the binary's PE
279    /// version resource on Windows, and by running `--version` on Unix (Windows
280    /// Chrome answers `--version` by launching a browser that never exits, so it
281    /// must not be executed there). Either probe failing falls back to a
282    /// baked-in default, so the resolver never errors solely on Chrome being
283    /// unavailable.
284    ///
285    /// # Errors
286    /// Returns [`StealthError::ChromeVersionDetect`] when the Chrome probe
287    /// fails *and* no override is provided, and [`StealthError::SystemInfo`]
288    /// when total-RAM detection fails.
289    // `StealthError` is large because `PatchFailed` wraps `CallError` (~152B).
290    // Boxing it would cross the Task 5 file scope; bypass per-fn instead.
291    #[allow(clippy::result_large_err)]
292    pub fn resolve_fingerprint(&self, chrome_exe: &Path) -> Result<Fingerprint, StealthError> {
293        let mut fp = match &self.fingerprint_override {
294            Some(fp) => fp.clone(),
295            None => Fingerprint::auto_detect(chrome_exe)?,
296        };
297        // The Accept-Encoding the *real* binary advertises, captured before the
298        // claimed-major override below (used only to warn on an uncorrectable skew).
299        let binary_major = fp.chrome_major;
300        if let Some(p) = self.per_field.platform {
301            fp.platform = p;
302        }
303        if let Some(c) = self.per_field.chrome_major {
304            fp.chrome_major = c;
305            fp.chrome_full = format!("{c}.0.6099.234"); // synthesize a full version if user only set major
306        }
307        if let Some(n) = self.per_field.cpu_count {
308            fp.cpu_count = n.clamp(2, 32);
309        }
310        if let Some(g) = self.per_field.memory_gb {
311            // Snap-down to the nearest spec-valid value Chrome exposes
312            // to JS — see `memory_gb` doc table.
313            let snapped = match g {
314                0..=1 => 1,
315                2..=3 => 2,
316                4..=7 => 4,
317                _ => 8,
318            };
319            if snapped != g {
320                tracing::debug!(
321                    requested = g,
322                    chosen = snapped,
323                    "stealth memory_gb snapped to nearest navigator.deviceMemory spec value",
324                );
325            }
326            fp.memory_gb = snapped;
327        }
328        // Always recompose so `ua_metadata.{platform_version, architecture,
329        // bitness}` track any `platform` / `chrome_major` overrides applied
330        // above. Then, if the user supplied an explicit UA string, replace
331        // the freshly composed `ua_string` with it (UAM remains coherent
332        // with the overridden platform).
333        fp.recompose();
334        if let Some(ref ua) = self.per_field.ua_string {
335            fp.ua_string = ua.clone();
336        }
337        if let Some(ref tz) = self.per_field.timezone {
338            fp.timezone = Some(tz.clone());
339        }
340        if let Some(ref locale) = self.per_field.locale {
341            fp.locale = Some(locale.clone());
342        }
343        if let Some(ref langs) = self.per_field.languages {
344            fp.languages = Some(langs.clone());
345        }
346        // Accept-Encoding coherence is observable but NOT correctable over CDP:
347        // Chrome's network service owns the header and ignores
348        // `Network.setExtraHTTPHeaders` for it (verified, Chrome 148). When a
349        // pinned `chrome_version` straddles the `zstd`/Chrome-123 boundary vs the
350        // launched binary, the request advertises the binary's encodings — an
351        // Accept-Encoding vs User-Agent mismatch we can only warn about.
352        if let Some(coherent) = crate::headers::accept_encoding_skew(binary_major, fp.chrome_major)
353        {
354            tracing::warn!(
355                claimed_major = fp.chrome_major,
356                binary_major,
357                coherent_accept_encoding = coherent,
358                "stealth: claimed Chrome major advertises a different Accept-Encoding \
359                 than the launched binary; Chrome controls this header and CDP cannot \
360                 override it, so requests will leak the binary's encodings — pin \
361                 chrome_version to the binary's major to stay coherent",
362            );
363        }
364        Ok(fp)
365    }

367    /// Composed Chrome launch flag list: per-profile defaults plus any
368    /// extras added via [`arg`](Self::arg) / [`args`](Self::args), with a
369    /// `--lang=<locale>` flag injected when [`locale`](Self::locale) is set.
370    ///
371    /// ```
372    /// use zendriver_stealth::StealthProfile;
373    /// let flags = StealthProfile::native().locale("fr-FR").build_flags();
374    /// assert!(flags.iter().any(|f| f == "--lang=fr-FR"));
375    /// ```
376    pub fn build_flags(&self) -> Vec<String> {
377        let mut flags = crate::flags::flags_for_profile(self.kind);
378        if let Some(ref locale) = self.per_field.locale {
379            flags.push(format!("--lang={locale}"));
380        }
381        flags.extend(self.extra_flags.iter().cloned());
382        flags
383    }
384
385    /// Whether `Page.setBypassCSP` will be sent for this profile. Defaults
386    /// to `true` for [`spoofed`](Self::spoofed) and `false` otherwise; the
387    /// [`bypass_csp`](Self::bypass_csp) setter toggles it explicitly.
388    pub fn bypass_csp_enabled(&self) -> bool {
389        self.bypass_csp
390    }

392    /// Returns the input-realism profile appropriate for this stealth profile.
393    /// `spoofed` returns realistic timings; `native` and `off` return zero-overhead.
394    #[must_use]
395    pub fn input_profile(&self) -> crate::InputProfile {
396        match self.kind {
397            ProfileKind::Spoofed => crate::InputProfile::spoofed(),
398            ProfileKind::Native | ProfileKind::Off => crate::InputProfile::native(),
399        }
400    }
