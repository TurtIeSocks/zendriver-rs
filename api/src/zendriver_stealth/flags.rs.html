8/// Flags ALL stealth profiles share (Native + Spoofed + Off-when-not-Off).
9/// Off profile uses an empty list (truly stock launch).
10///
11/// `native_isolation` selects the `--disable-features=...` entry: `false`
12/// (the default path) disables Chrome's render-process site isolation
13/// (`IsolateOrigins`/`site-per-process`) as today; `true` leaves those two
14/// features enabled — Chrome's stock site isolation stays on — while still
15/// disabling the unrelated `DisableLoadExtensionCommandLineSwitch` feature
16/// (needed for `--load-extension` regardless of the isolation choice). See
17/// [`StealthProfile::native_isolation`](crate::StealthProfile::native_isolation)
18/// for the caller-facing opt-in and its trade-off.
19fn shared_stealth_flags(native_isolation: bool) -> Vec<String> {
20    let disable_features = if native_isolation {
21        "--disable-features=DisableLoadExtensionCommandLineSwitch".to_string()
22    } else {
23        "--disable-features=IsolateOrigins,DisableLoadExtensionCommandLineSwitch,site-per-process"
24            .to_string()
25    };
26    vec![
27        "--no-first-run".into(),
28        "--no-service-autorun".into(),
29        "--no-default-browser-check".into(),
30        "--homepage=about:blank".into(),
31        "--no-pings".into(),
32        "--password-store=basic".into(),
33        "--disable-infobars".into(),
34        "--disable-breakpad".into(),
35        "--disable-component-update".into(),
36        "--disable-backgrounding-occluded-windows".into(),
37        "--disable-renderer-backgrounding".into(),
38        "--disable-background-networking".into(),
39        "--disable-dev-shm-usage".into(),
40        disable_features,
41        "--disable-session-crashed-bubble".into(),
42        "--disable-search-engine-choice-screen".into(),
43        "--remote-allow-origins=*".into(),
44        // Stop Blink injecting `navigator.webdriver` (native getter → `true`).
45        // py-zendriver sets this by default (config.py); ALL non-Off profiles
46        // need it. Previously it lived only in the Spoofed branch, so the
47        // Native profile leaked `navigator.webdriver === true` — an instant
48        // bot tell. The Spoofed JS webdriver shim also relies on this flag so
49        // Chrome's AutomationControlled hook can't re-inject over the shim.
50        "--disable-blink-features=AutomationControlled".into(),
51        // WebRTC IP-leak prevention (zendriver Python disable_webrtc=True default)
52        "--webrtc-ip-handling-policy=disable_non_proxied_udp".into(),
53        "--force-webrtc-ip-handling-policy".into(),
54    ]
55}

57/// Build the full flag list for a profile.
58///
59/// `native_isolation` is the opt-in from
60/// [`StealthProfile::native_isolation`](crate::StealthProfile::native_isolation):
61/// `false` (the default) is today's behavior, byte-identical for every
62/// [`ProfileKind`]. `true` omits the site-isolation-disabling feature names
63/// from `--disable-features=...`, leaving Chrome's real
64/// `IsolateOrigins`/`site-per-process` behavior in place.
65#[must_use]
66pub fn flags_for_profile(kind: ProfileKind, native_isolation: bool) -> Vec<String> {
67    match kind {
68        ProfileKind::Off => Vec::new(),
69        ProfileKind::Native => shared_stealth_flags(native_isolation),
70        ProfileKind::Spoofed => {
71            let mut v = shared_stealth_flags(native_isolation);
72            // (`--disable-blink-features=AutomationControlled` lives in
73            // `shared_stealth_flags` — both Native and Spoofed need it.)
74            // SwiftShader supplies a WebGL CONTEXT in headless. Chrome runs
75            // headless with `--headless=new --disable-gpu` (browser.rs), so with
76            // no software backend `canvas.getContext('webgl')` returns null —
77            // itself a bot tell (real browsers always have WebGL). This is
78            // about the WebGL CONTEXT existing at all, not the vendor/renderer
79            // *identity* it reports, so it stays regardless of
80            // `native_isolation` — the opt-in only skips the identity patch
81            // in `patches.rs`, not this context-enabling launch flag.
82            // `--enable-unsafe-swiftshader` re-enables SwiftShader on Chrome >=116.
83            v.push("--use-gl=angle".into());
84            v.push("--use-angle=swiftshader".into());
85            v.push("--enable-unsafe-swiftshader".into());
86            v
87        }
88    }
89}

91#[cfg(test)]
92#[allow(clippy::panic, clippy::unwrap_used)]
93mod tests {
94    use super::*;
95
96    #[test]
97    fn off_profile_emits_no_flags() {
98        assert!(flags_for_profile(ProfileKind::Off, false).is_empty());
99    }
100
101    #[test]
102    fn native_profile_includes_webrtc_disable() {
103        let flags = flags_for_profile(ProfileKind::Native, false);
104        assert!(
105            flags
106                .iter()
107                .any(|f| f.contains("webrtc-ip-handling-policy"))
108        );
109    }
110
111    #[test]
112    fn spoofed_profile_includes_isolate_origins_disable() {
113        let flags = flags_for_profile(ProfileKind::Spoofed, false);
114        assert!(flags.iter().any(|f| f.contains("IsolateOrigins")));
115    }
116
117    #[test]
118    fn shared_flags_snapshot_native() {
119        let flags = flags_for_profile(ProfileKind::Native, false);
120        insta::assert_yaml_snapshot!("native_profile_flags", flags);
121    }
122
123    #[test]
124    fn shared_flags_snapshot_spoofed() {
125        let flags = flags_for_profile(ProfileKind::Spoofed, false);
126        insta::assert_yaml_snapshot!("spoofed_profile_flags", flags);
127    }
128
129    #[test]
130    fn shared_flags_snapshot_off() {
131        let flags = flags_for_profile(ProfileKind::Off, false);
132        insta::assert_yaml_snapshot!("off_profile_flags", flags);
133    }

137    #[test]
138    fn native_isolation_flags_omit_isolate_origins_and_site_per_process() {
139        let flags = flags_for_profile(ProfileKind::Native, true);
140        assert!(
141            !flags
142                .iter()
143                .any(|f| f.contains("IsolateOrigins") || f.contains("site-per-process")),
144            "native_isolation=true must not disable Chrome's real site isolation, got: {flags:?}"
145        );
146    }
147
148    #[test]
149    fn native_isolation_flags_keep_unrelated_disable_load_extension_feature() {
150        // `DisableLoadExtensionCommandLineSwitch` is unrelated to site
151        // isolation (it controls whether `--load-extension` works) — it must
152        // stay disabled regardless of the native_isolation opt-in.
153        let flags = flags_for_profile(ProfileKind::Native, true);
154        assert!(
155            flags
156                .iter()
157                .any(|f| f.contains("DisableLoadExtensionCommandLineSwitch")),
158            "got: {flags:?}"
159        );
160    }
161
162    #[test]
163    fn native_isolation_spoofed_still_carries_swiftshader_context_flags() {
164        // The SwiftShader launch flags exist so headless has a *working*
165        // WebGL context at all — unrelated to the vendor/renderer identity
166        // patch that native_isolation skips in patches.rs. They must stay.
167        let flags = flags_for_profile(ProfileKind::Spoofed, true);
168        assert!(flags.iter().any(|f| f == "--enable-unsafe-swiftshader"));
169    }
170
171    #[test]
172    fn native_isolation_false_keeps_isolation_disabled_default_unchanged() {
173        // Regression guard: native_isolation=false (today's default, used by
174        // every call site that doesn't opt in) still disables site
175        // isolation — anchored byte-for-byte by the pre-existing
176        // `native_profile_flags`/`spoofed_profile_flags` snapshots above.
177        assert!(
178            flags_for_profile(ProfileKind::Native, false)
179                .iter()
180                .any(|f| f.contains("IsolateOrigins"))
181        );
182    }
