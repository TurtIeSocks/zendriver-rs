1//! Bundles individual patches into a single IIFE called with a serialized
2//! Fingerprint. Only one `Page.addScriptToEvaluateOnNewDocument` round-trip
3//! per nav instead of nine.
4
5use serde_json::json;
6
7use crate::Fingerprint;
8
9const WEBDRIVER: &str = include_str!("patches/webdriver.js");
10const PLUGINS: &str = include_str!("patches/plugins.js");
11const CHROME: &str = include_str!("patches/chrome.js");
12const WEBGL: &str = include_str!("patches/webgl.js");
13const PERMISSIONS: &str = include_str!("patches/permissions.js");
14const CODECS: &str = include_str!("patches/codecs.js");
15const NAVIGATOR_PROPS: &str = include_str!("patches/navigator_props.js");
16const USER_AGENT_DATA: &str = include_str!("patches/user_agent_data.js");
17const BROKEN_IMAGE: &str = include_str!("patches/broken_image.js");
18
19/// Build the bootstrap script for the spoofed profile.
20/// Order: webdriver first (most-probed), navigator_props last (touches most fields).
21#[must_use]
22pub fn bootstrap_script(fp: &Fingerprint) -> String {
23    let fp_json = json!({
24        "platformJs":      fp.platform.js_string(),
25        "chPlatform":      fp.platform.ch_platform(),
26        "platformVersion": fp.ua_metadata.platform_version,
27        "cpuCount":        fp.cpu_count,
28        "memoryGb":        fp.memory_gb,
29        "languages":       fp.locale.as_deref().map_or_else(
30            || vec!["en-US".to_string(), "en".to_string()],
31            |l| vec![l.to_string(), "en".to_string()],
32        ),
33        "architecture":    fp.ua_metadata.architecture,
34        "bitness":         fp.ua_metadata.bitness,
35        "brands":          fp.ua_metadata.brands,
36        "fullVersionList": fp.ua_metadata.full_version_list,
37    });
38
39    format!(
40        "(function(fp){{\n{WEBDRIVER}\n{PLUGINS}\n{CHROME}\n{WEBGL}\n{PERMISSIONS}\n{CODECS}\n{NAVIGATOR_PROPS}\n{USER_AGENT_DATA}\n{BROKEN_IMAGE}\n}})({fp_json});",
41    )
42}

44#[cfg(test)]
45#[allow(clippy::panic, clippy::unwrap_used)]
46mod tests {
47    use super::*;
48    use crate::{Platform, UserAgentMetadata};
49
50    fn mock_fp() -> Fingerprint {
51        Fingerprint {
52            platform: Platform::MacIntel,
53            chrome_major: 120,
54            chrome_full: "120.0.6099.234".into(),
55            cpu_count: 10,
56            memory_gb: 8,
57            ua_string: String::new(),
58            ua_metadata: UserAgentMetadata::realistic(Platform::MacIntel, 120, "120.0.6099.234"),
59            timezone: None,
60            locale: Some("en-US".into()),
61        }
62    }
63
64    #[test]
65    fn bootstrap_includes_all_nine_patches() {
66        let s = bootstrap_script(&mock_fp());
67        assert!(s.contains("webdriver"), "webdriver patch missing");
68        assert!(s.contains("PluginArray"), "plugins patch missing");
69        assert!(s.contains("window.chrome"), "chrome patch missing");
70        assert!(
71            s.contains("UNMASKED_VENDOR_WEBGL") || s.contains("37445"),
72            "webgl patch missing"
73        );
74        assert!(
75            s.contains("Notification.permission"),
76            "permissions patch missing"
77        );
78        assert!(s.contains("canPlayType"), "codecs patch missing");
79        assert!(
80            s.contains("hardwareConcurrency"),
81            "navigator_props patch missing"
82        );
83        assert!(s.contains("userAgentData"), "user_agent_data patch missing");
84        assert!(s.contains("naturalWidth"), "broken_image patch missing");
85    }
86
87    #[test]
88    fn bootstrap_is_an_iife_taking_fp() {
89        let s = bootstrap_script(&mock_fp());
90        assert!(s.starts_with("(function(fp){"));
91        assert!(s.contains("})({"), "fp arg JSON should follow");
92    }
93
94    #[test]
95    fn bootstrap_substitutes_platform_js_string() {
96        let s = bootstrap_script(&mock_fp());
97        assert!(s.contains("\"MacIntel\""));
98    }
