1//! Bundles individual patches into a bootstrap script driven by a [`Persona`]
2//! (the surface-spoofing config) plus a [`Fingerprint`] (coherent UA / Chrome
3//! identity probed at launch).
4//!
5//! The 9 identity patches run inside a single IIFE called with a serialized
6//! fingerprint object — one `Page.addScriptToEvaluateOnNewDocument` round-trip
7//! covers them. The persona-driven surface patches (canvas/audio/clientRects
8//! noise, webgl/fonts/hardware value substitution, webrtc policy) are appended
9//! after, each emitted only when its resolved [`Strategy`] is not `Native`.

30// --- Persona surface patches (appended after the identity IIFE) ----------
31//
32// `webgl.js` carries BOTH the hardcoded vendor/renderer fallback block and a
33// persona-driven value-substitution IIFE, so it is emitted exactly ONCE here
34// (not in the identity IIFE) to avoid a duplicate block + unsubstituted
35// `WEBGL_VENDOR`/`WEBGL_RENDERER` tokens. It references no `fp` fields, so
36// running it at the bootstrap's top level (rather than inside the fp IIFE) is
37// behavior-preserving; its top-level `const VENDOR`/`RENDERER` are script-
38// scoped and the nested IIFE's `const VENDOR` is function-scoped — no

58/// Build the bootstrap script for the spoofed profile.
59///
60/// `identity` supplies coherent UA-CH metadata and the *real* probed Chrome
61/// version; `persona` overrides simple identity fields (platform / hardware /
62/// locale) and drives the new fingerprint surfaces. Splitting the two keeps
63/// the UA coherent: when the persona changes platform we rebuild the UA-CH
64/// metadata against the real Chrome version instead of a stale fallback.
65///
66/// Order: identity IIFE first (webdriver-most-probed-first, navigator_props
67/// last), then the PRNG definition, then each persona surface.
68#[must_use]
69pub fn bootstrap_script(persona: &Persona, identity: &Fingerprint) -> String {
70    bootstrap_script_impl(persona, identity, true)
71}
72
73/// Like [`bootstrap_script`], but omits the WebGL vendor/renderer
74/// value-substitution patch (`patches/webgl.js`) entirely — the host's real
75/// `WebGLRenderingContext.getParameter`/`getSupportedExtensions` values pass
76/// through unpatched instead of the coherent ANGLE/Direct3D11 Intel identity
77/// the patch spoofs by default.
78///
79/// To keep WebGL and WebGPU reporting the *same* (real) GPU, the WebGPU
80/// **value** adapter spoof is omitted here too — otherwise a spoofed
81/// `navigator.gpu` adapter (derived from the WebGL renderer this variant does
82/// not apply) would disagree with the real WebGL renderer, a cross-API
83/// coherence tell. An explicit WebGPU `Block` (hiding `navigator.gpu`) is
84/// renderer-neutral and is still honored. Every other patch (identity IIFE,
85/// noise surfaces, fonts, hardware, webrtc, screen/mouse coherence) is
86/// unaffected. See
87/// [`StealthProfile::native_isolation`](crate::StealthProfile::native_isolation)
88/// for the caller-facing trade-off this backs.
89#[must_use]
90pub fn bootstrap_script_native_webgl(persona: &Persona, identity: &Fingerprint) -> String {
91    bootstrap_script_impl(persona, identity, false)
92}

94/// Shared implementation for [`bootstrap_script`] /
95/// [`bootstrap_script_native_webgl`]. `spoof_webgl` gates the `push_webgl`
96/// call and the WebGPU *value* spoof (which is skipped when WebGL is left
97/// real, so the two APIs stay coherent — see [`push_webgpu`]); every other
98/// patch is identical between the two public entry points.
99fn bootstrap_script_impl(persona: &Persona, identity: &Fingerprint, spoof_webgl: bool) -> String {
100    // Prelude first: installs the toString override + closure-local helpers
101    // that every patch below routes through.
102    let mut body = String::from(NATIVE);
103    body.push('\n');
104    body.push_str(&identity_iife(persona, identity));
105
106    let seed = persona.seed.unwrap_or_else(Seed::random).value();
107
108    // PRNG must be defined once, before any noise/font patch that references
109    // `__zdRng`. Emit it unconditionally — cheap, and harmless if every
110    // surface is Native (it just defines an unused function).
111    body.push('\n');
112    body.push_str(PRNG);
113
114    // Geometry coherence runs unconditionally (no persona spec) — it repairs the
115    // outer*/avail* props that the CDP metrics override cannot reach.
116    body.push('\n');
117    body.push_str(SCREEN);
118    // Synthetic pointer entropy, also unconditional.
119    body.push('\n');
120    body.push_str(MOUSE);
121
122    push_noise(
123        &mut body,
124        Surface::Canvas,
125        persona.canvas.as_ref(),
126        CANVAS,
127        seed,
128    );
129    push_noise(
130        &mut body,
131        Surface::Audio,
132        persona.audio.as_ref(),
133        AUDIO,
134        seed,
135    );
136    push_noise(
137        &mut body,
138        Surface::ClientRects,
139        persona.client_rects.as_ref(),
140        CLIENT_RECTS,
141        seed,
142    );
143
144    if spoof_webgl {
145        push_webgl(&mut body, persona.webgl.as_ref());
146    }
147    push_webgpu(
148        &mut body,
149        persona.webgpu.as_ref(),
150        persona.webgl.as_ref(),
151        spoof_webgl,
152    );
153    push_fonts(&mut body, persona.fonts.as_ref(), seed);
154    push_hardware(&mut body, persona.hardware.as_ref());
155    push_webrtc(&mut body, persona.webrtc.as_ref());
156
157    // Single outer IIFE: helpers stay closure-local (no globalThis leak); the
158    // Function.prototype.toString override inside still persists globally.
159    format!("(function(){{\n{body}\n}})();")
160}

162/// Emit the 9 identity patches wrapped in `(function(fp){ ... })(fpJson)`.
163/// Identity is resolved from `identity` with `persona` overrides applied,
164/// preserving UA coherence (see [`bootstrap_script`]).
165fn identity_iife(persona: &Persona, identity: &Fingerprint) -> String {
166    let platform = persona.platform.unwrap_or(identity.platform);
167
168    // If the persona changes the platform, rebuild UA-CH metadata coherently
169    // against the REAL probed Chrome version (never a fallback) so platform +
170    // platformVersion + brands all agree. Otherwise reuse the probed metadata.
171    let uam = if persona.platform.is_some_and(|p| p != identity.platform) {
172        UserAgentMetadata::realistic(platform, identity.chrome_major, &identity.chrome_full)
173    } else {
174        identity.ua_metadata.clone()
175    };
176
177    let cpu = persona.hardware_concurrency.unwrap_or(identity.cpu_count);
178    let mem = persona.device_memory_gb.unwrap_or(identity.memory_gb);
179    let languages = crate::lang::resolve_languages(persona, identity);
180
181    let fp_json = json!({
182        "platformJs":      platform.js_string(),
183        "chPlatform":      platform.ch_platform(),
184        "platformVersion": uam.platform_version,
185        "cpuCount":        cpu,
186        "memoryGb":        mem,
187        "languages":       languages,
188        "architecture":    uam.architecture,
189        "bitness":         uam.bitness,
190        "brands":          uam.brands,
191        "fullVersionList": uam.full_version_list,
192    });
193
194    format!(
195        "(function(fp){{\n{WEBDRIVER}\n{PLUGINS}\n{CHROME}\n{PERMISSIONS}\n{CODECS}\n{NAVIGATOR_PROPS}\n{USER_AGENT_DATA}\n{BROKEN_IMAGE}\n}})({fp_json});",
196    )
197}

199/// JS source for the resolved seed token of a noise surface, or `None` when
200/// the surface should be omitted entirely (`Native`).
201fn seed_token(strat: Strategy, seed: u64) -> Option<String> {
202    match strat {
203        Strategy::Native => None,
204        // Constant zero seed → every farble step deterministic & uniform; the
205        // `/*BLOCK*/` marker documents intent and lets tests assert on it.
206        Strategy::Block => Some("0/*BLOCK*/".to_string()),
207        Strategy::Seeded | Strategy::Value => Some(seed.to_string()),
208        Strategy::Random => Some("(Math.random()*4294967296)>>>0".to_string()),
209    }
210}

212/// Append a noise-surface patch (`SEED` token), respecting the resolved
213/// strategy. No-op under `Native`.
214fn push_noise(out: &mut String, surface: Surface, cfg: Option<&SurfaceCfg>, js: &str, seed: u64) {
215    let strat = surface.resolve_strategy(cfg.and_then(|c| c.strategy));
216    if let Some(tok) = seed_token(strat, seed) {
217        out.push('\n');
218        out.push_str(&js.replace("SEED", &tok));
219    }
220}

222/// JSON literal for an optional string value, or JS `null` when absent.
223fn json_or_null(v: Option<&String>) -> String {
224    v.map_or_else(
225        || "null".to_string(),
226        |s| serde_json::to_string(s).unwrap_or_else(|_| "null".to_string()),
227    )
228}

230/// Append the webgl value-substitution patch. The existing hardcoded webgl
231/// block (the first half of `webgl.js`) always runs; the appended IIFE's
232/// `WEBGL_VENDOR` / `WEBGL_RENDERER` args carry the persona values (or JS
233/// `null` when absent / `Native`, leaving the hardcoded block in charge).
234fn push_webgl(out: &mut String, spec: Option<&WebglSpec>) {
235    let strat = Surface::Webgl.resolve_strategy(spec.and_then(|s| s.strategy));
236    let (vendor, renderer) = match strat {
237        // Under Native the persona contributes nothing — pass null so the new
238        // IIFE delegates entirely to the hardcoded block (which still runs).
239        Strategy::Native => ("null".to_string(), "null".to_string()),
240        _ => (
241            json_or_null(spec.and_then(|s| s.unmasked_vendor.as_ref())),
242            json_or_null(spec.and_then(|s| s.unmasked_renderer.as_ref())),
243        ),
244    };
245    out.push('\n');
246    out.push_str(
247        &WEBGL
248            .replace("WEBGL_VENDOR", &vendor)
249            .replace("WEBGL_RENDERER", &renderer),
250    );
251}

253/// Append the WebGPU coherence patch. The adapter info is derived from the
254/// persona's WebGL renderer (or the hardcoded Intel default the webgl block
255/// falls back to), so navigator.gpu agrees with WebGL. Omitted under `Native`.
256///
257/// `spoof_webgl` reflects whether the WebGL value patch ran. When it is
258/// `false` (the native-WebGL opt-in — [`bootstrap_script_native_webgl`]) the
259/// real WebGL renderer passes through unpatched, so a spoofed WebGPU *value*
260/// adapter — derived from the WebGL renderer we did NOT apply, or the
261/// hardcoded default below — would disagree with the real GPU. That cross-API
262/// mismatch is the exact coherence tell the opt-in exists to avoid, so the
263/// value spoof is skipped too and the real `navigator.gpu` adapter passes
264/// through. An explicit `Block` (hide `navigator.gpu`) is renderer-neutral
265/// and stays honored regardless.
266fn push_webgpu(
267    out: &mut String,
268    cfg: Option<&SurfaceCfg>,
269    webgl: Option<&WebglSpec>,
270    spoof_webgl: bool,
271) {
272    use crate::persona::webgpu_adapter::adapter_for_renderer;
273    let strat = Surface::Webgpu.resolve_strategy(cfg.and_then(|c| c.strategy));
274    if strat == Strategy::Native {
275        return;
276    }
277    // Native-WebGL renderer coherence: skip the value adapter spoof when the
278    // real WebGL renderer is left unpatched (see the doc comment above). A
279    // `Block` is renderer-neutral, so it is still emitted.
280    if !spoof_webgl && strat != Strategy::Block {
281        return;
282    }
283    const DEFAULT_RENDERER: &str =
284        "ANGLE (Intel, Intel(R) UHD Graphics 630 Direct3D11 vs_5_0 ps_5_0, D3D11)";
285    let renderer = webgl
286        .and_then(|w| w.unmasked_renderer.as_deref())
287        .unwrap_or(DEFAULT_RENDERER);
288    let adapter = adapter_for_renderer(renderer);
289    let (vendor, arch, mode) = if strat == Strategy::Block {
290        (
291            "null".to_string(),
292            "null".to_string(),
293            "\"block\"".to_string(),
294        )
295    } else {
296        (
297            serde_json::to_string(&adapter.vendor).unwrap_or_else(|_| "null".into()),
298            serde_json::to_string(&adapter.architecture).unwrap_or_else(|_| "null".into()),
299            "\"value\"".to_string(),
300        )
301    };
302    out.push('\n');
303    out.push_str(
304        &WEBGPU
305            .replace("WEBGPU_VENDOR", &vendor)
306            .replace("WEBGPU_ARCHITECTURE", &arch)
307            .replace("WEBGPU_MODE", &mode),
308    );
309}

311/// Append the fonts patch (`FONT_ALLOW` + `SEED`). Omitted under `Native`.
312fn push_fonts(out: &mut String, spec: Option<&FontSpec>, seed: u64) {
313    let strat = Surface::Fonts.resolve_strategy(spec.and_then(|s| s.strategy));
314    let seed_tok = match seed_token(strat, seed) {
315        None => return, // Native → omit entirely.
316        Some(t) => t,
317    };
318    // Allow-list only meaningful when the persona supplies one AND we're not
319    // blocking; under Block, hide every font (empty allow-list).
320    let allow = match strat {
321        Strategy::Block => "[]".to_string(),
322        _ => spec.and_then(|s| s.available.as_ref()).map_or_else(
323            || "null".to_string(),
324            |v| serde_json::to_string(v).unwrap_or_else(|_| "null".to_string()),
325        ),
326    };
327    out.push('\n');
328    out.push_str(
329        &FONTS
330            .replace("FONT_ALLOW", &allow)
331            .replace("SEED", &seed_tok),
332    );
333}

335/// Append the hardware patch (`HW_BATTERY` / `HW_MEDIA_DEVICES` / `HW_VOICES`).
336/// Omitted under `Native`. Each field independently `null` when unset.
337fn push_hardware(out: &mut String, spec: Option<&HardwareSpec>) {
338    let strat = Surface::Hardware.resolve_strategy(spec.and_then(|s| s.strategy));
339    if strat == Strategy::Native {
340        return;
341    }
342    let (battery, media, voices) = if strat == Strategy::Block {
343        // Block → present a uniform, minimal hardware surface.
344        ("1".to_string(), "0".to_string(), "[]".to_string())
345    } else {
346        let battery = spec
347            .and_then(|s| s.battery_level)
348            .map_or_else(|| "null".to_string(), |b| b.to_string());
349        let media = spec
350            .and_then(|s| s.media_devices)
351            .map_or_else(|| "null".to_string(), |m| m.to_string());
352        let voices = spec.and_then(|s| s.speech_voices.as_ref()).map_or_else(
353            || "null".to_string(),
354            |v| serde_json::to_string(v).unwrap_or_else(|_| "null".to_string()),
355        );
356        (battery, media, voices)
357    };
358    out.push('\n');
359    out.push_str(
360        &HARDWARE
361            .replace("HW_BATTERY", &battery)
362            .replace("HW_MEDIA_DEVICES", &media)
363            .replace("HW_VOICES", &voices),
364    );
365}

367/// Append the webrtc policy patch (`WEBRTC_POLICY` + `WEBRTC_FAKE_IP`).
368/// The patch self-no-ops on the `"native"` policy, so we always emit it but
369/// pass the strategy-derived policy string.
370fn push_webrtc(out: &mut String, spec: Option<&WebrtcSpec>) {
371    let strat = Surface::Webrtc.resolve_strategy(spec.and_then(|s| s.strategy));
372    let policy = match strat {
373        Strategy::Native => "native",
374        Strategy::Value => "value",
375        // Block (the policy default) + any non-meaningful fallback.
376        _ => "block",
377    };
378    let fake_ip = json_or_null(spec.and_then(|s| s.fake_ip.as_ref()));
379    out.push('\n');
380    out.push_str(
381        &WEBRTC
382            .replace("WEBRTC_POLICY", &format!("\"{policy}\""))
383            .replace("WEBRTC_FAKE_IP", &fake_ip),
384    );
385}
