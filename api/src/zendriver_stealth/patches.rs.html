1//! Bundles individual patches into a bootstrap script driven by a [`Persona`]
2//! (the surface-spoofing config) plus a [`Fingerprint`] (coherent UA / Chrome
3//! identity probed at launch).
4//!
5//! The 9 identity patches run inside a single IIFE called with a serialized
6//! fingerprint object — one `Page.addScriptToEvaluateOnNewDocument` round-trip
7//! covers them. The persona-driven surface patches (canvas/audio/clientRects
8//! noise, webgl/fonts/hardware value substitution, webrtc policy) are appended
9//! after, each emitted only when its resolved [`Strategy`] is not `Native`.
10

19const PLUGINS: &str = include_str!("patches/plugins.js");
20const CHROME: &str = include_str!("patches/chrome.js");
21const PERMISSIONS: &str = include_str!("patches/permissions.js");
22const CODECS: &str = include_str!("patches/codecs.js");
23const NAVIGATOR_PROPS: &str = include_str!("patches/navigator_props.js");
24const USER_AGENT_DATA: &str = include_str!("patches/user_agent_data.js");
25const BROKEN_IMAGE: &str = include_str!("patches/broken_image.js");

27// --- Persona surface patches (appended after the identity IIFE) ----------
28//
29// `webgl.js` carries BOTH the hardcoded vendor/renderer fallback block and a
30// persona-driven value-substitution IIFE, so it is emitted exactly ONCE here
31// (not in the identity IIFE) to avoid a duplicate block + unsubstituted
32// `WEBGL_VENDOR`/`WEBGL_RENDERER` tokens. It references no `fp` fields, so
33// running it at the bootstrap's top level (rather than inside the fp IIFE) is
34// behavior-preserving; its top-level `const VENDOR`/`RENDERER` are script-
35// scoped and the nested IIFE's `const VENDOR` is function-scoped — no
36// redeclaration.
37const PRNG: &str = include_str!("patches/_prng.js");
38const WEBGL: &str = include_str!("patches/webgl.js");
39const CANVAS: &str = include_str!("patches/canvas.js");
40const AUDIO: &str = include_str!("patches/audio.js");
41const CLIENT_RECTS: &str = include_str!("patches/client_rects.js");
42const FONTS: &str = include_str!("patches/fonts.js");
43const WEBRTC: &str = include_str!("patches/webrtc.js");
44const HARDWARE: &str = include_str!("patches/hardware.js");
45const WEBGPU: &str = include_str!("patches/webgpu.js");

47/// Build the bootstrap script for the spoofed profile.
48///
49/// `identity` supplies coherent UA-CH metadata and the *real* probed Chrome
50/// version; `persona` overrides simple identity fields (platform / hardware /
51/// locale) and drives the new fingerprint surfaces. Splitting the two keeps
52/// the UA coherent: when the persona changes platform we rebuild the UA-CH
53/// metadata against the real Chrome version instead of a stale fallback.
54///
55/// Order: identity IIFE first (webdriver-most-probed-first, navigator_props
56/// last), then the PRNG definition, then each persona surface.
57#[must_use]
58pub fn bootstrap_script(persona: &Persona, identity: &Fingerprint) -> String {
59    let mut out = identity_iife(persona, identity);
60
61    let seed = persona.seed.unwrap_or_else(Seed::random).value();
62
63    // PRNG must be defined once, before any noise/font patch that references
64    // `__zdRng`. Emit it unconditionally — cheap, and harmless if every
65    // surface is Native (it just defines an unused function).
66    out.push('\n');
67    out.push_str(PRNG);
68
69    push_noise(
70        &mut out,
71        Surface::Canvas,
72        persona.canvas.as_ref(),
73        CANVAS,
74        seed,
75    );
76    push_noise(
77        &mut out,
78        Surface::Audio,
79        persona.audio.as_ref(),
80        AUDIO,
81        seed,
82    );
83    push_noise(
84        &mut out,
85        Surface::ClientRects,
86        persona.client_rects.as_ref(),
87        CLIENT_RECTS,
88        seed,
89    );
90
91    push_webgl(&mut out, persona.webgl.as_ref());
92    push_webgpu(&mut out, persona.webgpu.as_ref(), persona.webgl.as_ref());
93    push_fonts(&mut out, persona.fonts.as_ref(), seed);
94    push_hardware(&mut out, persona.hardware.as_ref());
95    push_webrtc(&mut out, persona.webrtc.as_ref());
96
97    out
98}

100/// Emit the 9 identity patches wrapped in `(function(fp){ ... })(fpJson)`.
101/// Identity is resolved from `identity` with `persona` overrides applied,
102/// preserving UA coherence (see [`bootstrap_script`]).
103fn identity_iife(persona: &Persona, identity: &Fingerprint) -> String {
104    let platform = persona.platform.unwrap_or(identity.platform);
105
106    // If the persona changes the platform, rebuild UA-CH metadata coherently
107    // against the REAL probed Chrome version (never a fallback) so platform +
108    // platformVersion + brands all agree. Otherwise reuse the probed metadata.
109    let uam = if persona.platform.is_some_and(|p| p != identity.platform) {
110        UserAgentMetadata::realistic(platform, identity.chrome_major, &identity.chrome_full)
111    } else {
112        identity.ua_metadata.clone()
113    };
114
115    let cpu = persona.hardware_concurrency.unwrap_or(identity.cpu_count);
116    let mem = persona.device_memory_gb.unwrap_or(identity.memory_gb);
117    let locale = persona.locale.clone().or_else(|| identity.locale.clone());
118
119    let fp_json = json!({
120        "platformJs":      platform.js_string(),
121        "chPlatform":      platform.ch_platform(),
122        "platformVersion": uam.platform_version,
123        "cpuCount":        cpu,
124        "memoryGb":        mem,
125        "languages":       locale.as_deref().map_or_else(
126            || vec!["en-US".to_string(), "en".to_string()],
127            |l| vec![l.to_string(), "en".to_string()],
128        ),
129        "architecture":    uam.architecture,
130        "bitness":         uam.bitness,
131        "brands":          uam.brands,
132        "fullVersionList": uam.full_version_list,
133    });
134
135    format!(
136        "(function(fp){{\n{WEBDRIVER}\n{PLUGINS}\n{CHROME}\n{PERMISSIONS}\n{CODECS}\n{NAVIGATOR_PROPS}\n{USER_AGENT_DATA}\n{BROKEN_IMAGE}\n}})({fp_json});",
137    )
138}

140/// JS source for the resolved seed token of a noise surface, or `None` when
141/// the surface should be omitted entirely (`Native`).
142fn seed_token(strat: Strategy, seed: u64) -> Option<String> {
143    match strat {
144        Strategy::Native => None,
145        // Constant zero seed → every farble step deterministic & uniform; the
146        // `/*BLOCK*/` marker documents intent and lets tests assert on it.
147        Strategy::Block => Some("0/*BLOCK*/".to_string()),
148        Strategy::Seeded | Strategy::Value => Some(seed.to_string()),
149        Strategy::Random => Some("(Math.random()*4294967296)>>>0".to_string()),
150    }
151}

153/// Append a noise-surface patch (`SEED` token), respecting the resolved
154/// strategy. No-op under `Native`.
155fn push_noise(out: &mut String, surface: Surface, cfg: Option<&SurfaceCfg>, js: &str, seed: u64) {
156    let strat = surface.resolve_strategy(cfg.and_then(|c| c.strategy));
157    if let Some(tok) = seed_token(strat, seed) {
158        out.push('\n');
159        out.push_str(&js.replace("SEED", &tok));
160    }
161}

163/// JSON literal for an optional string value, or JS `null` when absent.
164fn json_or_null(v: Option<&String>) -> String {
165    v.map_or_else(
166        || "null".to_string(),
167        |s| serde_json::to_string(s).unwrap_or_else(|_| "null".to_string()),
168    )
169}

171/// Append the webgl value-substitution patch. The existing hardcoded webgl
172/// block (the first half of `webgl.js`) always runs; the appended IIFE's
173/// `WEBGL_VENDOR` / `WEBGL_RENDERER` args carry the persona values (or JS
174/// `null` when absent / `Native`, leaving the hardcoded block in charge).
175fn push_webgl(out: &mut String, spec: Option<&WebglSpec>) {
176    let strat = Surface::Webgl.resolve_strategy(spec.and_then(|s| s.strategy));
177    let (vendor, renderer) = match strat {
178        // Under Native the persona contributes nothing — pass null so the new
179        // IIFE delegates entirely to the hardcoded block (which still runs).
180        Strategy::Native => ("null".to_string(), "null".to_string()),
181        _ => (
182            json_or_null(spec.and_then(|s| s.unmasked_vendor.as_ref())),
183            json_or_null(spec.and_then(|s| s.unmasked_renderer.as_ref())),
184        ),
185    };
186    out.push('\n');
187    out.push_str(
188        &WEBGL
189            .replace("WEBGL_VENDOR", &vendor)
190            .replace("WEBGL_RENDERER", &renderer),
191    );
192}

194/// Append the WebGPU coherence patch. The adapter info is derived from the
195/// persona's WebGL renderer (or the hardcoded Intel default the webgl block
196/// falls back to), so navigator.gpu agrees with WebGL. Omitted under `Native`.
197fn push_webgpu(out: &mut String, cfg: Option<&SurfaceCfg>, webgl: Option<&WebglSpec>) {
198    use crate::persona::webgpu_adapter::adapter_for_renderer;
199    let strat = Surface::Webgpu.resolve_strategy(cfg.and_then(|c| c.strategy));
200    if strat == Strategy::Native {
201        return;
202    }
203    const DEFAULT_RENDERER: &str =
204        "ANGLE (Intel, Intel(R) UHD Graphics 630 Direct3D11 vs_5_0 ps_5_0, D3D11)";
205    let renderer = webgl
206        .and_then(|w| w.unmasked_renderer.as_deref())
207        .unwrap_or(DEFAULT_RENDERER);
208    let adapter = adapter_for_renderer(renderer);
209    let (vendor, arch, mode) = if strat == Strategy::Block {
210        (
211            "null".to_string(),
212            "null".to_string(),
213            "\"block\"".to_string(),
214        )
215    } else {
216        (
217            serde_json::to_string(&adapter.vendor).unwrap_or_else(|_| "null".into()),
218            serde_json::to_string(&adapter.architecture).unwrap_or_else(|_| "null".into()),
219            "\"value\"".to_string(),
220        )
221    };
222    out.push('\n');
223    out.push_str(
224        &WEBGPU
225            .replace("WEBGPU_VENDOR", &vendor)
226            .replace("WEBGPU_ARCHITECTURE", &arch)
227            .replace("WEBGPU_MODE", &mode),
228    );
229}

231/// Append the fonts patch (`FONT_ALLOW` + `SEED`). Omitted under `Native`.
232fn push_fonts(out: &mut String, spec: Option<&FontSpec>, seed: u64) {
233    let strat = Surface::Fonts.resolve_strategy(spec.and_then(|s| s.strategy));
234    let seed_tok = match seed_token(strat, seed) {
235        None => return, // Native → omit entirely.
236        Some(t) => t,
237    };
238    // Allow-list only meaningful when the persona supplies one AND we're not
239    // blocking; under Block, hide every font (empty allow-list).
240    let allow = match strat {
241        Strategy::Block => "[]".to_string(),
242        _ => spec.and_then(|s| s.available.as_ref()).map_or_else(
243            || "null".to_string(),
244            |v| serde_json::to_string(v).unwrap_or_else(|_| "null".to_string()),
245        ),
246    };
247    out.push('\n');
248    out.push_str(
249        &FONTS
250            .replace("FONT_ALLOW", &allow)
251            .replace("SEED", &seed_tok),
252    );
253}

255/// Append the hardware patch (`HW_BATTERY` / `HW_MEDIA_DEVICES` / `HW_VOICES`).
256/// Omitted under `Native`. Each field independently `null` when unset.
257fn push_hardware(out: &mut String, spec: Option<&HardwareSpec>) {
258    let strat = Surface::Hardware.resolve_strategy(spec.and_then(|s| s.strategy));
259    if strat == Strategy::Native {
260        return;
261    }
262    let (battery, media, voices) = if strat == Strategy::Block {
263        // Block → present a uniform, minimal hardware surface.
264        ("1".to_string(), "0".to_string(), "[]".to_string())
265    } else {
266        let battery = spec
267            .and_then(|s| s.battery_level)
268            .map_or_else(|| "null".to_string(), |b| b.to_string());
269        let media = spec
270            .and_then(|s| s.media_devices)
271            .map_or_else(|| "null".to_string(), |m| m.to_string());
272        let voices = spec.and_then(|s| s.speech_voices.as_ref()).map_or_else(
273            || "null".to_string(),
274            |v| serde_json::to_string(v).unwrap_or_else(|_| "null".to_string()),
275        );
276        (battery, media, voices)
277    };
278    out.push('\n');
279    out.push_str(
280        &HARDWARE
281            .replace("HW_BATTERY", &battery)
282            .replace("HW_MEDIA_DEVICES", &media)
283            .replace("HW_VOICES", &voices),
284    );
285}

287/// Append the webrtc policy patch (`WEBRTC_POLICY` + `WEBRTC_FAKE_IP`).
288/// The patch self-no-ops on the `"native"` policy, so we always emit it but
289/// pass the strategy-derived policy string.
290fn push_webrtc(out: &mut String, spec: Option<&WebrtcSpec>) {
291    let strat = Surface::Webrtc.resolve_strategy(spec.and_then(|s| s.strategy));
292    let policy = match strat {
293        Strategy::Native => "native",
294        Strategy::Value => "value",
295        // Block (the policy default) + any non-meaningful fallback.
296        _ => "block",
297    };
298    let fake_ip = json_or_null(spec.and_then(|s| s.fake_ip.as_ref()));
299    out.push('\n');
300    out.push_str(
301        &WEBRTC
302            .replace("WEBRTC_POLICY", &format!("\"{policy}\""))
303            .replace("WEBRTC_FAKE_IP", &fake_ip),
304    );
305}
