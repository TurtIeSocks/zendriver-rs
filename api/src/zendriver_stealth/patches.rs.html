1//! Bundles individual patches into a bootstrap script driven by a [`Persona`]
2//! (the surface-spoofing config) plus a [`Fingerprint`] (coherent UA / Chrome
3//! identity probed at launch).
4//!
5//! The 9 identity patches run inside a single IIFE called with a serialized
6//! fingerprint object — one `Page.addScriptToEvaluateOnNewDocument` round-trip
7//! covers them. The persona-driven surface patches (canvas/audio/clientRects
8//! noise, webgl/fonts/hardware value substitution, webrtc policy) are appended
9//! after, each emitted only when its resolved [`Strategy`] is not `Native`.
10

31//
32// `webgl.js` carries BOTH the hardcoded vendor/renderer fallback block and a

34// (not in the identity IIFE) to avoid a duplicate block + unsubstituted
35// `WEBGL_VENDOR`/`WEBGL_RENDERER` tokens. It references no `fp` fields, so
36// running it at the bootstrap's top level (rather than inside the fp IIFE) is
37// behavior-preserving; its top-level `const VENDOR`/`RENDERER` are script-
38// scoped and the nested IIFE's `const VENDOR` is function-scoped — no

58/// Build the bootstrap script for the spoofed profile.
59///
60/// `identity` supplies coherent UA-CH metadata and the *real* probed Chrome
61/// version; `persona` overrides simple identity fields (platform / hardware /
62/// locale) and drives the new fingerprint surfaces. Splitting the two keeps
63/// the UA coherent: when the persona changes platform we rebuild the UA-CH
64/// metadata against the real Chrome version instead of a stale fallback.
65///
66/// Order: identity IIFE first (webdriver-most-probed-first, navigator_props
67/// last), then the PRNG definition, then each persona surface.
68#[must_use]
69pub fn bootstrap_script(persona: &Persona, identity: &Fingerprint) -> String {
70    // Prelude first: installs the toString override + closure-local helpers
71    // that every patch below routes through.
72    let mut body = String::from(NATIVE);
73    body.push('\n');
74    body.push_str(&identity_iife(persona, identity));
75
76    let seed = persona.seed.unwrap_or_else(Seed::random).value();
77
78    // PRNG must be defined once, before any noise/font patch that references
79    // `__zdRng`. Emit it unconditionally — cheap, and harmless if every
80    // surface is Native (it just defines an unused function).
81    body.push('\n');
82    body.push_str(PRNG);
83
84    // Geometry coherence runs unconditionally (no persona spec) — it repairs the
85    // outer*/avail* props that the CDP metrics override cannot reach.
86    body.push('\n');
87    body.push_str(SCREEN);
88    // Synthetic pointer entropy, also unconditional.
89    body.push('\n');
90    body.push_str(MOUSE);
91
92    push_noise(
93        &mut body,
94        Surface::Canvas,
95        persona.canvas.as_ref(),
96        CANVAS,
97        seed,
98    );
99    push_noise(
100        &mut body,
101        Surface::Audio,
102        persona.audio.as_ref(),
103        AUDIO,
104        seed,
105    );
106    push_noise(
107        &mut body,
108        Surface::ClientRects,
109        persona.client_rects.as_ref(),
110        CLIENT_RECTS,
111        seed,
112    );
113
114    push_webgl(&mut body, persona.webgl.as_ref());
115    push_webgpu(&mut body, persona.webgpu.as_ref(), persona.webgl.as_ref());
116    push_fonts(&mut body, persona.fonts.as_ref(), seed);
117    push_hardware(&mut body, persona.hardware.as_ref());
118    push_webrtc(&mut body, persona.webrtc.as_ref());
119
120    // Single outer IIFE: helpers stay closure-local (no globalThis leak); the
121    // Function.prototype.toString override inside still persists globally.
122    format!("(function(){{\n{body}\n}})();")
123}

125/// Emit the 9 identity patches wrapped in `(function(fp){ ... })(fpJson)`.
126/// Identity is resolved from `identity` with `persona` overrides applied,
127/// preserving UA coherence (see [`bootstrap_script`]).
128fn identity_iife(persona: &Persona, identity: &Fingerprint) -> String {
129    let platform = persona.platform.unwrap_or(identity.platform);
130
131    // If the persona changes the platform, rebuild UA-CH metadata coherently
132    // against the REAL probed Chrome version (never a fallback) so platform +
133    // platformVersion + brands all agree. Otherwise reuse the probed metadata.
134    let uam = if persona.platform.is_some_and(|p| p != identity.platform) {
135        UserAgentMetadata::realistic(platform, identity.chrome_major, &identity.chrome_full)
136    } else {
137        identity.ua_metadata.clone()
138    };
139
140    let cpu = persona.hardware_concurrency.unwrap_or(identity.cpu_count);
141    let mem = persona.device_memory_gb.unwrap_or(identity.memory_gb);
142    let languages = crate::lang::resolve_languages(persona, identity);
143
144    let fp_json = json!({
145        "platformJs":      platform.js_string(),
146        "chPlatform":      platform.ch_platform(),
147        "platformVersion": uam.platform_version,
148        "cpuCount":        cpu,
149        "memoryGb":        mem,
150        "languages":       languages,
151        "architecture":    uam.architecture,
152        "bitness":         uam.bitness,
153        "brands":          uam.brands,
154        "fullVersionList": uam.full_version_list,
155    });
156
157    format!(
158        "(function(fp){{\n{WEBDRIVER}\n{PLUGINS}\n{CHROME}\n{PERMISSIONS}\n{CODECS}\n{NAVIGATOR_PROPS}\n{USER_AGENT_DATA}\n{BROKEN_IMAGE}\n}})({fp_json});",
159    )
160}

162/// JS source for the resolved seed token of a noise surface, or `None` when
163/// the surface should be omitted entirely (`Native`).
164fn seed_token(strat: Strategy, seed: u64) -> Option<String> {
165    match strat {
166        Strategy::Native => None,
167        // Constant zero seed → every farble step deterministic & uniform; the
168        // `/*BLOCK*/` marker documents intent and lets tests assert on it.
169        Strategy::Block => Some("0/*BLOCK*/".to_string()),
170        Strategy::Seeded | Strategy::Value => Some(seed.to_string()),
171        Strategy::Random => Some("(Math.random()*4294967296)>>>0".to_string()),
172    }
173}

175/// Append a noise-surface patch (`SEED` token), respecting the resolved
176/// strategy. No-op under `Native`.
177fn push_noise(out: &mut String, surface: Surface, cfg: Option<&SurfaceCfg>, js: &str, seed: u64) {
178    let strat = surface.resolve_strategy(cfg.and_then(|c| c.strategy));
179    if let Some(tok) = seed_token(strat, seed) {
180        out.push('\n');
181        out.push_str(&js.replace("SEED", &tok));
182    }
183}

185/// JSON literal for an optional string value, or JS `null` when absent.
186fn json_or_null(v: Option<&String>) -> String {
187    v.map_or_else(
188        || "null".to_string(),
189        |s| serde_json::to_string(s).unwrap_or_else(|_| "null".to_string()),
190    )
191}

193/// Append the webgl value-substitution patch. The existing hardcoded webgl
194/// block (the first half of `webgl.js`) always runs; the appended IIFE's
195/// `WEBGL_VENDOR` / `WEBGL_RENDERER` args carry the persona values (or JS
196/// `null` when absent / `Native`, leaving the hardcoded block in charge).
197fn push_webgl(out: &mut String, spec: Option<&WebglSpec>) {
198    let strat = Surface::Webgl.resolve_strategy(spec.and_then(|s| s.strategy));
199    let (vendor, renderer) = match strat {
200        // Under Native the persona contributes nothing — pass null so the new
201        // IIFE delegates entirely to the hardcoded block (which still runs).
202        Strategy::Native => ("null".to_string(), "null".to_string()),
203        _ => (
204            json_or_null(spec.and_then(|s| s.unmasked_vendor.as_ref())),
205            json_or_null(spec.and_then(|s| s.unmasked_renderer.as_ref())),
206        ),
207    };
208    out.push('\n');
209    out.push_str(
210        &WEBGL
211            .replace("WEBGL_VENDOR", &vendor)
212            .replace("WEBGL_RENDERER", &renderer),
213    );
214}

216/// Append the WebGPU coherence patch. The adapter info is derived from the
217/// persona's WebGL renderer (or the hardcoded Intel default the webgl block
218/// falls back to), so navigator.gpu agrees with WebGL. Omitted under `Native`.
219fn push_webgpu(out: &mut String, cfg: Option<&SurfaceCfg>, webgl: Option<&WebglSpec>) {
220    use crate::persona::webgpu_adapter::adapter_for_renderer;
221    let strat = Surface::Webgpu.resolve_strategy(cfg.and_then(|c| c.strategy));
222    if strat == Strategy::Native {
223        return;
224    }
225    const DEFAULT_RENDERER: &str =
226        "ANGLE (Intel, Intel(R) UHD Graphics 630 Direct3D11 vs_5_0 ps_5_0, D3D11)";
227    let renderer = webgl
228        .and_then(|w| w.unmasked_renderer.as_deref())
229        .unwrap_or(DEFAULT_RENDERER);
230    let adapter = adapter_for_renderer(renderer);
231    let (vendor, arch, mode) = if strat == Strategy::Block {
232        (
233            "null".to_string(),
234            "null".to_string(),
235            "\"block\"".to_string(),
236        )
237    } else {
238        (
239            serde_json::to_string(&adapter.vendor).unwrap_or_else(|_| "null".into()),
240            serde_json::to_string(&adapter.architecture).unwrap_or_else(|_| "null".into()),
241            "\"value\"".to_string(),
242        )
243    };
244    out.push('\n');
245    out.push_str(
246        &WEBGPU
247            .replace("WEBGPU_VENDOR", &vendor)
248            .replace("WEBGPU_ARCHITECTURE", &arch)
249            .replace("WEBGPU_MODE", &mode),
250    );
251}

253/// Append the fonts patch (`FONT_ALLOW` + `SEED`). Omitted under `Native`.
254fn push_fonts(out: &mut String, spec: Option<&FontSpec>, seed: u64) {
255    let strat = Surface::Fonts.resolve_strategy(spec.and_then(|s| s.strategy));
256    let seed_tok = match seed_token(strat, seed) {
257        None => return, // Native → omit entirely.
258        Some(t) => t,
259    };
260    // Allow-list only meaningful when the persona supplies one AND we're not
261    // blocking; under Block, hide every font (empty allow-list).
262    let allow = match strat {
263        Strategy::Block => "[]".to_string(),
264        _ => spec.and_then(|s| s.available.as_ref()).map_or_else(
265            || "null".to_string(),
266            |v| serde_json::to_string(v).unwrap_or_else(|_| "null".to_string()),
267        ),
268    };
269    out.push('\n');
270    out.push_str(
271        &FONTS
272            .replace("FONT_ALLOW", &allow)
273            .replace("SEED", &seed_tok),
274    );
275}

277/// Append the hardware patch (`HW_BATTERY` / `HW_MEDIA_DEVICES` / `HW_VOICES`).
278/// Omitted under `Native`. Each field independently `null` when unset.
279fn push_hardware(out: &mut String, spec: Option<&HardwareSpec>) {
280    let strat = Surface::Hardware.resolve_strategy(spec.and_then(|s| s.strategy));
281    if strat == Strategy::Native {
282        return;
283    }
284    let (battery, media, voices) = if strat == Strategy::Block {
285        // Block → present a uniform, minimal hardware surface.
286        ("1".to_string(), "0".to_string(), "[]".to_string())
287    } else {
288        let battery = spec
289            .and_then(|s| s.battery_level)
290            .map_or_else(|| "null".to_string(), |b| b.to_string());
291        let media = spec
292            .and_then(|s| s.media_devices)
293            .map_or_else(|| "null".to_string(), |m| m.to_string());
294        let voices = spec.and_then(|s| s.speech_voices.as_ref()).map_or_else(
295            || "null".to_string(),
296            |v| serde_json::to_string(v).unwrap_or_else(|_| "null".to_string()),
297        );
298        (battery, media, voices)
299    };
300    out.push('\n');
301    out.push_str(
302        &HARDWARE
303            .replace("HW_BATTERY", &battery)
304            .replace("HW_MEDIA_DEVICES", &media)
305            .replace("HW_VOICES", &voices),
306    );
307}

309/// Append the webrtc policy patch (`WEBRTC_POLICY` + `WEBRTC_FAKE_IP`).
310/// The patch self-no-ops on the `"native"` policy, so we always emit it but
311/// pass the strategy-derived policy string.
312fn push_webrtc(out: &mut String, spec: Option<&WebrtcSpec>) {
313    let strat = Surface::Webrtc.resolve_strategy(spec.and_then(|s| s.strategy));
314    let policy = match strat {
315        Strategy::Native => "native",
316        Strategy::Value => "value",
317        // Block (the policy default) + any non-meaningful fallback.
318        _ => "block",
319    };
320    let fake_ip = json_or_null(spec.and_then(|s| s.fake_ip.as_ref()));
321    out.push('\n');
322    out.push_str(
323        &WEBRTC
324            .replace("WEBRTC_POLICY", &format!("\"{policy}\""))
325            .replace("WEBRTC_FAKE_IP", &fake_ip),
326    );
327}
