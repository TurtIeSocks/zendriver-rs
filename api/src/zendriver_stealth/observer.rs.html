1//! StealthObserver: applies a [`StealthProfile`] to each new attached target.
2//!
3//! Installed in the [`zendriver_transport`] actor's observer chain. On
4//! `Target.attachedToTarget`, the actor pauses the new target, walks every
5//! observer serially, then releases the debugger via
6//! `Runtime.runIfWaitingForDebugger`. The observer's job is to push every
7//! UA/screen/timezone/locale override and (for spoofed) the bootstrap script
8//! _before_ the debugger releases, so the first script the page runs sees the
9//! patched globals.
10
11use serde_json::json;
12use zendriver_transport::{ObserverError, PausedSession, TargetObserver};
13
14use crate::patches::bootstrap_script;
15use crate::persona::GeoPos;
16use crate::{Fingerprint, Persona, ProfileKind, StealthProfile};
17
18/// Observer that applies a [`StealthProfile`] + [`Fingerprint`] to every page
19/// target. Workers and iframes are skipped — workers have no DOM and iframes
20/// inherit patches from the parent target in flat session mode.
21#[derive(Debug)]
22pub struct StealthObserver {
23    profile: StealthProfile,
24    fingerprint: Fingerprint,
25    /// Pre-rendered bootstrap source. Empty for `Off`/`Native` — we never send
26    /// `Page.addScriptToEvaluateOnNewDocument` in those modes, so there is no
27    /// need to pay the patches-bundle cost for them.
28    bootstrap: String,
29    /// Mock geolocation coordinates from the resolved [`Persona`], sent via
30    /// `Emulation.setGeolocationOverride`. Unlike `timezone`/`locale` (carried
31    /// on [`Fingerprint`]), geolocation has no `Fingerprint` counterpart, so
32    /// it's captured here straight off the persona at construction time.
33    geolocation: Option<GeoPos>,
34}
35
36impl StealthObserver {
37    /// Build a new observer. Bootstrap source is composed eagerly so the
38    /// per-target hot path only pays a `clone`/borrow.
39    ///
40    /// The bootstrap is driven by a [`Persona`] (surface-spoofing config) plus
41    /// the [`Fingerprint`] (coherent UA / Chrome identity). This constructor
42    /// uses [`Persona::default`] — no surface overrides, identity-only patches
43    /// keep their current behavior. The launch path (a later task) will thread
44    /// a caller-supplied persona via [`StealthObserver::with_persona`].
45    #[must_use]
46    pub fn new(profile: StealthProfile, fingerprint: Fingerprint) -> Self {
47        Self::with_persona(profile, fingerprint, Persona::default())
48    }
49
50    /// Build a new observer with an explicit [`Persona`] driving the surface
51    /// patches. `identity` still supplies the coherent UA / Chrome version.
52    #[must_use]
53    pub fn with_persona(
54        profile: StealthProfile,
55        fingerprint: Fingerprint,
56        persona: Persona,
57    ) -> Self {
58        let bootstrap = if profile.kind() == ProfileKind::Spoofed {
59            bootstrap_script(&persona, &fingerprint)
60        } else {
61            String::new()
62        };
63        let geolocation = persona.geolocation;
64        Self {
65            profile,
66            fingerprint,
67            bootstrap,
68            geolocation,
69        }
70    }
71}
72
73#[async_trait::async_trait]
74impl TargetObserver for StealthObserver {
75    fn name(&self) -> &'static str {
76        "stealth"
77    }
78
79    async fn on_target_attached(&self, session: PausedSession<'_>) -> Result<(), ObserverError> {
80        // Workers + iframes are skipped — workers have no DOM; iframes inherit
81        // patches via the parent in flat mode.
82        if session.target_info.kind != "page" {
83            return Ok(());
84        }
85        if self.profile.kind() == ProfileKind::Off {
86            return Ok(());
87        }
88
89        session.call("Page.enable", json!({})).await?;
90
91        // UA override — Emulation.setUserAgentOverride carries the Client-Hints
92        // metadata too, so we don't have to send Network.setUserAgentOverride
93        // separately.
94        let accept_language = {
95            let langs = crate::lang::resolve_languages(&Persona::default(), &self.fingerprint);
96            // `Emulation.setUserAgentOverride.acceptLanguage` wants a PLAIN
97            // comma-separated locale list (e.g. `en-US,en`) — Chrome appends the
98            // `;q=` weights itself. Passing an already-weighted string (the
99            // `accept_language()` header form) makes Chrome double them, yielding
100            // a malformed `Accept-Language: en-US,en;q=0.9;q=0.9`. Send the bare
101            // list so the emitted header is a clean `en-US,en;q=0.9`.
102            langs.join(",")
103        };
104        session
105            .call(
106                "Emulation.setUserAgentOverride",
107                json!({
108                    "userAgent": &self.fingerprint.ua_string,
109                    "acceptLanguage": accept_language,
110                    "platform": self.fingerprint.platform.ch_platform(),
111                    "userAgentMetadata": &self.fingerprint.ua_metadata,
112                }),
113            )
114            .await?;
115
116        // Screen-size override + focus emulation: keeps headless from leaking
117        // an oddly-shaped viewport and from reporting `document.hasFocus()`
118        // false for the (always-backgrounded) headless tab.
119        session
120            .call(
121                "Emulation.setDeviceMetricsOverride",
122                json!({
123                    "width": 1920,
124                    "height": 1080,
125                    "deviceScaleFactor": 1.0,
126                    "mobile": false,
127                    "screenWidth": 1920,
128                    "screenHeight": 1080,
129                }),
130            )
131            .await?;
132
133        session
134            .call(
135                "Emulation.setFocusEmulationEnabled",
136                json!({ "enabled": true }),
137            )
138            .await?;
139
140        if let Some(ref tz) = self.fingerprint.timezone {
141            session
142                .call("Emulation.setTimezoneOverride", json!({ "timezoneId": tz }))
143                .await?;
144        }
145        if let Some(ref locale) = self.fingerprint.locale {
146            session
147                .call("Emulation.setLocaleOverride", json!({ "locale": locale }))
148                .await?;
149        }
150        if let Some(ref geo) = self.geolocation {
151            // Sets the value the Geolocation API *would* return — it does
152            // NOT grant the `geolocation` permission. Chrome still gates the
153            // API behind a permission prompt/grant; auto-granting it here
154            // would be a separate (and itself suspicious) signal, so we
155            // deliberately leave permissioning to the caller.
156            let mut params = json!({
157                "latitude": geo.latitude,
158                "longitude": geo.longitude,
159            });
160            if let Some(accuracy) = geo.accuracy {
161                params["accuracy"] = json!(accuracy);
162            }
163            session
164                .call("Emulation.setGeolocationOverride", params)
165                .await?;
166        }
167
168        if self.profile.kind() == ProfileKind::Spoofed {
169            if self.profile.bypass_csp_enabled() {
170                session
171                    .call("Page.setBypassCSP", json!({ "enabled": true }))
172                    .await?;
173            }
174            // Inject into the MAIN world (no `worldName`). The bootstrap's
175            // patches mutate `Navigator.prototype`, `window.chrome`,
176            // `WebGLRenderingContext.prototype`, etc. — every isolated
177            // world gets its own copy of these prototypes, so a patch
178            // applied in a named/isolated world is invisible to the
179            // page's own scripts (and to `evaluate_main`, the surface
180            // detection sites probe). Running the bootstrap in the main
181            // world is the only way these prototype mutations actually
182            // affect the document under test.
183            session
184                .call(
185                    "Page.addScriptToEvaluateOnNewDocument",
186                    json!({
187                        "source": &self.bootstrap,
188                        "includeCommandLineAPI": false,
189                        "runImmediately": true,
190                    }),
191                )
192                .await?;
193        }
194
195        Ok(())
196    }
197}

207    #[tokio::test]
208    async fn spoofed_observer_sends_expected_sequence_for_page_target() {
209        let fp = Fingerprint {
210            platform: Platform::MacIntel,
211            chrome_major: 120,
212            chrome_full: "120.0.6099.234".into(),
213            cpu_count: 10,
214            memory_gb: 8,
215            ua_string: crate::ua::compose_ua_string(Platform::MacIntel, "120.0.6099.234"),
216            ua_metadata: crate::UserAgentMetadata::realistic(
217                Platform::MacIntel,
218                120,
219                "120.0.6099.234",
220            ),
221            timezone: None,
222            locale: None,
223            languages: None,
224        };
225        let profile = StealthProfile::spoofed();
226        let observer = std::sync::Arc::new(StealthObserver::new(profile, fp));
227
228        let (mut mock, conn) = MockConnection::pair_with_observers(vec![observer.clone()]);
229
230        // Emit a Target.attachedToTarget event.
231        mock.emit_event(
232            "Target.attachedToTarget",
233            json!({
234                "sessionId": "S1",
235                "targetInfo": {
236                    "targetId": "T1",
237                    "type": "page",
238                    "url": "about:blank",
239                    "attached": true,
240                },
241                "waitingForDebugger": true,
242            }),
243        )
244        .await;
245
246        // Expected sequence (each followed by a reply so the observer
247        // continues). The closing Runtime.runIfWaitingForDebugger is the
248        // actor's debugger-release after every observer succeeds.
249        for expected in [
250            "Page.enable",
251            "Emulation.setUserAgentOverride",
252            "Emulation.setDeviceMetricsOverride",
253            "Emulation.setFocusEmulationEnabled",
254            "Page.setBypassCSP",
255            "Page.addScriptToEvaluateOnNewDocument",
256            "Runtime.runIfWaitingForDebugger",
257        ] {
258            let id =
259                tokio::time::timeout(std::time::Duration::from_secs(2), mock.expect_cmd(expected))
260                    .await
261                    .unwrap_or_else(|_| panic!("did not see {expected} within 2s"));
262            // `acceptLanguage` must be a PLAIN locale list — Chrome adds the
263            // `;q=` weights. A weighted value here makes Chrome double them into
264            // a malformed `en-US,en;q=0.9;q=0.9` header.
265            if expected == "Emulation.setUserAgentOverride" {
266                let al = mock.last_sent()["params"]["acceptLanguage"]
267                    .as_str()
268                    .unwrap_or_default()
269                    .to_string();
270                assert!(!al.is_empty(), "acceptLanguage must be set");
271                assert!(
272                    !al.contains(";q="),
273                    "acceptLanguage must be a bare locale list (no q-weights); got: {al}"
274                );
275            }
276            mock.reply(id, json!({})).await;
277        }
278
279        conn.shutdown();
280    }
