1//! StealthObserver: applies a [`StealthProfile`] to each new attached target.
2//!
3//! Installed in the [`zendriver_transport`] actor's observer chain. On
4//! `Target.attachedToTarget`, the actor pauses the new target, walks every
5//! observer serially, then releases the debugger via
6//! `Runtime.runIfWaitingForDebugger`. The observer's job is to push every
7//! UA/screen/timezone/locale override and (for spoofed) the bootstrap script
8//! _before_ the debugger releases, so the first script the page runs sees the
9//! patched globals.
10
11use serde_json::json;
12use zendriver_transport::{ObserverError, PausedSession, TargetObserver};
13
14use crate::patches::bootstrap_script;
15use crate::persona::GeoPos;
16use crate::persona::specs::{ScreenSpec, UaMetadata};
17use crate::{Fingerprint, Persona, ProfileKind, StealthProfile};

19/// Observer that applies a [`StealthProfile`] + [`Fingerprint`] to every page
20/// target. Workers and iframes are skipped — workers have no DOM and iframes
21/// inherit patches from the parent target in flat session mode.
22#[derive(Debug)]
23pub struct StealthObserver {
24    profile: StealthProfile,
25    fingerprint: Fingerprint,
26    /// Pre-rendered bootstrap source. Empty for `Off`/`Native` — we never send
27    /// `Page.addScriptToEvaluateOnNewDocument` in those modes, so there is no
28    /// need to pay the patches-bundle cost for them.
29    bootstrap: String,
30    /// Mock geolocation coordinates from the resolved [`Persona`], sent via
31    /// `Emulation.setGeolocationOverride`. Unlike `timezone`/`locale` (carried
32    /// on [`Fingerprint`]), geolocation has no `Fingerprint` counterpart, so
33    /// it's captured here straight off the persona at construction time.
34    geolocation: Option<GeoPos>,
35    /// Custom UA-CH from the resolved [`Persona`]'s `ua.ua_metadata`. When
36    /// set, [`UaMetadata::resolve`] fills any unset sub-field from
37    /// `fingerprint.ua_metadata` and the result drives
38    /// `Emulation.setUserAgentOverride.userAgentMetadata` in place of the
39    /// fingerprint-derived value outright.
40    ua_metadata: Option<UaMetadata>,
41    /// Custom screen / device-metrics from the resolved [`Persona`]. When
42    /// set, drives `Emulation.setDeviceMetricsOverride` in place of the
43    /// fixed 1920x1080 default.
44    screen: Option<ScreenSpec>,
45}
46
47impl StealthObserver {
48    /// Build a new observer. Bootstrap source is composed eagerly so the
49    /// per-target hot path only pays a `clone`/borrow.
50    ///
51    /// The bootstrap is driven by a [`Persona`] (surface-spoofing config) plus
52    /// the [`Fingerprint`] (coherent UA / Chrome identity). This constructor
53    /// uses [`Persona::default`] — no surface overrides, identity-only patches
54    /// keep their current behavior. The launch path (a later task) will thread
55    /// a caller-supplied persona via [`StealthObserver::with_persona`].
56    #[must_use]
57    pub fn new(profile: StealthProfile, fingerprint: Fingerprint) -> Self {
58        Self::with_persona(profile, fingerprint, Persona::default())
59    }
60
61    /// Build a new observer with an explicit [`Persona`] driving the surface
62    /// patches. `identity` still supplies the coherent UA / Chrome version.
63    #[must_use]
64    pub fn with_persona(
65        profile: StealthProfile,
66        fingerprint: Fingerprint,
67        persona: Persona,
68    ) -> Self {
69        let bootstrap = if profile.kind() == ProfileKind::Spoofed {
70            bootstrap_script(&persona, &fingerprint)
71        } else {
72            String::new()
73        };
74        let geolocation = persona.geolocation;
75        let ua_metadata = persona.ua.as_ref().and_then(|u| u.ua_metadata.clone());
76        let screen = persona.screen;
77        Self {
78            profile,
79            fingerprint,
80            bootstrap,
81            geolocation,
82            ua_metadata,
83            screen,
84        }
85    }
86}

88#[async_trait::async_trait]
89impl TargetObserver for StealthObserver {
90    fn name(&self) -> &'static str {
91        "stealth"
92    }
93
94    async fn on_target_attached(&self, session: PausedSession<'_>) -> Result<(), ObserverError> {
95        // Workers + iframes are skipped — workers have no DOM; iframes inherit
96        // patches via the parent in flat mode.
97        if session.target_info.kind != "page" {
98            return Ok(());
99        }
100        if self.profile.kind() == ProfileKind::Off {
101            return Ok(());
102        }
103
104        session.call("Page.enable", json!({})).await?;
105
106        // UA override — Emulation.setUserAgentOverride carries the Client-Hints
107        // metadata too, so we don't have to send Network.setUserAgentOverride
108        // separately.
109        let accept_language = {
110            let langs = crate::lang::resolve_languages(&Persona::default(), &self.fingerprint);
111            // `Emulation.setUserAgentOverride.acceptLanguage` wants a PLAIN
112            // comma-separated locale list (e.g. `en-US,en`) — Chrome appends the
113            // `;q=` weights itself. Passing an already-weighted string (the
114            // `accept_language()` header form) makes Chrome double them, yielding
115            // a malformed `Accept-Language: en-US,en;q=0.9;q=0.9`. Send the bare
116            // list so the emitted header is a clean `en-US,en;q=0.9`.
117            langs.join(",")
118        };
119        // Persona UA-CH wins when supplied — field-wise, falling back to the
120        // fingerprint-derived value for any sub-field the persona left
121        // unset. Absent persona UA-CH → today's behavior (fingerprint's
122        // UAM verbatim).
123        let user_agent_metadata = match &self.ua_metadata {
124            Some(custom) => custom.resolve(&self.fingerprint.ua_metadata),
125            None => self.fingerprint.ua_metadata.clone(),
126        };
127        session
128            .call(
129                "Emulation.setUserAgentOverride",
130                json!({
131                    "userAgent": &self.fingerprint.ua_string,
132                    "acceptLanguage": accept_language,
133                    "platform": self.fingerprint.platform.ch_platform(),
134                    "userAgentMetadata": &user_agent_metadata,
135                }),
136            )
137            .await?;
138
139        // Screen-size override + focus emulation: keeps headless from leaking
140        // an oddly-shaped viewport and from reporting `document.hasFocus()`
141        // false for the (always-backgrounded) headless tab. Persona screen
142        // wins when supplied; absent → today's fixed 1920x1080 default.
143        let (screen_width, screen_height, device_scale_factor) = match self.screen {
144            Some(s) => (s.width, s.height, s.device_pixel_ratio),
145            None => (1920, 1080, 1.0),
146        };
147        session
148            .call(
149                "Emulation.setDeviceMetricsOverride",
150                json!({
151                    "width": screen_width,
152                    "height": screen_height,
153                    "deviceScaleFactor": device_scale_factor,
154                    "mobile": false,
155                    "screenWidth": screen_width,
156                    "screenHeight": screen_height,
157                }),
158            )
159            .await?;
160
161        session
162            .call(
163                "Emulation.setFocusEmulationEnabled",
164                json!({ "enabled": true }),
165            )
166            .await?;
167
168        if let Some(ref tz) = self.fingerprint.timezone {
169            session
170                .call("Emulation.setTimezoneOverride", json!({ "timezoneId": tz }))
171                .await?;
172        }
173        if let Some(ref locale) = self.fingerprint.locale {
174            session
175                .call("Emulation.setLocaleOverride", json!({ "locale": locale }))
176                .await?;
177        }
178        if let Some(ref geo) = self.geolocation {
179            // Sets the value the Geolocation API *would* return — it does
180            // NOT grant the `geolocation` permission. Chrome still gates the
181            // API behind a permission prompt/grant; auto-granting it here
182            // would be a separate (and itself suspicious) signal, so we
183            // deliberately leave permissioning to the caller.
184            let mut params = json!({
185                "latitude": geo.latitude,
186                "longitude": geo.longitude,
187            });
188            if let Some(accuracy) = geo.accuracy {
189                params["accuracy"] = json!(accuracy);
190            }
191            session
192                .call("Emulation.setGeolocationOverride", params)
193                .await?;
194        }
195
196        if self.profile.kind() == ProfileKind::Spoofed {
197            if self.profile.bypass_csp_enabled() {
198                session
199                    .call("Page.setBypassCSP", json!({ "enabled": true }))
200                    .await?;
201            }
202            // Inject into the MAIN world (no `worldName`). The bootstrap's
203            // patches mutate `Navigator.prototype`, `window.chrome`,
204            // `WebGLRenderingContext.prototype`, etc. — every isolated
205            // world gets its own copy of these prototypes, so a patch
206            // applied in a named/isolated world is invisible to the
207            // page's own scripts (and to `evaluate_main`, the surface
208            // detection sites probe). Running the bootstrap in the main
209            // world is the only way these prototype mutations actually
210            // affect the document under test.
211            session
212                .call(
213                    "Page.addScriptToEvaluateOnNewDocument",
214                    json!({
215                        "source": &self.bootstrap,
216                        "includeCommandLineAPI": false,
217                        "runImmediately": true,
218                    }),
219                )
220                .await?;
221        }
222
223        Ok(())
224    }
225}

291            // `acceptLanguage` must be a PLAIN locale list — Chrome adds the

300                assert!(

302                    "acceptLanguage must be a bare locale list (no q-weights); got: {al}"

304            }
305            mock.reply(id, json!({})).await;
306        }
307
308        conn.shutdown();
309    }
