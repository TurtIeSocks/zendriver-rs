1//! StealthObserver: applies a [`StealthProfile`] to each new attached target.
2//!
3//! Installed in the [`zendriver_transport`] actor's observer chain. On
4//! `Target.attachedToTarget`, the actor pauses the new target, walks every
5//! observer serially, then releases the debugger via
6//! `Runtime.runIfWaitingForDebugger`. The observer's job is to push every
7//! UA/screen/timezone/locale override and (for spoofed) the bootstrap script
8//! _before_ the debugger releases, so the first script the page runs sees the
9//! patched globals.
10
11use serde_json::json;
12use zendriver_transport::{CallError, ObserverError, PausedSession, TargetObserver};
13
14use crate::patches::{bootstrap_script, bootstrap_script_native_webgl};
15use crate::persona::GeoPos;
16use crate::persona::specs::{ScreenSpec, UaMetadata};
17use crate::{Fingerprint, Persona, ProfileKind, StealthProfile};

19/// Observer that applies a [`StealthProfile`] + [`Fingerprint`] to every page
20/// target. Workers and iframes are skipped — workers have no DOM and iframes
21/// inherit patches from the parent target in flat session mode.
22#[derive(Debug)]
23pub struct StealthObserver {
24    profile: StealthProfile,
25    fingerprint: Fingerprint,
26    /// Pre-rendered bootstrap source. Empty for `Off`/`Native` — we never send
27    /// `Page.addScriptToEvaluateOnNewDocument` in those modes, so there is no
28    /// need to pay the patches-bundle cost for them.
29    bootstrap: String,
30    /// Mock geolocation coordinates from the resolved [`Persona`], sent via
31    /// `Emulation.setGeolocationOverride`. Unlike `timezone`/`locale` (carried
32    /// on [`Fingerprint`]), geolocation has no `Fingerprint` counterpart, so
33    /// it's captured here straight off the persona at construction time.
34    geolocation: Option<GeoPos>,
35    /// Custom UA-CH from the resolved [`Persona`]'s `ua.ua_metadata`. When
36    /// set, [`UaMetadata::resolve`] fills any unset sub-field from
37    /// `fingerprint.ua_metadata` and the result drives
38    /// `Emulation.setUserAgentOverride.userAgentMetadata` in place of the
39    /// fingerprint-derived value outright.
40    ua_metadata: Option<UaMetadata>,
41    /// Custom screen / device-metrics from the resolved [`Persona`]. When
42    /// set, drives `Emulation.setDeviceMetricsOverride` in place of the
43    /// fixed 1920x1080 default.
44    screen: Option<ScreenSpec>,
45}
46
47impl StealthObserver {
48    /// Build a new observer. Bootstrap source is composed eagerly so the
49    /// per-target hot path only pays a `clone`/borrow.
50    ///
51    /// The bootstrap is driven by a [`Persona`] (surface-spoofing config) plus
52    /// the [`Fingerprint`] (coherent UA / Chrome identity). This constructor
53    /// uses [`Persona::default`] — no surface overrides, identity-only patches
54    /// keep their current behavior. The launch path (a later task) will thread
55    /// a caller-supplied persona via [`StealthObserver::with_persona`].
56    #[must_use]
57    pub fn new(profile: StealthProfile, fingerprint: Fingerprint) -> Self {
58        Self::with_persona(profile, fingerprint, Persona::default())
59    }
60
61    /// Build a new observer with an explicit [`Persona`] driving the surface
62    /// patches. `identity` still supplies the coherent UA / Chrome version.
63    #[must_use]
64    pub fn with_persona(
65        profile: StealthProfile,
66        fingerprint: Fingerprint,
67        persona: Persona,
68    ) -> Self {
69        let bootstrap = if profile.kind() == ProfileKind::Spoofed {
70            if profile.native_webgl_enabled() {
71                bootstrap_script_native_webgl(&persona, &fingerprint)
72            } else {
73                bootstrap_script(&persona, &fingerprint)
74            }
75        } else {
76            String::new()
77        };
78        let geolocation = persona.geolocation;
79        let ua_metadata = persona.ua.as_ref().and_then(|u| u.ua_metadata.clone());
80        let screen = persona.screen;
81        Self {
82            profile,
83            fingerprint,
84            bootstrap,
85            geolocation,
86            ua_metadata,
87            screen,
88        }
89    }
90}

92#[async_trait::async_trait]
93impl TargetObserver for StealthObserver {
94    fn name(&self) -> &'static str {
95        "stealth"
96    }
97
98    async fn on_target_attached(&self, session: PausedSession<'_>) -> Result<(), ObserverError> {
99        // Workers + iframes are skipped — workers have no DOM; iframes inherit
100        // patches via the parent in flat mode.
101        if session.target_info.kind != "page" {
102            return Ok(());
103        }
104        if self.profile.kind() == ProfileKind::Off {
105            return Ok(());
106        }
107
108        session.call("Page.enable", json!({})).await?;
109
110        // UA override — Emulation.setUserAgentOverride carries the Client-Hints
111        // metadata too, so we don't have to send Network.setUserAgentOverride
112        // separately.
113        let accept_language = {
114            let langs = crate::lang::resolve_languages(&Persona::default(), &self.fingerprint);
115            // `Emulation.setUserAgentOverride.acceptLanguage` wants a PLAIN
116            // comma-separated locale list (e.g. `en-US,en`) — Chrome appends the
117            // `;q=` weights itself. Passing an already-weighted string (the
118            // `accept_language()` header form) makes Chrome double them, yielding
119            // a malformed `Accept-Language: en-US,en;q=0.9;q=0.9`. Send the bare
120            // list so the emitted header is a clean `en-US,en;q=0.9`.
121            langs.join(",")
122        };
123        // Persona UA-CH wins when supplied — field-wise, falling back to the
124        // fingerprint-derived value for any sub-field the persona left
125        // unset. Absent persona UA-CH → today's behavior (fingerprint's
126        // UAM verbatim).
127        let user_agent_metadata = match &self.ua_metadata {
128            Some(custom) => custom.resolve(&self.fingerprint.ua_metadata),
129            None => self.fingerprint.ua_metadata.clone(),
130        };
131        session
132            .call(
133                "Emulation.setUserAgentOverride",
134                json!({
135                    "userAgent": &self.fingerprint.ua_string,
136                    "acceptLanguage": accept_language,
137                    "platform": self.fingerprint.platform.ch_platform(),
138                    "userAgentMetadata": &user_agent_metadata,
139                }),
140            )
141            .await?;
142
143        // Screen-size override + focus emulation: keeps headless from leaking
144        // an oddly-shaped viewport and from reporting `document.hasFocus()`
145        // false for the (always-backgrounded) headless tab. Persona screen
146        // wins when supplied; absent → today's fixed 1920x1080 default.
147        let (screen_width, screen_height, device_scale_factor) = match self.screen {
148            Some(s) => (s.width, s.height, s.device_pixel_ratio),
149            None => (1920, 1080, 1.0),
150        };
151        session
152            .call(
153                "Emulation.setDeviceMetricsOverride",
154                json!({
155                    "width": screen_width,
156                    "height": screen_height,
157                    "deviceScaleFactor": device_scale_factor,
158                    "mobile": false,
159                    "screenWidth": screen_width,
160                    "screenHeight": screen_height,
161                }),
162            )
163            .await?;
164
165        session
166            .call(
167                "Emulation.setFocusEmulationEnabled",
168                json!({ "enabled": true }),
169            )
170            .await?;
171
172        if let Some(ref tz) = self.fingerprint.timezone {
173            session
174                .call("Emulation.setTimezoneOverride", json!({ "timezoneId": tz }))
175                .await?;
176        }
177        // Keep the JS-visible locale (navigator.language, Intl) coherent with
178        // the always-sent Accept-Language. Prefer an explicit fingerprint
179        // locale; otherwise, if the fingerprint pins a `languages` list, derive
180        // the locale from its primary entry — a `languages`-without-`locale`
181        // fingerprint must not leave the JS locale at Chrome's default while the
182        // Accept-Language header says otherwise. A pure-default fingerprint (no
183        // locale, no languages) keeps Chrome's native locale; we don't force one.
184        let effective_locale = self.fingerprint.locale.clone().or_else(|| {
185            self.fingerprint
186                .languages
187                .as_ref()
188                .and_then(|langs| langs.first())
189                .cloned()
190        });
191        if let Some(ref locale) = effective_locale {
192            // `Emulation.setLocaleOverride` is browser-global: only one override
193            // can be in effect at a time. A page with a cross-origin OOPIF (its
194            // own target -> its own session) attaches this observer per session,
195            // so a later session finds the override already set by the first and
196            // Chrome replies `[-32000] Another locale override is already in
197            // effect`. That is benign — the same coherent locale is already
198            // applied — and must NOT propagate: propagating detaches the observer
199            // and strips the frame's remaining patches (CSP bypass, the bootstrap
200            // fingerprint script). Tolerate that one reply; surface every other error.
201            match session
202                .call("Emulation.setLocaleOverride", json!({ "locale": locale }))
203                .await
204            {
205                Ok(_) => {}
206                Err(CallError::Rpc(_, ref message, _))
207                    if message.contains("locale override is already in effect") =>
208                {
209                    tracing::debug!(
210                        %locale,
211                        "locale override already in effect (set by an earlier session); tolerating"
212                    );
213                }
214                Err(e) => return Err(e.into()),
215            }
216        }
217        if let Some(ref geo) = self.geolocation {
218            // Sets the value the Geolocation API *would* return — it does
219            // NOT grant the `geolocation` permission. Chrome still gates the
220            // API behind a permission prompt/grant; auto-granting it here
221            // would be a separate (and itself suspicious) signal, so we
222            // deliberately leave permissioning to the caller.
223            let mut params = json!({
224                "latitude": geo.latitude,
225                "longitude": geo.longitude,
226            });
227            if let Some(accuracy) = geo.accuracy {
228                params["accuracy"] = json!(accuracy);
229            }
230            session
231                .call("Emulation.setGeolocationOverride", params)
232                .await?;
233        }
234
235        if self.profile.kind() == ProfileKind::Spoofed {
236            if self.profile.bypass_csp_enabled() {
237                session
238                    .call("Page.setBypassCSP", json!({ "enabled": true }))
239                    .await?;
240            }
241            // Inject into the MAIN world (no `worldName`). The bootstrap's
242            // patches mutate `Navigator.prototype`, `window.chrome`,
243            // `WebGLRenderingContext.prototype`, etc. — every isolated
244            // world gets its own copy of these prototypes, so a patch
245            // applied in a named/isolated world is invisible to the
246            // page's own scripts (and to `evaluate_main`, the surface
247            // detection sites probe). Running the bootstrap in the main
248            // world is the only way these prototype mutations actually
249            // affect the document under test.
250            session
251                .call(
252                    "Page.addScriptToEvaluateOnNewDocument",
253                    json!({
254                        "source": &self.bootstrap,
255                        "includeCommandLineAPI": false,
256                        "runImmediately": true,
257                    }),
258                )
259                .await?;
260        }
261
262        Ok(())
263    }
264}
