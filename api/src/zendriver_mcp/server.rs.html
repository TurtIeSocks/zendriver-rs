1//! rmcp server stack + tool router.
2//!
3//! Every `#[tool]` method on [`ZendriverServer`] is a one-liner that
4//! delegates to a free async fn in [`crate::tools`]. Tools are split
5//! across multiple `#[tool_router]` impl blocks: the always-on
6//! `base_tool_router`, and one cfg-gated block per feature
7//! (`interception`, `expect`, `cloudflare`, `fetcher`). The
8//! `combined_tool_router()` helper sums them (`ToolRouter` implements
9//! `Add`), and a hand-written `#[tool_handler]` impl wires that sum into
10//! [`rmcp::ServerHandler`]. This split is forced by [`tool_router`] not
11//! propagating per-method `#[cfg]` attributes — feature-gated tools must
12//! live in their own impl block that the macro can either generate or
13//! skip wholesale.
14//!
15//! Keeping the per-tool implementations in `tools/*.rs` (rather than
16//! inline here) makes the surface easy to grow: a new tool group adds a
17//! new module, a new wrapper here, and lands without touching the others.
18//!
19//! [`tool_router`]: rmcp::tool_router
20

47/// rmcp handler carrying the per-session [`SessionState`].
48///
49/// Clone is cheap (the only field is an `Arc<Mutex<_>>`).
50#[derive(Clone)]
51pub struct ZendriverServer {
52    pub state: Arc<Mutex<SessionState>>,
53}
54
55// Two `#[tool_router]` impl blocks: this one (always present) and the
56// `#[cfg(feature = "interception")]` one near the bottom of the file. The
57// `tool_router` macro can't see through `#[cfg]` attributes on individual
58// `#[tool]` methods, so we split feature-gated tools into a separate impl
59// block that the macro can either generate or skip wholesale. The two
60// per-block routers are then combined in `combined_tool_router()`, which
61// the manual `ServerHandler` impl below uses.
62#[tool_router(router = base_tool_router, vis = "pub")]
63impl ZendriverServer {
64    /// Construct a server bound to the given session state.
65    pub fn new(state: Arc<Mutex<SessionState>>) -> Self {
66        Self { state }
67    }

996        fetcher::install_chrome(self.state.clone(), input)
997            .await
998            .map(Json)
999    }
1000}

1002// ---------- combined router + ServerHandler -----------------------------
1003
1004impl ZendriverServer {
1005    /// Combine the always-on base router with every feature-gated router
1006    /// the build was compiled with (`interception`, `expect`,
1007    /// `cloudflare`, `fetcher`). The `tool_handler` impl below points at
1008    /// this so a single `ServerHandler::call_tool` / `list_tools` reaches
1009    /// every tool the build was compiled with.
1010    pub fn combined_tool_router() -> ToolRouter<Self> {
1011        let router = Self::base_tool_router();
1012        #[cfg(feature = "interception")]
1013        let router = router + Self::intercept_tool_router();
1014        #[cfg(feature = "expect")]
1015        let router = router + Self::expect_tool_router();
1016        #[cfg(feature = "cloudflare")]
1017        let router = router + Self::cloudflare_tool_router();
1018        #[cfg(feature = "imperva")]
1019        let router = router + Self::imperva_tool_router();
1020        #[cfg(feature = "fetcher")]
1021        let router = router + Self::fetcher_tool_router();
1022        router
1023    }
1024}
1025
1026#[tool_handler(router = Self::combined_tool_router())]
1027impl rmcp::ServerHandler for ZendriverServer {}

1029/// Build a fresh server handler bound to the given state.

1034/// Run the MCP server over stdio until the peer disconnects.
1035pub async fn run_stdio(state: Arc<Mutex<SessionState>>) -> Result<(), Box<dyn std::error::Error>> {
1036    let handler = build_handler(state);
1037    let service = handler.serve(stdio()).await?;
1038    service.waiting().await?;
1039    Ok(())
1040}
