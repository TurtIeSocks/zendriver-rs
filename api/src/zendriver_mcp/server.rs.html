1//! rmcp server stack + tool router.
2//!
3//! Every `#[tool]` method on [`ZendriverServer`] is a one-liner that
4//! delegates to a free async fn in [`crate::tools`]. Tools are split
5//! across multiple `#[tool_router]` impl blocks: the always-on
6//! `base_tool_router`, and one cfg-gated block per feature
7//! (`interception`, `expect`, `cloudflare`, `fetcher`). The
8//! `combined_tool_router()` helper sums them (`ToolRouter` implements
9//! `Add`), and a hand-written `#[tool_handler]` impl wires that sum into
10//! [`rmcp::ServerHandler`]. This split is forced by [`tool_router`] not
11//! propagating per-method `#[cfg]` attributes — feature-gated tools must
12//! live in their own impl block that the macro can either generate or
13//! skip wholesale.
14//!
15//! Keeping the per-tool implementations in `tools/*.rs` (rather than
16//! inline here) makes the surface easy to grow: a new tool group adds a
17//! new module, a new wrapper here, and lands without touching the others.
18//!
19//! [`tool_router`]: rmcp::tool_router
20

53/// rmcp handler carrying the per-session [`SessionState`].
54///
55/// Clone is cheap (the only field is an `Arc<Mutex<_>>`).
56#[derive(Clone)]
57pub struct ZendriverServer {
58    pub state: Arc<Mutex<SessionState>>,
59}
60
61// Two `#[tool_router]` impl blocks: this one (always present) and the
62// `#[cfg(feature = "interception")]` one near the bottom of the file. The
63// `tool_router` macro can't see through `#[cfg]` attributes on individual
64// `#[tool]` methods, so we split feature-gated tools into a separate impl
65// block that the macro can either generate or skip wholesale. The two
66// per-block routers are then combined in `combined_tool_router()`, which
67// the manual `ServerHandler` impl below uses.
68#[tool_router(router = base_tool_router, vis = "pub")]
69impl ZendriverServer {
70    /// Construct a server bound to the given session state.
71    pub fn new(state: Arc<Mutex<SessionState>>) -> Self {
72        Self { state }
73    }

1149/// Run the MCP server over stdio until the peer disconnects **or** a shutdown
1150/// signal arrives, then gracefully close any browser this session owns.
1151///
1152/// rmcp's `serve(stdio())` already returns when the peer closes stdin (EOF).
1153/// On that path process exit drops [`SessionState`], and an owned Chrome dies
1154/// via `kill_on_drop`. A SIGTERM/SIGINT, however — e.g. an MCP client
1155/// terminating the server on session end — kills the process before any async
1156/// `Drop` runs, orphaning the Chrome we launched. So we race the serve future
1157/// against a signal future and, on **either** path, take the browser out of the
1158/// session and [`Browser::close`] it (graceful SIGTERM→SIGKILL of the child).
1159/// `close` is a no-op for an attached/`connect`ed Chrome (the `owns_process`
1160/// guard), so a user's own browser is never killed.
1161///
1162/// Manual verification (needs a real Chrome): launch `zendriver-mcp`, drive a
1163/// `browser_open`, find the spawned Chrome PID (`pgrep -f
1164/// 'Chrome.*--headless'`), `kill -TERM` the server, then confirm that Chrome
1165/// PID is gone.
1166pub async fn run_stdio(state: Arc<Mutex<SessionState>>) -> Result<(), Box<dyn std::error::Error>> {
1167    let handler = build_handler(state.clone());
1168    let service = handler.serve(stdio()).await?;
1169    tokio::select! {
1170        res = service.waiting() => { res?; }
1171        () = shutdown_signal() => {
1172            tracing::info!("shutdown signal received; closing owned browser before exit");
1173        }
1174    }
1175    close_owned_browser(&state).await;
1176    Ok(())
1177}
1178
1179/// Resolve when the process receives a termination signal: SIGTERM or SIGINT
1180/// on Unix, Ctrl-C elsewhere.
1181async fn shutdown_signal() {
1182    #[cfg(unix)]
1183    {
1184        use tokio::signal::unix::{SignalKind, signal};
1185        let mut term = signal(SignalKind::terminate()).expect("install SIGTERM handler");
1186        let mut interrupt = signal(SignalKind::interrupt()).expect("install SIGINT handler");
1187        tokio::select! {
1188            _ = term.recv() => {}
1189            _ = interrupt.recv() => {}
1190        }
1191    }
1192    #[cfg(not(unix))]
1193    {
1194        let _ = tokio::signal::ctrl_c().await;
1195    }
1196}

1198/// Take the browser out of the session (if any) and gracefully close it.
1199/// Idempotent and panic-free when no browser was ever opened — it runs on
1200/// every shutdown path. [`Browser::close`] leaves an attached/`connect`ed
1201/// Chrome running (the `owns_process` guard), only the transport is torn down.
1202async fn close_owned_browser(state: &Arc<Mutex<SessionState>>) {
1203    let browser = { state.lock().await.browser.take() };
1204    if let Some(browser) = browser {
1205        if let Err(e) = browser.close().await {
1206            tracing::warn!("error closing browser during shutdown: {e}");
1207        }
1208    }
1209}
