1//! rmcp server stack + tool router.
2//!
3//! Every `#[tool]` method on [`ZendriverServer`] is a one-liner that
4//! delegates to a free async fn in [`crate::tools`]. Tools are split
5//! across multiple `#[tool_router]` impl blocks: the always-on
6//! `base_tool_router`, and one cfg-gated block per feature
7//! (`interception`, `expect`, `cloudflare`, `fetcher`). The
8//! `combined_tool_router()` helper sums them (`ToolRouter` implements
9//! `Add`), and a hand-written `#[tool_handler]` impl wires that sum into
10//! [`rmcp::ServerHandler`]. This split is forced by [`tool_router`] not
11//! propagating per-method `#[cfg]` attributes — feature-gated tools must
12//! live in their own impl block that the macro can either generate or
13//! skip wholesale.
14//!
15//! Keeping the per-tool implementations in `tools/*.rs` (rather than
16//! inline here) makes the surface easy to grow: a new tool group adds a
17//! new module, a new wrapper here, and lands without touching the others.
18//!
19//! [`tool_router`]: rmcp::tool_router
20

51/// rmcp handler carrying the per-session [`SessionState`].
52///
53/// Clone is cheap (the only field is an `Arc<Mutex<_>>`).
54#[derive(Clone)]
55pub struct ZendriverServer {
56    pub state: Arc<Mutex<SessionState>>,
57}
58
59// Two `#[tool_router]` impl blocks: this one (always present) and the
60// `#[cfg(feature = "interception")]` one near the bottom of the file. The
61// `tool_router` macro can't see through `#[cfg]` attributes on individual
62// `#[tool]` methods, so we split feature-gated tools into a separate impl
63// block that the macro can either generate or skip wholesale. The two
64// per-block routers are then combined in `combined_tool_router()`, which
65// the manual `ServerHandler` impl below uses.
66#[tool_router(router = base_tool_router, vis = "pub")]
67impl ZendriverServer {
68    /// Construct a server bound to the given session state.
69    pub fn new(state: Arc<Mutex<SessionState>>) -> Self {
70        Self { state }
71    }

1086// ---------- combined router + ServerHandler -----------------------------
1087
1088impl ZendriverServer {
1089    /// Combine the always-on base router with every feature-gated router
1090    /// the build was compiled with (`interception`, `expect`,
1091    /// `cloudflare`, `fetcher`). The `tool_handler` impl below points at
1092    /// this so a single `ServerHandler::call_tool` / `list_tools` reaches
1093    /// every tool the build was compiled with.
1094    pub fn combined_tool_router() -> ToolRouter<Self> {
1095        let router = Self::base_tool_router();
1096        #[cfg(feature = "interception")]
1097        let router = router + Self::intercept_tool_router();
1098        #[cfg(feature = "expect")]
1099        let router = router + Self::expect_tool_router();
1100        #[cfg(feature = "cloudflare")]
1101        let router = router + Self::cloudflare_tool_router();
1102        #[cfg(feature = "imperva")]
1103        let router = router + Self::imperva_tool_router();
1104        #[cfg(feature = "fetcher")]
1105        let router = router + Self::fetcher_tool_router();
1106        #[cfg(feature = "fingerprints")]
1107        let router = router + Self::fingerprints_tool_router();
1108        #[cfg(feature = "monitor")]
1109        let router = router + Self::monitor_tool_router();
1110        router
1111    }
1112}

1114#[tool_handler(router = Self::combined_tool_router())]
