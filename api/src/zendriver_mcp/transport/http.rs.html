1//! Streamable HTTP transport — per-session [`ZendriverServer`] factory
2//! mounted on axum.
3//!
4//! Each incoming MCP session triggers the factory, which mints a fresh
5//! [`SessionState`] (and therefore a fresh `Browser` slot). That keeps
6//! HTTP clients fully isolated — one client opening a browser does not
7//! affect another's tab state.
8
9use std::net::SocketAddr;
10use std::sync::Arc;
11
12use rmcp::transport::streamable_http_server::{
13    StreamableHttpServerConfig, StreamableHttpService, session::local::LocalSessionManager,
14};
15use tokio::sync::Mutex;
16
17use crate::server::ZendriverServer;
18use crate::state::{SessionState, StealthProfileChoice};
19
20/// Bind a streamable HTTP MCP server on `addr` and serve until the process
21/// is interrupted.
22///
23/// Each new MCP session is created with its own [`SessionState`] seeded
24/// with `default_profile` — clients never share browser state.
25pub async fn serve(
26    addr: SocketAddr,
27    default_profile: StealthProfileChoice,
28) -> Result<(), Box<dyn std::error::Error>> {
29    let service = StreamableHttpService::new(
30        move || {
31            // Fresh SessionState per session keeps each client's Browser
32            // / tabs isolated. The factory must be infallible — any
33            // failure here would surface as an `io::Error` to the client
34            // mid-handshake.
35            let state = Arc::new(Mutex::new(SessionState {
36                stealth_profile_choice: default_profile,
37                ..SessionState::new()
38            }));
39            Ok(ZendriverServer { state })
40        },
41        LocalSessionManager::default().into(),
42        StreamableHttpServerConfig::default(),
43    );
44
45    let router = axum::Router::new().nest_service("/mcp", service);
46    let listener = tokio::net::TcpListener::bind(addr).await?;
47    tracing::info!(%addr, "zendriver-mcp HTTP listening on /mcp");
48    axum::serve(listener, router).await?;
49    Ok(())
50}
