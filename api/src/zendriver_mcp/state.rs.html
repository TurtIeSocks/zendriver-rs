25/// Bounded capacity of each monitor's in-memory event ring.
26///
27/// Once a monitor's drain task has buffered this many unread events, the
28/// oldest is evicted on each new push and [`MonitorBuffer::dropped`] is
29/// incremented — so a slow `browser_monitor_read` caller bounds memory rather
30/// than growing it without limit. The next `read` surfaces the running
31/// `dropped` count so callers know events were lost.
32#[cfg(feature = "monitor")]
33pub const MONITOR_BUFFER_CAP: usize = 4096;

35/// Stealth profile choice carried over the MCP wire.
36///
37/// Concrete `StealthProfile` resolution happens inside the lifecycle
38/// handler (it depends on platform detection that only matters at launch
39/// time); the wire-level enum stays stable and platform-agnostic.
40#[derive(Debug, Clone, Copy, Default, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
41#[serde(rename_all = "snake_case")]
42pub enum StealthProfileChoice {
43    /// Defer to `zendriver_stealth::StealthProfile::native()` (auto-detect
44    /// platform via `sysinfo`).
45    #[default]
46    Auto,
47    /// Force native (same as `Auto`, but explicit).
48    Native,
49    /// Spoof a macOS fingerprint regardless of host platform.
50    SpoofMacos,
51    /// Spoof a Linux fingerprint regardless of host platform.
52    SpoofLinux,
53    /// Spoof a Windows fingerprint regardless of host platform.
54    SpoofWindows,
55}

62#[serde(rename_all = "snake_case")]

69    LinuxX86_64,

72/// Fine-grained stealth fingerprint overrides layered onto the chosen
73/// [`StealthProfileChoice`] at the next `browser_open`.
74///
75/// Every field is optional — an unset field leaves the base profile's value
76/// in place. Most meaningful paired with a `spoof_*` profile; applying these
77/// to `native` overrides the auto-detected real fingerprint and can *reduce*
78/// stealth if the values disagree with the host.
79#[derive(Debug, Clone, Default, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
80#[serde(deny_unknown_fields)]
81pub struct StealthOverrides {
82    /// Spoofed `navigator.platform` / UA platform token.
83    #[serde(default, skip_serializing_if = "Option::is_none")]
84    pub platform: Option<StealthPlatformChoice>,
85    /// Spoofed locale (e.g. `"en-US"`).
86    #[serde(default, skip_serializing_if = "Option::is_none")]
87    pub locale: Option<String>,
88    /// Spoofed timezone (IANA name, e.g. `"America/New_York"`).
89    #[serde(default, skip_serializing_if = "Option::is_none")]
90    pub timezone: Option<String>,
91    /// Spoofed `navigator.deviceMemory` in GiB.
92    #[serde(default, skip_serializing_if = "Option::is_none")]
93    pub memory_gb: Option<u32>,
94    /// Spoofed `navigator.hardwareConcurrency`.
95    #[serde(default, skip_serializing_if = "Option::is_none")]
96    pub cpu_count: Option<u32>,
97    /// Spoofed Chrome major version.
98    #[serde(default, skip_serializing_if = "Option::is_none")]
99    pub chrome_version: Option<u32>,
100    /// Full User-Agent string override.
101    #[serde(default, skip_serializing_if = "Option::is_none")]
102    pub user_agent: Option<String>,
103    /// Toggle Content-Security-Policy bypass.
104    #[serde(default, skip_serializing_if = "Option::is_none")]
105    pub bypass_csp: Option<bool>,
106    /// Opt in to Chrome's real site isolation (`IsolateOrigins`/
107    /// `site-per-process` stay enabled) and, for `spoof_*` profiles, skip
108    /// the WebGL vendor/renderer patch — the host's real WebGL renderer
109    /// passes through unpatched. **Trade-off, not a strict stealth
110    /// improvement**: dropping the WebGL patch removes an anti-WAF
111    /// coherence defense (see `StealthProfile::native_isolation` rustdoc).
112    /// Off (`false`/unset) by default — existing behavior unchanged.
113    #[serde(default, skip_serializing_if = "Option::is_none")]
114    pub native_isolation: Option<bool>,
115    /// Derive a coherent `locale` + `languages` from a country code
116    /// (ISO 3166-1 alpha-2, e.g. `"US"`). Overridden by an explicit `locale`.
117    ///
118    /// Always present in the schema so the wire shape is feature-stable; it
119    /// only takes effect when the server is built with the `geo` feature
120    /// (otherwise the field is accepted and ignored).
121    #[serde(default, skip_serializing_if = "Option::is_none")]
122    pub geo_country: Option<String>,
123}

125/// Input-timing profile choice carried over the MCP wire.
126///
127/// Wire mirror of `zendriver::stealth::InputProfile`'s two presets.
128/// Decoupled from [`StealthProfileChoice`] — an explicit choice here always
129/// wins over anything stealth would otherwise imply. When `browser_open`'s
130/// `input_profile` field is left unset, the *effective* profile instead
131/// follows the resolved stealth profile (spoofed stealth implies
132/// `Coherent`, `auto`/`native` stealth implies `Native`) — see
133/// `crate::tools::lifecycle::input_profile_choice_for`. This type's own
134/// `#[default]` (`Native`) is not consulted on that unset path; nothing in
135/// this crate currently calls `InputProfileChoice::default()`.
136#[derive(Debug, Clone, Copy, Default, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
137#[serde(rename_all = "snake_case")]
138pub enum InputProfileChoice {
139    /// Zero-overhead, deterministic timing (`InputProfile::native`). Default.
140    #[default]
141    Native,
142    /// Non-mechanical, humanized timing (`InputProfile::coherent`) — human-
143    /// paced typing and jittery mouse motion.
144    Coherent,
145}

167/// Live handle to a pending `expect_*` expectation.
168///
169/// The expectation is awaited inside a tokio task spawned by
170/// `browser_expect_register`; the task forwards the result through
171/// [`Self::rx`] (a `oneshot::Receiver`) carrying either the JSON-encoded
172/// matched-event or a textual error from the spawned task. The
173/// [`Self::task`] handle is retained so `browser_expect_cancel` can `.abort()`
174/// the in-flight `.matched()` future instead of leaving it orphaned until its
175/// inner timeout fires.
176///
177/// `kind` is a static label ("request" / "response" / "dialog" / "download")
178/// for diagnostics — not currently surfaced, but cheap to keep alongside.
179#[cfg(feature = "expect")]
180pub struct ExpectationHandle {
181    pub kind: &'static str,
182    pub task: tokio::task::JoinHandle<()>,
183    pub rx: tokio::sync::oneshot::Receiver<Result<serde_json::Value, String>>,
184}

186/// One MCP interception rule = one `zendriver_interception::InterceptHandle`.
187///
188/// Holding the handle is what keeps the rule live: dropping it (via
189/// [`HashMap::remove`] or [`HashMap::clear`]) cancels the actor and tears
190/// down `Fetch.enable` on that rule's session. `pattern` + `action_kind`
191/// are kept alongside so `browser_intercept_list_rules` can report back
192/// what each id corresponds to without poking at the handle's internals.
193#[cfg(feature = "interception")]
194pub struct InterceptRuleHandle {
195    pub pattern: String,
196    pub action_kind: &'static str,
197    pub _handle: zendriver::InterceptHandle,
198}

200/// A serde + JSON-Schema mirror of `zendriver::NetworkEvent` for the MCP wire.
201///
202/// One variant per observed network event. HTTP bodies are captured at
203/// observe-time by the drain task (before Chrome evicts them) when the monitor
204/// was started with `capture_bodies` — so `body` / `body_base64` (and the
205/// `body_truncated` / `body_full_bytes` / `body_capture_error` triple) are
206/// present only on `http` events from a body-capturing monitor.
207#[cfg(feature = "monitor")]
208#[derive(Debug, Clone, Serialize, JsonSchema, PartialEq, Eq)]
209#[serde(tag = "kind", rename_all = "snake_case")]
210pub enum MonitorEvent {
211    /// A completed HTTP request/response pair (or a failed request).
212    Http {
213        /// Request URL.
214        url: String,
215        /// HTTP method (e.g. `"GET"`, `"POST"`).
216        method: String,
217        /// Response status code, if a response was received before the request
218        /// finished (`None` for a network-level failure).
219        status: Option<u16>,
220        /// Network-level error text, if the request failed.
221        error: Option<String>,
222        /// UTF-8–lossy decode of the captured response body, bounded by
223        /// `capture_body_max_bytes`. Present only when the monitor captured
224        /// bodies and the fetch succeeded — a truncated body still decodes
225        /// (lossily) whatever prefix was kept.
226        #[serde(skip_serializing_if = "Option::is_none")]
227        body: Option<String>,
228        /// Base64 of the captured (possibly bounded) raw response body bytes.
229        /// Present only when the monitor captured bodies and the fetch
230        /// succeeded.
231        #[serde(skip_serializing_if = "Option::is_none")]
232        body_base64: Option<String>,
233        /// `true` if the body's full length exceeded `capture_body_max_bytes`
234        /// and `body` / `body_base64` hold only a prefix. `false` means the
235        /// entire body was captured. Present only alongside `body`.
236        #[serde(skip_serializing_if = "Option::is_none")]
237        body_truncated: Option<bool>,
238        /// The full (pre-truncation) response body length in bytes,
239        /// regardless of how much was kept in `body` / `body_base64`. Present
240        /// only alongside `body`.
241        #[serde(skip_serializing_if = "Option::is_none")]
242        body_full_bytes: Option<u64>,
243        /// Set instead of `body` / `body_base64` when body capture was
244        /// requested (`capture_bodies: true`) but the `getResponseBody` fetch
245        /// itself failed (e.g. Chrome already evicted the body) — distinct
246        /// from a genuinely empty body, which decodes to `body: ""` rather
247        /// than setting this field.
248        #[serde(skip_serializing_if = "Option::is_none")]
249        body_capture_error: Option<String>,
250    },
251    /// A new WebSocket connection was opened.
252    WebSocketOpen {
253        /// CDP request ID for this WebSocket connection.
254        request_id: String,
255        /// The WebSocket URL.
256        url: String,
257    },
258    /// A WebSocket frame was sent or received.
259    WebSocketFrame {
260        /// CDP request ID for the owning WebSocket connection.
261        request_id: String,
262        /// `"sent"` (page → server) or `"received"` (server → page).
263        direction: String,
264        /// WebSocket opcode (1 = text, 2 = binary, 8 = close, …).
265        opcode: u8,
266        /// Frame payload (text frames as UTF-8; binary frames as base64).
267        payload: String,
268    },
269    /// A WebSocket connection was closed.
270    WebSocketClose {
271        /// CDP request ID for the closed WebSocket connection.
272        request_id: String,
273    },
274    /// An SSE `EventSource` message was received.
275    EventSourceMessage {
276        /// CDP request ID for the `EventSource` stream.
277        request_id: String,
278        /// The SSE `event:` field (empty string if omitted).
279        event_name: String,
280        /// The SSE `id:` field (empty string if omitted).
281        event_id: String,
282        /// The SSE `data:` payload.
283        data: String,
284    },
285    /// A delivery-loss boundary on the monitor's underlying event stream (or
286    /// its own correlation bookkeeping) — a wire mirror of
287    /// `zendriver::NetworkDeliveryBoundary`. Previously these losses were
288    /// silent; they're now explicit events on the same stream. A
289    /// `disconnected` boundary means the underlying lib-side monitor's
290    /// correlator task has ended — no further events will be buffered for
291    /// this handle; start a new monitor to resume.
292    DeliveryBoundary {
293        /// `"lagged"` | `"reconnected"` | `"disconnected"` |
294        /// `"correlation_evicted"` | `"decode_failed"` | `"unknown"`.
295        boundary: String,
296        /// Present for `lagged` / `reconnected` / `disconnected`: the
297        /// transport connection generation active when the boundary occurred.
298        #[serde(skip_serializing_if = "Option::is_none")]
299        generation: Option<u64>,
300        /// Present for `lagged`: number of events this subscription missed.
301        #[serde(skip_serializing_if = "Option::is_none")]
302        missed: Option<u64>,
303        /// Present for `reconnected`: generation of the connection actor that
304        /// was replaced.
305        #[serde(skip_serializing_if = "Option::is_none")]
306        previous: Option<u64>,
307        /// Present for `correlation_evicted`: URL of the evicted in-flight
308        /// exchange.
309        #[serde(skip_serializing_if = "Option::is_none")]
310        url: Option<String>,
311    },
312}

314/// The bounded event ring shared between a monitor's drain task and its
315/// `read` handler.
316///
317/// Holding this behind its own `Arc<Mutex<_>>` (separate from [`MonitorState`])
318/// is what lets the drain task be spawned *before* [`MonitorState`] is
319/// constructed: the task captures a clone of the buffer `Arc`, and the real
320/// `JoinHandle` then goes into [`MonitorState`] with no placeholder spawn and
321/// no self-referential cycle.
322#[cfg(feature = "monitor")]
323#[derive(Debug, Default)]
324pub struct MonitorBuffer {
325    /// Bounded ring of buffered, not-yet-read events.
326    pub events: std::collections::VecDeque<MonitorEvent>,
327    /// Count of events evicted because the buffer was full since the last
328    /// `read`. Reset to 0 by `read`.
329    pub dropped: usize,
330}
331
332/// Live state for one running monitor.
333///
334/// A drain task (spawned by `browser_monitor_start`) owns the lib-side
335/// `NetworkMonitor` stream and pushes each converted [`MonitorEvent`] into the
336/// shared [`Self::buffer`] — a bounded ring capped at [`MONITOR_BUFFER_CAP`].
337/// When the buffer is full the oldest event is evicted and `dropped` is bumped.
338/// `browser_monitor_read` drains the buffer; `browser_monitor_stop` cancels the
339/// task via [`Self::cancel`] and aborts [`Self::task`].
340///
341/// Each mutex is held only briefly — the drain task locks the buffer per event,
342/// `read` locks it per drain, neither across an `.await` that waits on the
343/// other — so the drain/read pair cannot deadlock.
344#[cfg(feature = "monitor")]
345pub struct MonitorState {
346    /// Shared bounded ring (also held by the drain task).
347    pub buffer: std::sync::Arc<tokio::sync::Mutex<MonitorBuffer>>,
348    /// Cancels the drain task cooperatively (checked in its `select!`).
349    pub cancel: tokio_util::sync::CancellationToken,
350    /// The drain task's join handle, aborted on `stop`.
351    pub task: tokio::task::JoinHandle<()>,
352}
353
354#[cfg(feature = "monitor")]
355impl Drop for MonitorState {
356    /// Stop the drain task on drop (session teardown / `browser_close` /
357    /// map eviction). Dropping the `JoinHandle` alone detaches the task — it
358    /// holds its own `CancellationToken` clone and the live stream — so we
359    /// cancel (breaks the task's `select!`) and abort here.
360    fn drop(&mut self) {
361        self.cancel.cancel();
362        self.task.abort();
363    }
364}

366impl SessionState {
367    /// Construct an empty session — no browser, no tabs, default profile.

369        Self {
370            browser: None,
371            current_tab_id: None,
372            stealth_profile_choice: StealthProfileChoice::default(),
373            stealth_overrides: StealthOverrides::default(),
