1//! Shared helpers used by multiple tool modules.
2//!
3//! - [`EmptyInput`] — JSON-schema-bearing placeholder for tools that take no
4//!   arguments.
5//! - [`current_tab`] — resolves the current `zendriver::Tab` from a locked
6//!   [`SessionState`]. Returns an owned (cheap-to-clone) handle so callers
7//!   don't fight the borrow checker against `Browser::tabs(&self).await`'s
8//!   `Vec<Tab>`.
9
10use rmcp::ErrorData;
11use schemars::JsonSchema;
12use serde::Deserialize;
13
14use crate::errors::{McpServerError, map_error};
15use crate::state::SessionState;
16
17/// Placeholder input struct for tools with no arguments.
18///
19/// Required so rmcp can synthesize a JSON schema for the tool's args (an
20/// absent arg block would yield `null` schema).
21#[derive(Debug, Default, Deserialize, JsonSchema)]
22#[serde(deny_unknown_fields)]
23pub struct EmptyInput {}
24
25/// Resolve the currently-focused tab.
26///
27/// Returns an owned [`zendriver::Tab`] — `Tab` is an `Arc`-backed handle so
28/// cloning is cheap. The function takes ownership of the search rather than
29/// borrowing into the `Vec<Tab>` from `Browser::tabs()` because that vec
30/// gets dropped at the end of the function and we can't hand back a borrow.
31///
32/// # Errors
33///
34/// - [`McpServerError::BrowserNotOpen`] if `state.browser` is `None`.
35/// - [`McpServerError::NoCurrentTab`] if no tab matches
36///   `state.current_tab_id` (or `current_tab_id` is unset).
37pub async fn current_tab(s: &SessionState) -> Result<zendriver::Tab, ErrorData> {
38    let b = s
39        .browser
40        .as_ref()
41        .ok_or_else(|| map_error(McpServerError::BrowserNotOpen))?;
42    let id = s
43        .current_tab_id
44        .as_deref()
45        .ok_or_else(|| map_error(McpServerError::NoCurrentTab))?;
46    let tabs = b.tabs().await;
47    tabs.into_iter()
48        .find(|t| t.target_id() == id)
49        .ok_or_else(|| map_error(McpServerError::NoCurrentTab))
50}
