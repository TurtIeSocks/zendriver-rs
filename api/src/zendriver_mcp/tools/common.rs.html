1//! Shared helpers used by multiple tool modules.
2//!
3//! - [`EmptyInput`] — JSON-schema-bearing placeholder for tools that take no
4//!   arguments.
5//! - [`current_tab`] — resolves the current `zendriver::Tab` from a locked
6//!   [`SessionState`]. Returns an owned (cheap-to-clone) handle so callers
7//!   don't fight the borrow checker against `Browser::tabs(&self).await`'s
8//!   `Vec<Tab>`.
9

20/// Maximum blob size (bytes) returned inline as base64 before the caller is
21/// required to pass a `save_path`. 5 MiB keeps a single tool result well
22/// under typical MCP transport limits; larger captures (full-page PDFs,
23/// downloads) must go to disk.
24const MAX_INLINE_BYTES: usize = 5 * 1024 * 1024;
25
26/// Placeholder input struct for tools with no arguments.
27///
28/// Required so rmcp can synthesize a JSON schema for the tool's args (an
29/// absent arg block would yield `null` schema).
30#[derive(Debug, Default, Deserialize, JsonSchema)]
31#[serde(deny_unknown_fields)]
32pub struct EmptyInput {}

34/// Resolve the currently-focused tab.
35///
36/// Returns an owned [`zendriver::Tab`] — `Tab` is an `Arc`-backed handle so
37/// cloning is cheap. The function takes ownership of the search rather than
38/// borrowing into the `Vec<Tab>` from `Browser::tabs()` because that vec
39/// gets dropped at the end of the function and we can't hand back a borrow.
40///
41/// # Errors
42///
43/// - [`McpServerError::BrowserNotOpen`] if `state.browser` is `None`.
44/// - [`McpServerError::NoCurrentTab`] if no tab matches
45///   `state.current_tab_id` (or `current_tab_id` is unset).
46pub async fn current_tab(s: &SessionState) -> Result<zendriver::Tab, ErrorData> {
47    let b = s
48        .browser
49        .as_ref()
50        .ok_or_else(|| map_error(McpServerError::BrowserNotOpen))?;
51    let id = s
52        .current_tab_id
53        .as_deref()
54        .ok_or_else(|| map_error(McpServerError::NoCurrentTab))?;
55    let tabs = b.tabs().await;
56    tabs.into_iter()
57        .find(|t| t.target_id() == id)
58        .ok_or_else(|| map_error(McpServerError::NoCurrentTab))
59}

61/// Look up a [`zendriver::Frame`] on `tab` by id.
62///
63/// Surfaces [`ZendriverError::FrameNotFound`] through the standard error
64/// pipeline (which adds a `browser_frame_list` suggestion) when no frame
65/// matches. Shared by `eval`, `frames`, and `navigation`.
66pub async fn lookup_frame(
67    tab: &zendriver::Tab,
68    frame_id: &str,
69) -> Result<zendriver::Frame, ErrorData> {
70    let frames = tab
71        .frames()
72        .await
73        .map_err(|e| map_error(McpServerError::from(e)))?;
74    frames
75        .into_iter()
76        .find(|f| f.id() == frame_id)
77        .ok_or_else(|| {
78            map_error(McpServerError::from(
79                zendriver::ZendriverError::FrameNotFound(frame_id.to_string()),
80            ))
81        })
82}

84/// Collect a trimmed snapshot of the current rendered HTML.
85///
86/// Uses `document.documentElement.outerHTML` (rather than CDP's
87/// `Page.captureSnapshot`) so the result reflects post-script DOM mutations,
88/// then drops `<script>` / `<style>` and collapses whitespace. Shared by the
89/// tools (`scroll`, `mouse`) whose `return_snapshot` flag wants the same
90/// shape `navigation` / `actions` produce.
91pub async fn page_snapshot(tab: &zendriver::Tab) -> Result<String, ErrorData> {
92    let html: String = tab
93        .evaluate_main("document.documentElement.outerHTML")
94        .await
95        .map_err(|e| map_error(McpServerError::from(e)))?;
96    Ok(html_trim::trim(&html))
97}

99/// MCP-layer keyboard-modifier flag, mapped onto [`zendriver::KeyModifiers`].
100///
101/// Shared by coordinate-mouse clicks (`browser_mouse`) and keyboard chords
102/// (`browser_press` / `browser_key_sequence`).
103#[derive(Debug, Clone, Copy, Deserialize, Serialize, JsonSchema, PartialEq, Eq)]
104#[serde(rename_all = "lowercase")]
105pub enum ModifierArg {
106    /// `Alt` (Option on macOS).
107    Alt,
108    /// `Control`.
109    Ctrl,
110    /// `Meta` (Command on macOS, Windows key on Windows).
111    Meta,
112    /// `Shift`.
113    Shift,
114}
115
116/// Fold a slice of [`ModifierArg`] into a [`zendriver::KeyModifiers`] bitset.
117#[must_use]
118pub fn modifiers_to_bits(mods: &[ModifierArg]) -> zendriver::KeyModifiers {
119    let mut bits = zendriver::KeyModifiers::empty();
120    for m in mods {
121        bits |= match m {
122            ModifierArg::Alt => zendriver::KeyModifiers::ALT,
123            ModifierArg::Ctrl => zendriver::KeyModifiers::CTRL,
124            ModifierArg::Meta => zendriver::KeyModifiers::META,
125            ModifierArg::Shift => zendriver::KeyModifiers::SHIFT,
126        };
127    }
128    bits
129}

131/// Structured result for tools that produce a binary blob (PDF, MHTML,
132/// downloaded file).
133///
134/// When the caller passes a `save_path`, the bytes are written to disk on the
135/// MCP server host and only `{ saved_path, byte_len }` is returned. Otherwise
136/// the bytes are base64-inlined in `base64` (subject to [`MAX_INLINE_BYTES`]).
137#[derive(Debug, Serialize, JsonSchema)]
138pub struct BlobOutput {
139    /// Absolute/!relative path the bytes were written to, when `save_path`
140    /// was supplied. Path is on the MCP server host, not the client machine.
141    #[serde(skip_serializing_if = "Option::is_none")]
142    pub saved_path: Option<String>,
143    /// Size of the blob in bytes.
144    pub byte_len: usize,
145    /// Base64-encoded bytes. Populated only when no `save_path` was given and
146    /// the blob is within the inline size limit.
147    #[serde(skip_serializing_if = "Option::is_none")]
148    pub base64: Option<String>,
149}
150
151/// Build a [`BlobOutput`] from raw bytes + an optional save path.
152///
153/// # Errors
154///
155/// - The file write fails (`save_path` set).
156/// - The blob exceeds [`MAX_INLINE_BYTES`] and no `save_path` was given (the
157///   error message directs the caller to set one).
158pub fn blob_output(bytes: &[u8], save_path: Option<String>) -> Result<BlobOutput, ErrorData> {
159    let byte_len = bytes.len();
160    if let Some(path) = save_path {
161        std::fs::write(&path, bytes)
162            .map_err(|e| map_error(McpServerError::from(zendriver::ZendriverError::from(e))))?;
163        return Ok(BlobOutput {
164            saved_path: Some(path),
165            byte_len,
166            base64: None,
167        });
168    }
169    if byte_len > MAX_INLINE_BYTES {
170        return Err(ErrorData::invalid_params(
171            format!(
172                "Blob is {byte_len} bytes, over the {MAX_INLINE_BYTES}-byte inline limit. Pass `save_path` to write it to disk on the MCP server host instead of inlining it."
173            ),
174            None,
175        ));
176    }
177    Ok(BlobOutput {
178        saved_path: None,
179        byte_len,
180        base64: Some(BASE64.encode(bytes)),
181    })
182}
