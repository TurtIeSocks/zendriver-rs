1//! Expectation tools — `browser_expect_register / _await / _cancel`. Gated
2//! behind the `expect` feature.
3//!
4//! ## Register-then-await split
5//!
6//! The lib's API is one-call:
7//! `tab.expect_request(matcher).timeout(d).matched().await`. MCP's
8//! request/response cycle can't model an in-flight `.matched()` future across
9//! tool calls, so v0 splits the surface into three:
10//!
11//! - `browser_expect_register` constructs the expectation, applies an inner
12//!   `pre_await_timeout_ms` (default 60s — generous so the user has time to
13//!   call the trigger action tool in between), spawns a tokio task that
14//!   awaits `.matched()`, and pipes the result through a
15//!   `tokio::sync::oneshot::Sender`. The id is returned synchronously.
16//! - `browser_expect_await` takes ownership of the receiver from the
17//!   session's registry and `tokio::time::timeout`s on it.
18//! - `browser_expect_cancel` drops the registry entry + `.abort()`s the
19//!   spawned task so the lib-level `.matched()` future is killed promptly
20//!   rather than left to time out on its own.
21//!
22//! ## Wire-shape mappers + drive
23//!
24//! Each kind has a small async free fn that flattens its lib-side struct
25//! (`MatchedRequest` / `MatchedResponse` / `MatchedDialog` / `MatchedDownload`
26//! / `MatchedFileChooser`) into a `serde_json::Value` and, when asked,
27//! *drives* the matched object before it is dropped. The drive parameters
28//! live on `browser_expect_register` (not `_await`) because the spawned
29//! task owns the matched handle and several drive methods consume it
30//! (`accept` / `dismiss` / `save_to`); the decision must therefore be made
31//! at register time, before the triggering action runs:
32//!
33//! - **Dialog** — `dialog_action: accept|dismiss` (+ `dialog_prompt_text`)
34//!   drives the dialog so the page's blocking `alert/confirm/prompt` returns.
35//! - **Response** — `fetch_body: true` fetches the body and inlines it as
36//!   `body_base64` + `body_len`.
37//! - **Download** — `save_to: <path>` waits for completion and copies the file
38//!   to the MCP host, reporting `saved_path`.
39//! - **FileChooser** — `file_chooser_paths: [<path>, ...]` (required, must be
40//!   non-empty) is applied automatically the instant the chooser opens, via
41//!   `DOM.setFileInputFiles`; there is nothing left to drive by the time the
42//!   matched event is reported.
43

68/// Which `Page.*` / `Network.*` family the expectation listens on.
69#[derive(Debug, Clone, Copy, Deserialize, JsonSchema)]
70#[serde(rename_all = "snake_case")]
71pub enum ExpectKind {
72    /// `Network.requestWillBeSent` — outgoing request, no body fetch.
73    Request,
74    /// `Network.responseReceived` — incoming response headers (no body in v0).
75    Response,
76    /// `Page.javascriptDialogOpened` — alert/confirm/prompt/beforeunload.
77    Dialog,
78    /// `Page.downloadWillBegin` — start of a file download.
79    Download,
80    /// `Page.fileChooserOpened` — a button/label-triggered (or direct)
81    /// file picker, answered with `file_chooser_paths`.
82    FileChooser,
83}
84
85impl ExpectKind {
86    /// Static label used by [`ExpectationHandle::kind`] and surfaced
87    /// indirectly through the matched-event `kind` field for traceability.
88    fn label(self) -> &'static str {
89        match self {
90            Self::Request => "request",
91            Self::Response => "response",
92            Self::Dialog => "dialog",
93            Self::Download => "download",
94            Self::FileChooser => "file_chooser",
95        }
96    }
97}

99/// How to drive a matched JavaScript dialog (`kind: dialog` only).
100///
101/// Applied by the spawned task the instant the dialog opens — so the page's
102/// blocking `alert()` / `confirm()` / `prompt()` call returns promptly and
103/// the matched event carries the chosen action. Omitting `dialog_action`
104/// leaves the dialog untouched: the matched handle is dropped without
105/// dispatching `Page.handleJavaScriptDialog` (`MatchedDialog` has no `Drop`
106/// impl, so nothing is sent to Chrome), so the dialog stays open and the
107/// page's blocking JS call does not return until something else resolves
108/// it. Use this to merely observe that a dialog opened without acting on
109/// it.
110#[derive(Debug, Clone, Copy, Deserialize, Serialize, JsonSchema, PartialEq, Eq)]
111#[serde(rename_all = "snake_case")]
112pub enum DialogAction {
113    /// Accept the dialog. For a `prompt`, submit `dialog_prompt_text` (or the
114    /// dialog's default value when that is unset).
115    Accept,
116    /// Dismiss / cancel the dialog.
117    Dismiss,
118}

120/// URL match predicate for request/response expectations. Dialog and
121/// Download kinds ignore matcher fields entirely (the lib has no URL filter
122/// for them in v0).
123///
124/// Exactly zero or one of `url_substr` / `url_regex` may be set. Setting
125/// neither matches every URL (substring matcher with the empty string).
126#[derive(Debug, Default, Deserialize, JsonSchema)]
127#[serde(deny_unknown_fields)]
128pub struct ExpectMatcher {
129    /// Substring matcher — match if request URL contains this needle.
130    #[serde(default)]
131    pub url_substr: Option<String>,
132    /// Regex matcher — match if `regex::Regex::is_match(url)` returns true.
133    /// Compiled inside `_register`; an invalid regex surfaces as an
134    /// invalid-request error.
135    #[serde(default)]
136    pub url_regex: Option<String>,
137}
138
139impl ExpectMatcher {
140    /// Build a [`UrlMatcher`] from the wire-level matcher.
141    ///
142    /// Priority: `url_regex` wins over `url_substr` when both are set —
143    /// flagging that on the schema would be nicer, but a serde validator
144    /// would be intrusive; document the precedence here and let agents pick
145    /// one.
146    fn into_url_matcher(self) -> Result<UrlMatcher, ErrorData> {
147        if let Some(re) = self.url_regex {
148            let compiled = regex::Regex::new(&re).map_err(|e| {
149                ErrorData::invalid_request(format!("invalid url_regex `{re}`: {e}"), None)
150            })?;
151            return Ok(UrlMatcher::Regex(compiled));
152        }
153        if let Some(sub) = self.url_substr {
154            return Ok(UrlMatcher::Substring(sub));
155        }
156        // Empty-string substring matcher matches every URL — easiest "any"
157        // sentinel that doesn't require a separate variant.
158        Ok(UrlMatcher::Substring(String::new()))
159    }
160}

164/// Input for `browser_expect_register`.
165#[derive(Debug, Deserialize, JsonSchema)]
166#[serde(deny_unknown_fields)]
167pub struct RegisterInput {
168    /// Which event family to watch.
169    pub kind: ExpectKind,
170    /// URL matcher (request/response only — ignored for dialog/download).
171    #[serde(default)]
172    pub matcher: Option<ExpectMatcher>,
173    /// Inner timeout applied to the lib's `.timeout(d)` call. Generous by
174    /// default (60s) so the user has time to call the action tool that
175    /// triggers the event between `register` and `await`.
176    #[serde(default = "default_pre_timeout")]
177    pub pre_await_timeout_ms: u64,
178    /// Dialog only: drive the matched dialog (`accept` / `dismiss`) instead
179    /// of leaving it open for observation only. Decided here, at register
180    /// time, because the matched dialog handle is consumed by the spawned
181    /// task.
182    #[serde(default, skip_serializing_if = "Option::is_none")]
183    pub dialog_action: Option<DialogAction>,
184    /// Dialog only: prompt answer for `dialog_action: accept` on a `prompt`.
185    #[serde(default, skip_serializing_if = "Option::is_none")]
186    pub dialog_prompt_text: Option<String>,
187    /// Response only: fetch the response body and include it (base64) in the
188    /// matched event as `body_base64` + `body_len`. One extra CDP round-trip.
189    #[serde(default)]
190    pub fetch_body: bool,
191    /// Download only: copy the completed download to this path on the MCP
192    /// host; the matched event reports it as `saved_path`.
193    #[serde(default, skip_serializing_if = "Option::is_none")]
194    pub save_to: Option<String>,
195    /// File-chooser only: absolute paths (on the MCP host) to feed into the
196    /// chooser the instant it opens, via `DOM.setFileInputFiles`. Required
197    /// for `kind: file_chooser` — an empty list clears the input instead of
198    /// attaching anything, which is rarely what's wanted, so an empty/absent
199    /// list surfaces an `invalid_request` rather than silently no-op'ing.
200    #[serde(default)]
201    pub file_chooser_paths: Vec<String>,
202}
203
204fn default_pre_timeout() -> u64 {
205    60_000
206}

216/// Register a one-shot expectation against the current tab.
217///
218/// Spawns a tokio task that constructs the appropriate `tab.expect_*` call,
219/// applies the `pre_await_timeout_ms`, awaits `.matched()`, and pipes the
220/// result (JSON-encoded matched-event, or a textual error) through a
221/// `oneshot::Sender`. The task's `JoinHandle` is stored alongside the
222/// receiver so `_cancel` can `.abort()` it promptly.
223pub async fn register(
224    state: Arc<Mutex<SessionState>>,
225    input: RegisterInput,
226) -> Result<RegisterOutput, ErrorData> {
227    let pre_await = Duration::from_millis(input.pre_await_timeout_ms);
228    let matcher = input.matcher.unwrap_or_default().into_url_matcher()?;
229    let kind = input.kind;
230    // Drive params, captured before the per-kind spawn moves them in. Each is
231    // meaningful for exactly one kind (see field docs); the others are unused.
232    let dialog_action = input.dialog_action;
233    let dialog_prompt = input.dialog_prompt_text;
234    let fetch_body = input.fetch_body;
235    let save_to = input.save_to;
236    let file_chooser_paths = input.file_chooser_paths;
237    if matches!(kind, ExpectKind::FileChooser) && file_chooser_paths.is_empty() {
238        return Err(ErrorData::invalid_request(
239            "kind: file_chooser requires a non-empty file_chooser_paths",
240            None,
241        ));
242    }
243
244    let tab = {
245        let s = state.lock().await;
246        current_tab(&s).await?
247    };
248    // Captured before the per-kind spawn below (which may move/consume
249    // `tab`) so the registered handle can be reaped by `browser_tab_close`
250    // when this tab closes — see `state::ExpectationHandle::tab_id`.
251    let tab_id = tab.target_id().to_string();
252
253    let (tx, rx) = oneshot::channel::<Result<Value, String>>();
254
255    // One spawn per kind: each branch constructs its expectation against the
256    // tab handle (cloneable Arc-backed), applies the pre-await timeout, and
257    // forwards the matched event (or error string) over the oneshot. We
258    // can't share the spawned body across kinds because each kind's
259    // expectation type is different and `.matched()` returns a different
260    // matched-event struct.
261    let task: tokio::task::JoinHandle<()> = match kind {
262        ExpectKind::Request => {
263            let exp = tab.expect_request(matcher).timeout(pre_await);
264            tokio::spawn(async move {
265                let msg = exp.matched().await.map(request_to_json).map_err(err_to_str);
266                let _ = tx.send(msg);
267            })
268        }
269        ExpectKind::Response => {
270            let exp = tab.expect_response(matcher).timeout(pre_await);
271            tokio::spawn(async move {
272                let msg = match exp.matched().await {
273                    Ok(m) => response_to_json(m, fetch_body).await,
274                    Err(e) => Err(err_to_str(e)),
275                };
276                let _ = tx.send(msg);
277            })
278        }
279        ExpectKind::Dialog => {
280            let exp = tab.expect_dialog().timeout(pre_await);
281            tokio::spawn(async move {
282                let msg = match exp.matched().await {
283                    Ok(m) => dialog_to_json(m, dialog_action, dialog_prompt).await,
284                    Err(e) => Err(err_to_str(e)),
285                };
286                let _ = tx.send(msg);
287            })
288        }
289        ExpectKind::Download => {
290            // expect_download is async (lazy per-tab setup of the download
291            // coordinator + Browser.setDownloadBehavior dispatch). Surface
292            // its setup error inline instead of stashing the expectation —
293            // a registration we can never await isn't worth recording.
294            let exp = tab
295                .expect_download()
296                .await
297                .map_err(|e| map_error(McpServerError::from(e)))?
298                .timeout(pre_await);
299            tokio::spawn(async move {
300                let msg = match exp.matched().await {
301                    Ok(m) => download_to_json(m, save_to).await,
302                    Err(e) => Err(err_to_str(e)),
303                };
304                let _ = tx.send(msg);
305            })
306        }
307        ExpectKind::FileChooser => {
308            // expect_file_chooser is async (it must await
309            // Page.setInterceptFileChooserDialog{enabled:true} reaching
310            // Chrome before returning). Surface its setup error inline —
311            // same reasoning as the Download arm above.
312            let exp = tab
313                .expect_file_chooser(&file_chooser_paths)
314                .await
315                .map_err(|e| map_error(McpServerError::from(e)))?
316                .timeout(pre_await);
317            tokio::spawn(async move {
318                let msg = exp
319                    .matched()
320                    .await
321                    .map(file_chooser_to_json)
322                    .map_err(err_to_str);
323                let _ = tx.send(msg);
324            })
325        }
326    };
327
328    let id: ExpectationId = uuid::Uuid::new_v4().to_string();
329    {
330        let mut s = state.lock().await;
331        s.expectations.insert(
332            id.clone(),
333            ExpectationHandle {
334                kind: kind.label(),
335                task,
336                rx,
337                tab_id,
338            },
339        );
340    }
341    Ok(RegisterOutput { expectation_id: id })
342}

344/// Collapse a [`ZendriverError`] surfaced by the spawned task to a string
345/// suitable for the `Err` branch of the registry's `oneshot` channel. We
346/// flatten to a string because `serde_json::Value` cannot itself round-trip
347/// a `ZendriverError` and the error eventually surfaces through a plain
348/// invalid-request anyway.
349fn err_to_str(e: ZendriverError) -> String {
350    e.to_string()
351}

371/// Output of `browser_expect_await`.
372#[derive(Debug, Serialize, JsonSchema)]
373pub struct AwaitOutput {
374    /// Echoed id of the awaited expectation (matches input).
375    pub expectation_id: ExpectationId,
376    /// JSON-encoded matched event. Shape depends on the expectation's
377    /// `kind` — see the per-kind `*_to_json` helpers in this module.
378    pub event: Value,
379}
380
381/// Wait for a previously-registered expectation to resolve.
382///
383/// Removes the entry from `s.expectations` (taking ownership of the
384/// `oneshot::Receiver` + `JoinHandle`), then applies the caller's outer
385/// timeout to the channel. The spawned task's `pre_await_timeout_ms` is
386/// independent — if it fires first, the channel resolves with `Ok(Err(_))`
387/// carrying the lib's `ZendriverError::Timeout` as a string.
388pub async fn await_expectation(
389    state: Arc<Mutex<SessionState>>,
390    input: AwaitInput,
391) -> Result<AwaitOutput, ErrorData> {
392    let handle = {
393        let mut s = state.lock().await;
394        s.expectations
395            .remove(&input.expectation_id)
396            .ok_or_else(|| {
397                map_error(McpServerError::ExpectationNotFound(
398                    input.expectation_id.clone(),
399                ))
400            })?
401    };
402    let ExpectationHandle { rx, task, .. } = handle;
403
404    let outer = Duration::from_millis(input.timeout_ms);
405    match tokio::time::timeout(outer, rx).await {
406        Err(_elapsed) => {
407            // Outer timeout fired — abort the spawned task so its
408            // `.matched()` future stops holding a subscription stream open.
409            task.abort();
410            Err(ErrorData::invalid_request(
411                format!(
412                    "expectation `{}` await_timed_out after {:?}",
413                    input.expectation_id, outer
414                ),
415                None,
416            ))
417        }
418        Ok(Err(_recv)) => {
419            // Sender dropped without sending — spawned task panicked or was
420            // aborted externally. The task may still be in a transient
421            // state; abort defensively to be sure.
422            task.abort();
423            Err(ErrorData::invalid_request(
424                format!(
425                    "expectation `{}` channel_closed (spawned task ended without sending)",
426                    input.expectation_id
427                ),
428                None,
429            ))
430        }
431        Ok(Ok(Err(err_str))) => {
432            // Spawned task produced an error — lib-level failure or inner
433            // pre_await_timeout fired. Surface it on the wire.
434            Err(ErrorData::invalid_request(
435                format!("expectation `{}` failed: {err_str}", input.expectation_id),
436                None,
437            ))
438        }
439        Ok(Ok(Ok(event))) => Ok(AwaitOutput {
440            expectation_id: input.expectation_id,
441            event,
442        }),
443    }
444}

448/// Input for `browser_expect_cancel`.
449#[derive(Debug, Deserialize, JsonSchema)]
450#[serde(deny_unknown_fields)]
451pub struct CancelInput {
452    /// Id returned by an earlier `browser_expect_register` call.
453    pub expectation_id: ExpectationId,
454}
455
456/// Output of `browser_expect_cancel`.
457#[derive(Debug, Serialize, JsonSchema)]
458pub struct CancelOutput {
459    /// Always `true` on success. Cancellation of an unknown id returns
460    /// [`McpServerError::ExpectationNotFound`] rather than `false` here.
461    pub cancelled: bool,
462}
463
464/// Cancel a pending expectation, aborting its spawned task.
465///
466/// Drops the registry entry (which drops the `oneshot::Receiver`) and
467/// `.abort()`s the spawned task. Any in-flight `.matched()` future inside
468/// the task is dropped via abort, which tears down its lib-side subscriber.
469pub async fn cancel(
470    state: Arc<Mutex<SessionState>>,
471    input: CancelInput,
472) -> Result<CancelOutput, ErrorData> {
473    let mut s = state.lock().await;
474    let handle = s
475        .expectations
476        .remove(&input.expectation_id)
477        .ok_or_else(|| {
478            map_error(McpServerError::ExpectationNotFound(
479                input.expectation_id.clone(),
480            ))
481        })?;
482    handle.task.abort();
483    Ok(CancelOutput { cancelled: true })
484}

488/// Flatten a [`MatchedRequest`] to a wire JSON object.
489fn request_to_json(m: MatchedRequest) -> Value {
490    json!({
491        "kind": "request",
492        "url": m.url,
493        "method": m.method,
494        "headers": m.headers,
495        "request_id": m.request_id,
496        // post_data: emit only the byte length to avoid blowing up the
497        // wire for large POST bodies. Callers needing the bytes can use a
498        // future tool that fetches them explicitly.
499        "post_data_len": m.post_data.as_ref().map(Vec::len),
500    })
501}
502
503/// Flatten a [`MatchedResponse`] to a wire JSON object, optionally fetching
504/// the response body (base64) when `fetch_body` is set.
505///
506/// `body()` is an async CDP round-trip, so it is opt-in: agents that only
507/// need status/headers skip it.
508async fn response_to_json(m: MatchedResponse, fetch_body: bool) -> Result<Value, String> {
509    let mut v = json!({
510        "kind": "response",
511        "url": m.url,
512        "status": m.status,
513        "status_text": m.status_text,
514        "headers": m.headers,
515        "request_id": m.request_id,
516    });
517    if fetch_body {
518        let bytes = m.body().await.map_err(err_to_str)?;
519        v["body_len"] = json!(bytes.len());
520        v["body_base64"] = json!(BASE64.encode(&bytes));
521    }
522    Ok(v)
523}

535/// Flatten a [`MatchedDialog`] to a wire JSON object, driving the dialog
536/// (`accept` / `dismiss`) when `action` is set.
537///
538/// Fields are captured by reference first so the handle stays intact for the
539/// consuming `accept`/`dismiss` call. With no `action`, the handle is merely
540/// dropped here — `MatchedDialog` has no `Drop` impl, so no CDP call is
541/// made and the dialog is left open (observe-only).
542async fn dialog_to_json(
543    m: MatchedDialog,
544    action: Option<DialogAction>,
545    prompt: Option<String>,
546) -> Result<Value, String> {
547    let dialog_type = dialog_type_str(&m.dialog_type);
548    let message = m.message.clone();
549    let default_prompt = m.default_prompt.clone();
550    let url = m.url.clone();
551    let driven = match action {
552        Some(DialogAction::Accept) => {
553            m.accept(prompt).await.map_err(err_to_str)?;
554            "accept"
555        }
556        Some(DialogAction::Dismiss) => {
557            m.dismiss().await.map_err(err_to_str)?;
558            "dismiss"
559        }
560        None => {
561            drop(m);
562            "default"
563        }
564    };
565    Ok(json!({
566        "kind": "dialog",
567        "dialog_type": dialog_type,
568        "message": message,
569        "default_prompt": default_prompt,
570        "url": url,
571        "driven": driven,
572    }))
573}

575/// Flatten a [`MatchedDownload`] to a wire JSON object, optionally copying the
576/// completed download to `save_to` on the MCP host.
577///
578/// `MatchedDownload::save_to` waits for completion before copying, so the
579/// reported `saved_path` is a fully-written file.
580async fn download_to_json(m: MatchedDownload, save_to: Option<String>) -> Result<Value, String> {
581    let url = m.url.clone();
582    let suggested_filename = m.suggested_filename.clone();
583    let guid = m.guid.clone();
584    let download_dir = m.download_dir.to_string_lossy().into_owned();
585    let saved_path = if let Some(dest) = save_to {
586        m.save_to(PathBuf::from(&dest)).await.map_err(err_to_str)?;
587        Some(dest)
588    } else {
589        None
590    };
591    Ok(json!({
592        "kind": "download",
593        "url": url,
594        "suggested_filename": suggested_filename,
595        "guid": guid,
596        "download_dir": download_dir,
597        "saved_path": saved_path,
598    }))
599}

612fn file_chooser_to_json(m: MatchedFileChooser) -> Value {
613    json!({
614        "kind": "file_chooser",
615        "mode": file_chooser_mode_str(m.mode),
616    })

632    #[test]

634        let m = ExpectMatcher::default().into_url_matcher().unwrap();
635        match m {
636            UrlMatcher::Substring(s) => assert!(
637                s.is_empty(),
638                "default matcher must be empty substring (matches any url)",

640            UrlMatcher::Regex(_) => panic!("expected Substring(\"\")"),

644    #[test]
645    fn matcher_url_regex_wins_over_url_substr() {
646        // Documenting precedence: regex is checked first in
647        // `into_url_matcher`. Confirms agents picking both get the regex.
648        let m = ExpectMatcher {
649            url_substr: Some("/foo/".into()),
650            url_regex: Some(r"^https://api\.".into()),
651        }
652        .into_url_matcher()
653        .unwrap();
654        match m {
655            UrlMatcher::Regex(re) => assert!(re.is_match("https://api.example.com/v1")),
656            UrlMatcher::Substring(_) => panic!("expected Regex variant"),
657        }
658    }
