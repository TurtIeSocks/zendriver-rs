1//! Expectation tools — `browser_expect_register / _await / _cancel`. Gated
2//! behind the `expect` feature.
3//!
4//! ## Register-then-await split
5//!
6//! The lib's API is one-call:
7//! `tab.expect_request(matcher).timeout(d).matched().await`. MCP's
8//! request/response cycle can't model an in-flight `.matched()` future across
9//! tool calls, so v0 splits the surface into three:
10//!
11//! - `browser_expect_register` constructs the expectation, applies an inner
12//!   `pre_await_timeout_ms` (default 60s — generous so the user has time to
13//!   call the trigger action tool in between), spawns a tokio task that
14//!   awaits `.matched()`, and pipes the result through a
15//!   `tokio::sync::oneshot::Sender`. The id is returned synchronously.
16//! - `browser_expect_await` takes ownership of the receiver from the
17//!   session's registry and `tokio::time::timeout`s on it.
18//! - `browser_expect_cancel` drops the registry entry + `.abort()`s the
19//!   spawned task so the lib-level `.matched()` future is killed promptly
20//!   rather than left to time out on its own.
21//!
22//! ## Wire-shape mappers (`event_to_json`)
23//!
24//! Each kind has a small free fn that flattens its lib-side struct
25//! (`MatchedRequest` / `MatchedResponse` / `MatchedDialog` / `MatchedDownload`)
26//! into a `serde_json::Value`. Bodies (`MatchedResponse::body()`,
27//! `MatchedDownload::save_to(...)`) are NOT fetched in v0 — they require an
28//! extra round-trip and aren't part of the event itself. Dialog
29//! accept/dismiss is also not exposed; Chrome's default handling fires
30//! when the `MatchedDialog` is dropped at task end.
31

52/// Which `Page.*` / `Network.*` family the expectation listens on.
53#[derive(Debug, Clone, Copy, Deserialize, JsonSchema)]
54#[serde(rename_all = "snake_case")]
55pub enum ExpectKind {
56    /// `Network.requestWillBeSent` — outgoing request, no body fetch.
57    Request,
58    /// `Network.responseReceived` — incoming response headers (no body in v0).
59    Response,
60    /// `Page.javascriptDialogOpened` — alert/confirm/prompt/beforeunload.
61    Dialog,
62    /// `Page.downloadWillBegin` — start of a file download.
63    Download,
64}
65
66impl ExpectKind {
67    /// Static label used by [`ExpectationHandle::kind`] and surfaced
68    /// indirectly through the matched-event `kind` field for traceability.
69    fn label(self) -> &'static str {
70        match self {
71            Self::Request => "request",
72            Self::Response => "response",
73            Self::Dialog => "dialog",
74            Self::Download => "download",
75        }
76    }
77}

79/// URL match predicate for request/response expectations. Dialog and
80/// Download kinds ignore matcher fields entirely (the lib has no URL filter
81/// for them in v0).
82///
83/// Exactly zero or one of `url_substr` / `url_regex` may be set. Setting
84/// neither matches every URL (substring matcher with the empty string).
85#[derive(Debug, Default, Deserialize, JsonSchema)]
86#[serde(deny_unknown_fields)]
87pub struct ExpectMatcher {
88    /// Substring matcher — match if request URL contains this needle.
89    #[serde(default)]
90    pub url_substr: Option<String>,
91    /// Regex matcher — match if `regex::Regex::is_match(url)` returns true.
92    /// Compiled inside `_register`; an invalid regex surfaces as an
93    /// invalid-request error.
94    #[serde(default)]
95    pub url_regex: Option<String>,
96}
97
98impl ExpectMatcher {
99    /// Build a [`UrlMatcher`] from the wire-level matcher.
100    ///
101    /// Priority: `url_regex` wins over `url_substr` when both are set —
102    /// flagging that on the schema would be nicer, but a serde validator
103    /// would be intrusive; document the precedence here and let agents pick
104    /// one.
105    fn into_url_matcher(self) -> Result<UrlMatcher, ErrorData> {
106        if let Some(re) = self.url_regex {
107            let compiled = regex::Regex::new(&re).map_err(|e| {
108                ErrorData::invalid_request(format!("invalid url_regex `{re}`: {e}"), None)
109            })?;
110            return Ok(UrlMatcher::Regex(compiled));
111        }
112        if let Some(sub) = self.url_substr {
113            return Ok(UrlMatcher::Substring(sub));
114        }
115        // Empty-string substring matcher matches every URL — easiest "any"
116        // sentinel that doesn't require a separate variant.
117        Ok(UrlMatcher::Substring(String::new()))
118    }
119}

124#[derive(Debug, Deserialize, JsonSchema)]
125#[serde(deny_unknown_fields)]
126pub struct RegisterInput {
127    /// Which event family to watch.
128    pub kind: ExpectKind,

133    /// default (60s) so the user has time to call the action tool that
134    /// triggers the event between `register` and `await`.
135    #[serde(default = "default_pre_timeout")]
136    pub pre_await_timeout_ms: u64,
137}

139fn default_pre_timeout() -> u64 {

143/// Output of `browser_expect_register`.
144#[derive(Debug, Serialize, JsonSchema)]
145pub struct RegisterOutput {
146    /// Opaque id for the session's expectation registry. Pass to
147    /// `browser_expect_await` or `browser_expect_cancel`.
148    pub expectation_id: ExpectationId,
149}
150
151/// Register a one-shot expectation against the current tab.
152///
153/// Spawns a tokio task that constructs the appropriate `tab.expect_*` call,
154/// applies the `pre_await_timeout_ms`, awaits `.matched()`, and pipes the
155/// result (JSON-encoded matched-event, or a textual error) through a
156/// `oneshot::Sender`. The task's `JoinHandle` is stored alongside the
157/// receiver so `_cancel` can `.abort()` it promptly.
158pub async fn register(
159    state: Arc<Mutex<SessionState>>,
160    input: RegisterInput,
161) -> Result<RegisterOutput, ErrorData> {
162    let pre_await = Duration::from_millis(input.pre_await_timeout_ms);
163    let matcher = input.matcher.unwrap_or_default().into_url_matcher()?;
164    let kind = input.kind;
165
166    let tab = {
167        let s = state.lock().await;
168        current_tab(&s).await?
169    };
170
171    let (tx, rx) = oneshot::channel::<Result<Value, String>>();
172
173    // One spawn per kind: each branch constructs its expectation against the
174    // tab handle (cloneable Arc-backed), applies the pre-await timeout, and
175    // forwards the matched event (or error string) over the oneshot. We
176    // can't share the spawned body across kinds because each kind's
177    // expectation type is different and `.matched()` returns a different
178    // matched-event struct.
179    let task: tokio::task::JoinHandle<()> = match kind {
180        ExpectKind::Request => {
181            let exp = tab.expect_request(matcher).timeout(pre_await);
182            tokio::spawn(async move {
183                let msg = exp.matched().await.map(request_to_json).map_err(err_to_str);
184                let _ = tx.send(msg);
185            })
186        }
187        ExpectKind::Response => {
188            let exp = tab.expect_response(matcher).timeout(pre_await);
189            tokio::spawn(async move {
190                let msg = exp
191                    .matched()
192                    .await
193                    .map(response_to_json)
194                    .map_err(err_to_str);
195                let _ = tx.send(msg);
196            })
197        }
198        ExpectKind::Dialog => {
199            let exp = tab.expect_dialog().timeout(pre_await);
200            tokio::spawn(async move {
201                let msg = exp.matched().await.map(dialog_to_json).map_err(err_to_str);
202                let _ = tx.send(msg);
203            })
204        }
205        ExpectKind::Download => {
206            // expect_download is async (lazy per-tab setup of the download
207            // coordinator + Browser.setDownloadBehavior dispatch). Surface
208            // its setup error inline instead of stashing the expectation —
209            // a registration we can never await isn't worth recording.
210            let exp = tab
211                .expect_download()
212                .await
213                .map_err(|e| map_error(McpServerError::from(e)))?
214                .timeout(pre_await);
215            tokio::spawn(async move {
216                let msg = exp
217                    .matched()
218                    .await
219                    .map(download_to_json)
220                    .map_err(err_to_str);
221                let _ = tx.send(msg);
222            })
223        }
224    };
225
226    let id: ExpectationId = uuid::Uuid::new_v4().to_string();
227    {
228        let mut s = state.lock().await;
229        s.expectations.insert(
230            id.clone(),
231            ExpectationHandle {
232                kind: kind.label(),
233                task,
234                rx,
235            },
236        );
237    }
238    Ok(RegisterOutput { expectation_id: id })
239}
240
241/// Collapse a [`ZendriverError`] surfaced by the spawned task to a string
242/// suitable for the `Err` branch of the registry's `oneshot` channel. We
243/// flatten to a string because `serde_json::Value` cannot itself round-trip
244/// a `ZendriverError` and the error eventually surfaces through a plain
245/// invalid-request anyway.
246fn err_to_str(e: ZendriverError) -> String {
247    e.to_string()
248}

252/// Input for `browser_expect_await`.
253#[derive(Debug, Deserialize, JsonSchema)]
254#[serde(deny_unknown_fields)]
255pub struct AwaitInput {
256    /// Id returned by an earlier `browser_expect_register` call.
257    pub expectation_id: ExpectationId,
258    /// Outer timeout for waiting on the spawned task to send the matched
259    /// event. Defaults to 30s.
260    #[serde(default = "default_await_timeout")]
261    pub timeout_ms: u64,
262}
263
264fn default_await_timeout() -> u64 {
265    30_000
266}

268/// Output of `browser_expect_await`.
269#[derive(Debug, Serialize, JsonSchema)]
270pub struct AwaitOutput {
271    /// Echoed id of the awaited expectation (matches input).
272    pub expectation_id: ExpectationId,
273    /// JSON-encoded matched event. Shape depends on the expectation's
274    /// `kind` — see the per-kind `*_to_json` helpers in this module.
275    pub event: Value,
276}
277
278/// Wait for a previously-registered expectation to resolve.
279///
280/// Removes the entry from `s.expectations` (taking ownership of the
281/// `oneshot::Receiver` + `JoinHandle`), then applies the caller's outer
282/// timeout to the channel. The spawned task's `pre_await_timeout_ms` is
283/// independent — if it fires first, the channel resolves with `Ok(Err(_))`
284/// carrying the lib's `ZendriverError::Timeout` as a string.
285pub async fn await_expectation(
286    state: Arc<Mutex<SessionState>>,
287    input: AwaitInput,
288) -> Result<AwaitOutput, ErrorData> {
289    let handle = {
290        let mut s = state.lock().await;
291        s.expectations
292            .remove(&input.expectation_id)
293            .ok_or_else(|| {
294                map_error(McpServerError::ExpectationNotFound(
295                    input.expectation_id.clone(),
296                ))
297            })?
298    };
299    let ExpectationHandle { rx, task, .. } = handle;
300
301    let outer = Duration::from_millis(input.timeout_ms);
302    match tokio::time::timeout(outer, rx).await {
303        Err(_elapsed) => {
304            // Outer timeout fired — abort the spawned task so its
305            // `.matched()` future stops holding a subscription stream open.
306            task.abort();
307            Err(ErrorData::invalid_request(
308                format!(
309                    "expectation `{}` await_timed_out after {:?}",
310                    input.expectation_id, outer
311                ),
312                None,
313            ))
314        }
315        Ok(Err(_recv)) => {
316            // Sender dropped without sending — spawned task panicked or was
317            // aborted externally. The task may still be in a transient
318            // state; abort defensively to be sure.
319            task.abort();
320            Err(ErrorData::invalid_request(
321                format!(
322                    "expectation `{}` channel_closed (spawned task ended without sending)",
323                    input.expectation_id
324                ),
325                None,
326            ))
327        }
328        Ok(Ok(Err(err_str))) => {
329            // Spawned task produced an error — lib-level failure or inner
330            // pre_await_timeout fired. Surface it on the wire.
331            Err(ErrorData::invalid_request(
332                format!("expectation `{}` failed: {err_str}", input.expectation_id),
333                None,
334            ))
335        }
336        Ok(Ok(Ok(event))) => Ok(AwaitOutput {
337            expectation_id: input.expectation_id,
338            event,
339        }),
340    }
341}

345/// Input for `browser_expect_cancel`.
346#[derive(Debug, Deserialize, JsonSchema)]
347#[serde(deny_unknown_fields)]
348pub struct CancelInput {
349    /// Id returned by an earlier `browser_expect_register` call.
350    pub expectation_id: ExpectationId,
351}
352
353/// Output of `browser_expect_cancel`.
354#[derive(Debug, Serialize, JsonSchema)]
355pub struct CancelOutput {
356    /// Always `true` on success. Cancellation of an unknown id returns
357    /// [`McpServerError::ExpectationNotFound`] rather than `false` here.
358    pub cancelled: bool,
359}
360
361/// Cancel a pending expectation, aborting its spawned task.
362///
363/// Drops the registry entry (which drops the `oneshot::Receiver`) and
364/// `.abort()`s the spawned task. Any in-flight `.matched()` future inside
365/// the task is dropped via abort, which tears down its lib-side subscriber.
366pub async fn cancel(
367    state: Arc<Mutex<SessionState>>,
368    input: CancelInput,
369) -> Result<CancelOutput, ErrorData> {
370    let mut s = state.lock().await;
371    let handle = s
372        .expectations
373        .remove(&input.expectation_id)
374        .ok_or_else(|| {
375            map_error(McpServerError::ExpectationNotFound(
376                input.expectation_id.clone(),
377            ))
378        })?;
379    handle.task.abort();
380    Ok(CancelOutput { cancelled: true })
381}

385/// Flatten a [`MatchedRequest`] to a wire JSON object.
386fn request_to_json(m: MatchedRequest) -> Value {
387    json!({
388        "kind": "request",
389        "url": m.url,
390        "method": m.method,
391        "headers": m.headers,
392        "request_id": m.request_id,
393        // post_data: emit only the byte length to avoid blowing up the
394        // wire for large POST bodies. Callers needing the bytes can use a
395        // future tool that fetches them explicitly.
396        "post_data_len": m.post_data.as_ref().map(Vec::len),
397    })
398}
399
400/// Flatten a [`MatchedResponse`] to a wire JSON object.
401///
402/// Body is NOT fetched in v0 — `body()` is an async CDP round-trip and
403/// not part of the event itself.
404fn response_to_json(m: MatchedResponse) -> Value {
405    json!({
406        "kind": "response",
407        "url": m.url,
408        "status": m.status,
409        "status_text": m.status_text,
410        "headers": m.headers,
411        "request_id": m.request_id,
412    })
413}
414
415/// Flatten a [`MatchedDialog`] to a wire JSON object.
416///
417/// `dialog_type` is rendered as a lowercase string for stable wire output.
418/// `accept`/`dismiss` are NOT exposed in v0 — Chrome's default handling
419/// applies when the matched-dialog handle is dropped at task end.
420fn dialog_to_json(m: MatchedDialog) -> Value {
421    let kind = match m.dialog_type {
422        zendriver::DialogType::Alert => "alert",
423        zendriver::DialogType::Beforeunload => "beforeunload",
424        zendriver::DialogType::Confirm => "confirm",
425        zendriver::DialogType::Prompt => "prompt",
426    };
427    json!({
428        "kind": "dialog",
429        "dialog_type": kind,
430        "message": m.message,
431        "default_prompt": m.default_prompt,
432        "url": m.url,
433    })
434}
435
436/// Flatten a [`MatchedDownload`] to a wire JSON object.
437///
438/// `path()` is not awaited here — the download is only `InProgress` at
439/// `Page.downloadWillBegin` time. Agents needing the on-disk path can call
440/// a future tool that polls completion + reads the file.
441fn download_to_json(m: MatchedDownload) -> Value {
442    json!({
443        "kind": "download",
444        "url": m.url,
445        "suggested_filename": m.suggested_filename,
446        "guid": m.guid,
447        "download_dir": m.download_dir.to_string_lossy(),
448    })
449}

451#[cfg(test)]
452#[allow(clippy::panic, clippy::unwrap_used)]
453mod tests {
454    //! No-browser unit coverage.
455    //!
456    //! Browser-touching `register` paths need a live Chrome (they construct
457    //! a real `tab.expect_*` and spawn a real subscriber actor) — that path
458    //! is exercised in `tests/integration_expect.rs`. Here we cover the
459    //! bookkeeping: register / await / cancel of unknown ids surface
460    //! `ExpectationNotFound`, register without a browser surfaces
461    //! `BrowserNotOpen`, and the matcher conversion roundtrips.
462    use super::*;
463
464    #[test]
465    fn matcher_default_is_match_any_substring() {
466        let m = ExpectMatcher::default().into_url_matcher().unwrap();
467        match m {
468            UrlMatcher::Substring(s) => assert!(
469                s.is_empty(),
470                "default matcher must be empty substring (matches any url)",
471            ),
472            UrlMatcher::Regex(_) => panic!("expected Substring(\"\")"),
473        }
474    }
475
476    #[test]
477    fn matcher_url_regex_wins_over_url_substr() {
478        // Documenting precedence: regex is checked first in
479        // `into_url_matcher`. Confirms agents picking both get the regex.
480        let m = ExpectMatcher {
481            url_substr: Some("/foo/".into()),
482            url_regex: Some(r"^https://api\.".into()),
483        }
484        .into_url_matcher()
485        .unwrap();
486        match m {
487            UrlMatcher::Regex(re) => assert!(re.is_match("https://api.example.com/v1")),
488            UrlMatcher::Substring(_) => panic!("expected Regex variant"),
489        }
490    }
491
492    #[test]
493    fn matcher_invalid_regex_errors() {
494        let err = ExpectMatcher {
495            url_regex: Some("[".into()),
496            url_substr: None,
497        }
498        .into_url_matcher()
499        .expect_err("expected invalid regex");
500        assert!(err.message.contains("invalid url_regex"));
501    }
502
503    #[tokio::test]
504    async fn await_unknown_expectation_id_surfaces_not_found() {
505        let state = Arc::new(Mutex::new(SessionState::new()));
506        let err = await_expectation(
507            state,
508            AwaitInput {
509                expectation_id: "nope".into(),

511            },
512        )
513        .await
514        .expect_err("expected ExpectationNotFound");
515        let data = err.data.as_ref().expect("data populated");
516        assert_eq!(data["suggested_next"], "browser_expect_register");
517    }
