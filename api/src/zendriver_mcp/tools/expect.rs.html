1//! Expectation tools — `browser_expect_register / _await / _cancel`. Gated
2//! behind the `expect` feature.
3//!
4//! ## Register-then-await split
5//!
6//! The lib's API is one-call:
7//! `tab.expect_request(matcher).timeout(d).matched().await`. MCP's
8//! request/response cycle can't model an in-flight `.matched()` future across
9//! tool calls, so v0 splits the surface into three:
10//!
11//! - `browser_expect_register` constructs the expectation, applies an inner
12//!   `pre_await_timeout_ms` (default 60s — generous so the user has time to
13//!   call the trigger action tool in between), spawns a tokio task that
14//!   awaits `.matched()`, and pipes the result through a
15//!   `tokio::sync::oneshot::Sender`. The id is returned synchronously.
16//! - `browser_expect_await` takes ownership of the receiver from the
17//!   session's registry and `tokio::time::timeout`s on it.
18//! - `browser_expect_cancel` drops the registry entry + `.abort()`s the
19//!   spawned task so the lib-level `.matched()` future is killed promptly
20//!   rather than left to time out on its own.
21//!
22//! ## Wire-shape mappers + drive
23//!
24//! Each kind has a small async free fn that flattens its lib-side struct
25//! (`MatchedRequest` / `MatchedResponse` / `MatchedDialog` / `MatchedDownload`)
26//! into a `serde_json::Value` and, when asked, *drives* the matched object
27//! before it is dropped. The drive parameters live on
28//! `browser_expect_register` (not `_await`) because the spawned task owns the
29//! matched handle and several drive methods consume it (`accept` / `dismiss` /
30//! `save_to`); the decision must therefore be made at register time, before
31//! the triggering action runs:
32//!
33//! - **Dialog** — `dialog_action: accept|dismiss` (+ `dialog_prompt_text`)
34//!   drives the dialog so the page's blocking `alert/confirm/prompt` returns.
35//! - **Response** — `fetch_body: true` fetches the body and inlines it as
36//!   `body_base64` + `body_len`.
37//! - **Download** — `save_to: <path>` waits for completion and copies the file
38//!   to the MCP host, reporting `saved_path`.
39

64/// Which `Page.*` / `Network.*` family the expectation listens on.
65#[derive(Debug, Clone, Copy, Deserialize, JsonSchema)]
66#[serde(rename_all = "snake_case")]
67pub enum ExpectKind {
68    /// `Network.requestWillBeSent` — outgoing request, no body fetch.
69    Request,
70    /// `Network.responseReceived` — incoming response headers (no body in v0).
71    Response,
72    /// `Page.javascriptDialogOpened` — alert/confirm/prompt/beforeunload.
73    Dialog,
74    /// `Page.downloadWillBegin` — start of a file download.
75    Download,
76}
77
78impl ExpectKind {
79    /// Static label used by [`ExpectationHandle::kind`] and surfaced
80    /// indirectly through the matched-event `kind` field for traceability.
81    fn label(self) -> &'static str {
82        match self {
83            Self::Request => "request",
84            Self::Response => "response",
85            Self::Dialog => "dialog",
86            Self::Download => "download",
87        }
88    }
89}

91/// How to drive a matched JavaScript dialog (`kind: dialog` only).
92///
93/// Applied by the spawned task the instant the dialog opens — so the page's
94/// blocking `alert()` / `confirm()` / `prompt()` call returns promptly and
95/// the matched event carries the chosen action. Omitting `dialog_action`
96/// falls back to Chrome's default handling (the dialog is accepted when the
97/// matched handle is dropped at task end).
98#[derive(Debug, Clone, Copy, Deserialize, Serialize, JsonSchema, PartialEq, Eq)]
99#[serde(rename_all = "snake_case")]
100pub enum DialogAction {
101    /// Accept the dialog. For a `prompt`, submit `dialog_prompt_text` (or the
102    /// dialog's default value when that is unset).
103    Accept,
104    /// Dismiss / cancel the dialog.
105    Dismiss,
106}

108/// URL match predicate for request/response expectations. Dialog and
109/// Download kinds ignore matcher fields entirely (the lib has no URL filter
110/// for them in v0).
111///
112/// Exactly zero or one of `url_substr` / `url_regex` may be set. Setting
113/// neither matches every URL (substring matcher with the empty string).
114#[derive(Debug, Default, Deserialize, JsonSchema)]
115#[serde(deny_unknown_fields)]
116pub struct ExpectMatcher {
117    /// Substring matcher — match if request URL contains this needle.
118    #[serde(default)]
119    pub url_substr: Option<String>,
120    /// Regex matcher — match if `regex::Regex::is_match(url)` returns true.
121    /// Compiled inside `_register`; an invalid regex surfaces as an
122    /// invalid-request error.
123    #[serde(default)]
124    pub url_regex: Option<String>,
125}
126
127impl ExpectMatcher {
128    /// Build a [`UrlMatcher`] from the wire-level matcher.
129    ///
130    /// Priority: `url_regex` wins over `url_substr` when both are set —
131    /// flagging that on the schema would be nicer, but a serde validator
132    /// would be intrusive; document the precedence here and let agents pick
133    /// one.
134    fn into_url_matcher(self) -> Result<UrlMatcher, ErrorData> {
135        if let Some(re) = self.url_regex {
136            let compiled = regex::Regex::new(&re).map_err(|e| {
137                ErrorData::invalid_request(format!("invalid url_regex `{re}`: {e}"), None)
138            })?;
139            return Ok(UrlMatcher::Regex(compiled));
140        }
141        if let Some(sub) = self.url_substr {
142            return Ok(UrlMatcher::Substring(sub));
143        }
144        // Empty-string substring matcher matches every URL — easiest "any"
145        // sentinel that doesn't require a separate variant.
146        Ok(UrlMatcher::Substring(String::new()))
147    }
148}

152/// Input for `browser_expect_register`.
153#[derive(Debug, Deserialize, JsonSchema)]
154#[serde(deny_unknown_fields)]
155pub struct RegisterInput {
156    /// Which event family to watch.
157    pub kind: ExpectKind,
158    /// URL matcher (request/response only — ignored for dialog/download).
159    #[serde(default)]
160    pub matcher: Option<ExpectMatcher>,
161    /// Inner timeout applied to the lib's `.timeout(d)` call. Generous by
162    /// default (60s) so the user has time to call the action tool that
163    /// triggers the event between `register` and `await`.
164    #[serde(default = "default_pre_timeout")]
165    pub pre_await_timeout_ms: u64,
166    /// Dialog only: drive the matched dialog (`accept` / `dismiss`) instead
167    /// of leaving Chrome's default handling. Decided here, at register time,
168    /// because the matched dialog handle is consumed by the spawned task.
169    #[serde(default, skip_serializing_if = "Option::is_none")]
170    pub dialog_action: Option<DialogAction>,
171    /// Dialog only: prompt answer for `dialog_action: accept` on a `prompt`.
172    #[serde(default, skip_serializing_if = "Option::is_none")]
173    pub dialog_prompt_text: Option<String>,
174    /// Response only: fetch the response body and include it (base64) in the
175    /// matched event as `body_base64` + `body_len`. One extra CDP round-trip.
176    #[serde(default)]
177    pub fetch_body: bool,
178    /// Download only: copy the completed download to this path on the MCP
179    /// host; the matched event reports it as `saved_path`.
180    #[serde(default, skip_serializing_if = "Option::is_none")]
181    pub save_to: Option<String>,
182}

188/// Output of `browser_expect_register`.
189#[derive(Debug, Serialize, JsonSchema)]
190pub struct RegisterOutput {
191    /// Opaque id for the session's expectation registry. Pass to
192    /// `browser_expect_await` or `browser_expect_cancel`.
193    pub expectation_id: ExpectationId,
194}

196/// Register a one-shot expectation against the current tab.
197///
198/// Spawns a tokio task that constructs the appropriate `tab.expect_*` call,
199/// applies the `pre_await_timeout_ms`, awaits `.matched()`, and pipes the
200/// result (JSON-encoded matched-event, or a textual error) through a
201/// `oneshot::Sender`. The task's `JoinHandle` is stored alongside the
202/// receiver so `_cancel` can `.abort()` it promptly.
203pub async fn register(
204    state: Arc<Mutex<SessionState>>,
205    input: RegisterInput,
206) -> Result<RegisterOutput, ErrorData> {
207    let pre_await = Duration::from_millis(input.pre_await_timeout_ms);
208    let matcher = input.matcher.unwrap_or_default().into_url_matcher()?;
209    let kind = input.kind;
210    // Drive params, captured before the per-kind spawn moves them in. Each is
211    // meaningful for exactly one kind (see field docs); the others are unused.
212    let dialog_action = input.dialog_action;
213    let dialog_prompt = input.dialog_prompt_text;
214    let fetch_body = input.fetch_body;
215    let save_to = input.save_to;
216
217    let tab = {
218        let s = state.lock().await;
219        current_tab(&s).await?
220    };
221
222    let (tx, rx) = oneshot::channel::<Result<Value, String>>();
223
224    // One spawn per kind: each branch constructs its expectation against the
225    // tab handle (cloneable Arc-backed), applies the pre-await timeout, and
226    // forwards the matched event (or error string) over the oneshot. We
227    // can't share the spawned body across kinds because each kind's
228    // expectation type is different and `.matched()` returns a different
229    // matched-event struct.
230    let task: tokio::task::JoinHandle<()> = match kind {
231        ExpectKind::Request => {
232            let exp = tab.expect_request(matcher).timeout(pre_await);
233            tokio::spawn(async move {
234                let msg = exp.matched().await.map(request_to_json).map_err(err_to_str);
235                let _ = tx.send(msg);
236            })
237        }
238        ExpectKind::Response => {
239            let exp = tab.expect_response(matcher).timeout(pre_await);
240            tokio::spawn(async move {
241                let msg = match exp.matched().await {
242                    Ok(m) => response_to_json(m, fetch_body).await,
243                    Err(e) => Err(err_to_str(e)),
244                };
245                let _ = tx.send(msg);
246            })
247        }
248        ExpectKind::Dialog => {
249            let exp = tab.expect_dialog().timeout(pre_await);
250            tokio::spawn(async move {
251                let msg = match exp.matched().await {
252                    Ok(m) => dialog_to_json(m, dialog_action, dialog_prompt).await,
253                    Err(e) => Err(err_to_str(e)),
254                };
255                let _ = tx.send(msg);
256            })
257        }
258        ExpectKind::Download => {
259            // expect_download is async (lazy per-tab setup of the download
260            // coordinator + Browser.setDownloadBehavior dispatch). Surface
261            // its setup error inline instead of stashing the expectation —
262            // a registration we can never await isn't worth recording.
263            let exp = tab
264                .expect_download()
265                .await
266                .map_err(|e| map_error(McpServerError::from(e)))?
267                .timeout(pre_await);
268            tokio::spawn(async move {
269                let msg = match exp.matched().await {
270                    Ok(m) => download_to_json(m, save_to).await,
271                    Err(e) => Err(err_to_str(e)),
272                };
273                let _ = tx.send(msg);
274            })
275        }
276    };
277
278    let id: ExpectationId = uuid::Uuid::new_v4().to_string();
279    {
280        let mut s = state.lock().await;
281        s.expectations.insert(
282            id.clone(),
283            ExpectationHandle {
284                kind: kind.label(),
285                task,
286                rx,
287            },
288        );
289    }
290    Ok(RegisterOutput { expectation_id: id })
291}

293/// Collapse a [`ZendriverError`] surfaced by the spawned task to a string
294/// suitable for the `Err` branch of the registry's `oneshot` channel. We
295/// flatten to a string because `serde_json::Value` cannot itself round-trip
296/// a `ZendriverError` and the error eventually surfaces through a plain
297/// invalid-request anyway.
298fn err_to_str(e: ZendriverError) -> String {
299    e.to_string()
300}

320/// Output of `browser_expect_await`.
321#[derive(Debug, Serialize, JsonSchema)]
322pub struct AwaitOutput {
323    /// Echoed id of the awaited expectation (matches input).
324    pub expectation_id: ExpectationId,
325    /// JSON-encoded matched event. Shape depends on the expectation's
326    /// `kind` — see the per-kind `*_to_json` helpers in this module.
327    pub event: Value,
328}

330/// Wait for a previously-registered expectation to resolve.
331///
332/// Removes the entry from `s.expectations` (taking ownership of the
333/// `oneshot::Receiver` + `JoinHandle`), then applies the caller's outer
334/// timeout to the channel. The spawned task's `pre_await_timeout_ms` is
335/// independent — if it fires first, the channel resolves with `Ok(Err(_))`
336/// carrying the lib's `ZendriverError::Timeout` as a string.
337pub async fn await_expectation(
338    state: Arc<Mutex<SessionState>>,
339    input: AwaitInput,
340) -> Result<AwaitOutput, ErrorData> {
341    let handle = {
342        let mut s = state.lock().await;
343        s.expectations
344            .remove(&input.expectation_id)
345            .ok_or_else(|| {
346                map_error(McpServerError::ExpectationNotFound(
347                    input.expectation_id.clone(),
348                ))
349            })?
350    };
351    let ExpectationHandle { rx, task, .. } = handle;
352
353    let outer = Duration::from_millis(input.timeout_ms);
354    match tokio::time::timeout(outer, rx).await {
355        Err(_elapsed) => {
356            // Outer timeout fired — abort the spawned task so its
357            // `.matched()` future stops holding a subscription stream open.
358            task.abort();
359            Err(ErrorData::invalid_request(
360                format!(
361                    "expectation `{}` await_timed_out after {:?}",
362                    input.expectation_id, outer
363                ),
364                None,
365            ))
366        }
367        Ok(Err(_recv)) => {
368            // Sender dropped without sending — spawned task panicked or was
369            // aborted externally. The task may still be in a transient
370            // state; abort defensively to be sure.
371            task.abort();
372            Err(ErrorData::invalid_request(
373                format!(
374                    "expectation `{}` channel_closed (spawned task ended without sending)",
375                    input.expectation_id
376                ),
377                None,
378            ))
379        }
380        Ok(Ok(Err(err_str))) => {
381            // Spawned task produced an error — lib-level failure or inner
382            // pre_await_timeout fired. Surface it on the wire.
383            Err(ErrorData::invalid_request(
384                format!("expectation `{}` failed: {err_str}", input.expectation_id),
385                None,
386            ))
387        }
388        Ok(Ok(Ok(event))) => Ok(AwaitOutput {
389            expectation_id: input.expectation_id,
390            event,
391        }),
392    }
393}

397/// Input for `browser_expect_cancel`.
398#[derive(Debug, Deserialize, JsonSchema)]
399#[serde(deny_unknown_fields)]
400pub struct CancelInput {
401    /// Id returned by an earlier `browser_expect_register` call.
402    pub expectation_id: ExpectationId,
403}
404
405/// Output of `browser_expect_cancel`.
406#[derive(Debug, Serialize, JsonSchema)]
407pub struct CancelOutput {
408    /// Always `true` on success. Cancellation of an unknown id returns
409    /// [`McpServerError::ExpectationNotFound`] rather than `false` here.
410    pub cancelled: bool,
411}
412
413/// Cancel a pending expectation, aborting its spawned task.
414///
415/// Drops the registry entry (which drops the `oneshot::Receiver`) and
416/// `.abort()`s the spawned task. Any in-flight `.matched()` future inside
417/// the task is dropped via abort, which tears down its lib-side subscriber.
418pub async fn cancel(
419    state: Arc<Mutex<SessionState>>,
420    input: CancelInput,
421) -> Result<CancelOutput, ErrorData> {
422    let mut s = state.lock().await;
423    let handle = s
424        .expectations
425        .remove(&input.expectation_id)
426        .ok_or_else(|| {
427            map_error(McpServerError::ExpectationNotFound(
428                input.expectation_id.clone(),
429            ))
430        })?;
431    handle.task.abort();
432    Ok(CancelOutput { cancelled: true })
433}

437/// Flatten a [`MatchedRequest`] to a wire JSON object.
438fn request_to_json(m: MatchedRequest) -> Value {
439    json!({
440        "kind": "request",
441        "url": m.url,
442        "method": m.method,
443        "headers": m.headers,
444        "request_id": m.request_id,
445        // post_data: emit only the byte length to avoid blowing up the
446        // wire for large POST bodies. Callers needing the bytes can use a
447        // future tool that fetches them explicitly.
448        "post_data_len": m.post_data.as_ref().map(Vec::len),
449    })
450}
451
452/// Flatten a [`MatchedResponse`] to a wire JSON object, optionally fetching
453/// the response body (base64) when `fetch_body` is set.
454///
455/// `body()` is an async CDP round-trip, so it is opt-in: agents that only
456/// need status/headers skip it.
457async fn response_to_json(m: MatchedResponse, fetch_body: bool) -> Result<Value, String> {
458    let mut v = json!({
459        "kind": "response",
460        "url": m.url,
461        "status": m.status,
462        "status_text": m.status_text,
463        "headers": m.headers,
464        "request_id": m.request_id,
465    });
466    if fetch_body {
467        let bytes = m.body().await.map_err(err_to_str)?;
468        v["body_len"] = json!(bytes.len());
469        v["body_base64"] = json!(BASE64.encode(&bytes));
470    }
471    Ok(v)
472}

474/// Render a [`DialogType`] as a stable lowercase wire string.
475fn dialog_type_str(d: &DialogType) -> &'static str {
476    match d {
477        DialogType::Alert => "alert",
478        DialogType::Beforeunload => "beforeunload",
479        DialogType::Confirm => "confirm",
480        DialogType::Prompt => "prompt",
481    }
482}

484/// Flatten a [`MatchedDialog`] to a wire JSON object, driving the dialog
485/// (`accept` / `dismiss`) when `action` is set.
486///
487/// Fields are captured by reference first so the handle stays intact for the
488/// consuming `accept`/`dismiss` call. With no `action`, the handle is dropped
489/// here and Chrome's default handling applies.
490async fn dialog_to_json(
491    m: MatchedDialog,
492    action: Option<DialogAction>,
493    prompt: Option<String>,
494) -> Result<Value, String> {
495    let dialog_type = dialog_type_str(&m.dialog_type);
496    let message = m.message.clone();
497    let default_prompt = m.default_prompt.clone();
498    let url = m.url.clone();
499    let driven = match action {
500        Some(DialogAction::Accept) => {
501            m.accept(prompt).await.map_err(err_to_str)?;
502            "accept"
503        }
504        Some(DialogAction::Dismiss) => {
505            m.dismiss().await.map_err(err_to_str)?;
506            "dismiss"
507        }
508        None => {
509            drop(m);
510            "default"
511        }
512    };
513    Ok(json!({
514        "kind": "dialog",
515        "dialog_type": dialog_type,
516        "message": message,
517        "default_prompt": default_prompt,
518        "url": url,
519        "driven": driven,
520    }))
521}

523/// Flatten a [`MatchedDownload`] to a wire JSON object, optionally copying the
524/// completed download to `save_to` on the MCP host.
525///
526/// `MatchedDownload::save_to` waits for completion before copying, so the
527/// reported `saved_path` is a fully-written file.
528async fn download_to_json(m: MatchedDownload, save_to: Option<String>) -> Result<Value, String> {
529    let url = m.url.clone();
530    let suggested_filename = m.suggested_filename.clone();
531    let guid = m.guid.clone();
532    let download_dir = m.download_dir.to_string_lossy().into_owned();
533    let saved_path = if let Some(dest) = save_to {
534        m.save_to(PathBuf::from(&dest)).await.map_err(err_to_str)?;
535        Some(dest)
536    } else {
537        None
538    };
539    Ok(json!({
540        "kind": "download",
541        "url": url,
542        "suggested_filename": suggested_filename,
543        "guid": guid,
544        "download_dir": download_dir,
545        "saved_path": saved_path,
546    }))
547}
