1//! Imperva / Incapsula bypass tool — `browser_solve_imperva`. Gated behind
2//! the `imperva` feature.
3//!
4//! Mirrors [`tools/cloudflare.rs`][crate::tools::cloudflare]: the lib's
5//! [`ImpervaBypass::wait_for_clearance`] models its terminal state as
6//! `Result<ImpervaClearanceOutcome, ImpervaError>`. The MCP layer collapses
7//! the lib-side [`ImpervaClearanceOutcome::TimedOut`] terminal into a
8//! non-error [`Outcome::Timeout`] — a deadline in a bot-management flow is a
9//! normal "didn't finish, retry or give up" signal, not a server error — and
10//! keeps every `ImpervaError` (CAPTCHA-without-solver, CDP failure, JS error)
11//! as a real MCP error.
12//!
13//! Unlike Cloudflare's three states, Imperva reports a fourth —
14//! [`ImpervaClearanceOutcome::AlreadyClear`] (no surface present at call
15//! time, fast path) — surfaced as a distinct [`Outcome::AlreadyClear`] so an
16//! agent can skip redundant follow-up work.
17//!
18//! The lib's `on_captcha` solver hook is intentionally **not** wired over
19//! MCP (same class as the interception stream non-goal): a CAPTCHA surface
20//! without a registered solver surfaces as `ImpervaError::CaptchaRequired`,
21//! which becomes a real MCP error the agent must handle out-of-band.
22//!
23//! [`ImpervaBypass::wait_for_clearance`]: zendriver_imperva::ImpervaBypass::wait_for_clearance
24

40/// Input for `browser_solve_imperva`.
41#[derive(Debug, Deserialize, JsonSchema)]
42#[serde(deny_unknown_fields)]
43pub struct SolveImpervaInput {
44    /// Maximum total wait for a terminal outcome, in milliseconds. Default
45    /// 30_000 (30s). Maps to [`ImpervaBypass::timeout`].
46    ///
47    /// [`ImpervaBypass::timeout`]: zendriver_imperva::ImpervaBypass::timeout
48    #[serde(default = "default_timeout")]
49    pub timeout_ms: u64,
50    /// Override the bypass driver's internal poll cadence, in milliseconds.
51    /// Defaults to the lib's own default. Lowering speeds up detection at the
52    /// cost of more CDP `Runtime.evaluate` round-trips; raising is a safe
53    /// knob for slow / sandboxed environments.
54    #[serde(default, skip_serializing_if = "Option::is_none")]
55    pub poll_interval_ms: Option<u64>,
56    /// Enable the Fetch-domain interception fast-path
57    /// ([`ImpervaBypass::with_interception`]) for quicker reese84 capture on
58    /// modern surfaces. Default `false`.
59    ///
60    /// [`ImpervaBypass::with_interception`]: zendriver_imperva::ImpervaBypass::with_interception
61    #[serde(default)]
62    pub with_interception: bool,
63}
64
65fn default_timeout() -> u64 {
66    30_000
67}

69/// Terminal outcome of an Imperva bypass attempt.
70///
71/// `TokenAcquired` / `ChallengeGone` / `AlreadyClear` mirror the lib's
72/// [`ImpervaClearanceOutcome`] variants. `Timeout` mirrors the lib's
73/// [`ImpervaClearanceOutcome::TimedOut`] — a deadline in a bot-management
74/// flow is a normal outcome, not an error — surfacing it on the success
75/// channel so agents branch on `outcome` without try/catch around a timeout.
76#[derive(Debug, Serialize, JsonSchema, PartialEq, Eq)]
77#[serde(rename_all = "snake_case")]
78pub enum Outcome {
79    /// reese84 token acquired (value in [`SolveImpervaOutput::reese84`]).
80    TokenAcquired,
81    /// Body markers cleared without a reese84 token (e.g. legacy Incapsula
82    /// flow). `reese84` will be `None`.
83    ChallengeGone,
84    /// No Imperva surface was present at call time (fast path, no waiting).
85    /// `reese84` will be `None`.
86    AlreadyClear,
87    /// `timeout_ms` elapsed without reaching clearance. Not a hard error —
88    /// agents can retry or give up. `reese84` will be `None`.
89    Timeout,
90}
91
92/// Output of `browser_solve_imperva`.
93#[derive(Debug, Serialize, JsonSchema)]
94pub struct SolveImpervaOutput {
95    /// Which terminal state the bypass reached.
96    pub outcome: Outcome,
97    /// reese84 cookie value. Populated only when `outcome == token_acquired`;
98    /// `None` for every other outcome.
99    #[serde(skip_serializing_if = "Option::is_none")]
100    pub reese84: Option<String>,
101}
102
103/// Drive the Imperva clearance flow on the current tab, returning the
104/// terminal outcome within `timeout_ms`.
105///
106/// See module-level docs for outcome semantics.
107pub async fn solve_imperva(
108    state: Arc<Mutex<SessionState>>,
109    input: SolveImpervaInput,
110) -> Result<SolveImpervaOutput, ErrorData> {
111    let s = state.lock().await;
112    let tab = current_tab(&s).await?;
113    // `tab.imperva()` borrows `tab` for the bypass's lifetime; the builder
114    // methods consume + return `self`, and `wait_for_clearance` consumes it,
115    // so the bypass lives only for the single await below.
116    let mut bypass = tab
117        .imperva()
118        .timeout(Duration::from_millis(input.timeout_ms));
119    if let Some(p) = input.poll_interval_ms {
120        bypass = bypass.poll_interval(Duration::from_millis(p));
121    }
122    if input.with_interception {
123        bypass = bypass.with_interception();
124    }
125    match bypass.wait_for_clearance().await {
126        Ok(ImpervaClearanceOutcome::TokenAcquired { reese84, .. }) => Ok(SolveImpervaOutput {
127            outcome: Outcome::TokenAcquired,
128            reese84: Some(reese84),
129        }),
130        Ok(ImpervaClearanceOutcome::ChallengeGone) => Ok(SolveImpervaOutput {
131            outcome: Outcome::ChallengeGone,
132            reese84: None,
133        }),
134        Ok(ImpervaClearanceOutcome::AlreadyClear { reese84, .. }) => Ok(SolveImpervaOutput {
135            outcome: Outcome::AlreadyClear,
136            reese84,
137        }),
138        // TimedOut is a lib-side success terminal; collapse it into the
139        // success-channel `Outcome::Timeout` — see module docs.
140        Ok(ImpervaClearanceOutcome::TimedOut { .. }) => Ok(SolveImpervaOutput {
141            outcome: Outcome::Timeout,
142            reese84: None,
143        }),
144        // Everything else (CAPTCHA-without-solver, CDP failure, JS error) is a
145        // real error. Route through `From<ImpervaError> for ZendriverError` so
146        // the existing `map_error` knows how to format it.
147        Err(other) => Err(map_error(McpServerError::from(ZendriverError::from(other)))),
148    }
149}

151#[cfg(test)]
152#[allow(clippy::panic, clippy::unwrap_used)]
153mod tests {
154    //! No-browser unit coverage. The bypass flow itself needs a live Chrome +
155    //! an Imperva-protected page — exercised in the integration test gated
156    //! behind `integration-tests + imperva`. Here we cover the only branch
157    //! reachable without a browser: no browser open surfaces `BrowserNotOpen`.
158
159    use super::*;
160
161    #[tokio::test]
162    async fn solve_imperva_with_no_browser_errors() {
163        let state = Arc::new(Mutex::new(SessionState::new()));
164        let err = solve_imperva(
165            state,
166            SolveImpervaInput {
167                timeout_ms: 100,
168                poll_interval_ms: None,
169                with_interception: false,
170            },
171        )
172        .await
173        .expect_err("expected BrowserNotOpen");
174        assert!(err.message.contains("Browser not open"));
175    }
