1//! JS evaluation tools — `browser_evaluate`, `browser_evaluate_main`.
2//!
3//! The two flavors differ only in execution world:
4//! - [`evaluate`] → [`zendriver::Tab::evaluate`] (isolated world). Default
5//!   choice. Page globals set by page scripts are *not* visible, so
6//!   stealth fingerprint shims remain hidden from the page.
7//! - [`evaluate_main`] → [`zendriver::Tab::evaluate_main`] (main world).
8//!   Page globals are visible. Required when you must call functions the
9//!   page itself defined; **breaks stealth isolation** if used carelessly
10//!   because the page can observe the call.
11//!
12//! ## `await_promise` arg
13//!
14//! Currently observational — the lib's `evaluate*` methods send
15//! `awaitPromise: true` to CDP unconditionally (see
16//! `crates/zendriver/src/tab.rs:684` and `:757`). We surface the flag so
17//! the schema is stable when the lib gains a "don't await" variant; for
18//! now it's a documented no-op.
19//!
20//! ## `frame_id` routing
21//!
22//! When set, dispatches through [`zendriver::Frame::evaluate`] /
23//! `evaluate_main` instead of the tab-level helpers. The OOPIF case still
24//! works — `Frame::evaluate_main` lands in the frame's own main world.
25

37/// Input for `browser_evaluate` / `browser_evaluate_main`.
38#[derive(Debug, Deserialize, JsonSchema)]
39#[serde(deny_unknown_fields)]
40pub struct EvalInput {
41    /// JavaScript expression (NOT a statement block). Examples: `"1 + 2"`,
42    /// `"document.title"`. For multi-line logic, wrap in an IIFE:
43    /// `"(() => { /* ... */ return result; })()"`.
44    pub expression: String,
45    /// If the expression resolves to a promise, await it before returning
46    /// the value. Default `true`. Currently observational — the lib always
47    /// awaits promises in `Runtime.evaluate`.
48    #[serde(default = "default_await")]
49    pub await_promise: bool,
50    /// When set, evaluate inside the specified frame instead of the tab's
51    /// main frame.
52    #[serde(default, skip_serializing_if = "Option::is_none")]
53    pub frame_id: Option<String>,
54}
55
56const fn default_await() -> bool {
57    true
58}
59
60/// Output of `browser_evaluate` / `browser_evaluate_main`.
61#[derive(Debug, Serialize, JsonSchema)]
62pub struct EvalOutput {
63    /// Whatever the expression returned, serialized as JSON. `undefined` →
64    /// JSON `null`.
65    pub value: serde_json::Value,
66}
67
68/// Evaluate `expression` in the page's **isolated** world.
69///
70/// Preferred over [`evaluate_main`] for everything that doesn't require
71/// page globals — stays invisible to the page's own JS.
72pub async fn evaluate(
73    state: Arc<Mutex<SessionState>>,
74    input: EvalInput,
75) -> Result<EvalOutput, ErrorData> {
76    let s = state.lock().await;
77    let tab = current_tab(&s).await?;
78    let value: serde_json::Value = if let Some(fid) = input.frame_id.as_deref() {
79        let frame = lookup_frame(&tab, fid).await?;
80        frame
81            .evaluate(&input.expression)
82            .await
83            .map_err(|e| map_error(McpServerError::from(e)))?
84    } else {
85        tab.evaluate(&input.expression)
86            .await
87            .map_err(|e| map_error(McpServerError::from(e)))?
88    };
89    Ok(EvalOutput { value })
90}
91
92/// Evaluate `expression` in the page's **main** world.
93///
94/// Page globals are visible (and the page can observe the eval). Use only
95/// when isolated-world semantics don't fit — for stealth-sensitive flows
96/// prefer [`evaluate`].
97pub async fn evaluate_main(
98    state: Arc<Mutex<SessionState>>,
99    input: EvalInput,
100) -> Result<EvalOutput, ErrorData> {
101    let s = state.lock().await;
102    let tab = current_tab(&s).await?;
103    let value: serde_json::Value = if let Some(fid) = input.frame_id.as_deref() {
104        let frame = lookup_frame(&tab, fid).await?;
105        frame
106            .evaluate_main(&input.expression)
107            .await
108            .map_err(|e| map_error(McpServerError::from(e)))?
109    } else {
110        tab.evaluate_main(&input.expression)
111            .await
112            .map_err(|e| map_error(McpServerError::from(e)))?
113    };
114    Ok(EvalOutput { value })
115}

117#[cfg(test)]
118mod tests {
119    use super::*;
120
121    fn fresh() -> Arc<Mutex<SessionState>> {
122        Arc::new(Mutex::new(SessionState::new()))
123    }
124
125    #[tokio::test]
126    async fn evaluate_with_no_browser_suggests_browser_open() {
127        let err = evaluate(
128            fresh(),
129            EvalInput {
130                expression: "1 + 2".into(),
131                await_promise: true,
132                frame_id: None,
133            },
134        )
135        .await
136        .expect_err("must error without an open browser");
137        assert!(err.message.contains("browser_open"), "msg: {}", err.message);
138        let data = err.data.as_ref().expect("data populated");
139        assert_eq!(data["suggested_next"], "browser_open");
140    }
141
142    #[tokio::test]
143    async fn evaluate_main_with_no_browser_suggests_browser_open() {
144        let err = evaluate_main(
145            fresh(),
146            EvalInput {
147                expression: "document.title".into(),
148                await_promise: true,
149                frame_id: None,
150            },
151        )
152        .await
153        .expect_err("must error without an open browser");
154        assert!(err.message.contains("browser_open"), "msg: {}", err.message);
155        let data = err.data.as_ref().expect("data populated");
156        assert_eq!(data["suggested_next"], "browser_open");
157    }
