1//! Read-only state inspectors — `browser_element_state`.
2//!
3//! Resolves an element via the [`crate::tools::find`] bridge, then
4//! populates a sparse `ElementState` per the requested `include` preset.
5//! Every populated field is `Option<_>` so unrequested fields can be
6//! dropped from the wire payload via `skip_serializing_if`.
7

22/// Field preset for `browser_element_state`.
23///
24/// Each preset chooses which subset of the (potentially expensive) probes
25/// the handler runs. Defaults to [`ReadFieldsPreset::All`] — costs an
26/// extra eval + attrs + inner_html call but matches the most common
27/// agent inspection flow.
28#[derive(Debug, Clone, Copy, Deserialize, Serialize, JsonSchema, PartialEq, Eq)]
29#[serde(rename_all = "snake_case")]
30pub enum ReadFieldsPreset {
31    /// Every field (existence, visible/enabled, geometry, text/attrs/html).
32    /// `in_viewport` remains `None` in v0 — no zendriver primitive yet.
33    All,
34    /// Only `exists`. Cheapest preset — skips every probe past resolve.
35    ExistsOnly,
36    /// `exists`, `visible`, `enabled`.
37    VisibleEnabled,
38    /// `exists`, `bounding_box`. `in_viewport` stays `None` in v0.
39    Geometry,
40    /// `exists`, `text`, `attrs`, `inner_html`.
41    TextAttrs,
42}
43
44const fn default_preset() -> ReadFieldsPreset {
45    ReadFieldsPreset::All
46}
47
48/// Input for `browser_element_state`.
49#[derive(Debug, Deserialize, JsonSchema)]
50#[serde(deny_unknown_fields)]
51pub struct ElementStateInput {
52    #[serde(flatten)]
53    pub selector: Selector,
54    /// Field-subset preset. See [`ReadFieldsPreset`]. Default: `all`.
55    #[serde(default = "default_preset")]
56    pub include: ReadFieldsPreset,
57}
58
59/// Output of `browser_element_state`. Every field is `Option<_>` so only
60/// the ones a preset requested round-trip on the wire.
61#[derive(Debug, Serialize, JsonSchema, Default)]
62pub struct ElementState {
63    /// `true` when the selector resolved to an element. Always set (every
64    /// preset includes this).
65    pub exists: bool,
66    /// `true` when the lib's visibility predicate accepts the element.
67    /// Populated by `All` and `VisibleEnabled`.
68    #[serde(skip_serializing_if = "Option::is_none")]
69    pub visible: Option<bool>,
70    /// `true` when the element is not disabled. Populated by `All` and
71    /// `VisibleEnabled`.
72    #[serde(skip_serializing_if = "Option::is_none")]
73    pub enabled: Option<bool>,
74    /// `true` when the element's bounding box intersects the viewport.
75    /// Always `None` in v0 — no zendriver primitive yet. Field reserved
76    /// so a follow-up dispatch can add it without breaking the schema.
77    #[serde(skip_serializing_if = "Option::is_none")]
78    pub in_viewport: Option<bool>,
79    /// Viewport-relative bounding box. Populated by `All` and `Geometry`.
80    #[serde(skip_serializing_if = "Option::is_none")]
81    pub bounding_box: Option<BoundingBox>,
82    /// `innerText`. Populated by `All` and `TextAttrs`. NOT truncated —
83    /// callers wanting a snippet should clip on their side.
84    #[serde(skip_serializing_if = "Option::is_none")]
85    pub text: Option<String>,
86    /// All HTML attributes. Populated by `All` and `TextAttrs`.
87    #[serde(skip_serializing_if = "Option::is_none")]
88    pub attrs: Option<BTreeMap<String, String>>,
89    /// Serialized `innerHTML`. Populated by `All` and `TextAttrs`.
90    #[serde(skip_serializing_if = "Option::is_none")]
91    pub inner_html: Option<String>,
92}

94/// Probe an element's state per the requested field preset.
95///
96/// `exists: false` short-circuits to skip every other probe, so a
97/// missing-element check costs exactly one find dispatch.
98pub async fn element_state(
99    state: Arc<Mutex<SessionState>>,
100    input: ElementStateInput,
101) -> Result<ElementState, ErrorData> {
102    let s = state.lock().await;
103    let tab = current_tab(&s).await?;
104    let el = match resolve(&tab, &input.selector).await {
105        Ok(el) => el,
106        Err(err) if is_not_found(&err) => {
107            // Missing element → exists=false, no other fields.
108            return Ok(ElementState {
109                exists: false,
110                ..Default::default()
111            });
112        }
113        Err(err) => return Err(err),
114    };
115    let mut out = ElementState {
116        exists: true,
117        ..Default::default()
118    };
119    let want_visible = matches!(
120        input.include,
121        ReadFieldsPreset::All | ReadFieldsPreset::VisibleEnabled
122    );
123    let want_geometry = matches!(
124        input.include,
125        ReadFieldsPreset::All | ReadFieldsPreset::Geometry
126    );
127    let want_text_attrs = matches!(
128        input.include,
129        ReadFieldsPreset::All | ReadFieldsPreset::TextAttrs
130    );
131
132    if want_visible {
133        let visible = el
134            .is_visible()
135            .await
136            .map_err(|e| map_error(McpServerError::from(e)))?;
137        let enabled = el
138            .is_enabled()
139            .await
140            .map_err(|e| map_error(McpServerError::from(e)))?;
141        out.visible = Some(visible);
142        out.enabled = Some(enabled);
143    }
144    if want_geometry {
145        // `bounding_box` swallows `display:none` to `None` already; we
146        // surface that as a missing field rather than an error.
147        let bbox = el
148            .bounding_box()
149            .await
150            .ok()
151            .flatten()
152            .map(BoundingBox::from);
153        out.bounding_box = bbox;
154        // `in_viewport` reserved for v1 — no zendriver primitive yet.
155        out.in_viewport = None;
156    }
157    if want_text_attrs {
158        let text = el
159            .inner_text()
160            .await
161            .map_err(|e| map_error(McpServerError::from(e)))?;
162        let attrs_map = el
163            .attrs()
164            .await
165            .map_err(|e| map_error(McpServerError::from(e)))?;
166        let inner_html = el
167            .inner_html()
168            .await
169            .map_err(|e| map_error(McpServerError::from(e)))?;
170        out.text = Some(text);
171        out.attrs = Some(attrs_map.into_iter().collect());
172        out.inner_html = Some(inner_html);
173    }
174    Ok(out)
175}

177/// Mirror of the same predicate in [`crate::tools::find`] — kept private
178/// because the only consumer is this module's "missing element" branch.
179fn is_not_found(err: &ErrorData) -> bool {
180    err.data
181        .as_ref()
182        .and_then(|v| v.get("suggested_next"))
183        .and_then(|v| v.as_str())
184        == Some("browser_snapshot")
185        && err.message.contains("No element matched")
186}

188#[cfg(test)]
189mod tests {
190    use super::*;
191    use crate::selectors::Selector;
192
193    fn fresh() -> Arc<Mutex<SessionState>> {
194        Arc::new(Mutex::new(SessionState::new()))
195    }
196
197    fn css_sel(s: &str) -> Selector {
198        Selector {
199            css: Some(s.into()),
200            xpath: None,
201            text: None,
202            text_exact: None,
203            text_regex: None,
204            role: None,
205            role_name: None,
206            nth: None,
207            visible_only: true,
208            timeout_ms: 5000,
209            frame_id: None,
210        }
211    }
212
213    #[tokio::test]
214    async fn element_state_with_no_browser_suggests_browser_open() {
215        let err = element_state(
216            fresh(),
217            ElementStateInput {
218                selector: css_sel("h1"),
219                include: ReadFieldsPreset::All,
220            },
221        )
222        .await
223        .expect_err("must error without an open browser");
224        assert!(err.message.contains("browser_open"), "msg: {}", err.message);
225        let data = err.data.as_ref().expect("data populated");
226        assert_eq!(data["suggested_next"], "browser_open");
227    }
228
229    #[test]
230    fn default_preset_is_all() {
231        assert_eq!(default_preset(), ReadFieldsPreset::All);
232    }
