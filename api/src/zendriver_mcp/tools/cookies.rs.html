1//! Cookie handlers — `browser_cookies_get / _set / _delete / _clear / _persist`.
2//!
3//! All five tools route through [`zendriver::CookieJar`] (one shared
4//! browser-scope handle, [`zendriver::Browser::cookies`]) rather than going
5//! per-tab — Chrome's cookie store is browser-wide, so per-tab scoping is
6//! meaningless.
7//!
8//! ## Wire shape — why [`CookieDto`] exists
9//!
10//! `zendriver::Cookie` already derives `Serialize + Deserialize`, but **not**
11//! `schemars::JsonSchema` (the lib doesn't depend on `schemars`). rmcp needs
12//! `JsonSchema` for tool input/output schema synthesis, so we mirror the
13//! struct shape with a wire-only [`CookieDto`] and convert at the handler
14//! boundary. Round-trips losslessly via `From<Cookie>` / `From<CookieDto>`.
15//! Same for [`SameSiteDto`] mirroring `zendriver::SameSite`.
16//!
17//! `browser_cookies_persist` round-trips a `Vec<Cookie>` to disk via
18//! `serde_json` — there's no `save_to_file` / `load_from_file` on
19//! `CookieJar` itself (the lib doesn't take a stance on persistence
20//! formats), so the MCP layer owns the serialization shim.
21

35/// SameSite policy mirror of [`zendriver::SameSite`].
36///
37/// Serializes with the same `"Strict" / "Lax" / "None"` capitalization as
38/// the lib's enum (which matches the RFC and CDP), so on-the-wire payloads
39/// stay identical to what `zendriver::Cookie`'s own serde emits.
40#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
41pub enum SameSiteDto {
42    /// First-party only — never sent on cross-site requests.
43    Strict,
44    /// Sent on cross-site top-level navigations only.
45    Lax,
46    /// Always sent, including in third-party contexts.
47    None,
48}
49
50impl From<zendriver::SameSite> for SameSiteDto {
51    fn from(s: zendriver::SameSite) -> Self {
52        match s {
53            zendriver::SameSite::Strict => Self::Strict,
54            zendriver::SameSite::Lax => Self::Lax,
55            zendriver::SameSite::None => Self::None,
56        }
57    }
58}
59
60impl From<SameSiteDto> for zendriver::SameSite {
61    fn from(s: SameSiteDto) -> Self {
62        match s {
63            SameSiteDto::Strict => Self::Strict,
64            SameSiteDto::Lax => Self::Lax,
65            SameSiteDto::None => Self::None,
66        }
67    }
68}

166/// Wire-only mirror of [`zendriver::Cookie`] — see module docs.
167///
168/// Field set + serde shape match the lib's `Cookie` exactly, so a `Vec<Cookie>`
169/// serialized by the lib parses cleanly as `Vec<CookieDto>` and vice versa.
170/// Adding [`JsonSchema`] here (without touching the lib) lets rmcp synthesize
171/// tool schemas.
172#[derive(Debug, Clone, Default, PartialEq, Serialize, Deserialize, JsonSchema)]
173pub struct CookieDto {
174    /// Cookie name.
175    pub name: String,
176    /// Cookie value.
177    pub value: String,
178    /// Domain the cookie applies to. Leading dot (`.example.com`) matches
179    /// the domain and all subdomains.
180    pub domain: String,
181    /// URL path the cookie applies to (typically `"/"`).
182    pub path: String,
183    /// Expiration timestamp in seconds since Unix epoch. `None` for a
184    /// session cookie.
185    #[serde(default, skip_serializing_if = "Option::is_none")]
186    pub expires: Option<f64>,
187    /// `true` if the cookie has the `HttpOnly` flag.
188    #[serde(default)]
189    pub http_only: bool,
190    /// `true` if the cookie is only sent over HTTPS.
191    #[serde(default)]
192    pub secure: bool,
193    /// SameSite policy, if specified.
194    #[serde(default, skip_serializing_if = "Option::is_none")]
195    pub same_site: Option<SameSiteDto>,
196    /// Origin URL for set-side convenience. CDP infers `domain` + `path` +
197    /// `secure` when present. Always `None` on cookies returned by reads.
198    #[serde(default, skip_serializing_if = "Option::is_none")]
199    pub url: Option<String>,
200    /// Eviction-priority hint (`Network.CookiePriority`). `None` lets
201    /// Chrome apply its default (`Medium`).
202    #[serde(default, skip_serializing_if = "Option::is_none")]
203    pub priority: Option<CookiePriorityDto>,
204    /// First-Party Sets membership flag (`sameParty`). Rarely set by hand;
205    /// populated by CDP on reads when known.
206    #[serde(default, skip_serializing_if = "Option::is_none")]
207    pub same_party: Option<bool>,
208    /// Scheme of the origin that set the cookie
209    /// (`Network.CookieSourceScheme`). Drives schemeful same-site.
210    #[serde(default, skip_serializing_if = "Option::is_none")]
211    pub source_scheme: Option<CookieSourceSchemeDto>,
212    /// Source port of the origin that set the cookie. `-1` in CDP means
213    /// "unspecified"; modeled as a plain `Option<i32>` here.
214    #[serde(default, skip_serializing_if = "Option::is_none")]
215    pub source_port: Option<i32>,
216    /// CHIPS partition key — the top-level site the partitioned cookie is
217    /// scoped to, plus whether it was set from a cross-site context.
218    /// Mirrors CDP's `Network.CookiePartitionKey` (M119+).
219    #[serde(default, skip_serializing_if = "Option::is_none")]
220    pub partition_key: Option<CookiePartitionKeyDto>,
221}
222
223impl From<zendriver::Cookie> for CookieDto {
224    fn from(c: zendriver::Cookie) -> Self {
225        Self {
226            name: c.name,
227            value: c.value,
228            domain: c.domain,
229            path: c.path,
230            expires: c.expires,
231            http_only: c.http_only,
232            secure: c.secure,
233            same_site: c.same_site.map(SameSiteDto::from),
234            url: c.url,
235            priority: c.priority.map(CookiePriorityDto::from),
236            same_party: c.same_party,
237            source_scheme: c.source_scheme.map(CookieSourceSchemeDto::from),
238            source_port: c.source_port,
239            partition_key: c.partition_key.map(CookiePartitionKeyDto::from),
240        }
241    }
242}

267/// Input for `browser_cookies_get`.
268#[derive(Debug, Default, Deserialize, JsonSchema)]
269#[serde(deny_unknown_fields)]
270pub struct CookiesGetInput {
271    /// When set, returns only cookies that would be sent for this URL
272    /// (CDP `Network.getCookies`). When unset, returns every cookie in
273    /// the browser's store (`Storage.getCookies`).
274    #[serde(default)]
275    pub url: Option<String>,
276    /// When set, post-filters the result to cookies whose `name` matches
277    /// exactly. Combinable with `url`.
278    #[serde(default)]
279    pub name: Option<String>,
280}
281
282/// Output of `browser_cookies_get`.
283#[derive(Debug, Serialize, JsonSchema)]
284pub struct CookiesGetOutput {
285    /// Cookies matching the input filters. Order matches what CDP
286    /// returned (unspecified).
287    pub cookies: Vec<CookieDto>,
288}
289
290/// Fetch the browser's cookies, optionally filtered by URL and name.
291pub async fn cookies_get(
292    state: Arc<Mutex<SessionState>>,
293    input: CookiesGetInput,
294) -> Result<CookiesGetOutput, ErrorData> {
295    let s = state.lock().await;
296    let b = s
297        .browser
298        .as_ref()
299        .ok_or_else(|| map_error(McpServerError::BrowserNotOpen))?;
300    let jar = b.cookies();
301    let cs = match input.url.as_deref() {
302        Some(u) => {
303            let parsed = url::Url::parse(u)
304                .map_err(|e| ErrorData::invalid_params(format!("invalid url `{u}`: {e}"), None))?;
305            jar.for_url(&parsed)
306                .await
307                .map_err(|e| map_error(McpServerError::from(e)))?
308        }
309        None => jar
310            .all()
311            .await
312            .map_err(|e| map_error(McpServerError::from(e)))?,
313    };
314    let filtered: Vec<CookieDto> = match input.name {
315        Some(n) => cs
316            .into_iter()
317            .filter(|c| c.name == n)
318            .map(CookieDto::from)
319            .collect(),
320        None => cs.into_iter().map(CookieDto::from).collect(),
321    };
322    Ok(CookiesGetOutput { cookies: filtered })
323}

328#[derive(Debug, Deserialize, JsonSchema)]

330pub struct CookiesSetInput {

334}

336/// Output of `browser_cookies_set`.
337#[derive(Debug, Serialize, JsonSchema)]
338pub struct CookiesSetOutput {
339    /// Number of cookies dispatched to CDP (echoes input.cookies.len()).
340    pub added: usize,
341}
342
343/// Set many cookies in one CDP round-trip.
344pub async fn cookies_set(
345    state: Arc<Mutex<SessionState>>,
346    input: CookiesSetInput,
347) -> Result<CookiesSetOutput, ErrorData> {
348    let s = state.lock().await;
349    let b = s
350        .browser
351        .as_ref()
352        .ok_or_else(|| map_error(McpServerError::BrowserNotOpen))?;
353    let count = input.cookies.len();
354    let cookies: Vec<zendriver::Cookie> = input
355        .cookies
356        .into_iter()
357        .map(zendriver::Cookie::from)
358        .collect();
359    b.cookies()
360        .set_many(cookies)
361        .await
362        .map_err(|e| map_error(McpServerError::from(e)))?;
363    Ok(CookiesSetOutput { added: count })
364}

438/// Direction selector for `browser_cookies_persist`.
439#[derive(Debug, Clone, Copy, PartialEq, Eq, Deserialize, Serialize, JsonSchema)]
440#[serde(rename_all = "snake_case")]
441pub enum PersistDirection {
442    /// Snapshot every cookie in the browser store to `path` as pretty JSON.
443    Save,
444    /// Replace the in-memory cookies with those parsed from `path`.
445    Load,
446}
447
448/// Input for `browser_cookies_persist`.
449#[derive(Debug, Deserialize, JsonSchema)]
450#[serde(deny_unknown_fields)]
451pub struct CookiesPersistInput {
452    /// Whether to `save` (browser → disk) or `load` (disk → browser).
453    pub direction: PersistDirection,
454    /// Path on the MCP server host. Save writes pretty-printed JSON; load
455    /// expects the same shape (a JSON array of cookies).
456    pub path: String,
457}
458
459/// Output of `browser_cookies_persist`.
460#[derive(Debug, Serialize, JsonSchema)]
461pub struct CookiesPersistOutput {
462    /// Number of cookies written (save) or restored (load).
463    pub count: usize,
464    /// Echoes the input direction.
465    pub direction: PersistDirection,
466}
467
468/// Save the browser's cookies to disk or restore from disk.
469///
470/// The on-disk format is `serde_json` of a `Vec<CookieDto>` — same shape
471/// as `browser_cookies_get` returns. There's no `save_to_file` /
472/// `load_from_file` on `CookieJar` itself; the persistence shim lives at
473/// the MCP layer so the lib can stay format-agnostic.
474pub async fn cookies_persist(
475    state: Arc<Mutex<SessionState>>,
476    input: CookiesPersistInput,
477) -> Result<CookiesPersistOutput, ErrorData> {
478    let s = state.lock().await;
479    let b = s
480        .browser
481        .as_ref()
482        .ok_or_else(|| map_error(McpServerError::BrowserNotOpen))?;
483    let jar = b.cookies();
484    let path = PathBuf::from(&input.path);
485    let count = match input.direction {
486        PersistDirection::Save => {
487            let cookies = jar
488                .all()
489                .await
490                .map_err(|e| map_error(McpServerError::from(e)))?;
491            let dtos: Vec<CookieDto> = cookies.into_iter().map(CookieDto::from).collect();
492            let bytes = serde_json::to_vec_pretty(&dtos)
493                .map_err(|e| ErrorData::internal_error(format!("serialize cookies: {e}"), None))?;
494            tokio::fs::write(&path, &bytes).await.map_err(|e| {
495                ErrorData::internal_error(format!("write `{}`: {e}", path.display()), None)
496            })?;
497            dtos.len()
498        }
499        PersistDirection::Load => {
500            let bytes = tokio::fs::read(&path).await.map_err(|e| {
501                ErrorData::internal_error(format!("read `{}`: {e}", path.display()), None)
502            })?;
503            let dtos: Vec<CookieDto> = serde_json::from_slice(&bytes).map_err(|e| {
504                ErrorData::invalid_params(
505                    format!("parse `{}` as cookie JSON: {e}", path.display()),
506                    None,
507                )
508            })?;
509            let count = dtos.len();
510            let cookies: Vec<zendriver::Cookie> =
511                dtos.into_iter().map(zendriver::Cookie::from).collect();
512            jar.set_many(cookies)
513                .await
514                .map_err(|e| map_error(McpServerError::from(e)))?;
515            count
516        }
517    };
518    Ok(CookiesPersistOutput {
519        count,
520        direction: input.direction,
521    })
522}

674    /// `browser_cookies_persist save → load` round-trip via the serde shim
675    /// alone (no real Chrome). Validates the on-disk format the lib expects.
676    #[tokio::test]
677    async fn cookies_persist_save_load_round_trip_via_serde_shim() {
678        let tmp = std::env::temp_dir().join(format!(
679            "zendriver-mcp-cookies-persist-shim-{}.json",
680            std::process::id()
681        ));
682        let _ = tokio::fs::remove_file(&tmp).await;
683
684        let cookies = vec![
685            CookieDto {
686                name: "a".into(),
687                value: "1".into(),
688                domain: ".x.com".into(),
689                path: "/".into(),
690                expires: None,
691                http_only: false,
692                secure: false,
693                same_site: None,
694                url: None,
695                ..Default::default()
696            },
697            CookieDto {
698                name: "b".into(),
699                value: "2".into(),
700                domain: ".y.com".into(),
701                path: "/".into(),
702                expires: Some(1_700_000_000.0),
703                http_only: true,
704                secure: true,
705                same_site: Some(SameSiteDto::Strict),
706                url: None,
707                priority: Some(CookiePriorityDto::Low),
708                same_party: None,
709                source_scheme: Some(CookieSourceSchemeDto::NonSecure),
710                source_port: None,
711                partition_key: Some(CookiePartitionKeyDto {
712                    top_level_site: "https://top.example".into(),
713                    has_cross_site_ancestor: false,
714                }),
715            },
716        ];
717
718        let bytes = serde_json::to_vec_pretty(&cookies).unwrap();
719        tokio::fs::write(&tmp, &bytes).await.unwrap();
720
721        let raw = tokio::fs::read(&tmp).await.unwrap();
722        let loaded: Vec<CookieDto> = serde_json::from_slice(&raw).unwrap();

726        for dto in &loaded {

729            assert_eq!(*dto, back);
730        }
731
732        let _ = tokio::fs::remove_file(&tmp).await;
733    }
