1//! Network monitor tools — `browser_monitor_start` / `_read` / `_stop`.
2//!
3//! ## Drain-task model
4//!
5//! `browser_monitor_start` builds a lib-side `tab.monitor()` `Stream` and
6//! spawns a *drain task* that owns that stream. The task pulls events, converts
7//! each `zendriver::NetworkEvent` into the wire-level [`MonitorEvent`], and
8//! pushes them into a per-handle bounded ring ([`MonitorBuffer`], capped at
9//! [`MONITOR_BUFFER_CAP`]). The handle is returned synchronously; the task runs
10//! until cancelled.
11//!
12//! `browser_monitor_read` drains the ring (up to `max`) and reports the running
13//! `dropped` count (events evicted while the ring was full), then resets it.
14//! `browser_monitor_stop` cancels the task's [`CancellationToken`] and aborts
15//! its join handle, then drops the handle from the session registry.
16//!
17//! ## Why bodies are fetched in the drain task (not at read time)
18//!
19//! Chrome only retains a response body for a short window after the response
20//! completes. So when the monitor was started with `capture_bodies`, the drain
21//! task calls `exchange.body().await` *at observe-time* — the moment the
22//! exchange is pulled from the stream — and inlines the result on the
23//! [`MonitorEvent::Http`]. Deferring the fetch to `read` time would routinely
24//! race Chrome's eviction and return empty bodies.
25//!
26//! ## Concurrency
27//!
28//! The ring lives behind its own `Arc<Mutex<MonitorBuffer>>`, shared by the
29//! drain task and `read`. Each side locks it only briefly — the drain task per
30//! pushed event, `read` per drain — and never holds it across an `.await` that
31//! waits on the other side, so the pair cannot deadlock. Building the ring +
32//! cancel token *before* spawning the task lets the real `JoinHandle` go
33//! straight into [`MonitorState`] with no placeholder spawn.
34

55/// Input for `browser_monitor_start`.
56#[derive(Debug, Deserialize, JsonSchema)]
57#[serde(deny_unknown_fields)]
58pub struct StartInput {
59    /// Restrict buffered events to those whose URL contains this substring.
60    /// Omit to buffer every observed event.
61    #[serde(default, skip_serializing_if = "Option::is_none")]
62    pub url_pattern: Option<String>,
63    /// Fetch each HTTP response body at observe-time and inline it on the
64    /// `http` event (`body` + `body_base64`). One extra CDP round-trip per
65    /// exchange; off by default to keep the buffer light.
66    #[serde(default)]
67    pub capture_bodies: bool,
68}

70/// Output of `browser_monitor_start`.
71#[derive(Debug, Serialize, JsonSchema)]
72pub struct StartOutput {
73    /// Opaque handle for the running monitor. Pass to `browser_monitor_read`
74    /// and `browser_monitor_stop`.
75    pub handle: String,
76}

78/// Start a network monitor over the current tab and begin buffering events.
79///
80/// Resolves the current tab, builds the lib-side monitor stream, spawns the
81/// drain task, and registers the resulting [`MonitorState`] under a fresh
82/// handle. Returns immediately — events accumulate in the background.
83pub async fn start(
84    state: Arc<Mutex<SessionState>>,
85    input: StartInput,
86) -> Result<StartOutput, ErrorData> {
87    let tab = {
88        let s = state.lock().await;
89        current_tab(&s).await?
90    };
91
92    let mut mb = tab.monitor();
93    if let Some(p) = &input.url_pattern {
94        mb = mb.url_pattern(p.clone());
95    }
96    let mut stream = mb.start().await.map_err(map_error)?;
97
98    // Build the shared ring + cancel token FIRST, then spawn the drain task
99    // capturing clones, then construct `MonitorState` with the real join
100    // handle — no placeholder spawn, no self-referential cycle.
101    let buffer = Arc::new(Mutex::new(MonitorBuffer::default()));
102    let cancel = CancellationToken::new();
103    let capture = input.capture_bodies;
104
105    let task = {
106        let buffer = buffer.clone();
107        let cancel = cancel.clone();
108        tokio::spawn(async move {
109            loop {
110                tokio::select! {
111                    () = cancel.cancelled() => break,
112                    next = stream.next() => {
113                        let Some(ev) = next else { break };
114                        // Convert (and, when capturing, fetch the body) BEFORE
115                        // taking the buffer lock so the lock is never held
116                        // across the `body().await` CDP round-trip.
117                        let wire = convert(ev, capture).await;
118                        let mut buf = buffer.lock().await;
119                        push_capped(&mut buf, wire);
120                    }
121                }
122            }
123        })
124    };
125
126    let handle = uuid::Uuid::new_v4().to_string();
127    let mon = Arc::new(Mutex::new(MonitorState {
128        buffer,
129        cancel,
130        task,
131    }));
132    state.lock().await.monitors.insert(handle.clone(), mon);
133    Ok(StartOutput { handle })
134}
135
136/// Push `event` into the bounded ring, evicting the oldest (and bumping
137/// `dropped`) when the ring is already at [`MONITOR_BUFFER_CAP`].
138fn push_capped(buf: &mut MonitorBuffer, event: MonitorEvent) {
139    if buf.events.len() >= MONITOR_BUFFER_CAP {
140        buf.events.pop_front();
141        buf.dropped += 1;
142    }
143    buf.events.push_back(event);
144}

146/// Convert a lib-side [`NetworkEvent`] into the wire mirror, fetching the HTTP
147/// response body at observe-time when `capture` is set (before Chrome evicts
148/// it). A body-fetch failure degrades to a body-less event rather than an
149/// error — the exchange's metadata is still worth surfacing.
150async fn convert(ev: NetworkEvent, capture: bool) -> MonitorEvent {
151    match ev {
152        NetworkEvent::Http(ex) => {
153            let (body, body_base64) = if capture {
154                match ex.body().await {
155                    Ok(bytes) => (
156                        Some(String::from_utf8_lossy(&bytes).into_owned()),
157                        Some(BASE64.encode(&bytes)),
158                    ),
159                    Err(_) => (None, None),
160                }
161            } else {
162                (None, None)
163            };
164            MonitorEvent::Http {
165                url: ex.request.url.clone(),
166                method: ex.request.method.clone(),
167                status: ex.status(),
168                error: ex.error.clone(),
169                body,
170                body_base64,
171            }
172        }
173        NetworkEvent::WebSocketOpen { request_id, url } => {
174            MonitorEvent::WebSocketOpen { request_id, url }
175        }
176        NetworkEvent::WebSocketFrame {
177            request_id,
178            direction,
179            opcode,
180            payload,
181        } => MonitorEvent::WebSocketFrame {
182            request_id,
183            direction: match direction {
184                FrameDirection::Sent => "sent",
185                FrameDirection::Received => "received",
186            }
187            .to_string(),
188            opcode,
189            payload,
190        },
191        NetworkEvent::WebSocketClose { request_id } => MonitorEvent::WebSocketClose { request_id },
192        NetworkEvent::EventSourceMessage {
193            request_id,
194            event_name,
195            event_id,
196            data,
197        } => MonitorEvent::EventSourceMessage {
198            request_id,
199            event_name,
200            event_id,
201            data,
202        },
203    }
204}

208/// Input for `browser_monitor_read`.
209#[derive(Debug, Deserialize, JsonSchema)]
210#[serde(deny_unknown_fields)]
211pub struct ReadInput {
212    /// Handle returned by an earlier `browser_monitor_start`.
213    pub handle: String,
214    /// Maximum number of events to drain this call. Omit to drain all buffered
215    /// events.
216    #[serde(default, skip_serializing_if = "Option::is_none")]
217    pub max: Option<usize>,
218}
219
220/// Output of `browser_monitor_read`.
221#[derive(Debug, Serialize, JsonSchema)]
222pub struct ReadOutput {
223    /// Drained events, oldest first.
224    pub events: Vec<MonitorEvent>,
225    /// Events evicted (ring full) since the previous `read`. Non-zero means the
226    /// buffer overflowed — read more often or stop the monitor.
227    pub dropped: usize,
228}
229
230/// Drain up to `max` buffered events from a running monitor.
231///
232/// Looks the handle up in the session registry, then drains its ring. The
233/// running `dropped` count is returned and reset to 0.
234pub async fn read(
235    state: Arc<Mutex<SessionState>>,
236    input: ReadInput,
237) -> Result<ReadOutput, ErrorData> {
238    let mon = {
239        let s = state.lock().await;
240        s.monitors.get(&input.handle).cloned()
241    }
242    .ok_or_else(|| unknown_handle(&input.handle))?;
243
244    // Lock the MonitorState briefly to reach its shared ring, then the ring
245    // itself — neither lock is held across an `.await` that the drain task
246    // waits on.
247    let buffer = {
248        let m = mon.lock().await;
249        m.buffer.clone()
250    };
251    let mut buf = buffer.lock().await;
252    let (events, dropped) = drain(&mut buf, input.max.unwrap_or(usize::MAX));
253    Ok(ReadOutput { events, dropped })
254}
255
256/// Drain up to `max` events from `buf`, returning them (oldest first) plus the
257/// `dropped` count, which is reset to 0.
258///
259/// Extracted as a free fn so the drain + dropped-reset semantics are
260/// unit-testable without a browser, a `SessionState`, or the drain task.
261fn drain(buf: &mut MonitorBuffer, max: usize) -> (Vec<MonitorEvent>, usize) {
262    let take = max.min(buf.events.len());
263    let events: Vec<MonitorEvent> = buf.events.drain(..take).collect();
264    let dropped = std::mem::take(&mut buf.dropped);
265    (events, dropped)
266}

270/// Input for `browser_monitor_stop`.
271#[derive(Debug, Deserialize, JsonSchema)]
272#[serde(deny_unknown_fields)]
273pub struct StopInput {
274    /// Handle returned by an earlier `browser_monitor_start`.
275    pub handle: String,
276}
277
278/// Output of `browser_monitor_stop`.
279#[derive(Debug, Serialize, JsonSchema)]
280pub struct StopOutput {
281    /// `true` if a live monitor was found and stopped; `false` if the handle
282    /// was unknown (already stopped, or never started).
283    pub stopped: bool,
284}
285
286/// Stop a running monitor: cancel its drain task and drop its registry entry.
287///
288/// Removing the handle from the registry drops the `MonitorState` (and its
289/// shared ring) once this function's lock guard releases; cancelling + aborting
290/// the task first guarantees it stops touching the stream promptly. An unknown
291/// handle returns `{ stopped: false }` (idempotent stop) rather than an error.
292pub async fn stop(
293    state: Arc<Mutex<SessionState>>,
294    input: StopInput,
295) -> Result<StopOutput, ErrorData> {
296    let Some(mon) = state.lock().await.monitors.remove(&input.handle) else {
297        return Ok(StopOutput { stopped: false });
298    };
299    let m = mon.lock().await;
300    m.cancel.cancel();
301    m.task.abort();
302    Ok(StopOutput { stopped: true })
303}
304
305/// Wire error for a `read` / `stop` against a handle that isn't registered.
306fn unknown_handle(handle: &str) -> ErrorData {
307    ErrorData::invalid_params(
308        format!(
309            "unknown monitor handle `{handle}`. Start one with `browser_monitor_start`; the handle may have already been stopped."
310        ),
311        None,
312    )
313}

315#[cfg(test)]
316#[allow(clippy::panic, clippy::unwrap_used)]
317mod tests {
318    //! No-browser unit coverage of the buffer drain + cap-eviction logic.
319    //!
320    //! The drain-task path (which constructs a real `tab.monitor()` stream and
321    //! spawns the correlator) needs a live Chrome and is exercised in the gated
322    //! integration tests. Here we drive `drain` / `push_capped` directly against
323    //! a synthetic `MonitorBuffer`.
324    use std::collections::VecDeque;
325
326    use super::*;
327
328    /// A cheap synthetic event for buffer tests.
329    fn ev(n: usize) -> MonitorEvent {
330        MonitorEvent::Http {
331            url: format!("https://example.com/{n}"),
332            method: "GET".into(),
333            status: Some(200),
334            error: None,
335            body: None,
336            body_base64: None,
337        }
338    }
339
340    #[test]
341    fn drain_returns_up_to_max_and_clears_them() {
342        let mut buf = MonitorBuffer {
343            events: (0..5).map(ev).collect(),
344            dropped: 0,
345        };
346        let (events, dropped) = drain(&mut buf, 3);
347        assert_eq!(events.len(), 3, "drains exactly `max`");
348        assert_eq!(events[0], ev(0), "oldest first");
349        assert_eq!(events[2], ev(2));
350        assert_eq!(dropped, 0);
351        // Remaining events stay, in order, for the next read.
352        assert_eq!(buf.events.len(), 2);
353        assert_eq!(buf.events.front().unwrap(), &ev(3));
354    }
355
356    #[test]
357    fn drain_caps_at_buffer_len_when_max_exceeds_it() {
358        let mut buf = MonitorBuffer {
359            events: (0..2).map(ev).collect(),
360            dropped: 0,
361        };
362        let (events, _) = drain(&mut buf, usize::MAX);
363        assert_eq!(events.len(), 2, "never over-drains past what's buffered");
364        assert!(buf.events.is_empty());
365    }
366
367    #[test]
368    fn drain_returns_and_resets_dropped() {
369        let mut buf = MonitorBuffer {
370            events: VecDeque::new(),
371            dropped: 3,
372        };
373        let (events, dropped) = drain(&mut buf, usize::MAX);
374        assert!(events.is_empty());
375        assert_eq!(dropped, 3, "reports the running dropped count");
376        assert_eq!(buf.dropped, 0, "and resets it to zero");
377    }
378
379    #[test]
380    fn push_capped_evicts_oldest_and_counts_dropped_over_cap() {
381        let mut buf = MonitorBuffer::default();
382        // Fill exactly to capacity — no drops yet.
383        for n in 0..MONITOR_BUFFER_CAP {
384            push_capped(&mut buf, ev(n));
385        }
386        assert_eq!(buf.events.len(), MONITOR_BUFFER_CAP);
387        assert_eq!(buf.dropped, 0);
388
389        // One more over the cap evicts the oldest (event 0) and counts it.
390        push_capped(&mut buf, ev(MONITOR_BUFFER_CAP));
391        assert_eq!(buf.events.len(), MONITOR_BUFFER_CAP, "stays at the cap");
392        assert_eq!(buf.dropped, 1, "the evicted event is counted");
393        assert_eq!(
394            buf.events.front().unwrap(),
395            &ev(1),
396            "event 0 was evicted; event 1 is now oldest"
397        );
398        assert_eq!(
399            buf.events.back().unwrap(),
400            &ev(MONITOR_BUFFER_CAP),
401            "the newest push is retained"
402        );
403    }
