1//! Stealth-profile configuration — `browser_set_stealth_profile`.
2//!
3//! Mutates only the session-level `stealth_profile_choice` — the chosen
4//! profile takes effect on the NEXT `browser_open` call. We do NOT
5//! re-fingerprint an already-running browser; the underlying stealth
6//! patches are injected at launch and there's no live "switch profile"
7//! lever in the lib.
8
9use std::sync::Arc;
10
11use rmcp::ErrorData;
12use schemars::JsonSchema;
13use serde::{Deserialize, Serialize};
14use tokio::sync::Mutex;
15
16use crate::state::{SessionState, StealthProfileChoice};
17
18/// Input for `browser_set_stealth_profile`.
19#[derive(Debug, Deserialize, JsonSchema)]
20#[serde(deny_unknown_fields)]
21pub struct SetStealthProfileInput {
22    /// New default stealth profile for this session.
23    pub profile: StealthProfileChoice,
24}
25
26/// Output of `browser_set_stealth_profile`.
27#[derive(Debug, Serialize, JsonSchema)]
28pub struct SetStealthProfileOutput {
29    /// The profile that is now configured as the session default.
30    pub active_profile: StealthProfileChoice,
31    /// `true` iff a browser is currently open. When `true`, the new
32    /// profile only applies after `browser_close` + `browser_open`. When
33    /// `false`, the next `browser_open` call will pick it up directly.
34    pub takes_effect_on_next_open: bool,
35}
36
37/// Configure the session's default stealth profile.
38///
39/// Always succeeds — the operation only mutates `SessionState` and never
40/// touches the running browser. Callers wanting the change to apply to a
41/// currently-open browser must follow up with `browser_close` +
42/// `browser_open`.
43pub async fn set_stealth_profile(
44    state: Arc<Mutex<SessionState>>,
45    input: SetStealthProfileInput,
46) -> Result<SetStealthProfileOutput, ErrorData> {
47    let mut s = state.lock().await;
48    s.stealth_profile_choice = input.profile;
49    Ok(SetStealthProfileOutput {
50        active_profile: input.profile,
51        takes_effect_on_next_open: s.browser.is_some(),
52    })
53}

55#[cfg(test)]
56mod tests {
57    use super::*;
58
59    fn fresh() -> Arc<Mutex<SessionState>> {
60        Arc::new(Mutex::new(SessionState::new()))
61    }
62
63    #[tokio::test]
64    async fn set_profile_without_browser_reports_immediate_effect() {
65        let state = fresh();
66        let out = set_stealth_profile(
67            state.clone(),
68            SetStealthProfileInput {
69                profile: StealthProfileChoice::SpoofLinux,
70            },
71        )
72        .await
73        .expect("set profile ok");
74        assert_eq!(out.active_profile, StealthProfileChoice::SpoofLinux);
75        assert!(
76            !out.takes_effect_on_next_open,
77            "no browser open, so the next open call picks the new profile up directly"
78        );
79        // And the state was mutated.
80        let s = state.lock().await;
81        assert_eq!(s.stealth_profile_choice, StealthProfileChoice::SpoofLinux);
82    }
83
84    #[tokio::test]
85    async fn set_profile_overwrites_previous_choice() {
86        let state = fresh();
87        set_stealth_profile(
88            state.clone(),
89            SetStealthProfileInput {
90                profile: StealthProfileChoice::SpoofMacos,
91            },
92        )
93        .await
94        .expect("first set ok");
95        let out = set_stealth_profile(
96            state.clone(),
97            SetStealthProfileInput {
98                profile: StealthProfileChoice::SpoofWindows,
99            },
100        )
101        .await
102        .expect("second set ok");
103        assert_eq!(out.active_profile, StealthProfileChoice::SpoofWindows);
104        let s = state.lock().await;
105        assert_eq!(s.stealth_profile_choice, StealthProfileChoice::SpoofWindows);
106    }
107
108    #[tokio::test]
109    async fn set_profile_does_not_fail_when_choice_matches_existing() {
110        // No-op semantic: setting the same profile again should still
111        // succeed (and confirm "takes_effect_on_next_open=false").
112        let state = fresh();
113        let out = set_stealth_profile(
114            state,
115            SetStealthProfileInput {
116                profile: StealthProfileChoice::Auto,
117            },
118        )
119        .await
120        .expect("idempotent set ok");
121        assert_eq!(out.active_profile, StealthProfileChoice::Auto);
122        assert!(!out.takes_effect_on_next_open);
123    }
124}
