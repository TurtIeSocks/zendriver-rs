1//! Tab management handlers — `browser_tab_list`, `_new`, `_switch`,
2//! `_close`, `_activate`.
3//!
4//! Each handler is a free async fn that locks the shared
5//! [`SessionState`][crate::state::SessionState] internally and returns a
6//! typed output (or an [`rmcp::ErrorData`]). The thin `#[tool]` wrappers in
7//! [`crate::server`] forward to these.
8//!
9//! All tools use [`TabSummary`] as their per-tab projection. The `is_current`
10//! flag is computed against `state.current_tab_id`, so a freshly-listed
11//! `Vec<TabSummary>` always reflects the session's view of "the focused
12//! tab" — there is no separate `tabs().current()` accessor on `Browser`.
13//!
14//! ## API notes
15//!
16//! - `Browser::tabs()` is async and returns `Vec<Tab>` (owned, cheap to
17//!   clone — `Tab` is `Arc`-backed).
18//! - `Browser::new_tab()` opens about:blank; `new_tab_at(url)` accepts an
19//!   initial URL. Plan called for an `Option<String>` flag that branches
20//!   between the two.
21//! - `Tab::close(self)` consumes the receiver — we fetch the tab out of
22//!   `Browser::tabs()` and call `.close()` on the owned handle.
23//! - `Tab::activate(&self)` borrows. No consumption; the tab stays usable.
24//! - `Browser::tabs()` order is unspecified (HashMap-backed) — we lean on
25//!   `target_id` for lookup, never positional indexing.
26

40/// Per-tab projection returned by `browser_tab_list` + the create/switch
41/// tools.
42///
43/// `is_current` is computed against `SessionState::current_tab_id` at the
44/// time of the call. Returning it inline (rather than expecting the client
45/// to cross-reference against a separate "current tab" field) keeps the
46/// list response self-contained.
47#[derive(Debug, Serialize, JsonSchema, PartialEq, Eq)]
48pub struct TabSummary {
49    /// Stable CDP target id. Use this with `browser_tab_switch` /
50    /// `_close` / `_activate`.
51    pub id: String,
52    /// Live tab URL (best-effort; empty string on transient lookup
53    /// failure, e.g. a tab that's in the middle of closing).
54    pub url: String,
55    /// Live tab title (best-effort; empty string on transient lookup
56    /// failure).
57    pub title: String,
58    /// `true` iff this tab is the session's current focused tab.
59    pub is_current: bool,
60}
61
62/// Build a [`TabSummary`] from a `&Tab`, classifying `is_current` against
63/// the supplied `current_tab_id`.
64async fn summarize_tab(tab: &zendriver::Tab, current_tab_id: Option<&str>) -> TabSummary {
65    let id = tab.target_id().to_string();
66    let is_current = current_tab_id == Some(id.as_str());
67    let url = tab.url().await.map(|u| u.to_string()).unwrap_or_default();
68    let title = tab.title().await.unwrap_or_default();
69    TabSummary {
70        id,
71        url,
72        title,
73        is_current,
74    }
75}

79/// Output of `browser_tab_list`.
80#[derive(Debug, Serialize, JsonSchema)]
81pub struct TabListOutput {
82    /// All live tabs in unspecified order. Order is not guaranteed
83    /// stable across calls — `Browser::tabs()` is HashMap-backed.
84    pub tabs: Vec<TabSummary>,
85}
86
87/// Enumerate all live tabs.
88///
89/// Returns an empty list when no browser is open is NOT the contract —
90/// we return [`McpServerError::BrowserNotOpen`] instead, so the agent gets
91/// a `suggested_next: browser_open` hint.
92pub async fn list(
93    state: Arc<Mutex<SessionState>>,
94    _: EmptyInput,
95) -> Result<TabListOutput, ErrorData> {
96    let s = state.lock().await;
97    let b = s
98        .browser
99        .as_ref()
100        .ok_or_else(|| map_error(McpServerError::BrowserNotOpen))?;
101    let tabs = b.tabs().await;
102    let mut out = Vec::with_capacity(tabs.len());
103    let current = s.current_tab_id.as_deref();
104    for t in &tabs {
105        out.push(summarize_tab(t, current).await);
106    }
107    Ok(TabListOutput { tabs: out })
108}

112/// Input for `browser_tab_new`.
113#[derive(Debug, Deserialize, JsonSchema)]
114#[serde(deny_unknown_fields)]
115pub struct TabNewInput {
116    /// Initial URL for the new tab. When `None` the tab opens at
117    /// `about:blank` (via `Browser::new_tab`).
118    #[serde(default)]
119    pub url: Option<String>,
120    /// When `true` (the default), the new tab becomes the session's
121    /// current tab — subsequent navigation / find / action tools target
122    /// it. When `false` the focus stays on the previous current tab and
123    /// the caller can switch later via `browser_tab_switch`.
124    #[serde(default = "default_true")]
125    pub activate: bool,
126}
127
128const fn default_true() -> bool {
129    true
130}
131
132/// Open a new tab.
133///
134/// When `input.url` is `Some(_)` the underlying call is `new_tab_at(url)`;
135/// otherwise the tab opens at `about:blank`. If `input.activate` is `true`
136/// the session's current tab is updated to the freshly-opened tab.
137pub async fn new_tab(
138    state: Arc<Mutex<SessionState>>,
139    input: TabNewInput,
140) -> Result<TabSummary, ErrorData> {
141    let mut s = state.lock().await;
142    let b = s
143        .browser
144        .as_ref()
145        .ok_or_else(|| map_error(McpServerError::BrowserNotOpen))?;
146    let tab = match input.url.as_deref() {
147        Some(u) => b
148            .new_tab_at(u)
149            .await
150            .map_err(|e| map_error(McpServerError::from(e)))?,
151        None => b
152            .new_tab()
153            .await
154            .map_err(|e| map_error(McpServerError::from(e)))?,
155    };
156    let id = tab.target_id().to_string();
157    if input.activate {
158        s.current_tab_id = Some(id.clone());
159    }
160    // `summarize_tab` reads the current `s.current_tab_id` (possibly the
161    // freshly-set id above) so `is_current` lines up with the state we
162    // just wrote.
163    let current = s.current_tab_id.as_deref();
164    Ok(summarize_tab(&tab, current).await)
165}

169/// Input for `browser_tab_switch`.
170#[derive(Debug, Deserialize, JsonSchema)]
171#[serde(deny_unknown_fields)]
172pub struct TabSwitchInput {
173    /// CDP target id of the tab to switch to. Obtain from
174    /// `browser_tab_list`.
175    pub tab_id: String,
176}
177
178/// Switch the session's current tab to `tab_id`.
179///
180/// Validates the id against the live tab list first — if not found,
181/// returns [`McpServerError::NoCurrentTab`] (which suggests
182/// `browser_tab_new`) plus an error message pointing at
183/// `browser_tab_list`. We deliberately surface a richer message than the
184/// canonical `NoCurrentTab` carries: the agent's most likely next step is
185/// to re-enumerate tabs, not open a new one.
186pub async fn switch(
187    state: Arc<Mutex<SessionState>>,
188    input: TabSwitchInput,
189) -> Result<TabSummary, ErrorData> {
190    let mut s = state.lock().await;
191    let b = s
192        .browser
193        .as_ref()
194        .ok_or_else(|| map_error(McpServerError::BrowserNotOpen))?;
195    let tabs = b.tabs().await;
196    let tab = tabs
197        .iter()
198        .find(|t| t.target_id() == input.tab_id)
199        .cloned()
200        .ok_or_else(|| {
201            map_error(McpServerError::from(
202                zendriver::ZendriverError::TabNotFound(input.tab_id.clone()),
203            ))
204        })?;
205    s.current_tab_id = Some(input.tab_id.clone());
206    let current = s.current_tab_id.as_deref();
207    Ok(summarize_tab(&tab, current).await)
208}

212/// Input for `browser_tab_close`.
213#[derive(Debug, Default, Deserialize, JsonSchema)]
214#[serde(deny_unknown_fields)]
215pub struct TabCloseInput {
216    /// CDP target id of the tab to close. When `None`, the current tab
217    /// is closed.
218    #[serde(default)]
219    pub tab_id: Option<String>,
220}
221
222/// Output of `browser_tab_close`.
223#[derive(Debug, Serialize, JsonSchema)]
224pub struct TabCloseOutput {
225    /// CDP target id of the tab that was closed.
226    pub closed_id: String,
227    /// Session's current tab after the close. `None` when the last tab
228    /// was closed (or when the closed tab was the current tab and no
229    /// other tabs remained).
230    pub current_tab_id: Option<String>,
231}
232
233/// Close a tab.
234///
235/// When `tab_id` is `None`, the current tab is closed. After the close,
236/// if we just removed the current tab, the next live tab from
237/// `Browser::tabs()` becomes the current tab (deterministic-enough for
238/// agent flows that don't care which fallback wins; reorder via
239/// `browser_tab_switch` if you do).
240///
241/// `Tab::close(self)` consumes the receiver, so we have to fetch the
242/// owned `Tab` out of `Browser::tabs()` before calling `.close()`.
243///
244/// # v0 limitation: tab-scoped expectations and interception rules leak
245///
246/// `SessionState::expectations` and `SessionState::rules` are flat
247/// `HashMap`s keyed by opaque ids — neither tracks which tab a handle
248/// was originally bound to. A `browser_tab_close` here therefore does
249/// **not** reap expectations registered via `browser_expect_register`
250/// or rules registered via `browser_intercept_add_rule` against the
251/// closing tab; they continue running (the expectation's spawned task
252/// holds its `.matched()` future open until its inner
253/// `pre_await_timeout_ms` fires, the interception actor's
254/// `Fetch.disable` only dispatches when its handle drops) until
255/// either: (a) the agent explicitly calls `browser_expect_cancel` /
256/// `browser_intercept_remove_rule`, or (b) the whole browser is torn
257/// down via `browser_close` (which DOES drain both registries —
258/// see `crate::tools::lifecycle::close`).
259///
260/// Fixing this requires extending each handle struct with a `tab_id`
261/// field plus a per-tab filter pass on close — out of scope for v0
262/// (would also need to handle the "tab id mid-rotation" case where the
263/// CDP target id changes under us). Tracked for a follow-up release.
264pub async fn close(
265    state: Arc<Mutex<SessionState>>,
266    input: TabCloseInput,
267) -> Result<TabCloseOutput, ErrorData> {
268    let mut s = state.lock().await;
269    let b = s
270        .browser
271        .as_ref()
272        .ok_or_else(|| map_error(McpServerError::BrowserNotOpen))?;
273    let target_id = match input.tab_id {
274        Some(id) => id,
275        None => s
276            .current_tab_id
277            .clone()
278            .ok_or_else(|| map_error(McpServerError::NoCurrentTab))?,
279    };
280    let tabs = b.tabs().await;
281    let tab = tabs
282        .iter()
283        .find(|t| t.target_id() == target_id)
284        .cloned()
285        .ok_or_else(|| {
286            map_error(McpServerError::from(
287                zendriver::ZendriverError::TabNotFound(target_id.clone()),
288            ))
289        })?;
290    tab.close()
291        .await
292        .map_err(|e| map_error(McpServerError::from(e)))?;
293
294    // Reset / promote the current tab id.
295    let was_current = s.current_tab_id.as_deref() == Some(target_id.as_str());
296    if was_current {
297        // Re-read the tab list now that the close has landed; the
298        // browser's internal registrar may take a beat to remove the
299        // closed entry, so we filter explicitly by `target_id`.
300        let b = s.browser.as_ref().expect("browser still attached");
301        let remaining = b.tabs().await;
302        s.current_tab_id = remaining
303            .iter()
304            .find(|t| t.target_id() != target_id)
305            .map(|t| t.target_id().to_string());
306    }
307    Ok(TabCloseOutput {
308        closed_id: target_id,
309        current_tab_id: s.current_tab_id.clone(),
310    })
311}

330/// Bring `tab_id` to the foreground in Chrome.
331///
332/// Sends `Target.activateTarget` via [`zendriver::Tab::activate`]. Also
333/// updates `state.current_tab_id` so the session's "current tab" tracks
334/// the foreground tab — otherwise a caller could end up with a current
335/// tab that's hidden behind the foregrounded one, which is rarely what
336/// they want.
337pub async fn activate(
338    state: Arc<Mutex<SessionState>>,
339    input: TabActivateInput,
340) -> Result<TabActivateOutput, ErrorData> {
341    let mut s = state.lock().await;
342    let b = s
343        .browser
344        .as_ref()
345        .ok_or_else(|| map_error(McpServerError::BrowserNotOpen))?;
346    let tabs = b.tabs().await;
347    let tab = tabs
348        .iter()
349        .find(|t| t.target_id() == input.tab_id)
350        .cloned()
351        .ok_or_else(|| {
352            map_error(McpServerError::from(
353                zendriver::ZendriverError::TabNotFound(input.tab_id.clone()),
354            ))
355        })?;
356    tab.activate()
357        .await
358        .map_err(|e| map_error(McpServerError::from(e)))?;
359    s.current_tab_id = Some(input.tab_id.clone());
360    Ok(TabActivateOutput { id: input.tab_id })
361}
