1//! Element discovery handlers — `browser_find`, `browser_find_all`.
2//!
3//! Also hosts the [`Selector`][crate::selectors::Selector] →
4//! [`zendriver::FindBuilder`] bridge ([`resolve`] / [`resolve_all`]) consumed
5//! by every tool that takes a Selector arg block. The bridge lives here
6//! (rather than on `Selector` itself) because it depends on zendriver types
7//! that `selectors.rs` deliberately doesn't pull in — keeping the wire
8//! struct decoupled from CDP-shaped concerns.
9//!
10//! ## Frame routing
11//!
12//! When the caller sets `selector.frame_id`, we look up the matching
13//! [`zendriver::Frame`] from `tab.frames()` and call `.in_frame(&frame)`
14//! on the builder. Missing frame id maps to
15//! [`zendriver::ZendriverError::FrameNotFound`].
16//!
17//! ## ARIA role mapping
18//!
19//! The wire `role: String` is mapped to [`zendriver::AriaRole`] via a
20//! match against the same shortlist the lib enumerates. Unknown role
21//! strings yield [`rmcp::ErrorData::invalid_params`] listing the accepted
22//! variants — surfacing the typo to the agent rather than silently falling
23//! back to `AriaRole::Other` (which would compile to a literal
24//! `[role="<typo>"]` selector that matches nothing).
25

44/// Resolve a single element by [`Selector`].
45///
46/// Validates the selector, configures a [`FindBuilder`] with the chosen
47/// selector kind + modifiers (`nth`, `visible_only`, `timeout`,
48/// `in_frame`), then awaits `.one()`. Surfaces
49/// [`zendriver::ZendriverError::ElementNotFound`] to the caller — tools that want to
50/// swallow it (e.g. `browser_find` returning `found: false`) must do so
51/// themselves; the helper does NOT call `one_or_none` for them.
52pub async fn resolve(tab: &Tab, sel: &Selector) -> Result<Element, ErrorData> {
53    sel.validate()
54        .map_err(|e| ErrorData::invalid_params(e.to_string(), None))?;
55    // Frame lookup must happen BEFORE the builder is constructed so the
56    // borrow checker is happy — `in_frame(&frame)` takes a reference, and
57    // we need a stable owned `Frame` to borrow from.
58    let frame_handle = if let Some(fid) = sel.frame_id.as_deref() {
59        Some(lookup_frame(tab, fid).await?)
60    } else {
61        None
62    };
63    let builder = tab.find();
64    let builder = apply_selector(builder, sel)?;
65    let builder = apply_modifiers(builder, sel);
66    let builder = if let Some(ref f) = frame_handle {
67        builder.in_frame(f)
68    } else {
69        builder
70    };
71    builder
72        .one()
73        .await
74        .map_err(|e| map_error(McpServerError::from(e)))
75}

77/// Resolve up to `limit` elements by [`Selector`].
78///
79/// Same bridge as [`resolve`], but terminates with `.many_or_empty()` so
80/// "no matches" returns `Vec::new()` instead of an error. The caller's
81/// `limit` is applied after the lib returns its full match list — we don't
82/// have a per-call cap to push into the builder.
83pub async fn resolve_all(
84    tab: &Tab,
85    sel: &Selector,
86    limit: usize,
87) -> Result<Vec<Element>, ErrorData> {
88    sel.validate()
89        .map_err(|e| ErrorData::invalid_params(e.to_string(), None))?;
90    let frame_handle = if let Some(fid) = sel.frame_id.as_deref() {
91        Some(lookup_frame(tab, fid).await?)
92    } else {
93        None
94    };
95    let builder = tab.find_all();
96    let builder = apply_selector_all(builder, sel)?;
97    let builder = apply_modifiers_all(builder, sel);
98    let builder = if let Some(ref f) = frame_handle {
99        builder.in_frame(f)
100    } else {
101        builder
102    };
103    let mut all = builder
104        .many_or_empty()
105        .await
106        .map_err(|e| map_error(McpServerError::from(e)))?;
107    if all.len() > limit {
108        all.truncate(limit);
109    }
110    Ok(all)
111}

113/// Apply a [`Selector`]'s chosen kind onto a query builder, returning the
114/// configured builder (or an `invalid_params` `ErrorData`).
115///
116/// Generated once per builder type ([`FindBuilder`] for single-match,
117/// [`FindAllBuilder`] for multi-match) because the lib exposes the two as
118/// distinct concrete types with no shared trait — but they expose the
119/// *identical* selector/predicate method surface, so a single macro body
120/// covers both and a new selector kind only has to be added here once.
121///
122/// Two modes (mutually exclusive — enforced by [`Selector::validate`]):
123/// - **single-selector** (`css`/`xpath`/`text`/`text_exact`/`text_regex`/`role`)
124///   — exactly one is applied.
125/// - **predicate** (`tag` and/or `attrs`, plus optional `text*` post-filters,
126///   all AND-ed) — applies `.tag()`, the `.attr*()` / `.has_attr()` /
127///   `.attr_regex()` family, and `.containing_text()` / `.text_equals()` /
128///   `.text_matches()`.
129macro_rules! apply_selector_to {
130    ($builder:expr, $sel:expr) => {{
131        let mut builder = $builder;
132        let sel = $sel;
133        // Predicate mode: tag/attrs present; text* are optional AND-ed
134        // post-filters. Single-selector kinds (css/xpath/role) are absent
135        // here (validate() enforces the mutual exclusion).
136        if sel.tag.is_some() || !sel.attrs.is_empty() {
137            if let Some(tag) = sel.tag.as_deref() {
138                builder = builder.tag(tag);
139            }
140            for ap in &sel.attrs {
141                let name = ap.name.as_str();
142                let value = ap.value.as_deref().unwrap_or(""); // Has has no value
143                builder = match ap.op {
144                    AttrOp::Eq => builder.attr(name, value),
145                    AttrOp::Contains => builder.attr_contains(name, value),
146                    AttrOp::StartsWith => builder.attr_starts_with(name, value),
147                    AttrOp::EndsWith => builder.attr_ends_with(name, value),
148                    AttrOp::Has => builder.has_attr(name),
149                    AttrOp::Regex => builder.attr_regex(name, value),
150                };
151            }
152            if let Some(t) = sel.text.as_deref() {
153                builder = builder.containing_text(t);
154            }
155            if let Some(t) = sel.text_exact.as_deref() {
156                builder = builder.text_equals(t);
157            }
158            if let Some(pat) = sel.text_regex.as_deref() {
159                builder = builder.text_matches(pat);
160            }
161            Ok(builder)
162        } else if let Some(css) = sel.css.as_deref() {
163            Ok(builder.css(css))
164        } else if let Some(xp) = sel.xpath.as_deref() {
165            Ok(builder.xpath(xp))
166        } else if let Some(t) = sel.text.as_deref() {
167            Ok(builder.text(t))
168        } else if let Some(t) = sel.text_exact.as_deref() {
169            Ok(builder.text_exact(t))
170        } else if let Some(pat) = sel.text_regex.as_deref() {
171            let re = compile_regex(pat)?;
172            Ok(builder.text_regex(re))
173        } else if let Some(role_str) = sel.role.as_deref() {
174            let role = parse_role(role_str)?;
175            Ok(match sel.role_name.as_deref() {
176                Some(name) => builder.role_named(role, name),
177                None => builder.role(role),
178            })
179        } else {
180            // `validate()` already enforces exactly-one-of, so this branch is
181            // unreachable in practice — return invalid_params just in case.
182            Err(ErrorData::invalid_params(
183                "Selector has no selector kind set (validate() should have caught this)"
184                    .to_string(),
185                None,
186            ))
187        }
188    }};
189}

191/// Apply the Selector's chosen kind to a single-match [`FindBuilder`].
192fn apply_selector<'a>(
193    builder: FindBuilder<'a>,
194    sel: &Selector,
195) -> Result<FindBuilder<'a>, ErrorData> {
196    apply_selector_to!(builder, sel)
197}
198
199/// Apply the Selector's chosen kind to a [`FindAllBuilder`].
200fn apply_selector_all<'a>(
201    builder: FindAllBuilder<'a>,
202    sel: &Selector,
203) -> Result<FindAllBuilder<'a>, ErrorData> {
204    apply_selector_to!(builder, sel)
205}

207fn apply_modifiers<'a>(mut builder: FindBuilder<'a>, sel: &Selector) -> FindBuilder<'a> {
208    if let Some(n) = sel.nth {
209        builder = builder.nth(n);
210    }
211    builder = builder.visible_only(sel.visible_only);
212    builder = builder.timeout(Duration::from_millis(sel.timeout_ms));
213    builder
214}
215
216fn apply_modifiers_all<'a>(mut builder: FindAllBuilder<'a>, sel: &Selector) -> FindAllBuilder<'a> {
217    builder = builder.visible_only(sel.visible_only);
218    builder = builder.timeout(Duration::from_millis(sel.timeout_ms));
219    builder
220}

222fn compile_regex(pat: &str) -> Result<regex::Regex, ErrorData> {
223    regex::Regex::new(pat)
224        .map_err(|e| ErrorData::invalid_params(format!("Invalid `text_regex` pattern: {e}"), None))
225}

227/// Map the wire `role` string onto a [`zendriver::AriaRole`].
228///
229/// We accept lowercase ARIA role names. Unknown values surface an
230/// `invalid_params` error listing the accepted roles, so an agent that
231/// typos `"buttn"` gets a clean signal rather than `AriaRole::Other`
232/// silently compiling to a `[role="buttn"]` selector that matches nothing.
233fn parse_role(s: &str) -> Result<AriaRole, ErrorData> {
234    match s {
235        "button" => Ok(AriaRole::Button),
236        "link" => Ok(AriaRole::Link),
237        "textbox" => Ok(AriaRole::Textbox),
238        "combobox" => Ok(AriaRole::Combobox),
239        "checkbox" => Ok(AriaRole::Checkbox),
240        "radio" => Ok(AriaRole::Radio),
241        "tab" => Ok(AriaRole::Tab),
242        "menu" => Ok(AriaRole::Menu),
243        "menuitem" => Ok(AriaRole::Menuitem),
244        "dialog" => Ok(AriaRole::Dialog),
245        "heading" => Ok(AriaRole::Heading),
246        "banner" => Ok(AriaRole::Banner),
247        "navigation" => Ok(AriaRole::Navigation),
248        "main" => Ok(AriaRole::Main),
249        "article" => Ok(AriaRole::Article),
250        "list" => Ok(AriaRole::List),
251        "listitem" => Ok(AriaRole::Listitem),
252        "row" => Ok(AriaRole::Row),
253        "cell" => Ok(AriaRole::Cell),
254        "columnheader" => Ok(AriaRole::Columnheader),
255        "rowheader" => Ok(AriaRole::Rowheader),
256        other => Err(ErrorData::invalid_params(
257            format!(
258                "Unknown ARIA role `{other}`. Accepted: button, link, textbox, combobox, checkbox, radio, tab, menu, menuitem, dialog, heading, banner, navigation, main, article, list, listitem, row, cell, columnheader, rowheader."
259            ),
260            None,
261        )),
262    }
263}

291/// Element projection returned by `browser_find` (single) and
292/// `browser_find_all` (list).
293///
294/// `tag` is best-effort: the lib has no `Element::tag_name()` so we ask
295/// the page via `el.tagName.toLowerCase()`. If the eval call fails (e.g.
296/// the element became stale between the find and the describe), we leave
297/// `tag` as `None` rather than failing the entire describe — the rest of
298/// the projection (attrs, text, geometry) is still useful.
299#[derive(Debug, Clone, Serialize, JsonSchema)]
300pub struct ElementDescriptor {
301    /// Lowercase tag name, or `None` when the eval probe failed.
302    #[serde(skip_serializing_if = "Option::is_none")]
303    pub tag: Option<String>,
304    /// First 200 chars of `innerText` (post `chars().take(200)` so we
305    /// never split a multibyte codepoint).
306    pub text_snippet: String,
307    /// All attributes on the element, in stable key order.
308    pub attrs: BTreeMap<String, String>,
309    /// Viewport-relative bounding box, or `None` when the element has no
310    /// box (display: none, detached, etc.).
311    #[serde(skip_serializing_if = "Option::is_none")]
312    pub bounding_box: Option<BoundingBox>,
313    /// `true` when the lib's visibility predicate accepts the element.
314    pub visible: bool,
315    /// `true` when the element is not disabled (form-aware).
316    pub enabled: bool,
317}
318
319/// Build a descriptor from a live `Element`.
320///
321/// Attribute / visibility / enabled probes propagate their errors (those
322/// failures usually mean the element went stale and the caller should
323/// re-find). The tag probe is intentionally swallowed to `None` — see
324/// the [`ElementDescriptor::tag`] doc for the rationale.
325pub async fn describe(el: &Element) -> Result<ElementDescriptor, ErrorData> {
326    // Best-effort tag probe — falls back to None on any failure.
327    let tag: Option<String> = el.evaluate::<String>("el.tagName.toLowerCase()").await.ok();
328
329    // Text snippet: cap at 200 chars (not bytes) so we never split a
330    // multibyte codepoint, which would yield invalid UTF-8 on output.
331    let text_snippet: String = el
332        .inner_text()
333        .await
334        .unwrap_or_default()
335        .chars()
336        .take(200)
337        .collect();
338
339    let attrs_map = el
340        .attrs()
341        .await
342        .map_err(|e| map_error(McpServerError::from(e)))?;
343    let attrs: BTreeMap<String, String> = attrs_map.into_iter().collect();
344
345    let visible = el
346        .is_visible()
347        .await
348        .map_err(|e| map_error(McpServerError::from(e)))?;
349    let enabled = el
350        .is_enabled()
351        .await
352        .map_err(|e| map_error(McpServerError::from(e)))?;
353    let bounding_box = el
354        .bounding_box()
355        .await
356        .ok()
357        .flatten()
358        .map(BoundingBox::from);
359
360    Ok(ElementDescriptor {
361        tag,
362        text_snippet,
363        attrs,
364        bounding_box,
365        visible,
366        enabled,
367    })
368}

372/// Input for `browser_find`.
373#[derive(Debug, Deserialize, JsonSchema)]
374#[serde(deny_unknown_fields)]

380/// Output of `browser_find`.
381#[derive(Debug, Serialize, JsonSchema)]
382pub struct FindOutput {
383    /// `true` iff the selector matched within the timeout.
384    pub found: bool,
385    /// Element projection when `found` is `true`, otherwise `None`.
386    #[serde(skip_serializing_if = "Option::is_none")]
387    pub element: Option<ElementDescriptor>,
388}
389
390/// Resolve a Selector to a single element and return its descriptor.
391///
392/// Distinct from `browser_element_state` in that "not found" is reported
393/// as `{ found: false, element: null }` rather than an error — easier for
394/// agents that want to branch on existence without try/catch.
395pub async fn find(
396    state: Arc<Mutex<SessionState>>,
397    input: FindInput,
398) -> Result<FindOutput, ErrorData> {
399    let s = state.lock().await;
400    let tab = current_tab(&s).await?;
401    match resolve(&tab, &input.selector).await {
402        Ok(el) => {
403            let desc = describe(&el).await?;
404            Ok(FindOutput {
405                found: true,
406                element: Some(desc),
407            })
408        }
409        Err(err) if is_not_found(&err) => Ok(FindOutput {
410            found: false,
411            element: None,
412        }),
413        Err(err) => Err(err),
414    }
415}
416
417/// `true` when the `ErrorData` was produced by mapping an
418/// [`zendriver::ZendriverError::ElementNotFound`]. We rely on the
419/// `_meta.suggested_next == "browser_html"` marker that `map_error`
420/// attaches to that variant — checking the message would be brittle
421/// across translation tweaks.
422fn is_not_found(err: &ErrorData) -> bool {
423    err.data
424        .as_ref()
425        .and_then(|v| v.get("suggested_next"))
426        .and_then(|v| v.as_str())
427        == Some("browser_html")
428        && err.message.contains("No element matched")
429}

433/// Input for `browser_find_all`.
434#[derive(Debug, Deserialize, JsonSchema)]
435#[serde(deny_unknown_fields)]
436pub struct FindAllInput {
437    #[serde(flatten)]
438    pub selector: Selector,
439    /// Maximum number of matches to return. Default 50.
440    #[serde(default = "default_limit")]
441    pub limit: usize,
442}
443
444const fn default_limit() -> usize {
445    50
446}
447
448/// Output of `browser_find_all`.
449#[derive(Debug, Serialize, JsonSchema)]
450pub struct FindAllOutput {
451    /// All matched elements (capped at `limit`).
452    pub elements: Vec<ElementDescriptor>,
453}
454
455/// Resolve a Selector to ALL matches (up to `limit`).
456///
457/// "No matches" returns `elements: []` rather than an error — keeps the
458/// caller's branching uniform.
459pub async fn find_all(
460    state: Arc<Mutex<SessionState>>,
461    input: FindAllInput,
462) -> Result<FindAllOutput, ErrorData> {
463    let s = state.lock().await;
464    let tab = current_tab(&s).await?;
465    let els = resolve_all(&tab, &input.selector, input.limit).await?;
466    let mut out = Vec::with_capacity(els.len());
467    for el in &els {
468        out.push(describe(el).await?);
469    }
470    Ok(FindAllOutput { elements: out })
471}
