1//! Interception tools — `browser_intercept_add_rule / _remove_rule /
2//! _list_rules / _clear_rules`. Gated behind the `interception` feature.
3//!
4//! ## One MCP rule = one `InterceptHandle`
5//!
6//! The underlying `zendriver-interception` builder supports chaining many
7//! actions onto a single handle, but that shape doesn't map cleanly to a
8//! per-call MCP tool surface (each `add_rule` call must produce its own
9//! removable id). v0 simplification: every `browser_intercept_add_rule`
10//! invocation spawns a fresh `tab.intercept().pattern(pat).<action>.start()`
11//! chain and keeps the returned [`InterceptHandle`] in
12//! [`SessionState::rules`]. Removing the entry drops the handle, which the
13//! interception actor treats as a cancellation signal — so
14//! `browser_intercept_remove_rule` is the user-visible inverse of
15//! `_add_rule`.
16//!
17//! ## Action surface (`InterceptAction`)
18//!
19//! Four tagged variants mirroring the four `InterceptBuilder` rule kinds:
20//! `block`, `redirect`, `respond`, `modify_request`. `respond` carries a
21//! UTF-8 `body` plus an optional `content_type` (collapsed into the headers
22//! map when set) — binary bodies are out of scope for v0; agents that need
23//! them can fall back to `browser_intercept_add_rule` with a `redirect` to
24//! a server they control.
25//!
26//! `modify_request` accepts a headers map of replacements; CDP's
27//! `continueRequest.headers` is a *full replacement*, so the closure
28//! returns a [`RequestOverrides`] that preserves the original headers and
29//! overlays the caller-supplied entries. Other override fields (URL,
30//! method, post-data) are reserved for a follow-up.
31//!
32//! [`InterceptHandle`]: zendriver::InterceptHandle
33//! [`RequestOverrides`]: zendriver::RequestOverrides
34

52/// Wire-level interception action. One variant per supported rule kind.
53///
54/// Tagged with `kind` so a JSON payload looks like
55/// `{ "kind": "block" }` or `{ "kind": "redirect", "to": "..." }`, which
56/// is easier for an agent to construct than a flat field set with mutually
57/// exclusive shapes.
58#[derive(Debug, Deserialize, JsonSchema)]
59#[serde(deny_unknown_fields, tag = "kind", rename_all = "snake_case")]
60pub enum InterceptAction {
61    /// Fail every matching request with `BlockedByClient`.
62    Block,
63    /// Redirect every match to `to` (verbatim URL replacement).
64    Redirect {
65        /// Target URL. Sent to the actor as a string — no expansion of the
66        /// matched URL is performed; if you need per-request substitution
67        /// use `modify_request` or the manual stream API.
68        to: String,
69    },
70    /// Synthesize a response with the given status / body / headers.
71    Respond {
72        /// HTTP status code (e.g. `200`, `404`).
73        status: u16,
74        /// Response body as UTF-8 text.
75        body: String,
76        /// Convenience for the most common header: when set, a
77        /// `Content-Type: <content_type>` header is added (and overrides
78        /// any same-named entry in `headers`).
79        #[serde(default)]
80        content_type: Option<String>,
81        /// Additional response headers. Lexicographically sorted (`BTreeMap`)
82        /// for stable wire output.
83        #[serde(default)]
84        headers: BTreeMap<String, String>,
85    },
86    /// Continue the request with extra / replaced headers. Other fields
87    /// (URL, method, post-data) are left at Chrome's originals.
88    ModifyRequest {
89        /// Headers to merge over the request's originals. CDP's
90        /// `continueRequest.headers` is a full replacement, so the closure
91        /// rebuilds the header list from the intercepted request and
92        /// overlays these entries (case-insensitive on names).
93        #[serde(default)]
94        headers: BTreeMap<String, String>,
95    },
96    /// Continue the *response* with an overridden status and/or headers
97    /// (pauses at the response stage). Other fields are left at Chrome's
98    /// originals. The body is not rewritten — use `respond` to synthesize a
99    /// whole response instead.
100    ModifyResponse {
101        /// Override the HTTP status code. Omit to keep Chrome's original.
102        #[serde(default)]
103        status: Option<u16>,
104        /// Headers to merge over the response's originals (full-replacement
105        /// semantics, same as `modify_request`; case-insensitive names).
106        #[serde(default)]
107        headers: BTreeMap<String, String>,
108    },
109}

132/// Register one interception rule. Spawns a fresh
133/// `tab.intercept().pattern(...).<action>.start()` chain on the current
134/// tab and stashes the resulting handle.
135pub async fn add_rule(
136    state: Arc<Mutex<SessionState>>,
137    input: AddRuleInput,
138) -> Result<AddRuleOutput, ErrorData> {
139    let mut s = state.lock().await;
140    let tab = current_tab(&s).await?;
141    // Captured before the per-action match below so the registered handle
142    // can be reaped by `browser_tab_close` when this tab closes — see
143    // `state::InterceptRuleHandle::tab_id`.
144    let tab_id = tab.target_id().to_string();
145
146    // Each match-arm builds + starts its own InterceptBuilder. The action
147    // method (`block` / `redirect` / `respond` / `modify_request`) takes the
148    // URL pattern as its first arg and registers it as the rule's matcher;
149    // `InterceptBuilder::start` auto-injects a match-all `"*"` CDP
150    // `Fetch.RequestPattern` when none is added via `.pattern()`, so we
151    // don't need to add one ourselves. (Conflating the two would just
152    // register the same string twice as conceptually different things —
153    // the CDP server-side filter and the rule's per-request matcher.)
154    let (handle, action_kind): (zendriver::InterceptHandle, &'static str) = match input.action {
155        InterceptAction::Block => {
156            let h = tab
157                .intercept()
158                .block(input.pattern.clone())
159                .map_err(zendriver_err)?
160                .start();
161            (h, "block")
162        }
163        InterceptAction::Redirect { to } => {
164            let h = tab
165                .intercept()
166                .redirect(input.pattern.clone(), to)
167                .map_err(zendriver_err)?
168                .start();
169            (h, "redirect")
170        }
171        InterceptAction::Respond {
172            status,
173            body,
174            content_type,
175            mut headers,
176        } => {
177            if let Some(ct) = content_type {
178                // `Content-Type` is the common case; honor the convenience
179                // field by overlaying it into the explicit map.
180                headers.insert("content-type".into(), ct);
181            }
182            let header_vec: Vec<(String, String)> = headers.into_iter().collect();
183            let h = tab
184                .intercept()
185                .respond(input.pattern.clone(), status, header_vec, body.into_bytes())
186                .map_err(zendriver_err)?
187                .start();
188            (h, "respond")
189        }
190        InterceptAction::ModifyRequest { headers } => {
191            // Snapshot the overlay into an `Arc` so the per-request closure
192            // can be `Fn + Send + Sync + 'static` (the builder requires it
193            // — see InterceptBuilder::modify_request).
194            let overlay = Arc::new(headers);
195            let h = tab
196                .intercept()
197                .modify_request(input.pattern.clone(), move |req| {
198                    merge_headers(&req.headers, &overlay)
199                })
200                .map_err(zendriver_err)?
201                .start();
202            (h, "modify_request")
203        }
204        InterceptAction::ModifyResponse { status, headers } => {
205            // Same `Arc`-capture pattern as `modify_request`; the closure runs
206            // on the actor task per matching response.
207            let overlay = Arc::new(headers);
208            let h = tab
209                .intercept()
210                .modify_response(input.pattern.clone(), move |resp: &ResponseInfo| {
211                    let headers = if overlay.is_empty() {
212                        None
213                    } else {
214                        Some(merge_header_list(&resp.headers, &overlay))
215                    };
216                    ResponseOverrides {
217                        status,
218                        headers,
219                        ..ResponseOverrides::default()
220                    }
221                })
222                .map_err(zendriver_err)?
223                .start();
224            (h, "modify_response")
225        }
226    };
227
228    let id: RuleId = uuid::Uuid::new_v4().to_string();
229    s.rules.insert(
230        id.clone(),
231        InterceptRuleHandle {
232            pattern: input.pattern,
233            action_kind,
234            _handle: handle,
235            tab_id,
236        },
237    );
238    Ok(AddRuleOutput { rule_id: id })
239}

241/// Build the [`RequestOverrides`] payload for a `modify_request` rule.
242///
243/// CDP's `Fetch.continueRequest.headers` is a *full replacement* — there is
244/// no merge mode — so we rebuild the header list from the request as Chrome
245/// reported it and overlay the caller's entries by case-insensitive name
246/// match. Case-insensitive: HTTP header names are case-insensitive per RFC
247/// 7230, and Chrome's emission casing isn't stable across versions.
248fn merge_headers(
249    original: &[(String, String)],
250    overlay: &BTreeMap<String, String>,
251) -> RequestOverrides {
252    RequestOverrides {
253        headers: Some(merge_header_list(original, overlay)),
254        ..RequestOverrides::default()
255    }
256}
257
258/// Rebuild a full header list from `original`, dropping entries the `overlay`
259/// replaces (case-insensitive) and appending the overlay entries. Shared by
260/// `modify_request` (→ [`RequestOverrides`]) and `modify_response`
261/// (→ [`ResponseOverrides`]) because CDP's continue-* header fields are both
262/// full replacements with no merge mode.
263fn merge_header_list(
264    original: &[(String, String)],
265    overlay: &BTreeMap<String, String>,
266) -> Vec<(String, String)> {
267    let mut out: Vec<(String, String)> = Vec::with_capacity(original.len() + overlay.len());
268    let overlay_keys_lower: std::collections::HashSet<String> =
269        overlay.keys().map(|k| k.to_ascii_lowercase()).collect();
270    for (k, v) in original {
271        if !overlay_keys_lower.contains(&k.to_ascii_lowercase()) {
272            out.push((k.clone(), v.clone()));
273        }
274    }
275    for (k, v) in overlay {
276        out.push((k.clone(), v.clone()));
277    }
278    out
279}

281/// Wrap a `zendriver_interception::InterceptionError` (re-exported as
282/// `zendriver::InterceptionError`) into the MCP error wire format.
283fn zendriver_err(e: zendriver::InterceptionError) -> ErrorData {
284    map_error(McpServerError::from(ZendriverError::from(e)))
285}

291#[serde(deny_unknown_fields)]

293    /// Id returned by an earlier `browser_intercept_add_rule` call.
294    pub rule_id: RuleId,
295}
296
297/// Output of `browser_intercept_remove_rule`.
298#[derive(Debug, Serialize, JsonSchema)]
299pub struct RemoveRuleOutput {
300    /// Always `true` on success. Removal of an unknown id returns
301    /// [`McpServerError::RuleNotFound`] rather than a `false` here, so an
302    /// agent can branch on success without re-checking the body.
303    pub removed: bool,
304}
305
306/// Drop a previously-registered rule, tearing down its interception actor.
307pub async fn remove_rule(
308    state: Arc<Mutex<SessionState>>,
309    input: RemoveRuleInput,
310) -> Result<RemoveRuleOutput, ErrorData> {
311    let mut s = state.lock().await;
312    // `.remove(...)` returns the value (and thus its `_handle: InterceptHandle`),
313    // dropping it here cancels the actor. We don't need to await `stop()` —
314    // `Drop` fires-and-forgets the `Fetch.disable`.
315    s.rules
316        .remove(&input.rule_id)
317        .ok_or_else(|| map_error(McpServerError::RuleNotFound(input.rule_id)))?;
318    Ok(RemoveRuleOutput { removed: true })
319}

323/// Output of `browser_intercept_list_rules`.
324#[derive(Debug, Serialize, JsonSchema)]
325pub struct ListRulesOutput {
326    /// One entry per live rule. Sorted by `rule_id` so the output is stable
327    /// across runs (rather than HashMap-iteration-order chaos).
328    pub rules: Vec<RuleSummary>,
329}
330
331/// Description of a single live interception rule.
332#[derive(Debug, Serialize, JsonSchema)]
333pub struct RuleSummary {
334    /// Same id `browser_intercept_add_rule` returned.
335    pub rule_id: RuleId,
336    /// URL pattern the rule was registered with.
337    pub pattern: String,
338    /// `"block"`, `"redirect"`, `"respond"`, or `"modify_request"`.
339    pub action_kind: String,
340}
341
342/// Enumerate every live rule. Empty list when none are registered (never
343/// an error — a session with no rules is a valid steady state).
344pub async fn list_rules(
345    state: Arc<Mutex<SessionState>>,
346    _: crate::tools::common::EmptyInput,
347) -> Result<ListRulesOutput, ErrorData> {
348    let s = state.lock().await;
349    let mut rules: Vec<RuleSummary> = s
350        .rules
351        .iter()
352        .map(|(id, h)| RuleSummary {
353            rule_id: id.clone(),
354            pattern: h.pattern.clone(),
355            action_kind: h.action_kind.to_string(),
356        })
357        .collect();
358    rules.sort_by(|a, b| a.rule_id.cmp(&b.rule_id));
359    Ok(ListRulesOutput { rules })
360}

364/// Output of `browser_intercept_clear_rules`.
365#[derive(Debug, Serialize, JsonSchema)]
366pub struct ClearRulesOutput {
367    /// Number of rules that were dropped. `0` is a successful no-op
368    /// rather than an error — calling clear on an empty registry is the
369    /// agent's idempotent "make sure nothing is intercepting" lever.
370    pub cleared: usize,
371}
372
373/// Drop every live rule. Each handle's `Drop` cancels its actor; we don't
374/// wait for any of them — same semantics as `remove_rule`.
375pub async fn clear_rules(
376    state: Arc<Mutex<SessionState>>,
377    _: crate::tools::common::EmptyInput,
378) -> Result<ClearRulesOutput, ErrorData> {
379    let mut s = state.lock().await;
380    let cleared = s.rules.len();
381    s.rules.clear();
382    Ok(ClearRulesOutput { cleared })
383}

449    #[test]
450    fn merge_headers_overlays_case_insensitively_and_drops_originals() {
451        let original = vec![
452            ("Host".to_string(), "example.com".to_string()),
453            ("User-Agent".to_string(), "old".to_string()),
454            ("Accept".to_string(), "*/*".to_string()),
455        ];
456        let mut overlay = BTreeMap::new();
457        overlay.insert("user-agent".to_string(), "new".to_string());
458        overlay.insert("X-Marker".to_string(), "yes".to_string());
459
460        let ov = merge_headers(&original, &overlay);
461        let headers = ov.headers.expect("headers populated");
462
463        // Original "User-Agent" gone (case-insens), replaced by overlay's
464        // "user-agent: new". Host + Accept survive. Overlay's X-Marker is
465        // appended.
466        let names_lower: Vec<String> = headers
467            .iter()
468            .map(|(k, _)| k.to_ascii_lowercase())
469            .collect();
470        assert!(names_lower.contains(&"host".into()));
471        assert!(names_lower.contains(&"accept".into()));
472        assert!(names_lower.contains(&"user-agent".into()));
473        assert!(names_lower.contains(&"x-marker".into()));
474        // Exactly one "user-agent" (no dupes).
475        assert_eq!(names_lower.iter().filter(|n| *n == "user-agent").count(), 1);
476        // Replaced value is the overlay's.
477        let ua = headers
478            .iter()
479            .find(|(k, _)| k.eq_ignore_ascii_case("user-agent"))
480            .map(|(_, v)| v.as_str())
481            .expect("user-agent present");
482        assert_eq!(ua, "new");
483    }
