1//! Interception tools — `browser_intercept_add_rule / _remove_rule /
2//! _list_rules / _clear_rules`. Gated behind the `interception` feature.
3//!
4//! ## One MCP rule = one `InterceptHandle`
5//!
6//! The underlying `zendriver-interception` builder supports chaining many
7//! actions onto a single handle, but that shape doesn't map cleanly to a
8//! per-call MCP tool surface (each `add_rule` call must produce its own
9//! removable id). v0 simplification: every `browser_intercept_add_rule`
10//! invocation spawns a fresh `tab.intercept().pattern(pat).<action>.start()`
11//! chain and keeps the returned [`InterceptHandle`] in
12//! [`SessionState::rules`]. Removing the entry drops the handle, which the
13//! interception actor treats as a cancellation signal — so
14//! `browser_intercept_remove_rule` is the user-visible inverse of
15//! `_add_rule`.
16//!
17//! ## Action surface (`InterceptAction`)
18//!
19//! Four tagged variants mirroring the four `InterceptBuilder` rule kinds:
20//! `block`, `redirect`, `respond`, `modify_request`. `respond` carries a
21//! UTF-8 `body` plus an optional `content_type` (collapsed into the headers
22//! map when set) — binary bodies are out of scope for v0; agents that need
23//! them can fall back to `browser_intercept_add_rule` with a `redirect` to
24//! a server they control.
25//!
26//! `modify_request` accepts a headers map of replacements; CDP's
27//! `continueRequest.headers` is a *full replacement*, so the closure
28//! returns a [`RequestOverrides`] that preserves the original headers and
29//! overlays the caller-supplied entries. It also accepts optional `url` /
30//! `method` / `post_data` overrides, passed straight through to
31//! [`RequestOverrides`] (the core actor already dispatches all four
32//! fields — this is DTO-only wiring).
33//!
34//! [`InterceptHandle`]: zendriver::InterceptHandle
35//! [`RequestOverrides`]: zendriver::RequestOverrides
36

54/// Wire-level interception action. One variant per supported rule kind.
55///
56/// Tagged with `kind` so a JSON payload looks like
57/// `{ "kind": "block" }` or `{ "kind": "redirect", "to": "..." }`, which
58/// is easier for an agent to construct than a flat field set with mutually
59/// exclusive shapes.
60#[derive(Debug, Deserialize, JsonSchema)]
61#[serde(deny_unknown_fields, tag = "kind", rename_all = "snake_case")]
62pub enum InterceptAction {
63    /// Fail every matching request with `BlockedByClient`.
64    Block,
65    /// Redirect every match to `to` (verbatim URL replacement).
66    Redirect {
67        /// Target URL. Sent to the actor as a string — no expansion of the
68        /// matched URL is performed; if you need per-request substitution
69        /// use `modify_request` or the manual stream API.
70        to: String,
71    },
72    /// Synthesize a response with the given status / body / headers.
73    Respond {
74        /// HTTP status code (e.g. `200`, `404`).
75        status: u16,
76        /// Response body as UTF-8 text.
77        body: String,
78        /// Convenience for the most common header: when set, a
79        /// `Content-Type: <content_type>` header is added (and overrides
80        /// any same-named entry in `headers`).
81        #[serde(default)]
82        content_type: Option<String>,
83        /// Additional response headers. Lexicographically sorted (`BTreeMap`)
84        /// for stable wire output.
85        #[serde(default)]
86        headers: BTreeMap<String, String>,
87    },
88    /// Continue the request with extra / replaced headers and/or a replaced
89    /// URL / method / body. Any field left `None` keeps Chrome's original.
90    ModifyRequest {
91        /// Headers to merge over the request's originals. CDP's
92        /// `continueRequest.headers` is a full replacement, so the closure
93        /// rebuilds the header list from the intercepted request and
94        /// overlays these entries (case-insensitive on names).
95        #[serde(default)]
96        headers: BTreeMap<String, String>,
97        /// Replace the request URL. Omit to keep the original.
98        #[serde(default)]
99        url: Option<String>,
100        /// Replace the HTTP method (e.g. `"POST"`). Omit to keep the
101        /// original.
102        #[serde(default)]
103        method: Option<String>,
104        /// Replace the request body, as UTF-8 text. Omit to keep the
105        /// original. The core [`RequestOverrides::post_data`] field is
106        /// `Vec<u8>`; this is encoded via `String::into_bytes`, so it can't
107        /// carry a non-UTF-8 body — use the manual stream API for that.
108        #[serde(default)]
109        post_data: Option<String>,
110    },
111    /// Continue the *response* with an overridden status and/or headers
112    /// (pauses at the response stage). Other fields are left at Chrome's
113    /// originals. The body is not rewritten — use `respond` to synthesize a
114    /// whole response instead.
115    ModifyResponse {
116        /// Override the HTTP status code. Omit to keep Chrome's original.
117        #[serde(default)]
118        status: Option<u16>,
119        /// Headers to merge over the response's originals (full-replacement
120        /// semantics, same as `modify_request`; case-insensitive names).
121        #[serde(default)]
122        headers: BTreeMap<String, String>,
123    },
124}

147/// Register one interception rule. Spawns a fresh
148/// `tab.intercept().pattern(...).<action>.start()` chain on the current
149/// tab and stashes the resulting handle.
150pub async fn add_rule(
151    state: Arc<Mutex<SessionState>>,
152    input: AddRuleInput,
153) -> Result<AddRuleOutput, ErrorData> {
154    let mut s = state.lock().await;
155    let tab = current_tab(&s).await?;
156    // Captured before the per-action match below so the registered handle
157    // can be reaped by `browser_tab_close` when this tab closes — see
158    // `state::InterceptRuleHandle::tab_id`.
159    let tab_id = tab.target_id().to_string();
160
161    // Each match-arm builds + starts its own InterceptBuilder. The action
162    // method (`block` / `redirect` / `respond` / `modify_request`) takes the
163    // URL pattern as its first arg and registers it as the rule's matcher;
164    // `InterceptBuilder::start` auto-injects a match-all `"*"` CDP
165    // `Fetch.RequestPattern` when none is added via `.pattern()`, so we
166    // don't need to add one ourselves. (Conflating the two would just
167    // register the same string twice as conceptually different things —
168    // the CDP server-side filter and the rule's per-request matcher.)
169    let (handle, action_kind): (zendriver::InterceptHandle, &'static str) = match input.action {
170        InterceptAction::Block => {
171            let h = tab
172                .intercept()
173                .block(input.pattern.clone())
174                .map_err(zendriver_err)?
175                .start();
176            (h, "block")
177        }
178        InterceptAction::Redirect { to } => {
179            let h = tab
180                .intercept()
181                .redirect(input.pattern.clone(), to)
182                .map_err(zendriver_err)?
183                .start();
184            (h, "redirect")
185        }
186        InterceptAction::Respond {
187            status,
188            body,
189            content_type,
190            mut headers,
191        } => {
192            if let Some(ct) = content_type {
193                // `Content-Type` is the common case; honor the convenience
194                // field by overlaying it into the explicit map.
195                headers.insert("content-type".into(), ct);
196            }
197            let header_vec: Vec<(String, String)> = headers.into_iter().collect();
198            let h = tab
199                .intercept()
200                .respond(input.pattern.clone(), status, header_vec, body.into_bytes())
201                .map_err(zendriver_err)?
202                .start();
203            (h, "respond")
204        }
205        InterceptAction::ModifyRequest {
206            headers,
207            url,
208            method,
209            post_data,
210        } => {
211            // Snapshot the overlay + scalar overrides into an `Arc` so the
212            // per-request closure can be `Fn + Send + Sync + 'static` (the
213            // builder requires it — see InterceptBuilder::modify_request).
214            let overlay = Arc::new(ModifyRequestOverlay {
215                headers,
216                url,
217                method,
218                post_data,
219            });
220            let h = tab
221                .intercept()
222                .modify_request(input.pattern.clone(), move |req| {
223                    overlay.build(&req.headers)
224                })
225                .map_err(zendriver_err)?
226                .start();
227            (h, "modify_request")
228        }
229        InterceptAction::ModifyResponse { status, headers } => {
230            // Same `Arc`-capture pattern as `modify_request`; the closure runs
231            // on the actor task per matching response.
232            let overlay = Arc::new(headers);
233            let h = tab
234                .intercept()
235                .modify_response(input.pattern.clone(), move |resp: &ResponseInfo| {
236                    let headers = if overlay.is_empty() {
237                        None
238                    } else {
239                        Some(merge_header_list(&resp.headers, &overlay))
240                    };
241                    ResponseOverrides {
242                        status,
243                        headers,
244                        ..ResponseOverrides::default()
245                    }
246                })
247                .map_err(zendriver_err)?
248                .start();
249            (h, "modify_response")
250        }
251    };
252
253    let id: RuleId = uuid::Uuid::new_v4().to_string();
254    s.rules.insert(
255        id.clone(),
256        InterceptRuleHandle {
257            pattern: input.pattern,
258            action_kind,
259            _handle: handle,
260            tab_id,
261        },
262    );
263    Ok(AddRuleOutput { rule_id: id })
264}

266/// Snapshot of a `modify_request` rule's overrides, captured once at
267/// `add_rule` time and shared (via `Arc`) across every matching request.
268struct ModifyRequestOverlay {
269    headers: BTreeMap<String, String>,
270    url: Option<String>,
271    method: Option<String>,
272    post_data: Option<String>,
273}
274
275impl ModifyRequestOverlay {
276    /// Build the [`RequestOverrides`] payload for a `modify_request` rule.
277    ///
278    /// CDP's `Fetch.continueRequest.headers` is a *full replacement* — there
279    /// is no merge mode — so we rebuild the header list from the request as
280    /// Chrome reported it and overlay the caller's entries by
281    /// case-insensitive name match. Case-insensitive: HTTP header names are
282    /// case-insensitive per RFC 7230, and Chrome's emission casing isn't
283    /// stable across versions. `url` / `method` / `post_data` are passed
284    /// straight through — `None` leaves Chrome's original untouched.
285    fn build(&self, original_headers: &[(String, String)]) -> RequestOverrides {
286        RequestOverrides {
287            url: self.url.clone(),
288            method: self.method.clone(),
289            headers: Some(merge_header_list(original_headers, &self.headers)),
290            post_data: self.post_data.clone().map(String::into_bytes),
291        }
292    }
293}
294
295/// Rebuild a full header list from `original`, dropping entries the `overlay`
296/// replaces (case-insensitive) and appending the overlay entries. Shared by
297/// `modify_request` (→ [`RequestOverrides`]) and `modify_response`
298/// (→ [`ResponseOverrides`]) because CDP's continue-* header fields are both
299/// full replacements with no merge mode.
300fn merge_header_list(
301    original: &[(String, String)],
302    overlay: &BTreeMap<String, String>,
303) -> Vec<(String, String)> {
304    let mut out: Vec<(String, String)> = Vec::with_capacity(original.len() + overlay.len());
305    let overlay_keys_lower: std::collections::HashSet<String> =
306        overlay.keys().map(|k| k.to_ascii_lowercase()).collect();
307    for (k, v) in original {
308        if !overlay_keys_lower.contains(&k.to_ascii_lowercase()) {
309            out.push((k.clone(), v.clone()));
310        }
311    }
312    for (k, v) in overlay {
313        out.push((k.clone(), v.clone()));
314    }
315    out
316}

318/// Wrap a `zendriver_interception::InterceptionError` (re-exported as
319/// `zendriver::InterceptionError`) into the MCP error wire format.
320fn zendriver_err(e: zendriver::InterceptionError) -> ErrorData {
321    map_error(McpServerError::from(ZendriverError::from(e)))
322}

334/// Output of `browser_intercept_remove_rule`.
335#[derive(Debug, Serialize, JsonSchema)]
336pub struct RemoveRuleOutput {
337    /// Always `true` on success. Removal of an unknown id returns
338    /// [`McpServerError::RuleNotFound`] rather than a `false` here, so an
339    /// agent can branch on success without re-checking the body.
340    pub removed: bool,
341}
342
343/// Drop a previously-registered rule, tearing down its interception actor.
344pub async fn remove_rule(
345    state: Arc<Mutex<SessionState>>,
346    input: RemoveRuleInput,
347) -> Result<RemoveRuleOutput, ErrorData> {
348    let mut s = state.lock().await;
349    // `.remove(...)` returns the value (and thus its `_handle: InterceptHandle`),
350    // dropping it here cancels the actor. We don't need to await `stop()` —
351    // `Drop` fires-and-forgets the `Fetch.disable`.
352    s.rules
353        .remove(&input.rule_id)
354        .ok_or_else(|| map_error(McpServerError::RuleNotFound(input.rule_id)))?;
355    Ok(RemoveRuleOutput { removed: true })
356}

360/// Output of `browser_intercept_list_rules`.
361#[derive(Debug, Serialize, JsonSchema)]
362pub struct ListRulesOutput {
363    /// One entry per live rule. Sorted by `rule_id` so the output is stable
364    /// across runs (rather than HashMap-iteration-order chaos).
365    pub rules: Vec<RuleSummary>,
366}
367
368/// Description of a single live interception rule.
369#[derive(Debug, Serialize, JsonSchema)]
370pub struct RuleSummary {
371    /// Same id `browser_intercept_add_rule` returned.
372    pub rule_id: RuleId,
373    /// URL pattern the rule was registered with.
374    pub pattern: String,
375    /// `"block"`, `"redirect"`, `"respond"`, or `"modify_request"`.
376    pub action_kind: String,
377}
378
379/// Enumerate every live rule. Empty list when none are registered (never
380/// an error — a session with no rules is a valid steady state).
381pub async fn list_rules(
382    state: Arc<Mutex<SessionState>>,
383    _: crate::tools::common::EmptyInput,
384) -> Result<ListRulesOutput, ErrorData> {
385    let s = state.lock().await;
386    let mut rules: Vec<RuleSummary> = s
387        .rules
388        .iter()
389        .map(|(id, h)| RuleSummary {
390            rule_id: id.clone(),
391            pattern: h.pattern.clone(),
392            action_kind: h.action_kind.to_string(),
393        })
394        .collect();
395    rules.sort_by(|a, b| a.rule_id.cmp(&b.rule_id));
396    Ok(ListRulesOutput { rules })
397}

401/// Output of `browser_intercept_clear_rules`.
402#[derive(Debug, Serialize, JsonSchema)]
403pub struct ClearRulesOutput {
404    /// Number of rules that were dropped. `0` is a successful no-op
405    /// rather than an error — calling clear on an empty registry is the
406    /// agent's idempotent "make sure nothing is intercepting" lever.
407    pub cleared: usize,
408}
409
410/// Drop every live rule. Each handle's `Drop` cancels its actor; we don't
411/// wait for any of them — same semantics as `remove_rule`.
412pub async fn clear_rules(
413    state: Arc<Mutex<SessionState>>,
414    _: crate::tools::common::EmptyInput,
415) -> Result<ClearRulesOutput, ErrorData> {
416    let mut s = state.lock().await;
417    let cleared = s.rules.len();
418    s.rules.clear();
419    Ok(ClearRulesOutput { cleared })
420}

486    #[test]
487    fn merge_headers_overlays_case_insensitively_and_drops_originals() {
488        let original = vec![
489            ("Host".to_string(), "example.com".to_string()),
490            ("User-Agent".to_string(), "old".to_string()),
491            ("Accept".to_string(), "*/*".to_string()),
492        ];
493        let mut overlay = BTreeMap::new();
494        overlay.insert("user-agent".to_string(), "new".to_string());
495        overlay.insert("X-Marker".to_string(), "yes".to_string());
496
497        let snapshot = ModifyRequestOverlay {
498            headers: overlay,
499            url: None,
500            method: None,
501            post_data: None,
502        };
503        let ov = snapshot.build(&original);
504        assert!(ov.url.is_none());
505        assert!(ov.method.is_none());
506        assert!(ov.post_data.is_none());
507        let headers = ov.headers.expect("headers populated");
508
509        // Original "User-Agent" gone (case-insens), replaced by overlay's
510        // "user-agent: new". Host + Accept survive. Overlay's X-Marker is
511        // appended.
512        let names_lower: Vec<String> = headers
513            .iter()
514            .map(|(k, _)| k.to_ascii_lowercase())
515            .collect();
516        assert!(names_lower.contains(&"host".into()));
517        assert!(names_lower.contains(&"accept".into()));
518        assert!(names_lower.contains(&"user-agent".into()));
519        assert!(names_lower.contains(&"x-marker".into()));
520        // Exactly one "user-agent" (no dupes).
521        assert_eq!(names_lower.iter().filter(|n| *n == "user-agent").count(), 1);
522        // Replaced value is the overlay's.
523        let ua = headers
524            .iter()
525            .find(|(k, _)| k.eq_ignore_ascii_case("user-agent"))
526            .map(|(_, v)| v.as_str())
527            .expect("user-agent present");
528        assert_eq!(ua, "new");
529    }
