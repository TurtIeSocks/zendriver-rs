1//! Tiny HTML trimmer used by tools that bundle a "see the page" snapshot.
2//!
3//! - Strips `<script>...</script>` and `<style>...</style>` blocks (they
4//!   bloat snapshot payloads without helping the LLM reason about page
5//!   structure).
6//! - Collapses runs of whitespace into single spaces and trims the result.
7//!
8//! Implementation is intentionally crude — not a full HTML parser. A
9//! malformed or pathological input (e.g. `<script>` without a closing tag)
10//! cuts off the trailing content; we accept that for v0.
11
12/// Trim the rendered HTML for inclusion in an MCP tool response.
13pub fn trim(html: &str) -> String {
14    let no_scripts = strip_block(html, "script");
15    let no_styles = strip_block(&no_scripts, "style");
16    collapse_ws(&no_styles)
17}
18
19/// Remove every `<tag ...>...</tag>` block (case-sensitive) from `html`.
20fn strip_block(html: &str, tag: &str) -> String {
21    let open = format!("<{tag}");
22    let close = format!("</{tag}>");
23    let mut out = String::with_capacity(html.len());
24    let mut rest = html;
25    while let Some(start) = rest.find(&open) {
26        out.push_str(&rest[..start]);
27        match rest[start..].find(&close) {
28            Some(end_off) => {
29                rest = &rest[start + end_off + close.len()..];
30            }
31            None => {
32                // Unbalanced tag — drop everything from `start` onward.
33                rest = "";
34                break;
35            }
36        }
37    }
38    out.push_str(rest);
39    out
40}
41
42/// Collapse runs of whitespace into single spaces and trim the result.
43fn collapse_ws(s: &str) -> String {
44    let mut out = String::with_capacity(s.len());
45    let mut prev_space = false;
46    for ch in s.chars() {
47        if ch.is_whitespace() {
48            if !prev_space {
49                out.push(' ');
50                prev_space = true;
51            }
52        } else {
53            out.push(ch);
54            prev_space = false;
55        }
56    }
57    out.trim().to_string()
58}

60#[cfg(test)]
61mod tests {
62    use super::*;
63
64    #[test]
65    fn strips_scripts() {
66        let html = "<p>hi</p><script>alert(1)</script><p>bye</p>";
67        assert_eq!(trim(html), "<p>hi</p><p>bye</p>");
68    }
69
70    #[test]
71    fn strips_styles() {
72        let html = "<style>p{color:red}</style><p>hi</p>";
73        assert_eq!(trim(html), "<p>hi</p>");
74    }
75
76    #[test]
77    fn collapses_whitespace() {
78        let html = "<p>hi\n\n  there</p>";
79        assert_eq!(trim(html), "<p>hi there</p>");
80    }
81
82    #[test]
83    fn unbalanced_script_drops_trailing() {
84        let html = "<p>before</p><script>oops";
85        assert_eq!(trim(html), "<p>before</p>");
86    }
