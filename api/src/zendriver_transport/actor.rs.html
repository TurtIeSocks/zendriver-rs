22/// Outbound command sent from a `Connection` handle to the actor.
23#[derive(Debug)]
24pub(crate) struct OutboundCmd {
25    pub method: String,
26    pub params: Value,
27    pub session_id: Option<String>,
28    pub reply: oneshot::Sender<Result<Value, CdpRpcError>>,
29}

31/// Default broadcast bus capacity. Lagged subscribers drop frames.
32pub(crate) const EVENT_BUS_CAPACITY: usize = 1024;

34/// Accounted-bus plumbing for one actor run: the second, opt-in broadcast
35/// sender plus this run's generation number. Bundled into a struct (rather
36/// than two more [`run_actor`] parameters) to keep the function signature
37/// under clippy's `too_many_arguments` threshold and to make the bus/
38/// generation pairing explicit at each call site — mirrors the
39/// [`crate::connection`]-side rationale for bundling wide handshake inputs
40/// into a struct.
41pub(crate) struct AccountedBus {
42    /// Second broadcast bus behind
43    /// [`Connection::subscribe_raw_accounted`](crate::connection::Connection::subscribe_raw_accounted).
44    pub(crate) tx: broadcast::Sender<AccountedRawEvent>,
45    /// This actor run's generation number, stamped onto every
46    /// [`AccountedRawEvent`] it produces.
47    pub(crate) generation: u64,
48}

51#[derive(serde::Deserialize)]
52struct TargetAttached {
53    #[serde(rename = "sessionId")]
54    session_id: String,
55    #[serde(rename = "targetInfo")]

60#[derive(serde::Deserialize)]
61struct TargetDetached {

66/// Runs the actor loop until `shutdown` is cancelled or the WS dies.
67///
68/// Generic over the WS sink + stream so tests can drive in-memory streams
69/// instead of real WebSockets.
70///
71/// `observers` are invoked serially (in registration order) on each new
72/// `Target.attachedToTarget`. The handler runs in a spawned task so the actor
73/// loop stays responsive; `weak_inner` lets the handler reconstruct a
74/// `Connection` (without forming a strong cycle) to talk back through this
75/// same actor.
76///
77/// `accounted` bundles the second, opt-in broadcast bus behind
78/// [`Connection::subscribe_raw_accounted`](crate::connection::Connection::subscribe_raw_accounted)
79/// with this actor run's generation number, stamped onto every
80/// [`AccountedRawEvent`] it produces. Mirroring onto `accounted.tx` — the
81/// per-event clone of [`RawEvent`] and the send itself — is gated behind
82/// `accounted.tx.receiver_count() > 0` so a connection with no accounted
83/// subscribers pays nothing beyond the existing `event_tx` broadcast.
84pub(crate) async fn run_actor<S>(
85    mut ws: S,
86    mut cmd_rx: mpsc::Receiver<OutboundCmd>,
87    event_tx: broadcast::Sender<RawEvent>,
88    accounted: AccountedBus,
89    shutdown: CancellationToken,
90    observers: Vec<Arc<dyn TargetObserver>>,
91    weak_inner: Weak<ConnectionInner>,
92) where
93    S: futures::Sink<Message, Error = tokio_tungstenite::tungstenite::Error>
94        + futures::Stream<Item = Result<Message, tokio_tungstenite::tungstenite::Error>>
95        + Unpin,
96{
97    let mut pending: HashMap<u64, oneshot::Sender<Result<Value, CdpRpcError>>> = HashMap::new();
98    let mut next_id: u64 = 1;
99    // Per-generation monotonic counter for `AccountedRawEvent::Event::sequence`.
100    // Reset to 1 on every actor run (a fresh call to `run_actor`, i.e. every
101    // reconnect); only advances when `accounted_tx` actually has a
102    // subscriber, since it counts positions *on the accounted bus*.
103    let mut next_accounted_sequence: u64 = 1;
104
105    // Sentinel stamped onto drained pendings when the loop exits. Defaults to
106    // the clean-shutdown code; the unexpected-ws-death branches below flip it
107    // to `DISCONNECTED_CODE` so callers awaiting a CDP call during a dropped
108    // connection can distinguish "Chrome died / socket dropped" from a
109    // caller-requested `shutdown()`.
110    let mut drain_code = crate::connection::SHUTDOWN_DRAIN_CODE;
111
112    loop {
113        tokio::select! {
114            biased;
115            () = shutdown.cancelled() => {
116                debug!("actor shutdown received; draining {} pending", pending.len());
117                break;
118            }
119            cmd = cmd_rx.recv() => {
120                let Some(cmd) = cmd else {
121                    debug!("cmd channel closed; shutting down");
122                    break;
123                };
124                let id = next_id;
125                next_id = next_id.wrapping_add(1);
126                let frame = CdpCommand {
127                    id,
128                    method: &cmd.method,
129                    params: cmd.params,
130                    session_id: cmd.session_id.as_deref(),
131                };
132                match serde_json::to_string(&frame) {
133                    Ok(s) => {
134                        trace!(id, method = %cmd.method, "send");
135                        if let Err(e) = ws.send(Message::text(s)).await {
136                            error!("ws send failed: {e}");
137                            let _ = cmd.reply.send(Err(CdpRpcError {
138                                code: -32000,
139                                message: format!("ws send failed: {e}"),
140                                data: None,
141                            }));
142                            // A failed write means the socket died — treat the
143                            // remaining pendings as an unexpected disconnect,
144                            // consistent with the read-side death branches.
145                            drain_code = crate::connection::DISCONNECTED_CODE;
146                            break;
147                        }
148                        pending.insert(id, cmd.reply);
149                    }
150                    Err(e) => {
151                        let _ = cmd.reply.send(Err(CdpRpcError {
152                            code: -32700,
153                            message: format!("serialize: {e}"),
154                            data: None,
155                        }));
156                    }
157                }
158            }
159            frame = ws.next() => {
160                match frame {
161                    None => {
162                        debug!("ws stream ended");
163                        // Stream vanished without a caller-requested shutdown:
164                        // an unexpected disconnect, not a clean close.
165                        drain_code = crate::connection::DISCONNECTED_CODE;
166                        break;
167                    }
168                    Some(Err(e)) => {
169                        error!("ws read failed: {e}");
170                        drain_code = crate::connection::DISCONNECTED_CODE;
171                        break;
172                    }
173                    Some(Ok(Message::Text(text))) => {
174                        match serde_json::from_str::<CdpInbound>(&text) {
175                            Ok(CdpInbound::Response { id, result, error, .. }) => {
176                                match pending.remove(&id) { Some(reply) => {
177                                    let res = match error {
178                                        Some(e) => Err(e),
179                                        None    => Ok(result.unwrap_or(Value::Null)),
180                                    };
181                                    let _ = reply.send(res);
182                                } _ => {
183                                    warn!(id, "response for unknown id (caller dropped?)");
184                                }}
185                            }
186                            Ok(CdpInbound::Event { method, params, session_id }) => {
187                                // Branch on target-lifecycle events before broadcasting.
188                                if method == "Target.attachedToTarget" && !observers.is_empty() {
189                                    match serde_json::from_value::<TargetAttached>(params.clone()) {
190                                        Ok(ev) => {
191                                            match weak_inner.upgrade() { Some(strong) => {
192                                                let conn = Connection { inner: strong };
193                                                let timeout_dur = conn.observer_timeout();
194                                                let observers_clone = observers.clone();
195                                                tokio::spawn(async move {
196                                                    handle_target_attached(
197                                                        conn,
198                                                        ev,
199                                                        observers_clone,
200                                                        timeout_dur,
201                                                    )
202                                                    .await;
203                                                });
204                                            } _ => {
205                                                warn!(
206                                                    "Target.attachedToTarget arrived but \
207                                                     Connection has dropped; skipping observers"
208                                                );
209                                            }}
210                                        }
211                                        Err(e) => error!(
212                                            "bad Target.attachedToTarget payload: {e}"
213                                        ),
214                                    }
215                                } else if method == "Target.detachedFromTarget"
216                                    && !observers.is_empty()
217                                {
218                                    if let Ok(ev) =
219                                        serde_json::from_value::<TargetDetached>(params.clone())
220                                    {
221                                        // Mirror the `attachedToTarget` path:
222                                        // observers can panic in user code, so
223                                        // wrap each call in AssertUnwindSafe +
224                                        // catch_unwind and a soft timeout. We
225                                        // can't tear down the session on
226                                        // failure (it's already detaching),
227                                        // but logging gives users a fighting
228                                        // chance to find a regression instead
229                                        // of dropping the panic silently.
230                                        let timeout_dur = match weak_inner.upgrade()
231                                        { Some(strong) => {
232                                            Connection { inner: strong }.observer_timeout()
233                                        } _ => {
234                                            // Connection has dropped — pick a
235                                            // short default so the task can't
236                                            // hang forever.
237                                            Duration::from_secs(5)
238                                        }};
239                                        for obs in &observers {
240                                            let obs2 = obs.clone();
241                                            let sid = ev.session_id.clone();
242                                            let name = obs2.name();
243                                            tokio::spawn(async move {
244                                                let fut = obs2.on_target_detached(&sid);
245                                                match tokio::time::timeout(
246                                                    timeout_dur,
247                                                    AssertUnwindSafe(fut).catch_unwind(),
248                                                )
249                                                .await
250                                                {
251                                                    Ok(Ok(())) => {}
252                                                    Ok(Err(panic)) => {
253                                                        let msg = panic_payload(&panic);
254                                                        error!(
255                                                            observer = name,
256                                                            session_id = %sid,
257                                                            panic = %msg,
258                                                            "detached-target observer panicked",
259                                                        );
260                                                    }
261                                                    Err(_) => warn!(
262                                                        observer = name,
263                                                        session_id = %sid,
264                                                        "detached-target observer timed out",
265                                                    ),
266                                                }
267                                            });
268                                        }
269                                    } else {
270                                        warn!("bad Target.detachedFromTarget payload");
271                                    }
272                                }
273                                let ev = RawEvent { method, params, session_id };
274                                // Gate the per-event clone + accounted-bus send
275                                // behind an actual subscriber: cloning `ev`
276                                // (method String + params Value — potentially
277                                // large Network.* JSON) is wasted work when
278                                // nobody is listening on the accounted bus.
279                                if accounted.tx.receiver_count() > 0 {
280                                    let sequence = next_accounted_sequence;
281                                    next_accounted_sequence =
282                                        next_accounted_sequence.wrapping_add(1);
283                                    let _ = accounted.tx.send(AccountedRawEvent::Event {
284                                        generation: accounted.generation,
285                                        sequence,
286                                        event: ev.clone(),
287                                    });
288                                }
289                                // Ignore SendError: zero subscribers is fine.
290                                let _ = event_tx.send(ev);
291                            }
292                            Err(e) => warn!("frame parse failed: {e} (text: {text})"),
293                        }
294                    }
295                    Some(Ok(Message::Close(_))) => {
296                        debug!("ws close frame; disconnecting");
297                        // Chrome closed the socket on us — treat as an
298                        // unexpected disconnect so in-flight callers see
299                        // `Disconnected` rather than the clean-shutdown code.
300                        drain_code = crate::connection::DISCONNECTED_CODE;
301                        break;
302                    }
303                    Some(Ok(_)) => { /* ignore binary, ping, pong, frame */ }
304                }
305            }
306        }
307    }
308
309    // Drain pending into transport errors so callers don't hang. `drain_code`
310    // is a reserved sentinel matched by `Connection::call_raw`: it surfaces
311    // drained pendings as `TransportError::Shutdown` (clean shutdown) or
312    // `TransportError::Disconnected` (unexpected ws death) rather than a
313    // generic `Rpc` error.
314    let message = if drain_code == crate::connection::DISCONNECTED_CODE {
315        "connection lost"
316    } else {
317        "connection shut down"
318    };
319    // A genuine ws death (not a caller-requested `shutdown()` and not a
320    // `reconnect()` — both cancel via `shutdown.cancelled()`, which never
321    // sets `DISCONNECTED_CODE`) gets exactly one `Disconnected` on the
322    // accounted bus, gated the same as the per-event mirroring above.
323    if drain_code == crate::connection::DISCONNECTED_CODE && accounted.tx.receiver_count() > 0 {
324        let _ = accounted.tx.send(AccountedRawEvent::Disconnected {
325            generation: accounted.generation,
326        });
327    }
328    for (_id, reply) in pending.drain() {
329        let _ = reply.send(Err(CdpRpcError {
330            code: drain_code,
331            message: message.into(),
332            data: None,
333        }));
334    }
335    debug!(drain_code, "actor exit");
336}

338/// Run all `observers` serially against the just-attached session. Returns
339/// after either: (a) every observer succeeds — in which case we release the
340/// debugger via `Runtime.runIfWaitingForDebugger`; (b) one observer errors or
341/// panics — we detach via `Target.detachFromTarget` and return without
342/// releasing; or (c) an observer exceeds `observer_timeout` — behavior then
343/// depends on [`TargetObserver::failure_policy`]:
344/// [`ObserverFailurePolicy::Required`] (the default) detaches the same as
345/// (b), fail closed, so a hung observer can never leave the page running
346/// without its setup; [`ObserverFailurePolicy::BestEffort`] logs a warning
347/// and **skips only the timed-out observer**, then continues on to the
348/// remaining observers in the chain (if any) exactly as if this one had
349/// returned `Ok`. Once every remaining observer has run (or there were none
350/// left), the debugger is released the same as the success path. This
351/// matters because zendriver registers its tab-registrar observer last as a
352/// ready barrier (see `assemble_observer_chain` in `zendriver::browser`): a
353/// `BestEffort` timeout earlier in the chain must not skip the observers
354/// that come after it, or the registrar would never run and the target
355/// would leak — live and released, but absent from `BrowserInner::tabs`.
356///
357/// ## Cancellation
358///
359/// `AssertUnwindSafe + catch_unwind` covers panics inside the observer
360/// future, not cancellation of the outer task. In practice this function
361/// runs inside a `tokio::spawn`'d task that has no `JoinHandle` retained
362/// by the actor loop — the spawned task therefore cannot be cancelled by
363/// the actor dropping. The task owns the [`Connection`] via the moved
364/// `conn` field below, so the underlying transport stays alive at least
365/// until the task completes naturally. The only way for in-flight CDP
366/// calls to orphan is a runtime-wide shutdown, in which case the entire
367/// process is tearing down and the calls are moot.
368async fn handle_target_attached(
369    conn: Connection,
370    ev: TargetAttached,
371    observers: Vec<Arc<dyn TargetObserver>>,
372    observer_timeout: Duration,
373) {
374    let session_id = ev.session_id.clone();
375    for obs in &observers {
376        let paused = PausedSession {
377            session_id: &session_id,
378            target_info: &ev.target_info,
379            conn: &conn,
380        };
381        let name = obs.name();
382        let fut = obs.on_target_attached(paused);
383        match tokio::time::timeout(observer_timeout, AssertUnwindSafe(fut).catch_unwind()).await {
384            Ok(Ok(Ok(()))) => continue,
385            Ok(Ok(Err(e))) => {
386                error!(observer = name, %session_id, error = %e, "observer failed; detaching");
387                detach_target(&conn, &session_id).await;
388                return;
389            }
390            Ok(Err(panic)) => {
391                let msg = panic_payload(&panic);
392                error!(observer = name, %session_id, panic = %msg, "observer panicked; detaching");
393                detach_target(&conn, &session_id).await;
394                return;
395            }
396            Err(_) => match obs.failure_policy() {
397                ObserverFailurePolicy::Required => {
398                    // Same severity as the Err/panic branches above: a Required
399                    // observer timing out is a fail-closed detach, not a
400                    // best-effort skip — log it at error!, not warn!.
401                    error!(
402                        observer = name,
403                        %session_id,
404                        "observer timed out; detaching (fail-closed)"
405                    );
406                    detach_target(&conn, &session_id).await;
407                    return;
408                }
409                ObserverFailurePolicy::BestEffort => {
410                    warn!(
411                        observer = name,
412                        %session_id,
413                        "observer timed out; skipping (best-effort) and continuing the chain"
414                    );
415                    continue;
416                }
417            },
418        }
419    }
420    let _ = conn
421        .call_raw(
422            "Runtime.runIfWaitingForDebugger",
423            json!({}),
424            Some(session_id),
425        )
426        .await;
427}

429/// Shared detach path for every observer failure mode (`Err`, panic, or a
430/// `Required`-policy timeout): force-detach the session via
431/// `Target.detachFromTarget` so the target is never handed back to Chrome
432/// with `Runtime.runIfWaitingForDebugger`.
433async fn detach_target(conn: &Connection, session_id: &str) {
434    let _ = conn
435        .call_raw(
436            "Target.detachFromTarget",
437            json!({ "sessionId": session_id }),
438            None,
439        )
440        .await;
441}

443/// Best-effort extraction of a textual panic message from a `catch_unwind`
444/// payload. The standard library only guarantees a `Box<dyn Any + Send>`; we
445/// downcast to `&str` and `String` (the two cases the macros produce) and
446/// fall back to a placeholder for everything else.
447fn panic_payload(payload: &Box<dyn Any + Send>) -> String {
448    if let Some(s) = payload.downcast_ref::<&str>() {
449        (*s).to_string()
450    } else if let Some(s) = payload.downcast_ref::<String>() {
451        s.clone()
452    } else {
453        "<unknown panic payload>".to_string()
454    }
455}

862    enum ObserverBehavior {
863        Ok,
864        Err,
865        Panic,
866        Sleep(Duration),
867    }

869    /// Test observer that records the session id of every `on_target_attached`
870    /// invocation (under its `name`) and then executes the configured behavior.
871    struct RecordingObserver {
872        name: &'static str,
873        calls: Arc<Mutex<Vec<(&'static str, String)>>>,
874        behavior: ObserverBehavior,
875        policy: ObserverFailurePolicy,
876    }
877
878    #[async_trait::async_trait]
879    impl TargetObserver for RecordingObserver {
880        fn name(&self) -> &'static str {
881            self.name
882        }
883
884        async fn on_target_attached(
885            &self,
886            session: PausedSession<'_>,
887        ) -> Result<(), ObserverError> {
888            self.calls
889                .lock()
890                .unwrap()
891                .push((self.name, session.session_id.to_string()));
892            match &self.behavior {
893                ObserverBehavior::Ok => Ok(()),
894                ObserverBehavior::Err => Err(ObserverError::Other("boom".into())),
895                ObserverBehavior::Panic => panic!("observer panic"),
896                ObserverBehavior::Sleep(d) => {
897                    tokio::time::sleep(*d).await;
898                    Ok(())
899                }
900            }
901        }
902
903        fn failure_policy(&self) -> ObserverFailurePolicy {
904            self.policy
905        }
906    }

908    /// Read the next text frame from the driver-side and parse it as JSON.
909    async fn next_frame(rx: &mut mpsc::Receiver<Message>) -> Value {
910        let msg = rx.recv().await.expect("driver closed");
911        let text = match msg {
912            Message::Text(t) => t,
913            other => panic!("expected text frame, got {other:?}"),
914        };
915        serde_json::from_str(&text).expect("invalid frame json")
916    }
