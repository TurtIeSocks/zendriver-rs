2//! target is paused at the debugger.
3
4use crate::connection::Connection;
5use crate::error::CallError;

7/// Observer fired on every new [`Target.attachedToTarget`] event before the
8/// debugger releases the target.
9///
10/// The actor walks every registered observer serially (registration order)
11/// on each new target. An observer can fail three ways: it returns `Err`, it
12/// panics, or it exceeds the observer timeout. `Err` and panics always fail
13/// closed — the actor detaches the session via `Target.detachFromTarget` and
14/// stops the chain; no later observer runs. What happens on *timeout*
15/// depends on [`TargetObserver::failure_policy`]:
16/// [`ObserverFailurePolicy::Required`] (the default) also fails closed,
17/// detaches, and stops the chain the same as `Err`/panic;
18/// [`ObserverFailurePolicy::BestEffort`] fails open on timeout only — the
19/// actor skips just that one observer and continues on to the rest of the
20/// chain (registration order preserved), then releases the debugger once
21/// every remaining observer has run, so a non-critical, slow observer can't
22/// hang the target *or* block observers registered after it.
23///
24/// `zendriver-stealth::StealthObserver` implements this trait to install
25/// patches on every new page target before the page's first script runs.
26///
27/// [`Target.attachedToTarget`]: https://chromedevtools.github.io/devtools-protocol/tot/Target/#event-attachedToTarget
28#[async_trait::async_trait]
29pub trait TargetObserver: Send + Sync {
30    /// Called once per new target, after attach and before debugger release.
31    /// Observer MUST complete and return before the target resumes execution.
32    /// Observers run serially in registration order; returning Err leaves the
33    /// target paused (the actor logs + force-detaches the session).
34    async fn on_target_attached(&self, session: PausedSession<'_>) -> Result<(), ObserverError>;
35
36    /// Called when a target detaches. Default: no-op.
37    async fn on_target_detached(&self, _session_id: &str) {}
38
39    /// Stable identifier used in actor diagnostics (`error!` / `warn!` records).
40    fn name(&self) -> &'static str;
41
42    /// How the actor should treat this observer when its `on_target_attached`
43    /// future exceeds the observer timeout.
44    ///
45    /// Defaults to [`ObserverFailurePolicy::Required`] — fail closed. This is
46    /// a **behavior change** for any pre-existing impl that doesn't override
47    /// this method: a hung observer used to be silently skipped (the actor
48    /// released the debugger and the page ran without that observer's setup);
49    /// now it detaches the target instead, the same way an `Err` or panic
50    /// already did. Override to return [`ObserverFailurePolicy::BestEffort`]
51    /// to opt a specific, non-critical observer back into the old fail-open
52    /// timeout behavior: the actor logs and skips *only this observer*, then
53    /// continues running the rest of the chain (registration order
54    /// preserved) exactly as if it had returned `Ok`. Errors and panics are
55    /// unaffected by this policy — they always detach and stop the chain
56    /// regardless of which variant is returned here.
57    ///
58    /// # Examples
59    ///
60    /// ```no_run
61    /// use zendriver_transport::{
62    ///     ObserverError, ObserverFailurePolicy, PausedSession, TargetObserver,
63    /// };
64    ///
65    /// /// A telemetry hook: nice to have, but must never block a page from
66    /// /// loading if it hangs.
67    /// struct OptionalTelemetry;
68    ///
69    /// #[async_trait::async_trait]
70    /// impl TargetObserver for OptionalTelemetry {
71    ///     async fn on_target_attached(
72    ///         &self,
73    ///         _session: PausedSession<'_>,
74    ///     ) -> Result<(), ObserverError> {
75    ///         Ok(())
76    ///     }
77    ///
78    ///     fn name(&self) -> &'static str {
79    ///         "optional-telemetry"
80    ///     }
81    ///
82    ///     fn failure_policy(&self) -> ObserverFailurePolicy {
83    ///         ObserverFailurePolicy::BestEffort
84    ///     }
85    /// }
86    /// ```
87    fn failure_policy(&self) -> ObserverFailurePolicy {
88        ObserverFailurePolicy::Required
89    }
90}
91
92/// How the actor reacts when a [`TargetObserver`] exceeds the observer
93/// timeout — see [`TargetObserver::failure_policy`].
94#[derive(Debug, Clone, Copy, PartialEq, Eq)]
95#[non_exhaustive]
96pub enum ObserverFailurePolicy {
97    /// Fail closed (the default): a timeout is treated the same as an `Err`
98    /// or a panic — the actor sends `Target.detachFromTarget` and the target
99    /// is never handed out (the debugger is not released).
100    Required,
101    /// Fail open on timeout only: the actor logs a warning and skips *just
102    /// this observer*, then continues running any observers registered
103    /// after it (registration order preserved) exactly as if this one had
104    /// returned `Ok`. The debugger is released
105    /// (`Runtime.runIfWaitingForDebugger`) once the rest of the chain has
106    /// run, so the page runs without this observer's setup rather than
107    /// being detached — and observers later in the chain are not abandoned
108    /// on account of this one's timeout. Errors and panics from this
109    /// observer still detach and stop the chain — only the timeout branch
110    /// is relaxed.
111    BestEffort,
112}

114/// Scope passed to [`TargetObserver::on_target_attached`] — a session that's
115/// currently paused at the debugger, plus a back-reference to the connection
116/// for CDP calls scoped to that session.
117#[derive(Debug)]
118pub struct PausedSession<'a> {
119    /// CDP `sessionId` for the newly attached target.
120    pub session_id: &'a str,
121    /// Decoded `targetInfo` payload (target id, kind, url, ...).
122    pub target_info: &'a TargetInfo,
123    pub(crate) conn: &'a Connection,
124}
125
126impl<'a> PausedSession<'a> {
127    /// Send a CDP command scoped to this paused session's `sessionId`.
128    /// Convenience over reaching for [`PausedSession::connection`] manually.
129    pub async fn call(
130        &self,
131        method: impl Into<String>,
132        params: serde_json::Value,
133    ) -> Result<serde_json::Value, CallError> {
134        self.conn
135            .call_raw(method, params, Some(self.session_id.to_string()))
136            .await
137    }
138
139    /// The underlying [`Connection`]. Observers that need to spawn
140    /// additional [`crate::SessionHandle`]s (e.g. zendriver's
141    /// `TabRegistrar`) clone this to bind a fresh handle for the newly
142    /// attached `sessionId`.
143    #[must_use]
144    pub fn connection(&self) -> &'a Connection {
145        self.conn
146    }
147}

149/// Errors an observer may return to indicate it failed to set up its slice of
150/// the new target.
151#[derive(Debug, thiserror::Error)]
152#[non_exhaustive]
153pub enum ObserverError {
154    /// A CDP call dispatched from inside the observer failed.
155    #[error("call failed: {0}")]
156    Call(#[from] CallError),
157
158    /// The observer exceeded its per-target timeout. The actor surfaces this
159    /// when constructing diagnostic output; observers don't construct it
160    /// themselves.
161    #[error("observer timed out after {0:?}")]
162    Timeout(std::time::Duration),
163
164    /// The observer panicked. Carries the downcast panic payload.
165    #[error("observer panicked: {0}")]
166    Panicked(String),
167
168    /// Catch-all for observer-defined failures that don't fit the typed
169    /// variants above.
170    #[error("{0}")]
171    Other(String),
172}

174/// Decoded `targetInfo` payload from `Target.attachedToTarget` / `targetCreated`.
175///
176/// Mirrors CDP's [`Target.TargetInfo`] but only deserializes the fields used
177/// downstream by observers + zendriver core.
178///
179/// [`Target.TargetInfo`]: https://chromedevtools.github.io/devtools-protocol/tot/Target/#type-TargetInfo
180#[derive(Debug, Clone, serde::Deserialize)]
181pub struct TargetInfo {
182    /// CDP target id (stable across `attach` / `detach` cycles).
183    #[serde(rename = "targetId")]
184    pub target_id: String,
185    /// Target kind (`"page"`, `"iframe"`, `"worker"`, ...). The stealth
186    /// observer keys off this to skip workers + iframes.
187    #[serde(rename = "type")]
188    pub kind: String,
189    /// Initial URL the target is at — typically `about:blank` at attach time.
190    pub url: String,
191    /// Document title, when present.
192    #[serde(default)]
193    pub title: Option<String>,
194    /// Whether a debugger is currently attached.
195    #[serde(default)]
196    pub attached: bool,
197    /// Browser-context id this target belongs to (incognito / profile split).
198    #[serde(default, rename = "browserContextId")]
199    pub browser_context_id: Option<String>,
200    /// `frameId` of the iframe element that hosts this target, when present.
201    /// Chrome populates this for `kind == "iframe"` OOPIF targets (Chromium
202    /// 90+); used by [`crate::TargetObserver`] implementations to attach the
203    /// OOPIF's child session to its hosting frame in the parent tab's frame
204    /// tree. Not present for `kind == "page"` and may be absent on older
205    /// Chromium versions even for iframe targets, in which case attach
206    /// observers fall back to matching `target_id` against the frame tree.
207    #[serde(default, rename = "openerFrameId")]
208    pub opener_frame_id: Option<String>,
209}

212#[allow(clippy::panic, clippy::unwrap_used)]
213mod tests {
214    use super::*;
215
216    #[test]
217    fn display_observer_error_timeout_includes_duration() {
218        let e = ObserverError::Timeout(std::time::Duration::from_secs(5));
219        assert_eq!(e.to_string(), "observer timed out after 5s");
220    }
221
222    #[test]
223    fn display_observer_error_panicked_includes_message() {
224        let e = ObserverError::Panicked("oh no".into());
225        assert_eq!(e.to_string(), "observer panicked: oh no");
226    }
227
228    #[test]
229    fn target_info_deserializes_chrome_payload() {
230        let json = r#"{"targetId":"T1","type":"page","url":"about:blank","attached":true}"#;
231        let info: TargetInfo = serde_json::from_str(json).unwrap();
232        assert_eq!(info.target_id, "T1");
233        assert_eq!(info.kind, "page");
234        assert_eq!(info.url, "about:blank");
235        assert!(info.attached);
236    }
