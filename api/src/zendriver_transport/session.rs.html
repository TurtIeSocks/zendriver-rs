1//! `SessionHandle`: a [`Connection`] bound to a particular CDP `sessionId`.
2//! All commands sent through the handle are routed to that target.
3
4use std::sync::Arc;
5
6use futures::{Stream, StreamExt};
7use serde::de::DeserializeOwned;
8use serde_json::Value;
9
10use crate::connection::Connection;
11use crate::error::CallError;
12use crate::frame::RawEvent;
13
14/// Cheap-to-clone handle binding a [`Connection`] to a specific CDP session.
15///
16/// Every CDP target — page, OOPIF, worker — has its own `sessionId` after the
17/// browser fires `Target.attachedToTarget`. A `SessionHandle` couples a shared
18/// transport with one such id so callers can issue commands without repeating
19/// the id on every call.
20#[derive(Clone, Debug)]
21pub struct SessionHandle {
22    inner: Arc<Inner>,
23}
24
25#[derive(Debug)]
26struct Inner {
27    conn: Connection,
28    session_id: String,
29}
30
31impl SessionHandle {
32    /// Construct a handle around `conn` scoped to `session_id`. Used by
33    /// `zendriver` core when wiring a `Tab` to a newly attached page target.
34    pub fn new(conn: Connection, session_id: impl Into<String>) -> Self {
35        Self {
36            inner: Arc::new(Inner {
37                conn,
38                session_id: session_id.into(),
39            }),
40        }
41    }
42
43    /// The CDP `sessionId` this handle is scoped to.
44    pub fn session_id(&self) -> &str {
45        &self.inner.session_id
46    }
47
48    /// Borrow the underlying [`Connection`].
49    pub fn connection(&self) -> &Connection {
50        &self.inner.conn
51    }
52
53    /// Send a CDP command routed to this session.
54    ///
55    /// Forwards to [`Connection::call_raw`] with this handle's `sessionId`
56    /// — the surface most layers above this crate go through.
57    pub async fn call(&self, method: impl Into<String>, params: Value) -> Result<Value, CallError> {
58        self.inner
59            .conn
60            .call_raw(method, params, Some(self.inner.session_id.clone()))
61            .await
62    }
63
64    /// Subscribe to events of type `T` on this session, filtering out events
65    /// for every other target.
66    pub fn subscribe<T>(
67        &self,
68        method: &'static str,
69    ) -> impl Stream<Item = T> + Send + Unpin + use<T>
70    where
71        T: DeserializeOwned + Send + 'static,
72    {
73        let sid = self.inner.session_id.clone();
74        let raw = self.inner.conn.subscribe_raw();
75        Box::pin(raw.filter_map(move |ev: RawEvent| {
76            let sid = sid.clone();
77            async move {
78                if ev.session_id.as_deref() == Some(sid.as_str()) && ev.method == method {
79                    serde_json::from_value(ev.params).ok()
80                } else {
81                    None
82                }
83            }
84        }))
85    }
86}

88#[cfg(test)]
89#[allow(clippy::panic, clippy::unwrap_used)]
90mod tests {
91    use super::*;
92    use crate::connection::spawn_actor;
93    use crate::connection::test_only::DriverStream;
94    use serde_json::json;
95    use tokio::sync::mpsc;
96    use tokio_tungstenite::tungstenite::Message;
97
98    fn duplex_pair() -> (
99        DriverStream,
100        mpsc::Sender<Result<Message, tokio_tungstenite::tungstenite::Error>>,
101        mpsc::Receiver<Message>,
102    ) {
103        let (tx_out, rx_out) = mpsc::channel(32);
104        let (tx_in, rx_in) = mpsc::channel(32);
105        (
106            DriverStream {
107                tx: tx_out,
108                rx: rx_in,
109            },
110            tx_in,
111            rx_out,
112        )
113    }
114
115    #[tokio::test]
116    async fn session_call_includes_session_id() {
117        let (ws, _test_tx, mut test_rx) = duplex_pair();
118        let conn = spawn_actor(ws);
119        let sess = SessionHandle::new(conn.clone(), "S1");
120
121        let call = tokio::spawn({
122            let s = sess.clone();
123            async move {
124                s.call("Page.navigate", json!({ "url": "https://x.test" }))
125                    .await
126            }
127        });
128
129        let sent = test_rx.recv().await.unwrap();
130        let v: Value = serde_json::from_str(match &sent {
131            Message::Text(t) => t,
132            _ => panic!("expected text frame"),
133        })
134        .unwrap();
135        assert_eq!(v["sessionId"], "S1");
136
137        // Cancel via dropping
138        drop(call);
139        conn.shutdown();
140    }
