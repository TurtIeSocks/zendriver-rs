1//! Generative source (C3): a Bayesian-network persona generator.
2//!
3//! Ports the browserforge conditional-probability-table (CPT) form: each node is
4//! a fingerprint attribute whose distribution is conditioned on its parents'
5//! sampled values. Sampling is **deterministic** in the [`Seed`] and yields an
6//! internally **coherent** [`Persona`] (e.g. a `MacIntel` platform pairs with a
7//! Mac-plausible WebGL renderer, because the renderer node is conditioned on
8//! `platform`).
9//!
10//! The embedded [`network.json`] is intentionally small; expanding it from the
11//! full upstream browserforge dataset is tracked as a follow-up (spec §14). The
12//! loader and sampler are complete and correct regardless of network size.
13
14use std::collections::HashMap;
15
16use serde::Deserialize;
17use zendriver_stealth::{Persona, Platform, Seed, WebglSpec};

19/// A Bayesian network of conditional fingerprint attributes (browserforge form).
20#[derive(Debug, Clone, Deserialize)]
21pub struct Generator {
22    nodes: Vec<Node>,
23}
24
25#[derive(Debug, Clone, Deserialize)]
26struct Node {
27    name: String,
28    parents: Vec<String>,
29    /// Conditional distributions keyed by the parents' sampled values joined by
30    /// `|` (empty string for a root node). Each entry maps the key to a weighted
31    /// `(value, weight)` distribution.
32    cpt: HashMap<String, Vec<(String, f64)>>,
33}
34
35impl Generator {
36    /// Load the embedded, trimmed network.
37    pub fn embedded() -> Self {
38        serde_json::from_str(include_str!("network.json")).expect("valid embedded BN")
39    }
40
41    /// Sample a coherent persona deterministically from `seed`.
42    ///
43    /// Determinism is guaranteed by walking nodes in a stable topological order
44    /// (a vector, never a `HashMap`) and driving every pick from a single
45    /// [`Mulberry32`] stream seeded by `seed`. The same seed therefore always
46    /// produces byte-identical assignments and thus an identical [`Persona`].
47    pub fn generate(&self, seed: Seed) -> Persona {
48        let mut rng = Mulberry32::new(seed.value() as u32);
49        let mut assigned: HashMap<String, String> = HashMap::new();
50        for node in self.topo_order() {
51            // Build the conditioning key from already-assigned parents, joined
52            // by `|`. For a root node this is the empty string, matching the
53            // `""` CPT entry.
54            //
55            // A parent must always be assigned before a child reads it; the topo
56            // order guarantees this for the embedded network. If a parent is
57            // somehow missing (a malformed network), fall back to its empty key
58            // so sampling stays panic-free rather than producing a silently
59            // mismatched conditioning key.
60            debug_assert!(
61                node.parents.iter().all(|p| assigned.contains_key(p)),
62                "node `{}` sampled before parent assignment (bad topo order)",
63                node.name
64            );
65            let key = node
66                .parents
67                .iter()
68                .map(|p| assigned.get(p).cloned().unwrap_or_default())
69                .collect::<Vec<_>>()
70                .join("|");
71            // Prefer the conditioned distribution; fall back to the
72            // unconditioned (`""`) entry if the exact key is absent.
73            let Some(dist) = node.cpt.get(&key).or_else(|| node.cpt.get("")) else {
74                // No usable distribution for this node — skip it rather than
75                // panic. Downstream mapping treats absent attributes as `None`.
76                continue;
77            };
78            assigned.insert(node.name.clone(), weighted_pick(dist, rng.next_f64()));
79        }
80        persona_from_assignment(&assigned)
81    }
82
83    /// Nodes in parents-before-children order.
84    ///
85    /// Sorting by parent count is a defensive heuristic that is exact for the
86    /// shallow embedded network (roots have 0 parents, children have 1). The
87    /// sort is **stable**, so ties keep their `network.json` order — this keeps
88    /// [`generate`](Self::generate) deterministic regardless of `HashMap`
89    /// iteration order.
90    fn topo_order(&self) -> Vec<&Node> {
91        let mut order = self.nodes.iter().collect::<Vec<_>>();
92        order.sort_by_key(|n| n.parents.len());
93        order
94    }
95}

97/// Mulberry32 PRNG — tiny, fast, deterministic. Matches the JS reference used
98/// upstream so seeds are portable across the Rust and JS farble paths.
99struct Mulberry32(u32);
100
101impl Mulberry32 {
102    fn new(seed: u32) -> Self {
103        Self(seed)
104    }
105
106    /// Next value in `[0, 1)`.
107    fn next_f64(&mut self) -> f64 {
108        self.0 = self.0.wrapping_add(0x6D2B_79F5);
109        let mut t = self.0;
110        t = (t ^ (t >> 15)).wrapping_mul(1 | t);
111        t = t.wrapping_add((t ^ (t >> 7)).wrapping_mul(61 | t)) ^ t;
112        f64::from(t ^ (t >> 14)) / 4_294_967_296.0
113    }
114}

116/// Pick a value from a weighted distribution given a uniform draw `r` in
117/// `[0, 1)`. Weights need not be normalized.
118fn weighted_pick(dist: &[(String, f64)], r: f64) -> String {
119    let total: f64 = dist.iter().map(|(_, w)| w).sum();
120    if total <= 0.0 {
121        return dist.first().map(|(v, _)| v.clone()).unwrap_or_default();
122    }
123    let mut acc = 0.0;
124    for (v, w) in dist {
125        acc += w / total;
126        if r <= acc {
127            return v.clone();
128        }
129    }
130    // Floating-point slack: return the last value if `r` rounded past `acc`.
131    dist.last().map(|(v, _)| v.clone()).unwrap_or_default()
132}

134/// Map a sampled attribute assignment onto a [`Persona`]. Keys mirror the
135/// `network.json` node names (`platform`, `webglRenderer`, `deviceMemory`, …).
136/// Absent attributes leave the corresponding `Persona` field at its default.
137fn persona_from_assignment(a: &HashMap<String, String>) -> Persona {
138    let mut p = Persona::default();
139    if let Some(plat) = a.get("platform") {
140        p.platform = Some(match plat.as_str() {
141            "Win32" => Platform::Win32,
142            "MacIntel" => Platform::MacIntel,
143            _ => Platform::LinuxX86_64,
144        });
145    }
146    if let Some(r) = a.get("webglRenderer") {
147        p.webgl = Some(WebglSpec {
148            unmasked_renderer: Some(r.clone()),
149            ..Default::default()
150        });
151    }
152    if let Some(m) = a.get("deviceMemory") {
153        p.device_memory_gb = m.parse().ok();
154    }
155    p
156}

158#[cfg(test)]
159mod tests {
160    use super::*;

162    #[test]
163    fn generate_is_deterministic_and_coherent() {
164        let g = Generator::embedded();
165        let a = g.generate(Seed::from_u64(7));
166        let b = g.generate(Seed::from_u64(7));
167        // same seed → same persona
168        assert_eq!(a.platform, b.platform);
169        assert_eq!(
170            a.webgl
171                .as_ref()
172                .and_then(|w| w.unmasked_renderer.as_deref()),
173            b.webgl
174                .as_ref()
175                .and_then(|w| w.unmasked_renderer.as_deref())
176        );
177        assert_eq!(a.device_memory_gb, b.device_memory_gb);
178        // coherence: a populated persona
179        assert!(a.platform.is_some());
180        assert!(a.webgl.is_some());
181    }
182
183    #[test]
184    fn webgl_renderer_is_coherent_with_platform() {
185        // The renderer node is conditioned on `platform`, so a sampled persona
186        // must never pair a platform with a renderer from another platform.
187        let g = Generator::embedded();
188        for s in 0..64u64 {
189            let p = g.generate(Seed::from_u64(s));
190            let plat = p.platform.expect("platform always assigned");
191            let renderer = p
192                .webgl
193                .as_ref()
194                .and_then(|w| w.unmasked_renderer.as_deref())
195                .expect("renderer always assigned");
196            match plat {
197                Platform::MacIntel => assert!(
198                    renderer.contains("Apple"),
199                    "MacIntel paired with non-Apple renderer: {renderer}"
200                ),
201                Platform::Win32 => assert!(
202                    renderer.contains("Direct3D11"),
203                    "Win32 paired with non-D3D renderer: {renderer}"
204                ),
205                Platform::LinuxX86_64 => assert!(
206                    renderer.contains("Mesa"),
207                    "Linux paired with non-Mesa renderer: {renderer}"
208                ),
209            }
210        }
211    }

213    #[test]
214    fn different_seeds_can_differ() {
215        // Guard against a constant generator: across a handful of fixed seeds
216        // the sampled platforms must not all be identical.
217        let g = Generator::embedded();
218        let platforms: Vec<_> = (0..32u64)
219            .map(|s| g.generate(Seed::from_u64(s)).platform)
220            .collect();
221        let first = platforms[0];
222        assert!(
223            platforms.iter().any(|p| *p != first),
224            "generator produced the same platform for every seed"
225        );
226    }
