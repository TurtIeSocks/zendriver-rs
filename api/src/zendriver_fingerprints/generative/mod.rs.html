1//! Generative source (C3): a browserforge/apify Bayesian-network persona generator.
2//!
3//! Ports the canonical conditional-probability-table form: each node's
4//! distribution is conditioned on its parents' sampled values, walked through a
5//! nested `conditionalProbabilities` `deeper`/`skip` tree. Sampling is
6//! **deterministic** in the [`Seed`] (a seeded Mulberry32 stream) and yields an
7//! internally **coherent**, **desktop** [`Persona`]: the single root `userAgent`
8//! node is restricted to non-mobile UAs, and every other attribute is sampled
9//! conditioned on it.
10//!
11//! The network is fetched on first use and cached (see [`Generator::load_or_download`]).
12//! Determinism holds for a fixed cached network version; a cache refresh (the
13//! upstream file is regenerated ~monthly) may change per-seed output.
14

36/// Errors from loading or decompressing the network.
37#[derive(Debug, thiserror::Error)]
38pub enum GenError {
39    #[error("http error: {0}")]
40    Http(#[from] reqwest::Error),
41    #[error("io error: {0}")]
42    Io(#[from] std::io::Error),
43    #[error("zip error: {0}")]
44    Zip(#[from] zip::result::ZipError),
45    #[error("json error: {0}")]
46    Json(#[from] serde_json::Error),
47    #[error("empty network archive")]
48    EmptyArchive,
49}

56#[derive(Debug, Clone, Deserialize)]
57struct Node {

66/// A Bayesian network of conditional fingerprint attributes (browserforge form).
67#[derive(Debug, Clone)]
68pub struct Generator {
69    nodes: Arc<Vec<Node>>,
70}

72impl Generator {
73    /// Build from the inner network JSON (seam for fast unit tests).
74    pub fn from_network_json(json: &str) -> Result<Self, GenError> {
75        let raw: RawNetwork = serde_json::from_str(json)?;
76        Ok(Self {
77            nodes: Arc::new(raw.nodes),
78        })
79    }
80
81    /// Build from raw ZIP bytes (decompress the first entry, then parse).
82    pub fn from_zip_bytes(bytes: &[u8]) -> Result<Self, GenError> {
83        let mut archive = zip::ZipArchive::new(std::io::Cursor::new(bytes))?;
84        if archive.is_empty() {
85            return Err(GenError::EmptyArchive);
86        }
87        let mut entry = archive.by_index(0)?;
88        let mut json = String::new();
89        entry.read_to_string(&mut json)?;
90        Self::from_network_json(&json)
91    }
92
93    /// Fetch (or read cached) the network ZIP from `url`, then build.
94    pub async fn load_or_download(url: &str) -> Result<Self, GenError> {
95        let cache = download::cache_path();
96        let bytes = download::fetch_or_cached_bytes(url, &cache).await?;
97        Self::from_zip_bytes(&bytes)
98    }
99
100    /// Ergonomic [`load_or_download`](Self::load_or_download): uses
101    /// `ZENDRIVER_FP_NETWORK_URL` if set, else [`DEFAULT_NETWORK_URL`].
102    pub async fn load_or_download_default() -> Result<Self, GenError> {
103        let url = std::env::var("ZENDRIVER_FP_NETWORK_URL")
104            .unwrap_or_else(|_| DEFAULT_NETWORK_URL.to_string());
105        Self::load_or_download(&url).await
106    }
107
108    /// Sample a coherent **desktop** persona deterministically from `seed`.
109    pub fn generate(&self, seed: Seed) -> Persona {
110        let mut rng = Mulberry32::new(seed.value() as u32);
111        let mut assigned: HashMap<String, String> = HashMap::new();
112        for node in self.topo_order() {
113            let leaf = walk_cpt(&node.cpt, &node.parent_names, &assigned);
114            // Restrict the single root `userAgent` node to desktop UAs so the
115            // whole persona is desktop-coherent without backtracking.
116            let dist = leaf_distribution(leaf, node.name == "userAgent");
117            if dist.is_empty() {
118                continue;
119            }
120            assigned.insert(node.name.clone(), weighted_pick(&dist, rng.next_f64()));
121        }
122        mapping::persona_from_assignment(&assigned)
123    }
124
125    /// Nodes in parents-before-children order (stable sort by parent count;
126    /// ties keep JSON array order — keeps [`generate`](Self::generate)
127    /// deterministic).
128    fn topo_order(&self) -> Vec<&Node> {
129        let mut order: Vec<&Node> = self.nodes.iter().collect();
130        order.sort_by_key(|n| n.parent_names.len());
131        order
132    }
133}
134
135/// Walk the nested CPT to the leaf for the parents' sampled values. For each
136/// parent, descend `deeper[value]` if present, else `skip`. After
137/// `parents.len()` steps the result is the leaf `{ value: probability }` object
138/// (or `Null` for a missing/`null` `skip`).
139fn walk_cpt<'a>(
140    cpt: &'a Value,
141    parents: &[String],
142    assigned: &HashMap<String, String>,
143) -> &'a Value {
144    const NULL: Value = Value::Null;
145    let mut cur = cpt;
146    for parent in parents {
147        let pv = assigned.get(parent).map(String::as_str).unwrap_or_default();
148        // Descend `deeper[parent_value]` when present, else the `skip` branch
149        // (absent / `null` -> empty leaf). Plain match avoids any borrow ambiguity.
150        cur = match cur.get("deeper").and_then(|d| d.get(pv)) {
151            Some(next) => next,
152            None => cur.get("skip").unwrap_or(&NULL),
153        };
154    }
155    cur
156}
157
158/// Flatten a leaf object into a `(value, weight)` distribution. When
159/// `desktop_only`, keep only desktop-UA keys (used for the root node).
160fn leaf_distribution(leaf: &Value, desktop_only: bool) -> Vec<(String, f64)> {
161    let Some(obj) = leaf.as_object() else {
162        return Vec::new();
163    };
164    obj.iter()
165        .filter(|(k, _)| !desktop_only || mapping::is_desktop_ua(k))
166        .filter_map(|(k, v)| v.as_f64().map(|w| (k.clone(), w)))
167        .collect()
168}
169
170/// Weighted pick from a (value, weight) distribution given a uniform `r` in
171/// `[0, 1)`. Weights need not be normalized.
172fn weighted_pick(dist: &[(String, f64)], r: f64) -> String {
173    let total: f64 = dist.iter().map(|(_, w)| w).sum();
174    if total <= 0.0 {
175        return dist.first().map(|(v, _)| v.clone()).unwrap_or_default();
176    }
177    let mut acc = 0.0;
178    for (v, w) in dist {
179        acc += w / total;
180        if r <= acc {
181            return v.clone();
182        }
183    }
184    dist.last().map(|(v, _)| v.clone()).unwrap_or_default()
185}

187/// Mulberry32 PRNG — tiny, fast, deterministic.
188struct Mulberry32(u32);
189
190impl Mulberry32 {
191    fn new(seed: u32) -> Self {
192        Self(seed)
193    }
194
195    /// Next value in `[0, 1)`.
196    fn next_f64(&mut self) -> f64 {
197        self.0 = self.0.wrapping_add(0x6D2B_79F5);
198        let mut t = self.0;
199        t = (t ^ (t >> 15)).wrapping_mul(1 | t);
200        t = t.wrapping_add((t ^ (t >> 7)).wrapping_mul(61 | t)) ^ t;
201        f64::from(t ^ (t >> 14)) / 4_294_967_296.0
202    }
203}

234            let a = g.generate(Seed::from_u64(s));
235            let b = g.generate(Seed::from_u64(s));

243    #[test]
244    fn personas_are_desktop_coherent() {
245        let g = make_gen();
246        for s in 0..256u64 {
247            let p = g.generate(Seed::from_u64(s));
248            let plat = p.platform.expect("platform assigned");
249            assert!(
250                matches!(
251                    plat,
252                    Platform::Win32 | Platform::MacIntel | Platform::LinuxX86_64
253                ),
254                "non-desktop platform sampled: {plat:?}"
255            );
256            let w = p.webgl.as_ref().expect("webgl assigned");
257            let renderer = w.unmasked_renderer.as_deref().expect("renderer assigned");
258            assert!(w.unmasked_vendor.is_some(), "vendor assigned");
259            match plat {
260                Platform::Win32 => assert!(renderer.contains("D3D11"), "win renderer: {renderer}"),
261                Platform::MacIntel => {
262                    assert!(
263                        renderer.contains("Apple") || renderer.contains("Metal"),
264                        "mac renderer: {renderer}"
265                    )
266                }
267                Platform::LinuxX86_64 => {
268                    assert!(renderer.contains("Mesa"), "linux renderer: {renderer}")
269                }
270            }
271        }
272    }
