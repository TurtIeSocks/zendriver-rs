1//! Cloudflare Turnstile bypass driver.
2//!
3//! Public entry is [`CloudflareBypass`] — constructed via
4//! `Tab::cloudflare()` (zendriver crate, feature-gated). The driver runs a
5//! single CDP poll loop that, per tick:
6//!
7//! 1. Re-evaluates a unified shadow-DOM walker in the page's main world
8//!    (private [`POLL_JS`]) returning, in one round-trip:
9//!    - the `cf-turnstile-response` (or legacy `cf_challenge_response`) token
10//!      if present,
11//!    - the Turnstile challenge iframe's bounding box if mounted,
12//!    - whether *any* challenge marker exists on the page
13//!      (container, hidden input, or live iframe).
14//! 2. Resolves to [`ClearanceOutcome::TokenAcquired`] the first tick a
15//!    non-empty token is observed — including the **invisible Turnstile**
16//!    path where the iframe never mounts and the token is populated
17//!    directly.
18//! 3. If the interactive checkbox iframe is mounted *and we have not yet
19//!    clicked*, dispatches a single raw left-click at the canonical
20//!    `bbox.x + bbox.width * 0.15, bbox.y + bbox.height * 0.50` offset
21//!    (15% from left, 50% from top — Python's `cloudflare.py` convention).
22//! 4. If we previously observed an iframe + clicked, and the iframe is now
23//!    gone without a token, resolves to [`ClearanceOutcome::ChallengeGone`]
24//!    (clearance-cookie shortcut).
25//! 5. Resolves to [`ClearanceOutcome::TimedOut`] on deadline, carrying
26//!    `saw_challenge` — `true` when challenge markers were seen but never
27//!    resolved, `false` when the entire timeout window elapsed without
28//!    observing any challenge markers (the caller likely invoked the bypass
29//!    on a page that has no Cloudflare gate at all).

42/// Result of a clearance attempt.
43#[derive(Debug, Clone)]
44pub enum ClearanceOutcome {
45    /// Turnstile produced a token (value of `cf-turnstile-response`).
46    TokenAcquired(String),
47    /// The challenge container disappeared without yielding a token.
48    ChallengeGone,
49    /// Deadline elapsed without a terminal clearance state. `saw_challenge`
50    /// is `true` if any challenge marker (container, hidden input, or live
51    /// iframe) was ever observed; `false` if none ever appeared — the caller
52    /// likely invoked the bypass on a page that has no Cloudflare gate at all.
53    /// Not a fault: a deadline in a bot-management flow is a normal "didn't
54    /// finish, retry or give up" terminal.
55    TimedOut { saw_challenge: bool },
56}
57
58/// Default poll interval for `wait_for_clearance`.
59const DEFAULT_POLL_INTERVAL: Duration = Duration::from_millis(500);

61/// Drives a Cloudflare Turnstile clearance flow against a single tab's session.
62///
63/// Constructed via `Tab::cloudflare()`.
64#[derive(Debug)]
65pub struct CloudflareBypass<'a> {
66    pub(crate) session: &'a SessionHandle,
67    pub(crate) poll_interval: Duration,
68}

70impl<'a> CloudflareBypass<'a> {
71    /// Create a new bypass driver bound to `session` with a default 500ms
72    /// poll interval.
73    pub fn new(session: &'a SessionHandle) -> Self {
74        Self {
75            session,
76            poll_interval: DEFAULT_POLL_INTERVAL,
77        }
78    }
79
80    /// Override the default 500ms poll interval used by
81    /// [`wait_for_clearance`](Self::wait_for_clearance).
82    #[must_use]
83    pub fn poll_interval(mut self, dur: Duration) -> Self {
84        self.poll_interval = dur;
85        self
86    }

88    /// Poll the page until a Turnstile clearance terminal state is reached or
89    /// `timeout` elapses.
90    ///
91    /// Handles both the **interactive** flow (challenge iframe mounts → we
92    /// click it → token appears) and the **invisible** flow (no iframe;
93    /// token is populated directly by Cloudflare's loader script). Resolution
94    /// rules per tick are documented at module level.
95    ///
96    /// # Returns
97    /// - `Ok(ClearanceOutcome::TokenAcquired(token))` — the
98    ///   `cf-turnstile-response` input picked up a non-empty value, either
99    ///   after we clicked the interactive iframe or because the page uses
100    ///   invisible Turnstile.
101    /// - `Ok(ClearanceOutcome::ChallengeGone)` — the interactive iframe was
102    ///   present, we clicked it, and the challenge container then
103    ///   disappeared without yielding a token (e.g. a clearance cookie
104    ///   shortcut).
105    /// - `Ok(ClearanceOutcome::TimedOut { saw_challenge })` — `timeout`
106    ///   elapsed without a terminal clearance state. `saw_challenge` is
107    ///   `true` if challenge markers were observed (markers present but
108    ///   never resolved); `false` if none ever appeared (the caller likely
109    ///   invoked the bypass on a page that has no Cloudflare gate). Not a
110    ///   fault — a deadline here is a normal "retry or give up" signal.
111    ///
112    /// # Errors
113    /// - [`CloudflareError::Call`] / [`CloudflareError::JsError`] — CDP or
114    ///   in-page evaluator failure.
115    ///
116    /// ```no_run
117    /// # async fn ex(tab: &zendriver_transport::SessionHandle)
118    /// #   -> Result<(), zendriver_cloudflare::CloudflareError> {
119    /// use std::time::Duration;
120    /// use zendriver_cloudflare::CloudflareBypass;
121    ///
122    /// let bypass = CloudflareBypass::new(tab)
123    ///     .poll_interval(Duration::from_millis(250));
124    /// let outcome = bypass.wait_for_clearance(Duration::from_secs(15)).await?;
125    /// println!("{outcome:?}");
126    /// # Ok(()) }
127    /// ```
128    pub async fn wait_for_clearance(
129        self,
130        timeout: Duration,
131    ) -> Result<ClearanceOutcome, CloudflareError> {
132        let deadline = Instant::now() + timeout;
133        let mut clicked = false;
134        let mut ever_seen_markers = false;
135
136        let mut stall_ticks: u32 = 0;
137        let mut warned_stall = false;
138
139        loop {
140            let state = poll_state(self.session).await?;
141            if state.has_markers {
142                ever_seen_markers = true;
143            }
144
145            if let Some(token) = state.token {
146                return Ok(ClearanceOutcome::TokenAcquired(token));
147            }
148
149            match state.bbox {
150                Some(bbox) if !clicked => {
151                    let click_x = bbox.x + bbox.width * 0.15;
152                    let click_y = bbox.y + bbox.height * 0.50;
153                    click_at(self.session, click_x, click_y).await?;
154                    clicked = true;
155                }
156                None if clicked => {
157                    // Iframe was present, we clicked, and now the challenge
158                    // container has been torn down without a token — the
159                    // clearance-cookie shortcut.
160                    return Ok(ClearanceOutcome::ChallengeGone);
161                }
162                _ => {
163                    stall_ticks += 1;
164                    if stall_ticks == 10 && !warned_stall {
165                        tracing::warn!(
166                            poll_interval_ms = self.poll_interval.as_millis() as u64,
167                            "cloudflare clearance stalled — is BrowserBuilder::stealth enabled?"
168                        );
169                        warned_stall = true;
170                    }
171                }
172            }
173
174            if Instant::now() >= deadline {
175                return Ok(ClearanceOutcome::TimedOut {
176                    saw_challenge: ever_seen_markers,
177                });
178            }
179
180            tokio::select! {
181                () = tokio::time::sleep(self.poll_interval) => {}
182                () = tokio::time::sleep_until(deadline) => {}
183            }
184        }
185    }
186}

188/// Decoded payload from the unified poll evaluator. Combines token check,
189/// iframe bbox, and a coarse "any challenge marker present" flag so each
190/// tick is a single CDP round-trip.
191#[derive(Debug, Deserialize)]
192struct PollState {
193    /// Non-empty `cf-turnstile-response` / `cf_challenge_response` value, if
194    /// any.
195    #[serde(default)]
196    token: Option<String>,
197    /// Bounding box of the Turnstile challenge iframe, if mounted. Walks
198    /// shadow roots so closed shadow trees still surface the iframe.
199    #[serde(default)]
200    bbox: Option<BoundingBox>,
201    /// `true` when the page has any of: a `.cf-turnstile` / `.turnstile`
202    /// container, a `cf-turnstile-response` hidden input, or a live
203    /// challenge iframe. Used to distinguish "wrong page" (no markers ever
204    /// seen) from "real challenge that timed out".
205    #[serde(default, rename = "hasMarkers")]
206    has_markers: bool,
207}
208
209/// Unified in-page evaluator. Returns:
210/// - `token` — non-empty `cf-turnstile-response` (or legacy
211///   `cf_challenge_response`) input value, else null.
212/// - `bbox` — the Turnstile challenge iframe's bounding box (shadow-DOM
213///   aware walk), else null.
214/// - `hasMarkers` — true when *any* Cloudflare challenge marker is present
215///   (container, hidden input, or live iframe).
216///
217/// The shadow-DOM walk is necessary because Cloudflare sometimes hosts the
218/// challenge iframe inside an open shadow root attached to a host element on
219/// the page. The token input itself is in light DOM by design (page JS reads
220/// it to submit forms), so `document.querySelector` is sufficient there.
221const POLL_JS: &str = r#"
222(function () {
223    function findBbox(root) {
224        var iframes = root.querySelectorAll ? root.querySelectorAll("iframe") : [];
225        for (var i = 0; i < iframes.length; i++) {
226            var f = iframes[i];
227            if (f.src && f.src.includes("challenges.cloudflare.com")) {
228                var r = f.getBoundingClientRect();
229                return { x: r.left, y: r.top, width: r.width, height: r.height };
230            }
231        }
232        var all = root.querySelectorAll ? root.querySelectorAll("*") : [];
233        for (var j = 0; j < all.length; j++) {
234            if (all[j].shadowRoot) {
235                var sub = findBbox(all[j].shadowRoot);
236                if (sub) return sub;
237            }
238        }
239        return null;
240    }
241    var bbox = findBbox(document);
242    var input =
243        document.querySelector('[name="cf-turnstile-response"]') ||
244        document.querySelector('[name="cf_challenge_response"]');
245    var token = (input && input.value) ? input.value : null;
246    var hasContainer = !!document.querySelector('.cf-turnstile, .turnstile, [data-sitekey]');
247    var hasMarkers = hasContainer || !!input || !!bbox;
248    return { token: token, bbox: bbox, hasMarkers: hasMarkers };
249})()
250"#;
251
252/// Run [`POLL_JS`] against `session`'s main world and decode the result.
253async fn poll_state(session: &SessionHandle) -> Result<PollState, CloudflareError> {
254    let res = session
255        .call(
256            "Runtime.evaluate",
257            json!({
258                "expression": POLL_JS,
259                "returnByValue": true,
260                "awaitPromise": true,
261            }),
262        )
263        .await?;
264
265    if let Some(details) = res.get("exceptionDetails") {
266        let msg = details
267            .get("exception")
268            .and_then(|e| e.get("description"))
269            .and_then(|d| d.as_str())
270            .unwrap_or("unknown")
271            .to_string();
272        return Err(CloudflareError::JsError(msg));
273    }
274
275    let value = res
276        .get("result")
277        .and_then(|r| r.get("value"))
278        .cloned()
279        .unwrap_or(Value::Null);
280
281    serde_json::from_value(value)
282        .map_err(|e| CloudflareError::JsError(format!("invalid poll payload: {e}")))
283}

291    /// Interactive happy path: poll #1 yields a bbox → click_at fires the
292    /// three mouse events at (15% × width, 50% × height) inside the bbox →
293    /// poll #2 observes the token → TokenAcquired.
294    #[tokio::test]
295    async fn wait_for_clearance_clicks_at_bbox_offset_then_returns_token() {
296        let (mut mock, conn) = MockConnection::pair();
297        let sess = SessionHandle::new(conn.clone(), "S1");
298
299        // bbox at (100, 200) with 60×40 → click at
300        //   x = 100 + 60 * 0.15 = 109
301        //   y = 200 + 40 * 0.50 = 220
302        const BBOX_X: f64 = 100.0;
303        const BBOX_Y: f64 = 200.0;
304        const BBOX_W: f64 = 60.0;
305        const BBOX_H: f64 = 40.0;
306        const EXPECTED_CLICK_X: f64 = BBOX_X + BBOX_W * 0.15;
307        const EXPECTED_CLICK_Y: f64 = BBOX_Y + BBOX_H * 0.50;
308
309        let fut = tokio::spawn({
310            let s = sess.clone();
311            async move {
312                let b = CloudflareBypass::new(&s).poll_interval(Duration::from_millis(1));
313                b.wait_for_clearance(Duration::from_secs(5)).await
314            }
315        });
316
317        // Poll #1: unified evaluator returns bbox, no token, markers present.
318        let id_poll1 = mock.expect_cmd("Runtime.evaluate").await;
319        assert!(
320            mock.last_sent()["params"]["expression"]
321                .as_str()
322                .unwrap()
323                .contains("challenges.cloudflare.com"),
324            "poll eval should walk for challenges.cloudflare.com iframe"
325        );
326        assert!(
327            mock.last_sent()["params"]["expression"]
328                .as_str()
329                .unwrap()
330                .contains("cf-turnstile-response"),
331            "poll eval should look at the cf-turnstile-response input"
332        );
333        mock.reply(
334            id_poll1,
335            json!({
336                "result": {
337                    "type": "object",
338                    "value": {
339                        "token": null,
340                        "bbox": { "x": BBOX_X, "y": BBOX_Y, "width": BBOX_W, "height": BBOX_H },
341                        "hasMarkers": true,
342                    }
343                }
344            }),
345        )
346        .await;
347
348        // click_at → mouseMoved → mousePressed → mouseReleased.
349        let id_move = mock.expect_cmd("Input.dispatchMouseEvent").await;
350        let sent = mock.last_sent();
351        assert_eq!(sent["params"]["type"], "mouseMoved");
352        assert_eq!(sent["params"]["x"], EXPECTED_CLICK_X);
353        assert_eq!(sent["params"]["y"], EXPECTED_CLICK_Y);
354        mock.reply(id_move, json!({})).await;
355
356        let id_press = mock.expect_cmd("Input.dispatchMouseEvent").await;
357        let sent = mock.last_sent();
358        assert_eq!(sent["params"]["type"], "mousePressed");
359        assert_eq!(sent["params"]["button"], "left");
360        assert_eq!(sent["params"]["clickCount"], 1);
361        assert_eq!(sent["params"]["x"], EXPECTED_CLICK_X);
362        assert_eq!(sent["params"]["y"], EXPECTED_CLICK_Y);
363        mock.reply(id_press, json!({})).await;
364
365        let id_rel = mock.expect_cmd("Input.dispatchMouseEvent").await;
366        let sent = mock.last_sent();
367        assert_eq!(sent["params"]["type"], "mouseReleased");
368        assert_eq!(sent["params"]["button"], "left");
369        assert_eq!(sent["params"]["clickCount"], 1);
370        mock.reply(id_rel, json!({})).await;
371
372        // Poll #2: token appears → TokenAcquired terminates the loop.
373        let id_poll2 = mock.expect_cmd("Runtime.evaluate").await;
374        mock.reply(
375            id_poll2,
376            json!({
377                "result": {
378                    "type": "object",
379                    "value": {
380                        "token": "TOKEN_XYZ",
381                        "bbox": null,
382                        "hasMarkers": true,
383                    }
384                }
385            }),
386        )
387        .await;
388
389        let outcome = fut.await.unwrap().unwrap();
390        match outcome {
391            ClearanceOutcome::TokenAcquired(t) => assert_eq!(t, "TOKEN_XYZ"),
392            other => panic!("expected TokenAcquired, got {other:?}"),
393        }
394        conn.shutdown();
395    }
