1//! Fetch-domain fast-path for DataDome clearance detection.
2//!
3//! Opt-in via [`DataDomeBypass::with_interception`]. Subscribes to
4//! Fetch responses matching `captcha-delivery.com` or `datadome` URL
5//! patterns; signals the waiter via a oneshot when a 2xx is observed.
6//! Polling continues in parallel — first signal wins.
7//!
8//! [`DataDomeBypass::with_interception`]: crate::bypass::DataDomeBypass::with_interception
9
10use futures::StreamExt;
11use tokio::sync::oneshot;
12use tokio_util::sync::CancellationToken;
13use zendriver_interception::InterceptBuilder;
14use zendriver_transport::SessionHandle;
15
16/// Spawn a background task that signals on first 2xx DataDome-sensor
17/// response and returns the receiver half of a oneshot.
18///
19/// Infallible: the `InterceptBuilder` chain used here (`new` → `pattern` →
20/// `at_response` → `subscribe`) is pure sync setup with no `Result`-returning
21/// step. The actual `Fetch.enable` CDP round-trip is fire-and-forget inside
22/// `subscribe()`; transport errors there are surfaced as a warn log + an
23/// empty stream, not as a `spawn_signal` failure. Treating this as `-> _`
24/// rather than `Result<_, DataDomeError>` avoids the
25/// `clippy::result_large_err` warning that `DataDomeError`'s 136-byte
26/// `CallError` variant would otherwise trip for a tiny `Ok` payload.
27///
28/// Caller must keep the returned [`InterceptionGuard`] alive until they are
29/// done with the receiver. Dropping the guard cancels the spawned task
30/// cooperatively via a `CancellationToken` checked at every loop boundary,
31/// then aborts it as a backstop. Cooperative cancel is preferred over a
32/// bare `abort()` because abort is asynchronous: the task may run one more
33/// `paused.continue_().await` after the abort signal lands — harmless, but
34/// the token lets the loop exit cleanly between events instead.
35pub(crate) fn spawn_signal(session: &SessionHandle) -> (oneshot::Receiver<()>, InterceptionGuard) {
36    let (tx, rx) = oneshot::channel();
37    let cancel = CancellationToken::new();
38    let task_cancel = cancel.clone();
39
40    // `subscribe()` is sync and returns `impl Stream<Item = PausedRequest>`.
41    // `pattern()` returns `Self` (not a `Result`), so no `?` needed on chain.
42    let stream = InterceptBuilder::new(session)
43        .pattern("*captcha-delivery.com*")
44        .at_response()
45        .pattern("*datadome*")
46        .at_response()
47        .subscribe();
48
49    let handle = tokio::spawn(async move {
50        let mut stream = Box::pin(stream);
51        let mut tx = Some(tx);
52        loop {
53            tokio::select! {
54                biased;
55                () = task_cancel.cancelled() => break,
56                next = stream.next() => {
57                    let Some(paused) = next else { break };
58                    let is_2xx = paused
59                        .response
60                        .as_ref()
61                        .map(|r| (200..300).contains(&r.status))
62                        .unwrap_or(false);
63                    // Always release the pause so the page keeps loading.
64                    let _ = paused.continue_().await;
65                    if is_2xx {
66                        if let Some(t) = tx.take() {
67                            let _ = t.send(());
68                        }
69                        break;
70                    }
71                }
72            }
73        }
74    });
75
76    (
77        rx,
78        InterceptionGuard {
79            cancel,
80            handle: Some(handle),
81        },
82    )
83}

85/// Guard for the background interception task. On drop, signals
86/// cooperative cancellation first (clean exit between events) and then
87/// aborts as a backstop in case the task is parked on a non-cancellable
88/// future.
89pub(crate) struct InterceptionGuard {
90    cancel: CancellationToken,
91    handle: Option<tokio::task::JoinHandle<()>>,
92}
93
94impl Drop for InterceptionGuard {
95    fn drop(&mut self) {
96        self.cancel.cancel();
97        if let Some(h) = self.handle.take() {
98            h.abort();
99        }
100    }
101}
