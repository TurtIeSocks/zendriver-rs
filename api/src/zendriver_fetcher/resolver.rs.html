1//! Version + platform → download URL resolution.
2//!
3//! Given a parsed Chrome-for-Testing manifest plus a
4//! [`VersionSpec`]/[`Platform`] pair, picks the matching `(version, url)`
5//! tuple — or returns a specific [`FetcherError`] when no match is possible.
6//!
7//! Split out from manifest fetching so unit tests can drive the resolver
8//! with hand-built manifests, without touching the network.
9
10use crate::error::FetcherError;
11use crate::manifest::KnownGoodVersionsResponse;
12use crate::platform::Platform;
13use crate::version::{Channel, VersionSpec};
14
15/// Resolve a `(version_string, download_url)` pair from a parsed manifest.
16///
17/// # Errors
18///
19/// - [`FetcherError::VersionNotFound`] if [`VersionSpec::Explicit`] does not
20///   match any entry in the manifest.
21/// - [`FetcherError::UnsupportedPlatform`] if the manifest has no download
22///   for `platform`, or if a non-stable channel is requested (the
23///   `latest-versions-per-milestone.json` endpoint is not wired in P5).
24pub(crate) async fn resolve_download_url(
25    manifest: &KnownGoodVersionsResponse,
26    spec: &VersionSpec,
27    platform: Platform,
28) -> Result<(String, String), FetcherError> {
29    // Pick the manifest entry matching the version spec.
30    let entry = match spec {
31        VersionSpec::Latest | VersionSpec::Stable | VersionSpec::Channel(Channel::Stable) => {
32            manifest
33                .versions
34                .last()
35                .ok_or_else(|| FetcherError::VersionNotFound("manifest is empty".to_string()))?
36        }
37        VersionSpec::Channel(Channel::Beta | Channel::Dev | Channel::Canary) => {
38            return Err(FetcherError::UnsupportedPlatform);
39        }
40        VersionSpec::Explicit(want) => manifest
41            .versions
42            .iter()
43            .find(|v| &v.version == want)
44            .ok_or_else(|| FetcherError::VersionNotFound(want.clone()))?,
45    };
46
47    // Walk that entry's downloads.chrome for a matching platform.
48    let key = platform.as_cft_str();
49    let download = entry
50        .downloads
51        .chrome
52        .iter()
53        .find(|d| d.platform == key)
54        .ok_or(FetcherError::UnsupportedPlatform)?;
55
56    Ok((entry.version.clone(), download.url.clone()))
57}

59#[cfg(test)]
60#[allow(clippy::panic, clippy::unwrap_used)]
61mod tests {
62    use super::*;
63
64    fn fixture_manifest() -> KnownGoodVersionsResponse {
65        // Hand-built equivalent of the wiremock fixture in manifest.rs:
66        // one version with linux64 + mac-x64 downloads.
67        let json = r#"{
68            "versions": [
69                {
70                    "version": "120.0.6099.234",
71                    "revision": "1234",
72                    "downloads": {
73                        "chrome": [
74                            {"platform": "linux64", "url": "https://example.com/chrome-linux64.zip"},
75                            {"platform": "mac-x64", "url": "https://example.com/chrome-mac-x64.zip"}
76                        ]
77                    }
78                }
79            ]
80        }"#;
81        serde_json::from_str(json).unwrap()
82    }
83
84    #[tokio::test]
85    async fn latest_returns_last_entry_url_for_matching_platform() {
86        let manifest = fixture_manifest();
87        let (version, url) =
88            resolve_download_url(&manifest, &VersionSpec::Latest, Platform::LinuxX64)
89                .await
90                .unwrap();
91        assert_eq!(version, "120.0.6099.234");
92        assert_eq!(url, "https://example.com/chrome-linux64.zip");
93    }
94
95    #[tokio::test]
96    async fn explicit_unknown_version_returns_version_not_found() {
97        let manifest = fixture_manifest();
98        let err = resolve_download_url(
99            &manifest,
100            &VersionSpec::Explicit("999.0".to_string()),
101            Platform::LinuxX64,
102        )
103        .await
104        .unwrap_err();
105        match err {
106            FetcherError::VersionNotFound(v) => assert_eq!(v, "999.0"),
107            other => panic!("expected VersionNotFound, got {other:?}"),
108        }
109    }
110
111    #[tokio::test]
112    async fn beta_channel_returns_unsupported_platform() {
113        let manifest = fixture_manifest();
114        let err = resolve_download_url(
115            &manifest,
116            &VersionSpec::Channel(Channel::Beta),
117            Platform::LinuxX64,
118        )
119        .await
120        .unwrap_err();
121        assert!(matches!(err, FetcherError::UnsupportedPlatform));
122    }
123}
