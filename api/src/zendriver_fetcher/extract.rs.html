1//! Zip archive extraction.
2//!
3//! Unzips a Chrome for Testing archive into a destination directory.
4//! Chrome for Testing ships a `.zip` on every platform (Linux, macOS, Windows),
5//! so a single sync `zip::ZipArchive` walk wrapped in
6//! [`tokio::task::spawn_blocking`] handles all three.
7//!
8//! On Unix, executable bits from the archive's `unix_mode()` are preserved
9//! so the extracted Chrome binary stays runnable without a chmod pass.
10//!
11//! # Trust boundary
12//!
13//! The archives we accept come from the Chrome for Testing CDN (Google).
14//! We trust the *content* of those archives — they may include arbitrary
15//! files and executable bits because that's what running Chrome requires.
16//! We do **not** trust the archive's *paths*: a malicious or corrupt zip
17//! could ship absolute paths, `..` segments, or symlinks that try to
18//! write outside `dest_dir`. The extractor defends against those classes
19//! of attack:
20//!
21//! 1. `zip::read::ZipFile::enclosed_name()` rejects any entry whose
22//!    normalized path escapes the archive root (absolute, parent-traversal).
23//! 2. After joining with `dest_dir`, the resolved path is verified to still
24//!    sit under `dest_dir` — defense in depth against any future change
25//!    to `enclosed_name`'s semantics.
26//! 3. Symlink entries (detected via `unix_mode() & S_IFMT == S_IFLNK`) are
27//!    skipped — Chrome for Testing archives never ship symlinks, and
28//!    accepting them would let an attacker plant a follow-on write that
29//!    escapes the directory via symlink resolution.
30//! 4. Optional `expected_top_prefix` parameter requires every non-empty
31//!    entry to live under a single named top-level directory (e.g.
32//!    `chrome-linux64/`); enforced from the fetcher to lock the archive
33//!    to the CfT layout we expect.
34
35use std::io;
36use std::path::{Path, PathBuf};
37
38use crate::error::FetcherError;

40/// Unzips `archive_path` into `dest_dir`, preserving directory layout and
41/// (on Unix) executable bits from the archive.
42///
43/// `dest_dir` must already exist. If `expected_top_prefix` is `Some`, every
44/// entry must live under that single top-level directory (matches the CfT
45/// `chrome-<platform>/...` layout). Per-entry errors (corrupt zip, IO
46/// failure, unsafe path, missing prefix) surface as
47/// [`FetcherError::Extraction`] or [`FetcherError::Io`].
48pub(crate) async fn extract(
49    archive_path: &Path,
50    dest_dir: &Path,
51    expected_top_prefix: Option<&str>,
52) -> Result<(), FetcherError> {
53    let archive_path = archive_path.to_path_buf();
54    let dest_dir = dest_dir.to_path_buf();
55    let expected_top_prefix = expected_top_prefix.map(str::to_owned);
56
57    tokio::task::spawn_blocking(move || {
58        extract_blocking(&archive_path, &dest_dir, expected_top_prefix.as_deref())
59    })
60    .await
61    .map_err(|e| FetcherError::Extraction(format!("join error: {e}")))?
62}
63
64/// Synchronous unzip body — runs on a blocking thread.
65fn extract_blocking(
66    archive_path: &Path,
67    dest_dir: &Path,
68    expected_top_prefix: Option<&str>,
69) -> Result<(), FetcherError> {
70    let file = std::fs::File::open(archive_path)?;
71    let mut archive =
72        zip::ZipArchive::new(file).map_err(|e| FetcherError::Extraction(e.to_string()))?;
73
74    // Canonicalize dest_dir up front for the containment check below.
75    // `dest_dir` is created by the fetcher before invoking extract, so
76    // canonicalize is expected to succeed.
77    let dest_canonical = std::fs::canonicalize(dest_dir)
78        .map_err(|e| FetcherError::Extraction(format!("dest_dir canonicalize: {e}")))?;
79
80    for i in 0..archive.len() {
81        let mut entry = archive
82            .by_index(i)
83            .map_err(|e| FetcherError::Extraction(e.to_string()))?;
84
85        // Reject entries with unsafe paths (absolute / parent-traversal).
86        let Some(rel_path) = entry.enclosed_name() else {
87            return Err(FetcherError::Extraction(format!(
88                "zip entry has unsafe path: {}",
89                entry.name()
90            )));
91        };
92
93        // Enforce CfT top-level directory when the caller supplies one.
94        if let Some(prefix) = expected_top_prefix {
95            let top = rel_path.components().next().and_then(|c| match c {
96                std::path::Component::Normal(s) => s.to_str(),
97                _ => None,
98            });
99            if top != Some(prefix) {
100                return Err(FetcherError::Extraction(format!(
101                    "zip entry {:?} not under expected top-level {:?}",
102                    rel_path, prefix
103                )));
104            }
105        }
106
107        // Refuse symlink entries — CfT archives don't use them and they
108        // are the primary follow-on vector for zip-slip on Unix.
109        #[cfg(unix)]
110        if let Some(mode) = entry.unix_mode() {
111            const S_IFMT: u32 = 0o170_000;
112            const S_IFLNK: u32 = 0o120_000;
113            if mode & S_IFMT == S_IFLNK {
114                return Err(FetcherError::Extraction(format!(
115                    "zip entry {:?} is a symlink; refusing for safety",
116                    rel_path
117                )));
118            }
119        }
120
121        let out_path: PathBuf = dest_dir.join(&rel_path);
122
123        if entry.is_dir() {
124            std::fs::create_dir_all(&out_path)?;
125            assert_under_dest(&out_path, &dest_canonical)?;
126            continue;
127        }
128
129        if let Some(parent) = out_path.parent() {
130            std::fs::create_dir_all(parent)?;
131        }
132
133        // Defense in depth: verify the resolved path is still under
134        // `dest_canonical`. Catches future regressions in `enclosed_name`
135        // semantics or surprise from filesystem-level path resolution.
136        let mut probe_path = out_path.clone();
137        while !probe_path.exists() {
138            // Walk up until we hit a directory that exists, so canonicalize
139            // can succeed. The leaf file doesn't exist yet (we're about to
140            // create it), so canonicalize its parent chain.
141            if !probe_path.pop() {
142                break;
143            }
144        }
145        assert_under_dest(&probe_path, &dest_canonical)?;
146
147        let mut out_file = std::fs::File::create(&out_path)?;
148        io::copy(&mut entry, &mut out_file)?;
149
150        #[cfg(unix)]
151        if let Some(mode) = entry.unix_mode() {
152            use std::os::unix::fs::PermissionsExt as _;
153            std::fs::set_permissions(&out_path, std::fs::Permissions::from_mode(mode))?;
154        }
155    }
156
157    Ok(())
158}
159
160/// Verify `path`, after `canonicalize`, still has `dest_canonical` as a
161/// prefix. Used to enforce the dest_dir containment invariant.
162fn assert_under_dest(path: &Path, dest_canonical: &Path) -> Result<(), FetcherError> {
163    let resolved = std::fs::canonicalize(path)
164        .map_err(|e| FetcherError::Extraction(format!("path canonicalize: {e}")))?;
165    if !resolved.starts_with(dest_canonical) {
166        return Err(FetcherError::Extraction(format!(
167            "zip entry resolves to {:?} which is outside dest_dir {:?}",
168            resolved, dest_canonical
169        )));
170    }
171    Ok(())
172}

174#[cfg(test)]
175#[allow(clippy::panic, clippy::unwrap_used)]
176mod tests {
177    use super::*;
178    use std::io::Write as _;
179
180    #[tokio::test]
181    async fn extract_recovers_single_file_contents() {
182        let dir = tempfile::tempdir().unwrap();
183        let zip_path = dir.path().join("test.zip");
184        let dest_dir = dir.path().join("out");
185        std::fs::create_dir_all(&dest_dir).unwrap();
186
187        // Build an in-memory zip with one file "test.txt" -> "hello world".
188        let mut buf = std::io::Cursor::new(Vec::new());
189        {
190            let mut writer = zip::ZipWriter::new(&mut buf);
191            writer
192                .start_file("test.txt", zip::write::SimpleFileOptions::default())
193                .unwrap();
194            writer.write_all(b"hello world").unwrap();
195            writer.finish().unwrap();
196        }
197        std::fs::write(&zip_path, buf.into_inner()).unwrap();
198
199        extract(&zip_path, &dest_dir, None).await.unwrap();
200
201        let extracted = std::fs::read(dest_dir.join("test.txt")).unwrap();
202        assert_eq!(extracted, b"hello world");
203    }
204
205    /// `expected_top_prefix` enforces every entry lives under the named
206    /// top-level dir. A bare-file zip (no leading directory) is rejected.
207    #[tokio::test]
208    async fn extract_rejects_entries_outside_expected_top_prefix() {
209        let dir = tempfile::tempdir().unwrap();
210        let zip_path = dir.path().join("badroot.zip");
211        let dest_dir = dir.path().join("out");
212        std::fs::create_dir_all(&dest_dir).unwrap();
213
214        let mut buf = std::io::Cursor::new(Vec::new());
215        {
216            let mut writer = zip::ZipWriter::new(&mut buf);
217            writer
218                .start_file(
219                    "not-chrome/chrome",
220                    zip::write::SimpleFileOptions::default(),
221                )
222                .unwrap();
223            writer.write_all(b"x").unwrap();
224            writer.finish().unwrap();
225        }
226        std::fs::write(&zip_path, buf.into_inner()).unwrap();
227
228        let err = extract(&zip_path, &dest_dir, Some("chrome-linux64"))
229            .await
230            .unwrap_err();
231        let msg = err.to_string();
232        assert!(
233            msg.contains("expected top-level"),
234            "unexpected error: {msg}"
235        );
236    }

238    /// A symlink entry (Unix mode bits include `S_IFLNK`) is rejected even
239    /// when its name is a safe relative path. CfT archives never use them.
240    #[cfg(unix)]
241    #[tokio::test]
242    async fn extract_rejects_symlink_entries() {
243        use zip::write::SimpleFileOptions;
244        let dir = tempfile::tempdir().unwrap();
245        let zip_path = dir.path().join("symlink.zip");
246        let dest_dir = dir.path().join("out");
247        std::fs::create_dir_all(&dest_dir).unwrap();
248
249        let mut buf = std::io::Cursor::new(Vec::new());
250        {
251            let mut writer = zip::ZipWriter::new(&mut buf);
252            // 0o120_000 | 0o777 marks the entry as a symlink in the
253            // archive's Unix mode bits.
254            let opts = SimpleFileOptions::default().unix_permissions(0o777);
255            writer
256                .add_symlink("chrome-linux64/link", "../escape", opts)
257                .unwrap();
258            writer.finish().unwrap();
259        }
260        std::fs::write(&zip_path, buf.into_inner()).unwrap();
261
262        let err = extract(&zip_path, &dest_dir, Some("chrome-linux64"))
263            .await
264            .unwrap_err();
265        assert!(
266            err.to_string().contains("symlink"),
267            "unexpected error: {err}"
268        );
