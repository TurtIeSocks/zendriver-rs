1//! [`RequestExpectation`] + [`MatchedRequest`] + [`crate::Tab::expect_request`]
2//! (gated `expect`).
3//!
4//! Registers a one-shot subscription against `Network.requestWillBeSent` on
5//! a tab's session, filters by [`UrlMatcher`], and resolves with the first
6//! matching event via a `oneshot` channel. The subscriber task self-cancels
7//! after sending so each `expect_request` call is observably one-shot.
8//!
9//! `Network.enable` is already on for every Tab via the per-Tab in-flight
10//! network tracker, so this module does not re-enable the domain.
11

27/// Default outer timeout for a [`RequestExpectation`] — matches the rest of
28/// the high-level surface (`wait_for_load`, etc).
29const DEFAULT_EXPECT_TIMEOUT: Duration = Duration::from_secs(30);
30
31/// A network request observed via `Network.requestWillBeSent` that matched
32/// an [`UrlMatcher`] passed to [`crate::Tab::expect_request`].
33///
34/// Decoded from the CDP event payload. `post_data` is the raw request body
35/// bytes if Chrome reported any (POST/PUT/etc).
36#[derive(Debug, Clone)]
37pub struct MatchedRequest {
38    /// The request URL.
39    pub url: String,
40    /// HTTP method (GET, POST, ...).
41    pub method: String,
42    /// Request headers as reported by Chrome.
43    pub headers: HashMap<String, String>,
44    /// Request body if present. `None` for methods without a body.
45    pub post_data: Option<Vec<u8>>,
46    /// CDP `requestId` — same id reported by `Network.responseReceived`
47    /// for the matching response.
48    pub request_id: String,
49}
50
51/// Awaitable handle returned by [`crate::Tab::expect_request`]. Resolves
52/// with the first matched [`MatchedRequest`] or [`ZendriverError::Timeout`]
53/// if no match arrives within the configured timeout.
54///
55/// Implements [`Future`] directly — `.await` works without calling
56/// `.matched()`. The `.matched()` accessor exists for parity with the
57/// Playwright-style fluent API.
58#[derive(Debug)]
59pub struct RequestExpectation {
60    rx: oneshot::Receiver<MatchedRequest>,
61    timeout: Duration,
62    sleep: Option<Pin<Box<Sleep>>>,
63}
64
65impl RequestExpectation {
66    /// Override the default 30s timeout.
67    ///
68    /// # Examples
69    ///
70    /// ```no_run
71    /// # use std::time::Duration;
72    /// # async fn ex() -> zendriver::Result<()> {
73    /// # let browser = zendriver::Browser::builder().launch().await?;
74    /// # let tab = browser.main_tab();
75    /// let exp = tab.expect_request("/api/users").timeout(Duration::from_secs(5));
76    /// let req = exp.await?;
77    /// # let _ = req;
78    /// # Ok(()) }
79    /// ```
80    #[must_use]
81    pub fn timeout(mut self, dur: Duration) -> Self {
82        self.timeout = dur;
83        // Reset any already-armed sleep — the next poll will rebuild it
84        // with the new deadline.
85        self.sleep = None;
86        self
87    }
88
89    /// Playwright-style alias for `.await`.
90    ///
91    /// Functionally identical to awaiting the expectation directly.
92    ///
93    /// # Examples
94    ///
95    /// ```no_run
96    /// # async fn ex() -> zendriver::Result<()> {
97    /// # let browser = zendriver::Browser::builder().launch().await?;
98    /// # let tab = browser.main_tab();
99    /// let req = tab.expect_request("/api/users").matched().await?;
100    /// # let _ = req;
101    /// # Ok(()) }
102    /// ```
103    pub async fn matched(self) -> Result<MatchedRequest> {
104        self.await
105    }
106}
107
108impl Future for RequestExpectation {
109    type Output = Result<MatchedRequest>;
110
111    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
112        // Poll the oneshot first — if the subscriber already sent, return
113        // without ever arming the sleep timer.
114        match Pin::new(&mut self.rx).poll(cx) {
115            Poll::Ready(Ok(req)) => return Poll::Ready(Ok(req)),
116            Poll::Ready(Err(_)) => {
117                // Sender dropped without sending — subscriber task exited
118                // (transport closed). Surface as timeout: same observable
119                // shape (no event arrived), avoids inventing a new error.
120                return Poll::Ready(Err(ZendriverError::Timeout(self.timeout)));
121            }
122            Poll::Pending => {}
123        }
124
125        // Lazily arm the timer on first poll so `timeout(...)` overrides
126        // take effect.
127        let timeout = self.timeout;
128        let sleep = self
129            .sleep
130            .get_or_insert_with(|| Box::pin(tokio::time::sleep(timeout)));
131        match sleep.as_mut().poll(cx) {
132            Poll::Ready(()) => Poll::Ready(Err(ZendriverError::Timeout(timeout))),
133            Poll::Pending => Poll::Pending,
134        }
135    }
136}

138/// CDP `Network.requestWillBeSent` payload subset we care about. Field
139/// names follow the protocol (camelCase) via serde rename.
140#[derive(Debug, Deserialize)]
141struct RequestWillBeSentEvent {
142    #[serde(rename = "requestId")]
143    request_id: String,
144    request: RequestPayload,
145}
146
147#[derive(Debug, Deserialize)]
148struct RequestPayload {
149    url: String,
150    method: String,
151    #[serde(default)]
152    headers: HashMap<String, String>,
153    #[serde(rename = "postData", default)]
154    post_data: Option<String>,
155}
156
157/// Spawn a one-shot subscriber that watches `Network.requestWillBeSent` on
158/// `session`, sends the first event whose URL matches `matcher` through
159/// the `tx`, and exits. Subscription registers synchronously before the
160/// returned [`RequestExpectation`] is constructed so events fired
161/// immediately after a trigger action cannot slip past us.
162pub(crate) fn register(session: &SessionHandle, matcher: UrlMatcher) -> RequestExpectation {
163    let (tx, rx) = oneshot::channel();
164    let mut stream = session.subscribe::<RequestWillBeSentEvent>("Network.requestWillBeSent");
165    tokio::spawn(async move {
166        while let Some(ev) = stream.next().await {
167            if matcher.matches(&ev.request.url) {
168                let matched = MatchedRequest {
169                    url: ev.request.url,
170                    method: ev.request.method,
171                    headers: ev.request.headers,
172                    post_data: ev.request.post_data.map(String::into_bytes),
173                    request_id: ev.request_id,
174                };
175                // Send is fallible only if the receiver was dropped; in
176                // that case the caller no longer cares and we just exit.
177                let _ = tx.send(matched);
178                return;
179            }
180        }
181    });
182    RequestExpectation {
183        rx,
184        timeout: DEFAULT_EXPECT_TIMEOUT,
185        sleep: None,
186    }
187}

189#[cfg(test)]
190#[allow(clippy::panic, clippy::unwrap_used)]
191mod tests {
192    use super::*;
193    use serde_json::json;
194    use zendriver_transport::testing::MockConnection;
195
196    /// Register an expectation for `*/api/*`, emit a matching
197    /// `Network.requestWillBeSent`, and assert the expectation resolves
198    /// with the decoded [`MatchedRequest`].
199    #[tokio::test]
200    async fn expect_request_resolves_on_matching_event() {
201        let (mock, conn) = MockConnection::pair();
202        let session = SessionHandle::new(conn.clone(), "S1");
203
204        let expectation = register(&session, UrlMatcher::from("/api/"));
205
206        // Emit a non-matching event first to verify the matcher actually
207        // filters; the subscriber should ignore it.
208        mock.emit_event_for_session(
209            "Network.requestWillBeSent",
210            json!({
211                "requestId": "R0",
212                "request": {
213                    "url": "https://example.com/static/app.js",
214                    "method": "GET",
215                    "headers": {},
216                },
217            }),
218            "S1",
219        )
220        .await;
221
222        // Now emit the matching event.
223        mock.emit_event_for_session(
224            "Network.requestWillBeSent",
225            json!({
226                "requestId": "R1",
227                "request": {
228                    "url": "https://example.com/api/users",
229                    "method": "POST",
230                    "headers": {
231                        "content-type": "application/json",
232                    },
233                    "postData": "{\"name\":\"x\"}",
234                },
235            }),
236            "S1",
237        )
238        .await;
239
240        let matched = tokio::time::timeout(Duration::from_secs(2), expectation)
241            .await
242            .expect("expectation did not resolve within 2s")
243            .expect("expectation returned Err");
244
245        assert_eq!(matched.request_id, "R1");
246        assert_eq!(matched.url, "https://example.com/api/users");
247        assert_eq!(matched.method, "POST");
248        assert_eq!(
249            matched.headers.get("content-type").map(String::as_str),
250            Some("application/json"),
251        );
252        assert_eq!(
253            matched.post_data.as_deref(),
254            Some(b"{\"name\":\"x\"}".as_slice())
255        );
256
257        conn.shutdown();
258    }
259
260    /// Register an expectation with a 50ms timeout, emit no events, and
261    /// assert it returns [`ZendriverError::Timeout`] carrying that
262    /// duration.
263    #[tokio::test]
264    async fn expect_request_times_out() {
265        let (_mock, conn) = MockConnection::pair();
266        let session = SessionHandle::new(conn.clone(), "S1");
267
268        let expectation =
269            register(&session, UrlMatcher::from("/api/")).timeout(Duration::from_millis(50));
270
271        let res = expectation.await;
272        match res {
273            Err(ZendriverError::Timeout(d)) => {
274                assert_eq!(d, Duration::from_millis(50));
275            }
276            other => panic!("expected Timeout(50ms), got {other:?}"),
277        }
278
279        conn.shutdown();
280    }
