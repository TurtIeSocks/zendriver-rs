1//! Event expectation helpers (`expect_request` / `expect_response` /
2//! `expect_dialog` / `expect_download` / `expect_file_chooser`).
3//!
4//! Each helper registers a one-shot subscription on a Tab's CDP event stream
5//! and resolves with the first matching event. [`UrlMatcher`] is the shared
6//! pattern type used by request/response expectations.
7//!
8//! ```no_run
9//! # async fn ex() -> zendriver::Result<()> {
10//! # let browser = zendriver::Browser::builder().launch().await?;
11//! # let tab = browser.main_tab();
12//! // Pre-register before triggering the action that causes the request.
13//! let exp = tab.expect_response("/api/users");
14//! tab.find().css("button#load").one().await?.click().await?;
15//! let resp = exp.await?;
16//! let body = resp.body().await?;
17//! # let _ = body;
18//! # Ok(()) }
19//! ```

34/// Subscribe to `method`-shaped events scoped to `session`, decoded via
35/// `T`'s [`serde::de::DeserializeOwned`] impl. Backs every `expect_*`
36/// subscriber task ‚Äî riding [`Connection::subscribe_raw_accounted`]
37/// (rather than the plain [`SessionHandle::subscribe`]) so a delivery-loss
38/// boundary crossed while a wait is in progress is visible instead of
39/// silently vanishing.
40///
41/// [`Connection::subscribe_raw_accounted`]: zendriver_transport::Connection::subscribe_raw_accounted
42///
43/// Yields `Ok(T)` for every successfully-decoded `method` event on this
44/// session. Yields `Err(`[`ZendriverError::EventStreamIncomplete`]`)` on:
45///
46/// - [`AccountedRawEvent::Disconnected`] / [`AccountedRawEvent::Reconnected`]
47///   ‚Äî the watched stream is gone, so the caller can no longer prove the
48///   awaited condition didn't occur just before the boundary.
49/// - [`AccountedRawEvent::Lagged`] ‚Äî this subscriber missed frames it can't
50///   recover. There's no way to prove the awaited event wasn't among them,
51///   so a "no show" (`Timeout`) would be a possibly-wrong claim. Treated the
52///   same as an outright disconnect ‚Äî mirrors the `IdleLossPolicy::Strict`
53///   precedent in `Tab::wait_for_idle_opts`, which makes the identical call
54///   for the same reason.
55/// - A decode failure on an event that otherwise matches `method` +
56///   `session` ‚Äî the event we were waiting for arrived in a shape we can't
57///   parse, which leaves us just as unable to confirm the awaited condition
58///   as an outright loss would.
59///
60/// Events for any other method or session are silently skipped: they were
61/// never candidates for the awaited condition, so their absence from the
62/// output isn't a loss of anything this caller cares about.
63pub(crate) fn watch<T>(
64    session: &SessionHandle,
65    method: &'static str,
66) -> impl Stream<Item = Result<T>> + Send + Unpin + use<T>
67where
68    T: serde::de::DeserializeOwned + Send + 'static,
69{
70    let sid = session.session_id().to_string();
71    let raw = session.connection().subscribe_raw_accounted();
72    Box::pin(raw.filter_map(move |acc| {
73        let sid = sid.clone();
74        async move {
75            match acc {
76                AccountedRawEvent::Event { event, .. } => {
77                    if event.session_id.as_deref() == Some(sid.as_str()) && event.method == method {
78                        match serde_json::from_value::<T>(event.params) {
79                            Ok(v) => Some(Ok(v)),
80                            Err(_) => Some(Err(ZendriverError::EventStreamIncomplete)),
81                        }
82                    } else {
83                        None
84                    }
85                }
86                // coverage: `Reconnected` shares this arm with `Lagged` /
87                // `Disconnected`. Each boundary has its own end-to-end test
88                // (`expect_response_returns_event_stream_incomplete_on_{reconnect,disconnect}`,
89                // plus the `Lagged` capacity harness) ‚Äî if this arm is ever
90                // split (e.g. reconnect-specific retry semantics), keep those
91                // per-boundary tests so coverage doesn't silently regress.
92                AccountedRawEvent::Lagged { .. }
93                | AccountedRawEvent::Reconnected { .. }
94                | AccountedRawEvent::Disconnected { .. } => {
95                    Some(Err(ZendriverError::EventStreamIncomplete))
96                }
97            }
98        }
99    }))
