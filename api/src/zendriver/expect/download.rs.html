1//! [`DownloadExpectation`] + [`MatchedDownload`] + [`crate::Tab::expect_download`]
2//! (gated `expect`).
3//!
4//! Registers a one-shot subscription against `Page.downloadWillBegin` on a
5//! tab's session, resolves with the first download event, and exposes
6//! [`MatchedDownload::path`] / [`MatchedDownload::save_to`] for accessing the
7//! downloaded file once Chrome reports the transfer complete.
8//!
9//! Per-Tab setup (allocated lazily on the first [`crate::Tab::expect_download`]
10//! call):
11//!
12//! 1. A [`tempfile::TempDir`] under the OS temp dir to receive Chrome's
13//!    writes. Files land there named by their CDP `guid`.
14//! 2. `Browser.setDownloadBehavior { behavior: "allowAndName", downloadPath
15//!    }` dispatched at browser scope — Chrome routes downloads from this
16//!    target into the tempdir. `allowAndName` (vs `allow`) makes Chrome
17//!    write to `<downloadPath>/<guid>` rather than the suggested filename,
18//!    which avoids collisions when the same name is downloaded twice.
19//! 3. A long-running `Page.downloadProgress` subscriber that mutates each
20//!    matched download's [`DownloadState`] in place. Lives until the Tab is
21//!    dropped.
22//!
23//! The coordinator is reused across every `expect_download` call on the same
24//! tab — held in a [`tokio::sync::OnceCell`] on `TabInner` — so the
25//! per-target wiring is paid once.
26//!
27//! `Page.enable` is already on for every Tab via the P4 frame-lifecycle
28//! subscriber, so this module does not re-enable the domain.
29

48/// Default outer timeout for a [`DownloadExpectation`] — matches the rest of
49/// the high-level surface (`wait_for_load`, etc).
50const DEFAULT_EXPECT_TIMEOUT: Duration = Duration::from_secs(30);
51
52/// Terminal/in-progress lifecycle marker for a download. Updated on every
53/// `Page.downloadProgress` event by the per-Tab subscriber.
54#[derive(Debug, Clone, Copy, PartialEq, Eq)]
55pub enum DownloadProgressState {
56    /// Chrome is actively transferring bytes.
57    InProgress,
58    /// Transfer finished successfully; file at `download_dir/guid` is final.
59    Completed,
60    /// Transfer was canceled (user cancel, browser shutdown, etc).
61    Canceled,
62}
63
64impl DownloadProgressState {
65    fn from_cdp(s: &str) -> Self {
66        match s {
67            "completed" => Self::Completed,
68            "canceled" => Self::Canceled,
69            // CDP only emits the three above; treat anything else as
70            // still-receiving rather than inventing an Unknown variant.
71            _ => Self::InProgress,
72        }
73    }
74}

76/// Snapshot of a download's progress at a point in time. Held behind a
77/// [`tokio::sync::Mutex`] on [`MatchedDownload::state`] and mutated in place
78/// by the per-Tab `Page.downloadProgress` subscriber.
79#[derive(Debug, Clone, Copy)]
80pub struct DownloadState {
81    /// Bytes received so far.
82    pub received_bytes: u64,
83    /// Expected total bytes. `0` until Chrome reports a size (some downloads
84    /// don't carry a `Content-Length`).
85    pub total_bytes: u64,
86    /// Lifecycle marker.
87    pub state: DownloadProgressState,
88}
89
90impl Default for DownloadState {
91    fn default() -> Self {
92        Self {
93            received_bytes: 0,
94            total_bytes: 0,
95            state: DownloadProgressState::InProgress,
96        }
97    }
98}

100/// Type alias for the shared state cell used by both [`MatchedDownload`] and
101/// the per-Tab progress subscriber's routing map.
102pub(crate) type SharedDownloadState = Arc<tokio::sync::Mutex<DownloadState>>;

104/// Per-Tab download coordinator: owns the tempdir Chrome writes into, the
105/// routing map from CDP `guid` → shared [`DownloadState`] cells, and is held
106/// alive (via the OnceCell on `TabInner`) until the Tab is dropped.
107///
108/// Constructed lazily on the first [`crate::Tab::expect_download`] call via
109/// [`ensure_download_setup`]. The constructor:
110///
111/// - Allocates a [`TempDir`].
112/// - Dispatches `Browser.setDownloadBehavior { behavior: "allowAndName",
113///   downloadPath }` at browser scope so Chrome routes downloads from this
114///   target into the tempdir.
115/// - Spawns a long-running subscriber on `Page.downloadProgress` that walks
116///   the routing map on every event and mutates the matching
117///   [`DownloadState`] in place.
118#[derive(Debug)]
119pub(crate) struct DownloadCoordinator {
120    /// Backing tempdir. Held to keep the directory alive — drops on Tab
121    /// teardown clean up the temp files.
122    _tempdir: TempDir,
123    /// Resolved download path (a snapshot of `_tempdir.path()`). Cached so
124    /// [`MatchedDownload::path`] / [`MatchedDownload::save_to`] don't need
125    /// to walk the tempdir handle again.
126    download_dir: PathBuf,
127    /// Per-guid shared state cells. Inserted on `Page.downloadWillBegin` by
128    /// the expectation subscriber; read + mutated by the long-running
129    /// progress subscriber.
130    states: Arc<tokio::sync::Mutex<HashMap<String, SharedDownloadState>>>,
131}
132
133impl DownloadCoordinator {
134    fn download_dir(&self) -> &PathBuf {
135        &self.download_dir
136    }
137}

139/// Ensure the per-Tab download coordinator is set up, returning a clone of
140/// the shared `Arc`. Idempotent — once-cell initialization makes the
141/// `Browser.setDownloadBehavior` dispatch + tempdir allocation + progress
142/// subscriber spawn happen exactly once per Tab.
143///
144/// Called from `Tab::expect_download` on every invocation; only the first
145/// call performs work.
146pub(crate) async fn ensure_download_setup(
147    cell: &tokio::sync::OnceCell<Arc<DownloadCoordinator>>,
148    session: &SessionHandle,
149) -> Result<Arc<DownloadCoordinator>> {
150    let coord = cell
151        .get_or_try_init(|| async {
152            // Allocate a tempdir under the OS default. Naming with a
153            // `zendriver-downloads-` prefix makes manual inspection easier
154            // when debugging a failing test.
155            let tempdir = tempfile::Builder::new()
156                .prefix("zendriver-downloads-")
157                .tempdir()
158                .map_err(ZendriverError::Io)?;
159            let download_dir = tempdir.path().to_path_buf();
160
161            // Tell Chrome where to write. `allowAndName` writes files with
162            // their CDP `guid` as the on-disk name; that avoids
163            // suggested-filename collisions if two downloads share a name.
164            // Browser-scope dispatch (no session_id) so the policy applies
165            // to all targets — Chrome doesn't honor per-session
166            // `setDownloadBehavior` reliably across versions.
167            let _ = session
168                .connection()
169                .call_raw(
170                    "Browser.setDownloadBehavior",
171                    json!({
172                        "behavior": "allowAndName",
173                        "downloadPath": download_dir.to_string_lossy().to_string(),
174                    }),
175                    None,
176                )
177                .await?;
178
179            let states: Arc<tokio::sync::Mutex<HashMap<String, SharedDownloadState>>> =
180                Arc::new(tokio::sync::Mutex::new(HashMap::new()));
181
182            // Spawn the long-running progress subscriber. Lives until the
183            // Tab is dropped (which drops the OnceCell + this Arc, and the
184            // subscriber exits when the underlying stream closes).
185            let states_for_task = states.clone();
186            let mut progress_stream =
187                session.subscribe::<DownloadProgressEvent>("Page.downloadProgress");
188            tokio::spawn(async move {
189                while let Some(ev) = progress_stream.next().await {
190                    let map = states_for_task.lock().await;
191                    if let Some(cell) = map.get(&ev.guid) {
192                        let mut s = cell.lock().await;
193                        s.received_bytes = ev.received_bytes;
194                        s.total_bytes = ev.total_bytes;
195                        s.state = DownloadProgressState::from_cdp(&ev.state);
196                    }
197                }
198            });
199
200            Ok::<_, ZendriverError>(Arc::new(DownloadCoordinator {
201                _tempdir: tempdir,
202                download_dir,
203                states,
204            }))
205        })
206        .await?;
207    Ok(coord.clone())
208}

210/// A download observed via `Page.downloadWillBegin`.
211///
212/// The internal shared state is tracked in the per-Tab download coordinator's
213/// routing map; the long-running progress subscriber mutates it as
214/// `Page.downloadProgress` events arrive.
215///
216/// `Debug` is manually implemented since [`SessionHandle`] and
217/// [`tokio::sync::Mutex`] don't have useful default debug renderings.
218#[derive(Clone)]
219pub struct MatchedDownload {
220    /// Source URL Chrome started downloading from.
221    pub url: String,
222    /// Filename Chrome would have used in the user's downloads folder
223    /// (extracted from `Content-Disposition` / URL). Useful for callers
224    /// that want to preserve the original name in [`Self::save_to`].
225    pub suggested_filename: String,
226    /// CDP `guid` — Chrome uses this as the on-disk filename inside
227    /// [`Self::download_dir`] (because we configured `allowAndName`).
228    pub guid: String,
229    /// Shared progress state. Mutated in place by the per-Tab progress
230    /// subscriber; cloneable so the same handle can be observed concurrently.
231    pub state: SharedDownloadState,
232    /// Session this download fired on. Retained so future per-tab APIs
233    /// (e.g. `cancel`) can dispatch against the correct target — currently
234    /// unused but documented for stability.
235    pub session: SessionHandle,
236    /// Directory Chrome is writing to (the per-Tab tempdir). The final file
237    /// lives at `download_dir.join(guid)` once `state.state == Completed`.
238    pub download_dir: PathBuf,
239}
240
241impl std::fmt::Debug for MatchedDownload {
242    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
243        f.debug_struct("MatchedDownload")
244            .field("url", &self.url)
245            .field("suggested_filename", &self.suggested_filename)
246            .field("guid", &self.guid)
247            .field("download_dir", &self.download_dir)
248            .field("state", &"<Mutex<DownloadState>>")
249            .field("session", &"<SessionHandle>")
250            .finish()
251    }
252}

254impl MatchedDownload {
255    /// Path to the completed download on disk, or `None` if pending.
256    ///
257    /// Locks the internal lifecycle state; the path is only returned once
258    /// the download has completed. For canceled or in-progress downloads
259    /// this returns `None` — callers waiting on completion should use
260    /// [`Self::save_to`] which polls under the hood.
261    ///
262    /// # Examples
263    ///
264    /// ```no_run
265    /// # async fn ex() -> zendriver::Result<()> {
266    /// # let browser = zendriver::Browser::builder().launch().await?;
267    /// # let tab = browser.main_tab();
268    /// let exp = tab.expect_download().await?;
269    /// let dl = exp.await?;
270    /// if let Some(p) = dl.path().await {
271    ///     println!("downloaded to {}", p.display());
272    /// }
273    /// # Ok(()) }
274    /// ```
275    pub async fn path(&self) -> Option<PathBuf> {
276        let s = self.state.lock().await;
277        if s.state == DownloadProgressState::Completed {
278            Some(self.download_dir.join(&self.guid))
279        } else {
280            None
281        }
282    }
283
284    /// Wait for completion, then copy the bytes to `dest`.
285    ///
286    /// `dest` is interpreted as a full filename — join a directory +
287    /// [`Self::suggested_filename`] if you want to preserve Chrome's name.
288    ///
289    /// Polls the internal lifecycle state every 100ms. A 30s outer cap
290    /// protects against downloads that hang indefinitely without Chrome
291    /// reporting progress.
292    ///
293    /// # Errors
294    ///
295    /// Returns [`ZendriverError::Navigation`] if the download is canceled;
296    /// [`ZendriverError::Timeout`] if the 30s cap elapses;
297    /// [`ZendriverError::Io`] on file-copy failure.
298    ///
299    /// # Examples
300    ///
301    /// ```no_run
302    /// # use std::path::PathBuf;
303    /// # async fn ex() -> zendriver::Result<()> {
304    /// # let browser = zendriver::Browser::builder().launch().await?;
305    /// # let tab = browser.main_tab();
306    /// let exp = tab.expect_download().await?;
307    /// let dl = exp.await?;
308    /// dl.save_to(PathBuf::from("/tmp/downloaded.zip")).await?;
309    /// # Ok(()) }
310    /// ```
311    pub async fn save_to(self, dest: PathBuf) -> Result<()> {
312        let deadline = tokio::time::Instant::now() + Duration::from_secs(30);
313        loop {
314            let snapshot = *self.state.lock().await;
315            match snapshot.state {
316                DownloadProgressState::Completed => break,
317                DownloadProgressState::Canceled => {
318                    return Err(ZendriverError::Navigation(format!(
319                        "download {} canceled before save_to completed",
320                        self.guid
321                    )));
322                }
323                DownloadProgressState::InProgress => {
324                    if tokio::time::Instant::now() >= deadline {
325                        return Err(ZendriverError::Timeout(Duration::from_secs(30)));
326                    }
327                    tokio::time::sleep(Duration::from_millis(100)).await;
328                }
329            }
330        }
331        let src = self.download_dir.join(&self.guid);
332        tokio::fs::copy(&src, &dest)
333            .await
334            .map_err(ZendriverError::Io)?;
335        Ok(())
336    }
337}

339/// Awaitable handle returned by [`crate::Tab::expect_download`]. Resolves
340/// with the first matched [`MatchedDownload`] or [`ZendriverError::Timeout`]
341/// if no download begins within the configured timeout.
342///
343/// Implements [`Future`] directly — `.await` works without calling
344/// `.matched()`. The `.matched()` accessor exists for parity with the
345/// Playwright-style fluent API.
346#[derive(Debug)]
347pub struct DownloadExpectation {
348    rx: oneshot::Receiver<Result<MatchedDownload>>,
349    timeout: Duration,
350    sleep: Option<Pin<Box<Sleep>>>,
351}
352
353impl DownloadExpectation {
354    /// Override the default 30s timeout.
355    ///
356    /// # Examples
357    ///
358    /// ```no_run
359    /// # use std::time::Duration;
360    /// # async fn ex() -> zendriver::Result<()> {
361    /// # let browser = zendriver::Browser::builder().launch().await?;
362    /// # let tab = browser.main_tab();
363    /// let exp = tab.expect_download().await?.timeout(Duration::from_secs(60));
364    /// # let _ = exp;
365    /// # Ok(()) }
366    /// ```
367    #[must_use]
368    pub fn timeout(mut self, dur: Duration) -> Self {
369        self.timeout = dur;
370        // Reset any already-armed sleep — the next poll will rebuild it
371        // with the new deadline.
372        self.sleep = None;
373        self
374    }
375
376    /// Playwright-style alias for `.await`.
377    ///
378    /// Functionally identical to awaiting the expectation directly.
379    ///
380    /// # Examples
381    ///
382    /// ```no_run
383    /// # async fn ex() -> zendriver::Result<()> {
384    /// # let browser = zendriver::Browser::builder().launch().await?;
385    /// # let tab = browser.main_tab();
386    /// let dl = tab.expect_download().await?.matched().await?;
387    /// # let _ = dl;
388    /// # Ok(()) }
389    /// ```
390    pub async fn matched(self) -> Result<MatchedDownload> {
391        self.await
392    }
393}
394
395impl Future for DownloadExpectation {
396    type Output = Result<MatchedDownload>;
397
398    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
399        // Poll the oneshot first — if the subscriber already sent, return
400        // without ever arming the sleep timer.
401        match Pin::new(&mut self.rx).poll(cx) {
402            Poll::Ready(Ok(Ok(d))) => return Poll::Ready(Ok(d)),
403            // Subscriber observed a delivery-loss boundary (disconnect,
404            // reconnect, lag, or a same-method decode failure) before a
405            // download began — see `crate::expect::watch`. We can no
406            // longer prove a download didn't begin just before the
407            // boundary, so reporting a plain timeout would be a lie.
408            Poll::Ready(Ok(Err(e))) => return Poll::Ready(Err(e)),
409            Poll::Ready(Err(_)) => {
410                // Sender dropped without sending anything — the accounted
411                // stream itself ended (e.g. a clean `Connection::shutdown`,
412                // which never emits a `Disconnected` boundary) with no
413                // boundary and no match observed. Surface as timeout: same
414                // observable shape as a genuine no-show.
415                return Poll::Ready(Err(ZendriverError::Timeout(self.timeout)));
416            }
417            Poll::Pending => {}
418        }
419
420        let timeout = self.timeout;
421        let sleep = self
422            .sleep
423            .get_or_insert_with(|| Box::pin(tokio::time::sleep(timeout)));
424        match sleep.as_mut().poll(cx) {
425            Poll::Ready(()) => Poll::Ready(Err(ZendriverError::Timeout(timeout))),
426            Poll::Pending => Poll::Pending,
427        }
428    }
429}

431/// CDP `Page.downloadWillBegin` payload. Field names follow the protocol
432/// (camelCase) via serde rename.
433#[derive(Debug, Deserialize)]
434struct DownloadWillBeginEvent {
435    guid: String,
436    url: String,
437    #[serde(rename = "suggestedFilename", default)]
438    suggested_filename: String,
439}

441/// CDP `Page.downloadProgress` payload. Drives the long-running subscriber
442/// in [`DownloadCoordinator`].
443#[derive(Debug, Deserialize)]
444struct DownloadProgressEvent {
445    guid: String,
446    #[serde(rename = "receivedBytes", default)]
447    received_bytes: u64,
448    #[serde(rename = "totalBytes", default)]
449    total_bytes: u64,
450    state: String,
451}

453/// Spawn a one-shot subscriber for `Page.downloadWillBegin` on `session`,
454/// wire the matched download's state cell into the coordinator's routing
455/// map, and return a [`DownloadExpectation`].
456///
457/// Subscription registers synchronously before the returned expectation is
458/// constructed so downloads triggered immediately after a click cannot slip
459/// past us.
460pub(crate) fn register(
461    session: &SessionHandle,
462    coordinator: Arc<DownloadCoordinator>,
463) -> DownloadExpectation {
464    let (tx, rx) = oneshot::channel();
465    let mut stream =
466        crate::expect::watch::<DownloadWillBeginEvent>(session, "Page.downloadWillBegin");
467    let session_for_match = session.clone();
468    tokio::spawn(async move {
469        if let Some(res) = stream.next().await {
470            let outcome = match res {
471                Ok(ev) => {
472                    // Register the per-download state cell BEFORE handing
473                    // off the MatchedDownload. The progress subscriber
474                    // walks this map on every Page.downloadProgress event;
475                    // missing the registration would drop progress for
476                    // races where progress fires before the caller awaits
477                    // the expectation.
478                    let state: SharedDownloadState =
479                        Arc::new(tokio::sync::Mutex::new(DownloadState::default()));
480                    coordinator
481                        .states
482                        .lock()
483                        .await
484                        .insert(ev.guid.clone(), state.clone());
485
486                    Ok(MatchedDownload {
487                        url: ev.url,
488                        suggested_filename: ev.suggested_filename,
489                        guid: ev.guid,
490                        state,
491                        session: session_for_match,
492                        download_dir: coordinator.download_dir().clone(),
493                    })
494                }
495                Err(e) => Err(e),
496            };
497            // Send is fallible only if the receiver was dropped; in that
498            // case the caller no longer cares and we just exit.
499            let _ = tx.send(outcome);
500        }
501    });
502    DownloadExpectation {
503        rx,
504        timeout: DEFAULT_EXPECT_TIMEOUT,
505        sleep: None,
506    }
507}

516    /// Drive a `register(...)` end-to-end against a [`MockConnection`]:
517    ///
518    /// 1. Build a coordinator manually (skips the `Browser.setDownloadBehavior`
519    ///    + tempdir + subscriber spin-up — those are exercised separately
520    ///    via the `Tab::expect_download` integration path).
521    /// 2. Register the expectation, emit a `Page.downloadWillBegin`, assert
522    ///    the matched download carries the correct `suggested_filename` /
523    ///    `guid` / `url`.
524    /// 3. Assert `path()` returns `None` before any `Page.downloadProgress`
525    ///    arrives (state stays `InProgress`).
526    #[tokio::test]
527    async fn expect_download_resolves_on_download_will_begin_and_path_is_none_until_completed() {
528        let (mock, conn) = MockConnection::pair();
529        let session = SessionHandle::new(conn.clone(), "S1");
530
531        // Hand-build a coordinator that mirrors the production shape but
532        // doesn't dispatch any CDP setup — keeps the test focused on the
533        // event → MatchedDownload routing.
534        let tempdir = tempfile::tempdir().unwrap();
535        let download_dir = tempdir.path().to_path_buf();
536        let coord = Arc::new(DownloadCoordinator {
537            _tempdir: tempdir,
538            download_dir: download_dir.clone(),
539            states: Arc::new(tokio::sync::Mutex::new(HashMap::new())),
540        });
541
542        let expectation = register(&session, coord);
543
544        mock.emit_event_for_session(
545            "Page.downloadWillBegin",
546            json!({
547                "frameId": "F0",
548                "guid": "GUID-123",
549                "url": "https://example.com/report.pdf",
550                "suggestedFilename": "report.pdf",
551            }),
552            "S1",
553        )
554        .await;
555
556        let matched = tokio::time::timeout(Duration::from_secs(2), expectation)
557            .await
558            .expect("expectation did not resolve within 2s")
559            .expect("expectation returned Err");
560
561        assert_eq!(matched.guid, "GUID-123");
562        assert_eq!(matched.url, "https://example.com/report.pdf");
563        assert_eq!(matched.suggested_filename, "report.pdf");
564        assert_eq!(matched.download_dir, download_dir);
565
566        // No Page.downloadProgress yet → state stays InProgress → path()
567        // must return None even though download_dir/guid would resolve to
568        // a syntactically valid path.
569        assert!(
570            matched.path().await.is_none(),
571            "path() must be None until state == Completed",
572        );
573
574        conn.shutdown();
575    }
