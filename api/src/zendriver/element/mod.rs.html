1//! Handle to a DOM node via CDP `RemoteObjectId` / `BackendNodeId`.
2//!
3//! [`Element`] is the result of a [`crate::Tab::find`] /
4//! [`crate::Tab::find_all`] / [`crate::Element::find`] query (or any of the
5//! traversal helpers). Actions sit on submodules:
6//!
7//! - [`mod@actions`] — click / hover / focus / scroll / set_value / clear /
8//!   upload_files.
9//! - [`mod@input`] — type_text / type_text_fast / press / press_with.
10//! - [`mod@reads`] — attribute access, innerText, outerHTML, bounding box,
11//!   visibility / enabled state.
12//! - [`mod@traversal`] — parent / nth_child.
13//! - [`mod@isolated_eval`] — true isolated-world `evaluate` (with the
14//!   element bound as `el`).
15//! - [`mod@screenshot`] — element-scoped PNG capture.
16//! - [`mod@refresh`] — auto-refresh-on-stale-handle support.
17//!
18//! ```no_run
19//! # async fn ex() -> zendriver::Result<()> {
20//! # let browser = zendriver::Browser::builder().launch().await?;
21//! # let tab = browser.main_tab();
22//! tab.goto("https://example.com").await?;
23//! let h1 = tab.find().css("h1").one().await?;
24//! assert_eq!(h1.inner_text().await?, "Example Domain");
25//! # Ok(()) }
26//! ```

46/// Handle to a DOM node in a [`Tab`].
47///
48/// `Element` is `Clone` (cheap — wraps an `Arc`) and `Send + Sync`. Methods
49/// are grouped into thematic submodules — see the [module-level docs](self)
50/// for the map.
51///
52/// Get one via [`Tab::find`](crate::Tab::find) / [`Tab::find_all`](crate::Tab::find_all),
53/// frame queries, or element traversal helpers.
54#[derive(Clone, Debug)]
55pub struct Element {
56    pub(crate) inner: Arc<ElementInner>,
57}
58
59#[derive(Debug)]
60pub(crate) struct ElementInner {
61    pub(crate) tab: Tab,
62    /// `None` once the element has been observed stale; refilled by
63    /// `Element::refresh` (T17). Reads + actions lock briefly to clone
64    /// the inner value, then proceed without holding the lock across
65    /// `.await` on the CDP session.
66    pub(crate) backend_node_id: Mutex<Option<i64>>,
67    pub(crate) remote_object_id: Mutex<Option<String>>,
68    /// How this element was first obtained — drives `Element::refresh`'s
69    /// re-resolution path (`element::refresh::resolve_origin`).
70    pub(crate) origin: ElementOrigin,
71}
72
73/// How an `Element` was obtained. Drives `Element::refresh`: a
74/// `Query`-origin element re-runs its selector against its original
75/// scope; a `Traversal`-origin element re-traverses from its parent
76/// (which itself may need refreshing recursively); an `Evaluation`
77/// origin has no way to re-resolve and surfaces `NotRefreshable`.
78#[derive(Debug, Clone)]
79pub(crate) enum ElementOrigin {
80    Query {
81        scope_kind: ScopeKind,
82        selector: SelectorKind,
83        nth: usize,
84    },
85    Traversal {
86        parent: Box<ElementOrigin>,
87        kind: TraversalKind,
88    },
89    /// Returned from a raw JS expression (e.g. `Tab::evaluate` that
90    /// yields a node handle). No selector to replay → not refreshable.
91    Evaluation,
92}
93
94/// The root context against which a `Query` origin's selector was
95/// originally resolved. Kept coarse — we only need to know "tab vs
96/// subtree" to decide where refresh should run. Re-resolving an
97/// element-subtree origin against a stale parent is not yet supported
98/// (see `ElementOrigin::Query { scope_kind: ElementSubtree, .. }`'s
99/// `NotRefreshable` arm in `element::refresh`).
100#[derive(Debug, Clone, Copy, PartialEq, Eq)]
101pub(crate) enum ScopeKind {
102    TabMain,
103    ElementSubtree,
104}
105
106/// The traversal step that produced a `Traversal`-origin element from
107/// its parent. Covers `Parent` + `NthChild`; richer relationships
108/// (sibling indices, etc.) can extend the enum without churn.
109#[derive(Debug, Clone, Copy, PartialEq, Eq)]
110pub(crate) enum TraversalKind {
111    Parent,
112    NthChild(usize),
113}

115impl Element {
116    /// Construct an `Element` whose origin is a tracked query against
117    /// `scope`. T17's `Element::refresh` re-runs `selector` against
118    /// that scope and re-picks `nth` to recover from stale handles.
119    pub(crate) fn synthesize_query(
120        r: RemoteRef,
121        scope: &QueryScope<'_>,
122        selector: &SelectorKind,
123        nth: usize,
124    ) -> Self {
125        let scope_kind = match scope {
126            // Frame queries are document-level within a frame's own
127            // session — refresh-wise they look like TabMain (re-resolve
128            // the selector against the frame's document root) rather
129            // than an element-subtree walk. The dedicated FrameMain
130            // variant lands when T17 wires Frame-aware refresh.
131            QueryScope::Tab(_) | QueryScope::Frame(_) => ScopeKind::TabMain,
132            QueryScope::Element(_) => ScopeKind::ElementSubtree,
133        };
134        Self {
135            inner: Arc::new(ElementInner {
136                tab: scope.synthesize_tab(),
137                backend_node_id: Mutex::new(Some(r.backend_node_id)),
138                remote_object_id: Mutex::new(Some(r.remote_object_id)),
139                origin: ElementOrigin::Query {
140                    scope_kind,
141                    selector: selector.clone(),
142                    nth,
143                },
144            }),
145        }
146    }
147
148    /// Construct an `Element` returned from a JS expression (e.g. a
149    /// `Runtime.evaluate` that yielded a node handle, or a predicate
150    /// query's match — see `Resolver::synthesize` in `query::mod`). No
151    /// selector to replay → `Element::refresh` errors with
152    /// `NotRefreshable` for elements built this way.
153    pub(crate) fn from_jsret(tab: Tab, backend_node_id: i64, remote_object_id: String) -> Self {
154        Self {
155            inner: Arc::new(ElementInner {
156                tab,
157                backend_node_id: Mutex::new(Some(backend_node_id)),
158                remote_object_id: Mutex::new(Some(remote_object_id)),
159                origin: ElementOrigin::Evaluation,
160            }),
161        }
162    }
163
164    /// Construct an `Element` produced by traversing from `parent_origin`
165    /// via `kind` (e.g. `Parent` or `NthChild(i)`). P3 stores the origin
166    /// for completeness; full chain-refresh lands in P4 (today, T17's
167    /// `refresh` returns `NotRefreshable` for `Traversal` origins).
168    pub(crate) fn synthesize_traversal(
169        tab: Tab,
170        backend_node_id: i64,
171        remote_object_id: String,
172        parent_origin: ElementOrigin,
173        kind: TraversalKind,
174    ) -> Self {
175        Self {
176            inner: Arc::new(ElementInner {
177                tab,
178                backend_node_id: Mutex::new(Some(backend_node_id)),
179                remote_object_id: Mutex::new(Some(remote_object_id)),
180                origin: ElementOrigin::Traversal {
181                    parent: Box::new(parent_origin),
182                    kind,
183                },
184            }),
185        }
186    }
187
188    /// The parent [`Tab`] this element was queried from.
189    ///
190    /// # Examples
191    ///
192    /// ```no_run
193    /// # async fn ex() -> zendriver::Result<()> {
194    /// # let browser = zendriver::Browser::builder().launch().await?;
195    /// # let tab = browser.main_tab();
196    /// let el = tab.find().css("button").one().await?;
197    /// let _: &zendriver::Tab = el.tab();
198    /// # Ok(()) }
199    /// ```
200    #[must_use]
201    pub fn tab(&self) -> &Tab {
202        &self.inner.tab
203    }
204
205    /// Lock + clone the current `remote_object_id`, erroring with
206    /// `ElementStale` if it has been cleared (which T17's refresh path
207    /// does between a stale-error observation and the re-resolve).
208    /// Used everywhere a CDP call needs the raw object id.
209    pub(crate) async fn remote_object_id_cloned(&self) -> Result<String> {
210        self.inner
211            .remote_object_id
212            .lock()
213            .await
214            .clone()
215            .ok_or(ZendriverError::ElementStale)
216    }
217
218    /// Lock + clone the current `backend_node_id`, erroring with
219    /// `ElementStale` if it has been cleared. Symmetric with
220    /// `remote_object_id_cloned`; used by DOM-domain calls keyed by
221    /// backend id (e.g. `DOM.setFileInputFiles`, `DOM.getBoxModel`).
222    pub(crate) async fn backend_node_id_cloned(&self) -> Result<i64> {
223        self.inner
224            .backend_node_id
225            .lock()
226            .await
227            .as_ref()
228            .copied()
229            .ok_or(ZendriverError::ElementStale)
230    }
231
232    /// Call a JS function on this element's remote object. The function
233    /// signature MUST take exactly one parameter (the element); use
234    /// `function(el){ ... }`.
235    pub(crate) async fn call_on(&self, function: &str, args: Value) -> Result<Value> {
236        let object_id = self.remote_object_id_cloned().await?;
237        let res = self
238            .inner
239            .tab
240            .call(
241                "Runtime.callFunctionOn",
242                json!({
243                    "objectId": object_id,
244                    "functionDeclaration": function,
245                    "arguments": args,
246                    "returnByValue": true,
247                    "awaitPromise": true,
248                }),
249            )
250            .await?;
251        if let Some(details) = res.get("exceptionDetails") {
252            let msg = details
253                .get("exception")
254                .and_then(|e| e.get("description"))
255                .and_then(|d| d.as_str())
256                .unwrap_or("unknown")
257                .to_string();
258            return Err(ZendriverError::JsException(msg));
259        }
260        Ok(res["result"].clone())
261    }
262
263    /// Invoke a JS function in the main world with this element bound as
264    /// the first positional argument. Accepts a function declaration whose
265    /// first parameter is the element handle (`function(el, ...rest){...}`)
266    /// and an `args` JSON array of additional `Runtime.callFunctionOn`
267    /// argument descriptors that follow the element. Returns the raw
268    /// `result` RemoteObject (caller picks `value` if `returnByValue`).
269    ///
270    /// Locks the remote-object Mutex once at the top, then routes
271    /// through `call_on` (which re-locks once more). The double-lock is
272    /// cheap — `tokio::sync::Mutex` is uncontended in the common case
273    /// and the guard is dropped before any `.await`.
274    pub(crate) async fn call_on_main(&self, function: &str, args: Value) -> Result<Value> {
275        let object_id = self.remote_object_id_cloned().await?;
276        let mut full_args = vec![json!({ "objectId": object_id })];
277        if let Some(extra) = args.as_array() {
278            full_args.extend(extra.iter().cloned());
279        }
280        self.call_on(function, Value::Array(full_args)).await
281    }
282
283    /// Evaluate a JS expression in the main world with `el` bound to this
284    /// element handle.
285    ///
286    /// Uses `Runtime.callFunctionOn` against the element's remote object,
287    /// which lives in whatever world it was created in (main world if found
288    /// via `document.querySelector`).
289    ///
290    /// For stealth-safe isolated-world evaluation, see [`Element::evaluate`].
291    ///
292    /// # Errors
293    ///
294    /// Returns [`ZendriverError::JsException`] when the expression raises;
295    /// [`ZendriverError::Serde`] when the result cannot be decoded into `T`.
296    ///
297    /// # Examples
298    ///
299    /// ```no_run
300    /// # async fn ex() -> zendriver::Result<()> {
301    /// # let browser = zendriver::Browser::builder().launch().await?;
302    /// # let tab = browser.main_tab();
303    /// let el = tab.find().css("input").one().await?;
304    /// let value: String = el.evaluate_main("el.value").await?;
305    /// # let _ = value;
306    /// # Ok(()) }
307    /// ```
308    pub async fn evaluate_main<T: DeserializeOwned>(&self, js: impl AsRef<str>) -> Result<T> {
309        let function = format!("function(el){{ return ({}) }}", js.as_ref());
310        let result = self.call_on_main(&function, json!([])).await?;
311        let value = result.get("value").cloned().unwrap_or(Value::Null);
312        serde_json::from_value(value).map_err(ZendriverError::Serde)
313    }
314}

326#[allow(clippy::panic, clippy::unwrap_used)]
327mod tests {
328    use super::*;
329    use zendriver_transport::SessionHandle;
330    use zendriver_transport::testing::MockConnection;

332    #[tokio::test]
333    async fn from_jsret_yields_evaluation_origin() {
334        let (_mock, conn) = MockConnection::pair();
335        let sess = SessionHandle::new(conn.clone(), "S1");
336        let tab = Tab::new_for_test(sess);
337        let el = Element::from_jsret(tab, 7, "R7".to_string());
338        assert!(matches!(el.inner.origin, ElementOrigin::Evaluation));
339        conn.shutdown();
340    }
341
342    #[tokio::test]
343    async fn remote_object_id_cloned_errors_after_clear() {
344        let (_mock, conn) = MockConnection::pair();
345        let sess = SessionHandle::new(conn.clone(), "S1");
346        let tab = Tab::new_for_test(sess);
347        let el = Element::from_jsret(tab, 1, "R1".to_string());
348
349        // Initially OK.
350        assert_eq!(el.remote_object_id_cloned().await.unwrap(), "R1");
351
352        // Clear → simulates the T17 refresh path mid-flight.
353        *el.inner.remote_object_id.lock().await = None;
354        let err = el.remote_object_id_cloned().await.unwrap_err();
355        assert!(matches!(err, ZendriverError::ElementStale));
356        conn.shutdown();
357    }
