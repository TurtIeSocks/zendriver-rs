1//! [`Element`] actions: `click` / `hover` / `focus` / `scroll_into_view` /
2//! `set_value` / `set_text` / `clear` / `upload_files` / `mouse_drag` /
3//! `flash` / `highlight_overlay`.
4//!
5//! Each action wraps its CDP dispatch sequence in an internal "refresh on
6//! stale" wrapper so a stale handle (post-navigation, post-React-rerender)
7//! transparently re-resolves once and retries.
8//!
9//! `hover` / `hover_fast`:
10//!   1. `scroll_into_view` — bring the element into the viewport so its
11//!      bbox center is a real, dispatchable coordinate.
12//!   2. Actionability gate (`visible + stable + receives_pointer`) — avoids
13//!      mid-transition hover races and overlay occlusion. Pointer events
14//!      are dispatched at the geometric center, so the element must be the
15//!      actual hit-test target there. `enabled` is left off — hover doesn't
16//!      activate the element, so disabled controls still accept mouseover.
17//!   3. Compute bbox center (`x + width / 2`, `y + height / 2`).
18//!   4. Dispatch the mouse-move via the shared [`crate::input::InputController`].
19//!      `hover` uses a realistic Bezier path; `hover_fast` uses a single
20//!      teleport dispatch for test/automation paths.
21//!
22//! `focus`: actionability gate (visible + enabled — no pointer or stability
23//! requirement; focus routes through the focused element, not the cursor's
24//! position), then `el.focus()` in the main world.
25//!
26//! `scroll_into_view`: no actionability gate (this *is* the visibility
27//! prereq other actions wait for). Calls `el.scrollIntoView({ block:
28//! 'center', behavior: 'instant' })`. `block: 'center'` matches Playwright
29//! (avoids sticky headers/footers obscuring the element after the scroll);
30//! `behavior: 'instant'` skips animation so the post-scroll bbox is final
31//! by the time the next CDP call runs.

44/// Default deadline for the actionability gate before each action. Matches
45/// the value the spec calls out for P3; per-call override land in P4 when
46/// the per-action options structs grow.
47const DEFAULT_ACTIONABILITY_TIMEOUT: Duration = Duration::from_secs(5);

49/// JS body that assigns `arguments[1].value` through the *native* prototype
50/// value-setter and fires bubbled `input` + `change` events.
51///
52/// React installs a per-instance `_valueTracker` whose own `value` setter
53/// shadows the prototype's. A direct `this.value = v` goes through that
54/// tracker, which updates its cached value and the DOM together — so the
55/// `input` event it then sees reports "no change" and React skips `onChange`,
56/// reverting the field on the next render. Calling the *prototype's*
57/// `value` setter (resolved via `Object.getOwnPropertyDescriptor`) bypasses
58/// the tracker, leaving its cached value stale so React fires `onChange`
59/// correctly. Falls back to a plain assignment when the descriptor has no
60/// setter (non-input elements). Resolves the `<textarea>` prototype for
61/// textareas and the `<input>` prototype otherwise.
62const NATIVE_VALUE_SETTER_JS: &str = "function(el, v){ \
63        const proto = this instanceof HTMLTextAreaElement \
64            ? HTMLTextAreaElement.prototype \
65            : HTMLInputElement.prototype; \
66        const d = Object.getOwnPropertyDescriptor(proto, 'value'); \
67        if (d && d.set) { d.set.call(this, v); } else { this.value = v; } \
68        this.dispatchEvent(new Event('input', {bubbles: true})); \
69        this.dispatchEvent(new Event('change', {bubbles: true})); \
70    }";

72/// Extra Backspace presses [`Element::clear_by_deleting`] issues beyond the

82/// Per-call knobs for [`Element::click_with`].
83///
84/// `Default` matches the behavior of [`Element::click`]: a left, single,
85/// realistic click at the element's bbox center with no modifiers held and
86/// full actionability gating. Override fields individually for richer
87/// dispatches (right-click, modifier-held click, raw teleport, etc.).
88///
89/// # Examples
90///
91/// ```no_run
92/// # async fn ex() -> zendriver::Result<()> {
93/// use zendriver::{ClickOptions, MouseButton};
94/// # let browser = zendriver::Browser::builder().launch().await?;
95/// # let tab = browser.main_tab();
96/// let el = tab.find().css("button").one().await?;
97/// el.click_with(ClickOptions {
98///     button: MouseButton::Right,
99///     click_count: 2,
100///     ..Default::default()
101/// }).await?;
102/// # Ok(()) }
103/// ```
104#[derive(Debug, Clone, Copy)]
105pub struct ClickOptions {
106    /// Which mouse button to dispatch. [`MouseButton::Left`] by default.
107    pub button: MouseButton,
108    /// Modifier keys held during the dispatch. Empty by default.
109    pub modifiers: KeyModifiers,
110    /// `clickCount` for the CDP dispatch. `1` by default; set `2` for a
111    /// double-click in a single `click_with` call.
112    pub click_count: u32,
113    /// Skip the actionability gate when true. Use sparingly — bypasses the
114    /// visibility/stability/pointer checks. Mirrors Playwright's
115    /// `force: true`.
116    pub force: bool,
117    /// Bezier-interpolated cursor path (`true`) vs single teleport
118    /// dispatch (`false`). `true` by default; the `click_fast` shortcut
119    /// flips this for deterministic test paths.
120    pub realistic: bool,
121    /// Click position relative to the element's bbox top-left
122    /// (`(dx, dy)`). `None` clicks at the bbox center.
123    pub position: Option<(f64, f64)>,
124}
125
126impl Default for ClickOptions {
127    fn default() -> Self {
128        Self {
129            button: MouseButton::Left,
130            modifiers: KeyModifiers::empty(),
131            click_count: 1,
132            force: false,
133            realistic: true,
134            position: None,
135        }
136    }
137}

194    /// Click this element with explicit [`ClickOptions`].
195    ///
196    /// See module docs for the dispatch sequence — same shape as `hover`
197    /// (scroll → gate → bbox math → pointer dispatch) but emits the
198    /// `mousePressed` + `mouseReleased` pair after the cursor arrives.
199    /// `opts.force` skips the actionability gate; `opts.position` shifts
200    /// the click point off bbox-center.
201    ///
202    /// # Examples
203    ///
204    /// ```no_run
205    /// # use zendriver::{ClickOptions, MouseButton};
206    /// # async fn ex() -> zendriver::Result<()> {
207    /// # let browser = zendriver::Browser::builder().launch().await?;
208    /// # let tab = browser.main_tab();
209    /// let row = tab.find().css("tr.contact").one().await?;
210    /// row.click_with(ClickOptions {
211    ///     button: MouseButton::Right,
212    ///     ..Default::default()
213    /// }).await?;
214    /// # Ok(()) }
215    /// ```
216    pub async fn click_with(&self, opts: ClickOptions) -> Result<()> {
217        self.with_refresh(|| async move {
218            self.scroll_into_view().await?;
219            if !opts.force {
220                actionability::wait_actionable(
221                    self,
222                    ActionabilityCheck::FULL,
223                    DEFAULT_ACTIONABILITY_TIMEOUT,
224                )
225                .await?;
226            }
227            let bbox = self
228                .bounding_box()
229                .await?
230                .ok_or_else(|| ZendriverError::Navigation("element has no bounding box".into()))?;
231            let (tx, ty) = match opts.position {
232                Some((dx, dy)) => (bbox.x + dx, bbox.y + dy),
233                None => (bbox.x + bbox.width / 2.0, bbox.y + bbox.height / 2.0),
234            };
235            let input = self.inner.tab.input().clone();
236            mouse::click_at(
237                &input,
238                &self.inner.tab,
239                tx,
240                ty,
241                opts.button,
242                opts.click_count,
243                opts.realistic,
244            )
245            .await
246        })
247        .await
248    }

250    /// Hover the cursor over this element's bbox center.
251    ///
252    /// Uses a realistic Bezier-interpolated mouse path. See module docs for
253    /// the full sequence (`scroll_into_view` → actionability gate → bbox
254    /// center → dispatch). Use [`Element::hover_fast`] when the cursor path
255    /// doesn't matter.
256    ///
257    /// # Examples
258    ///
259    /// ```no_run
260    /// # async fn ex() -> zendriver::Result<()> {
261    /// # let browser = zendriver::Browser::builder().launch().await?;
262    /// # let tab = browser.main_tab();
263    /// let nav = tab.find().css("nav.dropdown").one().await?;
264    /// nav.hover().await?;
265    /// # Ok(()) }
266    /// ```
267    pub async fn hover(&self) -> Result<()> {
268        self.with_refresh(|| async move {
269            self.scroll_into_view().await?;
270            actionability::wait_actionable(
271                self,
272                ActionabilityCheck {
273                    visible: true,
274                    stable: true,
275                    enabled: false,
276                    receives_pointer: true,
277                },
278                DEFAULT_ACTIONABILITY_TIMEOUT,
279            )
280            .await?;
281            let bbox = self
282                .bounding_box()
283                .await?
284                .ok_or_else(|| ZendriverError::Navigation("element has no bounding box".into()))?;
285            let cx = bbox.x + bbox.width / 2.0;
286            let cy = bbox.y + bbox.height / 2.0;
287            let input = self.inner.tab.input().clone();
288            mouse::move_realistic(&input, &self.inner.tab, cx, cy).await
289        })
290        .await
291    }

403    /// Set this element's `value` + fire bubbled `input` and `change` events.
404    ///
405    /// Assigns through the element's *native* prototype value-setter rather
406    /// than a direct `this.value =` so React's per-instance `_valueTracker`
407    /// can't silently revert the change on the next render. See
408    /// [`NATIVE_VALUE_SETTER_JS`] for the full rationale. Bypasses
409    /// keydown/keyup (use [`Element::type_text`] for a real keystroke
410    /// sequence) but the bubbled `input` + `change` events let React-style
411    /// controlled inputs see the update through their onChange handlers.
412    ///
413    /// No actionability gate — `set_value` is the fast-path for tests +
414    /// automation flows that don't care about visibility/enabledness. If
415    /// you need the gate, focus the element first then call this.
416    ///
417    /// # Examples
418    ///
419    /// ```no_run
420    /// # async fn ex() -> zendriver::Result<()> {
421    /// # let browser = zendriver::Browser::builder().launch().await?;
422    /// # let tab = browser.main_tab();
423    /// let input = tab.find().css("input[name=q]").one().await?;
424    /// input.set_value("rust async").await?;
425    /// # Ok(()) }
426    /// ```
427    pub async fn set_value(&self, value: impl AsRef<str>) -> Result<()> {
428        let value = value.as_ref().to_string();
429        self.with_refresh(|| {
430            let value = value.clone();
431            async move {
432                let _ = self
433                    .call_on_main(NATIVE_VALUE_SETTER_JS, json!([{ "value": value }]))
434                    .await?;
435                Ok(())
436            }
437        })
438        .await
439    }

517    /// Clear this element by keystroke: focus, select-all, then press
518    /// Backspace until the field empties.
519    ///
520    /// A real-keystroke fallback for inputs that ignore a programmatic
521    /// value-set (e.g. fields with custom `onKeyDown` handling that
522    /// [`Element::clear`] can't drive). Sequence:
523    ///
524    /// 1. [`Element::focus`] the element.
525    /// 2. Select-all chord — `Cmd+A` on macOS, `Ctrl+A` elsewhere — so the
526    ///    whole value is selected before deletion.
527    /// 3. Read the current `value.length`, then press
528    ///    [`SpecialKey::Backspace`] that many times plus a small slack
529    ///    ([`CLEAR_BY_DELETING_SLACK`]), bounded by [`CLEAR_BY_DELETING_MAX`].
530    ///
531    /// Deletes backward (Backspace) only — never forward-Delete — because
532    /// `VK_DELETE` at caret position 0 is treated as a backward delete on some
533    /// VM environments, which would no-op and spin forever.
534    ///
535    /// # Examples
536    ///
537    /// ```no_run
538    /// # async fn ex() -> zendriver::Result<()> {
539    /// # let browser = zendriver::Browser::builder().launch().await?;
540    /// # let tab = browser.main_tab();
541    /// let input = tab.find().css("input.react-controlled").one().await?;
542    /// input.clear_by_deleting().await?;
543    /// # Ok(()) }
544    /// ```
545    pub async fn clear_by_deleting(&self) -> Result<()> {
546        self.with_refresh(|| async move {
547            self.focus().await?;
548            // Select-all so the field's whole value is highlighted; Backspace
549            // then clears the selection in one stroke before falling back to
550            // per-character deletion for any stragglers.
551            let select_all = if cfg!(target_os = "macos") {
552                KeyModifiers::META
553            } else {
554                KeyModifiers::CTRL
555            };
556            self.press_with(Key::Char('a'), select_all).await?;
557
558            let len: usize = {
559                let res = self
560                    .call_on_main("function(){ return (this.value || '').length; }", json!([]))
561                    .await?;
562                res.get("value")
563                    .and_then(serde_json::Value::as_u64)
564                    .and_then(|n| usize::try_from(n).ok())
565                    .unwrap_or(0)
566            };
567            let presses = len
568                .saturating_add(CLEAR_BY_DELETING_SLACK)
569                .min(CLEAR_BY_DELETING_MAX);
570            for _ in 0..presses {
571                self.press(Key::Special(SpecialKey::Backspace)).await?;
572            }
573            Ok(())
574        })
575        .await
576    }

578    /// Attach files to this `<input type="file">` element via
579    /// `DOM.setFileInputFiles`.
580    ///
581    /// Bypasses the OS file picker entirely — CDP wires the paths straight
582    /// into the input's `FileList`, and the page sees a normal `change`
583    /// event from the input.
584    ///
585    /// Scope is direct `<input type="file">` only — the element this is
586    /// called on must be the input itself. For a button/label that opens
587    /// the picker indirectly (e.g. a JS handler that calls
588    /// `hiddenInput.click()`, or any custom upload widget), use
589    /// [`Tab::expect_file_chooser`](crate::Tab::expect_file_chooser)
590    /// (feature `expect`) instead — it intercepts `Page.fileChooserOpened`
591    /// so it works regardless of what triggered the picker. Paths are
592    /// passed as their lossy `to_string_lossy()` representation, matching
593    /// CDP's UTF-8 string contract.
594    ///
595    /// # Examples
596    ///
597    /// ```no_run
598    /// # async fn ex() -> zendriver::Result<()> {
599    /// # let browser = zendriver::Browser::builder().launch().await?;
600    /// # let tab = browser.main_tab();

631    /// Flash a transient colored dot at this element's bbox center.
632    ///
633    /// Scrolls the element into view, computes its bbox center, then injects
634    /// a small absolutely-positioned `<div>` there that removes itself after
635    /// `duration`. A debugging aid for eyeballing where pointer actions land
636    /// against a headful Chrome — no input state changes, not for production.
637    ///
638    /// Ports nodriver's `Element.flash` (element.py:913): nodriver resolves
639    /// the node, takes its position, and calls `Tab.flash_point(*center)`
640    /// (its richer self-animating overlay is commented out and its `duration`
641    /// argument unused). This mirrors the same dot-at-center behavior but
642    /// honors `duration` for the self-removal timer rather than hardcoding it,
643    /// matching the sibling [`crate::Tab::flash_point`] dispatch shape.
644    ///
645    /// No-ops gracefully (returns `Ok`) when the element has no box (e.g.
646    /// `display: none`), since there is nothing to point at.
647    ///
648    /// # Examples
649    ///
650    /// ```no_run
651    /// # async fn ex() -> zendriver::Result<()> {
652    /// # use std::time::Duration;
653    /// # let browser = zendriver::Browser::builder().launch().await?;
654    /// # let tab = browser.main_tab();
655    /// let btn = tab.find().css("button").one().await?;
656    /// btn.flash(Duration::from_millis(500)).await?;
657    /// # Ok(()) }
658    /// ```
659    pub async fn flash(&self, duration: Duration) -> Result<()> {
660        let ms = duration.as_millis();
661        self.with_refresh(|| async move {
662            self.scroll_into_view().await?;
663            let bbox = match self.bounding_box().await? {
664                Some(b) => b,
665                // No box to point at — nothing to flash.
666                None => return Ok(()),
667            };
668            let cx = bbox.x + bbox.width / 2.0;
669            let cy = bbox.y + bbox.height / 2.0;
670            // Inject the dot in the main world with the element bound (house
671            // pattern); the args carry the viewport-center coords + lifetime.
672            // The dot is `position:fixed` so the coords are viewport-relative,
673            // matching `bounding_box`'s frame and `Tab::flash_point`.
674            let js = "function(cx, cy, ms){ \
675                    const d = document.createElement('div'); \
676                    d.style.cssText = 'position:fixed;left:'+cx+'px;top:'+cy+'px;\
677width:10px;height:10px;margin:-5px 0 0 -5px;border-radius:50%;background:red;\
678z-index:2147483647;pointer-events:none;opacity:0.85;'; \
679                    document.body.appendChild(d); \
680                    setTimeout(() => d.remove(), ms); \
681                }";
682            let _ = self
683                .call_on_main(
684                    js,
685                    json!([{ "value": cx }, { "value": cy }, { "value": ms }]),
686                )
687                .await?;
688            Ok(())
689        })
690        .await
691    }

696    /// with this element's backend node id and a default

748    /// Drag this element's bbox center to the viewport point `to`, holding
749    /// the left button, over `steps` interpolated hops.
750    ///
751    /// Scrolls the element into view, computes its bbox center, then drives
752    /// the drag via [`crate::Tab::mouse_drag`] (press at the center → linearly
753    /// interpolated `mouseMoved` frames → release at `to`). A larger `steps`
754    /// makes the drag look smoother / more human (nodriver suggests 50–100 for
755    /// "very smooth"); `0` or `1` collapses to a single hop straight to `to`.
756    ///
757    /// Ports nodriver's `Element.mouse_drag` (element.py:596): nodriver takes
758    /// the element's center, resolves the destination (another element's
759    /// center or a coordinate, optionally relative), and forwards to
760    /// `Tab.mouse_drag`. This is the coordinate-destination form; the source
761    /// is always this element's center.
762    ///
763    /// # Errors
764    ///
765    /// Returns [`ZendriverError::Navigation`] when the element has no bounding
766    /// box (nothing to drag from); propagates transport / CDP errors from the
767    /// underlying dispatches.
768    ///
769    /// # Examples
770    ///
771    /// ```no_run
772    /// # async fn ex() -> zendriver::Result<()> {
773    /// # let browser = zendriver::Browser::builder().launch().await?;
774    /// # let tab = browser.main_tab();
775    /// // Drag a slider handle to (320, 300) over 40 smooth steps.
776    /// let handle = tab.find().css(".slider-handle").one().await?;
777    /// handle.mouse_drag((320.0, 300.0), 40).await?;
778    /// # Ok(()) }
779    /// ```
780    pub async fn mouse_drag(&self, to: (f64, f64), steps: usize) -> Result<()> {
781        self.with_refresh(|| async move {
782            self.scroll_into_view().await?;
783            let bbox = self
784                .bounding_box()
785                .await?
786                .ok_or_else(|| ZendriverError::Navigation("element has no bounding box".into()))?;
787            let cx = bbox.x + bbox.width / 2.0;
788            let cy = bbox.y + bbox.height / 2.0;
789            // Reuse the Tab's drag primitive (don't duplicate the
790            // press/move/release dispatch sequence).
791            self.inner.tab.mouse_drag((cx, cy), to, steps).await
792        })
793        .await
794    }
795}
