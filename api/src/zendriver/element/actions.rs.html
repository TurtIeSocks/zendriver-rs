1//! [`Element`] actions: `click` / `hover` / `focus` / `scroll_into_view` /
2//! `set_value` / `clear` / `upload_files`.
3//!
4//! Each action wraps its CDP dispatch sequence in an internal "refresh on
5//! stale" wrapper so a stale handle (post-navigation, post-React-rerender)
6//! transparently re-resolves once and retries.
7//!
8//! `hover` / `hover_fast`:
9//!   1. `scroll_into_view` — bring the element into the viewport so its
10//!      bbox center is a real, dispatchable coordinate.
11//!   2. Actionability gate (`visible + stable + receives_pointer`) — avoids
12//!      mid-transition hover races and overlay occlusion. Pointer events
13//!      are dispatched at the geometric center, so the element must be the
14//!      actual hit-test target there. `enabled` is left off — hover doesn't
15//!      activate the element, so disabled controls still accept mouseover.
16//!   3. Compute bbox center (`x + width / 2`, `y + height / 2`).
17//!   4. Dispatch the mouse-move via the shared [`crate::input::InputController`].
18//!      `hover` uses a realistic Bezier path; `hover_fast` uses a single
19//!      teleport dispatch for test/automation paths.
20//!
21//! `focus`: actionability gate (visible + enabled — no pointer or stability
22//! requirement; focus routes through the focused element, not the cursor's
23//! position), then `el.focus()` in the main world.
24//!
25//! `scroll_into_view`: no actionability gate (this *is* the visibility
26//! prereq other actions wait for). Calls `el.scrollIntoView({ block:
27//! 'center', behavior: 'instant' })`. `block: 'center'` matches Playwright
28//! (avoids sticky headers/footers obscuring the element after the scroll);
29//! `behavior: 'instant'` skips animation so the post-scroll bbox is final
30//! by the time the next CDP call runs.

43/// Default deadline for the actionability gate before each action. Matches
44/// the value the spec calls out for P3; per-call override land in P4 when
45/// the per-action options structs grow.
46const DEFAULT_ACTIONABILITY_TIMEOUT: Duration = Duration::from_secs(5);

48/// Per-call knobs for [`Element::click_with`].
49///
50/// `Default` matches the behavior of [`Element::click`]: a left, single,
51/// realistic click at the element's bbox center with no modifiers held and
52/// full actionability gating. Override fields individually for richer
53/// dispatches (right-click, modifier-held click, raw teleport, etc.).
54///
55/// # Examples
56///
57/// ```no_run
58/// # async fn ex() -> zendriver::Result<()> {
59/// use zendriver::{ClickOptions, MouseButton};
60/// # let browser = zendriver::Browser::builder().launch().await?;
61/// # let tab = browser.main_tab();
62/// let el = tab.find().css("button").one().await?;
63/// el.click_with(ClickOptions {
64///     button: MouseButton::Right,
65///     click_count: 2,
66///     ..Default::default()
67/// }).await?;
68/// # Ok(()) }
69/// ```
70#[derive(Debug, Clone, Copy)]
71pub struct ClickOptions {
72    /// Which mouse button to dispatch. [`MouseButton::Left`] by default.
73    pub button: MouseButton,
74    /// Modifier keys held during the dispatch. Empty by default.
75    pub modifiers: KeyModifiers,
76    /// `clickCount` for the CDP dispatch. `1` by default; set `2` for a
77    /// double-click in a single `click_with` call.
78    pub click_count: u32,
79    /// Skip the actionability gate when true. Use sparingly — bypasses the
80    /// visibility/stability/pointer checks. Mirrors Playwright's
81    /// `force: true`.
82    pub force: bool,
83    /// Bezier-interpolated cursor path (`true`) vs single teleport
84    /// dispatch (`false`). `true` by default; the `click_fast` shortcut
85    /// flips this for deterministic test paths.
86    pub realistic: bool,
87    /// Click position relative to the element's bbox top-left
88    /// (`(dx, dy)`). `None` clicks at the bbox center.
89    pub position: Option<(f64, f64)>,
90}
91
92impl Default for ClickOptions {
93    fn default() -> Self {
94        Self {
95            button: MouseButton::Left,
96            modifiers: KeyModifiers::empty(),
97            click_count: 1,
98            force: false,
99            realistic: true,
100            position: None,
101        }
102    }
103}

105impl Element {
106    /// Click this element with realistic defaults.
107    ///
108    /// Left button, single click, Bezier-path cursor approach, and the full
109    /// actionability gate. Equivalent to
110    /// `click_with(ClickOptions::default())`. For right-click / modifier-held
111    /// / double-click / raw-teleport variations, use [`Element::click_with`].
112    ///
113    /// # Errors
114    ///
115    /// Returns [`ZendriverError::NotActionable`] when the actionability gate
116    /// times out, [`ZendriverError::ElementStale`] when the handle can't be
117    /// refreshed.
118    ///
119    /// # Examples
120    ///
121    /// ```no_run
122    /// # async fn ex() -> zendriver::Result<()> {
123    /// # let browser = zendriver::Browser::builder().launch().await?;
124    /// # let tab = browser.main_tab();
125    /// let btn = tab.find().css("button#submit").one().await?;
126    /// btn.click().await?;
127    /// # Ok(()) }
128    /// ```
129    pub async fn click(&self) -> Result<()> {
130        self.click_with(ClickOptions::default()).await
131    }

133    /// Click this element with a deterministic raw teleport.
134    ///
135    /// Skips the Bezier interpolation [`Element::click`] uses and bypasses
136    /// the actionability gate. Equivalent to
137    /// `click_with(ClickOptions { realistic: false, force: true, ..Default::default() })`.
138    /// Intended for test paths and fast automation flows where realism
139    /// and per-action gating get in the way.
140    ///
141    /// # Examples
142    ///
143    /// ```no_run
144    /// # async fn ex() -> zendriver::Result<()> {
145    /// # let browser = zendriver::Browser::builder().launch().await?;
146    /// # let tab = browser.main_tab();
147    /// let btn = tab.find().css("button").one().await?;
148    /// btn.click_fast().await?;
149    /// # Ok(()) }
150    /// ```
151    pub async fn click_fast(&self) -> Result<()> {
152        self.click_with(ClickOptions {
153            realistic: false,
154            force: true,
155            ..Default::default()
156        })
157        .await
158    }

160    /// Click this element with explicit [`ClickOptions`].
161    ///
162    /// See module docs for the dispatch sequence — same shape as `hover`
163    /// (scroll → gate → bbox math → pointer dispatch) but emits the
164    /// `mousePressed` + `mouseReleased` pair after the cursor arrives.
165    /// `opts.force` skips the actionability gate; `opts.position` shifts
166    /// the click point off bbox-center.
167    ///
168    /// # Examples
169    ///
170    /// ```no_run
171    /// # use zendriver::{ClickOptions, MouseButton};
172    /// # async fn ex() -> zendriver::Result<()> {
173    /// # let browser = zendriver::Browser::builder().launch().await?;
174    /// # let tab = browser.main_tab();
175    /// let row = tab.find().css("tr.contact").one().await?;
176    /// row.click_with(ClickOptions {
177    ///     button: MouseButton::Right,
178    ///     ..Default::default()
179    /// }).await?;
180    /// # Ok(()) }
181    /// ```
182    pub async fn click_with(&self, opts: ClickOptions) -> Result<()> {
183        self.with_refresh(|| async move {
184            self.scroll_into_view().await?;
185            if !opts.force {
186                actionability::wait_actionable(
187                    self,
188                    ActionabilityCheck::FULL,
189                    DEFAULT_ACTIONABILITY_TIMEOUT,
190                )
191                .await?;
192            }
193            let bbox = self
194                .bounding_box()
195                .await?
196                .ok_or_else(|| ZendriverError::Navigation("element has no bounding box".into()))?;
197            let (tx, ty) = match opts.position {
198                Some((dx, dy)) => (bbox.x + dx, bbox.y + dy),
199                None => (bbox.x + bbox.width / 2.0, bbox.y + bbox.height / 2.0),
200            };
201            let input = self.inner.tab.input().clone();
202            mouse::click_at(
203                &input,
204                &self.inner.tab,
205                tx,
206                ty,
207                opts.button,
208                opts.click_count,
209                opts.realistic,
210            )
211            .await
212        })
213        .await
214    }

216    /// Hover the cursor over this element's bbox center.
217    ///
218    /// Uses a realistic Bezier-interpolated mouse path. See module docs for
219    /// the full sequence (`scroll_into_view` → actionability gate → bbox
220    /// center → dispatch). Use [`Element::hover_fast`] when the cursor path
221    /// doesn't matter.
222    ///
223    /// # Examples
224    ///
225    /// ```no_run
226    /// # async fn ex() -> zendriver::Result<()> {
227    /// # let browser = zendriver::Browser::builder().launch().await?;
228    /// # let tab = browser.main_tab();
229    /// let nav = tab.find().css("nav.dropdown").one().await?;
230    /// nav.hover().await?;
231    /// # Ok(()) }
232    /// ```
233    pub async fn hover(&self) -> Result<()> {
234        self.with_refresh(|| async move {
235            self.scroll_into_view().await?;
236            actionability::wait_actionable(
237                self,
238                ActionabilityCheck {
239                    visible: true,
240                    stable: true,
241                    enabled: false,
242                    receives_pointer: true,
243                },
244                DEFAULT_ACTIONABILITY_TIMEOUT,
245            )
246            .await?;
247            let bbox = self
248                .bounding_box()
249                .await?
250                .ok_or_else(|| ZendriverError::Navigation("element has no bounding box".into()))?;
251            let cx = bbox.x + bbox.width / 2.0;
252            let cy = bbox.y + bbox.height / 2.0;
253            let input = self.inner.tab.input().clone();
254            mouse::move_realistic(&input, &self.inner.tab, cx, cy).await
255        })
256        .await
257    }

260    /// teleport.
261    ///
262    /// Skips the Bezier interpolation [`Element::hover`] does — same
263    /// actionability gate + bbox math, but no human-pointer modeling.
264    /// Intended for paths where deterministic timing matters more than
265    /// realism.
266    ///
267    /// # Examples
268    ///
269    /// ```no_run
270    /// # async fn ex() -> zendriver::Result<()> {
271    /// # let browser = zendriver::Browser::builder().launch().await?;
272    /// # let tab = browser.main_tab();
273    /// let el = tab.find().css("button").one().await?;
274    /// el.hover_fast().await?;
275    /// # Ok(()) }
276    /// ```
277    pub async fn hover_fast(&self) -> Result<()> {
278        self.with_refresh(|| async move {
279            self.scroll_into_view().await?;
280            actionability::wait_actionable(
281                self,
282                ActionabilityCheck {
283                    visible: true,
284                    stable: true,
285                    enabled: false,
286                    receives_pointer: true,
287                },
288                DEFAULT_ACTIONABILITY_TIMEOUT,
289            )
290            .await?;
291            let bbox = self
292                .bounding_box()
293                .await?
294                .ok_or_else(|| ZendriverError::Navigation("element has no bounding box".into()))?;
295            let cx = bbox.x + bbox.width / 2.0;
296            let cy = bbox.y + bbox.height / 2.0;
297            let input = self.inner.tab.input().clone();
298            mouse::move_raw(&input, &self.inner.tab, cx, cy).await
299        })
300        .await
301    }

303    /// Move keyboard focus to this element by calling `el.focus()`.
304    ///
305    /// Gated by an actionability check (visible + enabled) so disabled
306    /// controls + hidden elements surface a [`ZendriverError::NotActionable`]
307    /// error rather than silently no-op on the page side. Reused by
308    /// [`Element::type_text`] / [`Element::press`] — they focus first so
309    /// keystrokes reach this element.
310    ///
311    /// # Examples
312    ///
313    /// ```no_run
314    /// # async fn ex() -> zendriver::Result<()> {
315    /// # let browser = zendriver::Browser::builder().launch().await?;
316    /// # let tab = browser.main_tab();
317    /// let input = tab.find().css("input[name=email]").one().await?;
318    /// input.focus().await?;
319    /// # Ok(()) }
320    /// ```
321    pub async fn focus(&self) -> Result<()> {
322        self.with_refresh(|| async move {
323            actionability::wait_actionable(
324                self,
325                ActionabilityCheck::TEXT_INPUT,
326                DEFAULT_ACTIONABILITY_TIMEOUT,
327            )
328            .await?;
329            let _ = self
330                .call_on_main("function(){ this.focus(); }", json!([]))
331                .await?;
332            Ok(())
333        })
334        .await
335    }

337    /// Scroll this element into view, centered in its scroll container.
338    ///
339    /// Synchronous (`behavior: 'instant'`) so the post-scroll bbox is final
340    /// by the time the next CDP call (e.g. `bounding_box`) runs — important
341    /// because subsequent action steps assume the layout is settled.
342    ///
343    /// No actionability gate: this method IS the visibility prerequisite
344    /// for the other actions; gating it on visibility would deadlock.
345    ///
346    /// # Examples
347    ///
348    /// ```no_run
349    /// # async fn ex() -> zendriver::Result<()> {
350    /// # let browser = zendriver::Browser::builder().launch().await?;
351    /// # let tab = browser.main_tab();
352    /// let footer = tab.find().css("footer").one().await?;
353    /// footer.scroll_into_view().await?;
354    /// # Ok(()) }
355    /// ```
356    pub async fn scroll_into_view(&self) -> Result<()> {
357        self.with_refresh(|| async move {
358            let _ = self
359                .call_on_main(
360                    "function(){ this.scrollIntoView({block:'center',behavior:'instant'}); }",
361                    json!([]),
362                )
363                .await?;
364            Ok(())
365        })
366        .await
367    }

369    /// Set this element's `value` directly + fire `input` and `change`
370    /// events.
371    ///
372    /// Bypasses keydown/keyup (use [`Element::type_text`] for a real
373    /// keystroke sequence) but dispatches the bubbled `input` + `change`
374    /// events so React-style controlled inputs see the update through their
375    /// onChange handlers.
376    ///
377    /// No actionability gate — `set_value` is the fast-path for tests +
378    /// automation flows that don't care about visibility/enabledness. If
379    /// you need the gate, focus the element first then call this.
380    ///
381    /// # Examples
382    ///
383    /// ```no_run
384    /// # async fn ex() -> zendriver::Result<()> {
385    /// # let browser = zendriver::Browser::builder().launch().await?;
386    /// # let tab = browser.main_tab();
387    /// let input = tab.find().css("input[name=q]").one().await?;
388    /// input.set_value("rust async").await?;
389    /// # Ok(()) }
390    /// ```
391    pub async fn set_value(&self, value: impl AsRef<str>) -> Result<()> {
392        let value = value.as_ref().to_string();
393        self.with_refresh(|| {
394            let value = value.clone();
395            async move {
396                let _ = self
397                    .call_on_main(
398                        "function(el, v){ \
399                            this.value = v; \
400                            this.dispatchEvent(new Event('input', {bubbles: true})); \
401                            this.dispatchEvent(new Event('change', {bubbles: true})); \
402                        }",
403                        json!([{ "value": value }]),
404                    )
405                    .await?;
406                Ok(())
407            }
408        })
409        .await
410    }

447    /// Attach files to this `<input type="file">` element via
448    /// `DOM.setFileInputFiles`.
449    ///
450    /// Bypasses the OS file picker entirely — CDP wires the paths straight
451    /// into the input's `FileList`, and the page sees a normal `change`
452    /// event from the input.
453    ///
454    /// Scope is direct `<input type="file">` only; routing through a hidden
455    /// input clicked by a label / button wrapper is the page's
456    /// responsibility. Paths are passed as their lossy `to_string_lossy()`
457    /// representation, matching CDP's UTF-8 string contract.
458    ///
459    /// # Examples
460    ///
461    /// ```no_run
462    /// # async fn ex() -> zendriver::Result<()> {
463    /// # let browser = zendriver::Browser::builder().launch().await?;
464    /// # let tab = browser.main_tab();
465    /// let file_input = tab.find().css("input[type=file]").one().await?;
466    /// file_input.upload_files(&["/tmp/photo.jpg"]).await?;
467    /// # Ok(()) }
468    /// ```
469    pub async fn upload_files<P: AsRef<Path>>(&self, paths: &[P]) -> Result<()> {
470        let files: Vec<String> = paths
471            .iter()
472            .map(|p| p.as_ref().to_string_lossy().into_owned())
473            .collect();
474        self.with_refresh(|| {
475            let files = files.clone();
476            async move {
477                let backend_node_id = self.backend_node_id_cloned().await?;
478                let _ = self
479                    .inner
480                    .tab
481                    .call(
482                        "DOM.setFileInputFiles",
483                        json!({
484                            "files": files,
485                            "backendNodeId": backend_node_id,
486                        }),
487                    )
488                    .await?;
489                Ok(())
490            }
491        })
492        .await
493    }

507        let sess = SessionHandle::new(conn.clone(), "S1");
508        // `Tab::new_for_test` seeds an `InputController` from the native

634            mock.reply(
635                id,
636                json!({ "result": { "value": true, "type": "boolean" } }),
637            )
638            .await;
639        }
640
641        // Step 3: bounding_box → DOM.getBoxModel.
642        let id = mock.expect_cmd("DOM.getBoxModel").await;
643        mock.reply(
644            id,
645            json!({
646                "model": {
647                    "content": [10.0, 20.0, 110.0, 20.0, 110.0, 70.0, 10.0, 70.0],
648                    "padding": [10.0, 20.0, 110.0, 20.0, 110.0, 70.0, 10.0, 70.0],
649                    "border":  [10.0, 20.0, 110.0, 20.0, 110.0, 70.0, 10.0, 70.0],
650                    "margin":  [10.0, 20.0, 110.0, 20.0, 110.0, 70.0, 10.0, 70.0],
651                    "width":  100,
652                    "height": 50
653                }
654            }),
655        )
656        .await;
657
658        // Step 4: dispatch sequence. Bezier emits N mouseMoved frames, then
659        // exactly one mousePressed + one mouseReleased. Walk all
660        // `Input.dispatchMouseEvent` calls and assert ordering: every
661        // mouseMoved precedes mousePressed, which precedes mouseReleased.
662        let mut saw_pressed = false;
663        let mut saw_released = false;
664        let mut last_kind = String::new();
665        loop {
666            let next = tokio::time::timeout(
667                Duration::from_millis(500),
668                mock.expect_cmd("Input.dispatchMouseEvent"),
669            )
670            .await;
671            match next {
672                Ok(id) => {
673                    let sent = mock.last_sent();
674                    let kind = sent["params"]["type"].as_str().unwrap_or("").to_string();
675                    match kind.as_str() {
676                        "mouseMoved" => {
677                            assert!(
678                                !saw_pressed && !saw_released,
679                                "mouseMoved arrived after mousePressed/Released"
680                            );
681                        }
682                        "mousePressed" => {
683                            assert!(!saw_pressed, "duplicate mousePressed");
684                            assert!(!saw_released, "mousePressed after mouseReleased");
685                            saw_pressed = true;
686                        }
687                        "mouseReleased" => {
688                            assert!(saw_pressed, "mouseReleased before mousePressed");
689                            assert!(!saw_released, "duplicate mouseReleased");
690                            saw_released = true;
691                        }
692                        other => panic!("unexpected dispatch type: {other}"),
693                    }
694                    last_kind = kind;
695                    mock.reply(id, json!({})).await;
696                }
697                Err(_) => break,
698            }
699        }
700
701        let res = fut.await.unwrap();
702        res.unwrap();
703        assert!(saw_pressed, "expected a mousePressed dispatch");
704        assert!(saw_released, "expected a mouseReleased dispatch");
705        assert_eq!(
706            last_kind, "mouseReleased",
707            "final dispatch should be mouseReleased"
708        );
709        conn.shutdown();
710    }
