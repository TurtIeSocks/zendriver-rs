1//! [`Element`] traversal: walk to parents + children.
2//!
3//! Both methods preserve origin so the auto-refresh chain can re-resolve
4//! the result against its parent. They dispatch `Runtime.callFunctionOn`
5//! with `returnByValue: false` so the result carries an `objectId` we can
6//! follow up on.
7
8use serde_json::json;
9
10use crate::element::{Element, TraversalKind};
11use crate::error::Result;
12use crate::query::selectors::{extract_array_refs, extract_node_ref};
13
14impl Element {
15    /// Return this element's `parentElement`, or `Ok(None)` if absent.
16    ///
17    /// Returns `None` for the root `<html>`, detached nodes, or text nodes
18    /// with no parent. The returned `Element`'s origin tracks back to this
19    /// element so a stale parent will recursively re-resolve via
20    /// [`Element::refresh`].
21    ///
22    /// # Examples
23    ///
24    /// ```no_run
25    /// # async fn ex() -> zendriver::Result<()> {
26    /// # let browser = zendriver::Browser::builder().launch().await?;
27    /// # let tab = browser.main_tab();
28    /// let el = tab.find().css("h1").one().await?;
29    /// if let Some(parent) = el.parent().await? {
30    ///     println!("parent: {}", parent.inner_text().await?);
31    /// }
32    /// # Ok(()) }
33    /// ```
34    pub async fn parent(&self) -> Result<Option<Element>> {
35        self.with_refresh(|| async move {
36            let object_id = self.remote_object_id_cloned().await?;
37            let raw = self
38                .inner
39                .tab
40                .call(
41                    "Runtime.callFunctionOn",
42                    json!({
43                        "objectId": object_id,
44                        "functionDeclaration": "function(){ return this.parentElement; }",
45                        "arguments": [],
46                        "returnByValue": false,
47                        "awaitPromise": true,
48                    }),
49                )
50                .await?;
51            let Some(remote_ref) =
52                extract_node_ref(self.inner.tab.session(), &raw["result"]).await?
53            else {
54                return Ok(None);
55            };
56            let parent_origin = self.inner.origin.clone();
57            Ok(Some(Element::synthesize_traversal(
58                self.inner.tab.clone(),
59                remote_ref.backend_node_id,
60                remote_ref.remote_object_id,
61                parent_origin,
62                TraversalKind::Parent,
63            )))
64        })
65        .await
66    }
67
68    /// Begin a subtree-scoped chainable query against this element.
69    ///
70    /// Pick a selector kind, optionally apply modifiers, then terminate
71    /// with `.one()` or `.one_or_none()`. Matches outside this element's
72    /// subtree are not considered. The terminal dispatches
73    /// `Runtime.callFunctionOn` against this element's remote object,
74    /// distinguishing it from [`crate::Tab::find`] which scans the whole
75    /// document.
76    ///
77    /// # Examples
78    ///
79    /// ```no_run
80    /// # async fn ex() -> zendriver::Result<()> {
81    /// # let browser = zendriver::Browser::builder().launch().await?;
82    /// # let tab = browser.main_tab();
83    /// let card = tab.find().css(".card").one().await?;
84    /// let title = card.find().css("h2").one().await?;
85    /// # let _ = title;
86    /// # Ok(()) }
87    /// ```
88    pub fn find(&self) -> crate::query::FindBuilder<'_> {
89        crate::query::FindBuilder::new_for_element(self)
90    }
91
92    /// Begin a subtree-scoped chainable query that returns ALL matches.
93    ///
94    /// Mirrors [`Element::find`] selectors + modifiers (no `.nth`);
95    /// terminate with `.many()` (errors on empty) or `.many_or_empty()`
96    /// (returns an empty `Vec`).
97    ///
98    /// # Examples
99    ///
100    /// ```no_run
101    /// # async fn ex() -> zendriver::Result<()> {
102    /// # let browser = zendriver::Browser::builder().launch().await?;
103    /// # let tab = browser.main_tab();
104    /// let list = tab.find().css("ul").one().await?;
105    /// let items = list.find_all().css("li").many_or_empty().await?;
106    /// println!("{} items", items.len());
107    /// # Ok(()) }
108    /// ```
109    pub fn find_all(&self) -> crate::query::FindAllBuilder<'_> {
110        crate::query::FindAllBuilder::new_for_element(self)
111    }
112
113    /// Find one descendant element by CSS selector. Python-parity convenience
114    /// for `find().css(sel).one()`. Scoped to this element's subtree.
115    ///
116    /// # Errors
117    ///
118    /// Returns [`crate::error::ZendriverError::ElementNotFound`] if no element
119    /// matches.
120    ///
121    /// # Examples
122    ///
123    /// ```no_run
124    /// # async fn ex() -> zendriver::Result<()> {
125    /// # let browser = zendriver::Browser::builder().launch().await?;
126    /// # let tab = browser.main_tab();
127    /// let card = tab.find().css(".card").one().await?;
128    /// let title = card.select("h2").await?;
129    /// # let _ = title;
130    /// # Ok(()) }
131    /// ```
132    pub async fn select(&self, css: &str) -> crate::error::Result<crate::Element> {
133        self.find().css(css).one().await
134    }
135
136    /// Find all descendant elements by CSS selector. Python-parity convenience
137    /// for `find_all().css(sel).many()`. Scoped to this element's subtree.
138    ///
139    /// # Errors
140    ///
141    /// Returns [`crate::error::ZendriverError::ElementNotFound`] if no elements
142    /// match.
143    ///
144    /// # Examples
145    ///
146    /// ```no_run
147    /// # async fn ex() -> zendriver::Result<()> {
148    /// # let browser = zendriver::Browser::builder().launch().await?;
149    /// # let tab = browser.main_tab();
150    /// let list = tab.find().css("ul").one().await?;
151    /// let items = list.select_all("li").await?;
152    /// println!("{} items", items.len());
153    /// # Ok(()) }
154    /// ```
155    pub async fn select_all(&self, css: &str) -> crate::error::Result<Vec<crate::Element>> {
156        self.find_all().css(css).many().await
157    }
158
159    /// Return this element's child elements as a `Vec<Element>`.
160    ///
161    /// HTMLCollection is materialized via `Array.from`. The returned
162    /// elements are tagged with `NthChild(i)` so auto-refresh can re-pick
163    /// the same index after a re-render. Empty children yield an empty
164    /// `Vec`, not an error.
165    ///
166    /// # Examples
167    ///
168    /// ```no_run
169    /// # async fn ex() -> zendriver::Result<()> {
170    /// # let browser = zendriver::Browser::builder().launch().await?;
171    /// # let tab = browser.main_tab();
172    /// let nav = tab.find().css("nav").one().await?;
173    /// for link in nav.children().await? {
174    ///     println!("{}", link.inner_text().await?);
175    /// }
176    /// # Ok(()) }
177    /// ```
178    pub async fn children(&self) -> Result<Vec<Element>> {
179        self.with_refresh(|| async move {
180            let object_id = self.remote_object_id_cloned().await?;
181            let raw = self
182                .inner
183                .tab
184                .call(
185                    "Runtime.callFunctionOn",
186                    json!({
187                        "objectId": object_id,
188                        "functionDeclaration": "function(){ return Array.from(this.children); }",
189                        "arguments": [],
190                        "returnByValue": false,
191                        "awaitPromise": true,
192                    }),
193                )
194                .await?;
195            let refs = extract_array_refs(self.inner.tab.session(), &raw["result"]).await?;
196            let parent_origin = self.inner.origin.clone();
197            let out = refs
198                .into_iter()
199                .enumerate()
200                .map(|(idx, r)| {
201                    Element::synthesize_traversal(
202                        self.inner.tab.clone(),
203                        r.backend_node_id,
204                        r.remote_object_id,
205                        parent_origin.clone(),
206                        TraversalKind::NthChild(idx),
207                    )
208                })
209                .collect();
210            Ok(out)
211        })
212        .await
213    }

225    #[tokio::test]
226    async fn parent_calls_runtime_callfunctionon_with_parent_element() {
227        let (mut mock, conn) = MockConnection::pair();
228        let sess = SessionHandle::new(conn.clone(), "S1");
229        let tab = Tab::new_for_test(sess);
230        let el = Element::from_jsret(tab, 7, "R7".to_string());
231
232        let fut = tokio::spawn({
233            let e = el.clone();
234            async move { e.parent().await }
235        });
236
237        // 1. Runtime.callFunctionOn with this.parentElement.
238        let id_q = mock.expect_cmd("Runtime.callFunctionOn").await;
239        let sent = mock.last_sent();
240        assert_eq!(sent["params"]["objectId"], "R7");
241        let decl = sent["params"]["functionDeclaration"].as_str().unwrap();
242        assert!(
243            decl.contains("this.parentElement"),
244            "parent() must call this.parentElement, got: {decl}"
245        );
246        assert_eq!(sent["params"]["returnByValue"], false);
247        mock.reply(
248            id_q,
249            json!({ "result": { "objectId": "RP", "type": "object", "subtype": "node" } }),
250        )
251        .await;
252
253        // 2. DOM.describeNode resolves backendNodeId.
254        let id_d = mock.expect_cmd("DOM.describeNode").await;
255        assert_eq!(mock.last_sent()["params"]["objectId"], "RP");
256        mock.reply(id_d, json!({ "node": { "backendNodeId": 314 } }))
257            .await;
258
259        let parent = fut.await.unwrap().unwrap().expect("parent present");
260        assert_eq!(parent.remote_object_id_cloned().await.unwrap(), "RP");
261        assert_eq!(parent.backend_node_id_cloned().await.unwrap(), 314);
262        assert!(matches!(
263            parent.inner.origin,
264            ElementOrigin::Traversal {
265                kind: TraversalKind::Parent,
266                ..
267            }
268        ));
269        conn.shutdown();
270    }

272    #[tokio::test]
273    async fn find_css_one_dispatches_callfunctionon_against_element_scope() {
274        // Element::find().css(...).one() must resolve via
275        // Runtime.callFunctionOn against the element's remote object
276        // (`this.querySelectorAll(...)`) rather than Runtime.evaluate
277        // against `document` — that's the contract that distinguishes
278        // subtree-scoped queries from tab-scoped queries.
279        let (mut mock, conn) = MockConnection::pair();
280        let sess = SessionHandle::new(conn.clone(), "S1");
281        let tab = Tab::new_for_test(sess);
282        let el = Element::from_jsret(tab, 9, "ROOT".to_string());
283
284        let fut = tokio::spawn({
285            let e = el.clone();
286            async move { e.find().css("button").one().await }
287        });
288
289        // 1. Element-scoped CSS many → Runtime.callFunctionOn on the
290        //    element's remote object, NOT Runtime.evaluate.
291        let id_q = mock.expect_cmd("Runtime.callFunctionOn").await;
292        let sent = mock.last_sent();
293        assert_eq!(
294            sent["params"]["objectId"], "ROOT",
295            "element-scoped query must dispatch against the element's remote object"
296        );
297        let decl = sent["params"]["functionDeclaration"].as_str().unwrap();
298        assert!(
299            decl.contains("this.querySelectorAll"),
300            "element-scoped CSS must use this.querySelectorAll, got: {decl}"
301        );
302        assert_eq!(sent["params"]["arguments"][0]["value"], "button");
303        mock.reply(
304            id_q,
305            json!({ "result": { "objectId": "RArr", "type": "object", "subtype": "array" } }),
306        )
307        .await;
308
309        // 2. Enumerate the array — one match at index 0.
310        let id_p = mock.expect_cmd("Runtime.getProperties").await;
311        assert_eq!(mock.last_sent()["params"]["objectId"], "RArr");
312        mock.reply(
313            id_p,
314            json!({
315                "result": [
316                    { "name": "0", "value": { "objectId": "RB", "type": "object", "subtype": "node" } },
317                    { "name": "length", "value": { "value": 1, "type": "number" } }
318                ]
319            }),
320        )
321        .await;
322
323        // 3. DOM.describeNode resolves backendNodeId of the matched node.
324        let id_d = mock.expect_cmd("DOM.describeNode").await;
325        assert_eq!(mock.last_sent()["params"]["objectId"], "RB");
326        mock.reply(id_d, json!({ "node": { "backendNodeId": 77 } }))
327            .await;
328
329        let found = fut.await.unwrap().unwrap();
330        assert_eq!(found.remote_object_id_cloned().await.unwrap(), "RB");
331        assert_eq!(found.backend_node_id_cloned().await.unwrap(), 77);
332        // Element-scoped origin tracks ScopeKind::ElementSubtree so T17
333        // refresh can re-resolve via this element's subtree.
334        assert!(matches!(
335            found.inner.origin,
336            ElementOrigin::Query {
337                scope_kind: crate::element::ScopeKind::ElementSubtree,
338                ..
339            }
340        ));
341        conn.shutdown();
342    }
