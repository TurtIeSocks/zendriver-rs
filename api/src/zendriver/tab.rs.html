1//! Per-page handle to a single CDP target session.
2//!
3//! [`Tab`] is the primary interaction surface in zendriver — most workflows
4//! are some sequence of `goto`, `find().css(...).one()`, `evaluate`,
5//! `screenshot`, and `wait_for_idle`. Each [`Tab`] owns its own
6//! [`InputController`] (cursor + held-modifier state), its own per-tab
7//! frame registry, and its own in-flight network tracker, so multiple tabs
8//! in the same [`crate::Browser`] don't interfere with one another.
9//!
10//! ```no_run
11//! # async fn ex() -> zendriver::Result<()> {
12//! let browser = zendriver::Browser::builder().launch().await?;
13//! let tab = browser.main_tab();
14//! tab.goto("https://example.com").await?;
15//! tab.wait_for_load().await?;
16//! let title: String = tab.evaluate_main("document.title").await?;
17//! assert_eq!(title, "Example Domain");
18//! # Ok(()) }
19//! ```

53/// Per-call knobs for [`Tab::reload_with`].
54///
55/// `Default` reloads with `ignore_cache: false` and no injected script —
56/// the same behavior as the plain [`Tab::reload`] shortcut. Set
57/// `ignore_cache: true` for a hard refresh, and/or
58/// `script_to_evaluate_on_load` to inject a script that runs on every frame
59/// load triggered by the reload.
60///
61/// # Examples
62///
63/// ```no_run
64/// # async fn ex() -> zendriver::Result<()> {
65/// use zendriver::ReloadOptions;
66/// # let browser = zendriver::Browser::builder().launch().await?;
67/// # let tab = browser.main_tab();
68/// tab.reload_with(ReloadOptions {
69///     ignore_cache: true,
70///     ..Default::default()
71/// }).await?;
72/// # Ok(()) }
73/// ```
74#[derive(Debug, Clone, Default)]
75pub struct ReloadOptions {
76    /// Bypass the HTTP cache for the reload (`Page.reload.ignoreCache`).
77    /// `false` by default — a soft refresh.
78    pub ignore_cache: bool,
79    /// Script source injected before any other page script on each frame
80    /// loaded by the reload (`Page.reload.scriptToEvaluateOnLoad`). Omitted
81    /// from the dispatch entirely when `None`.
82    pub script_to_evaluate_on_load: Option<String>,
83}

85/// Per-call knobs for [`Tab::scroll_with`].
86///
87/// `dx` / `dy` are signed pixel distances forwarded verbatim to
88/// `Input.synthesizeScrollGesture`'s `xDistance` / `yDistance`. Following
89/// the CDP convention a **negative** `dy` scrolls the page *down* (content
90/// moves up); a positive `dy` scrolls up. `speed` (px/s) plumbs through to
91/// the gesture's `speed` field when `Some`, and is omitted otherwise (Chrome
92/// picks its default).
93///
94/// For the common cases prefer the [`Tab::scroll_down`] / [`Tab::scroll_up`]
95/// shortcuts, which take an unsigned pixel amount and pick the sign for you.
96///
97/// # Examples
98///
99/// ```no_run
100/// # async fn ex() -> zendriver::Result<()> {
101/// use zendriver::ScrollOptions;
102/// # let browser = zendriver::Browser::builder().launch().await?;
103/// # let tab = browser.main_tab();
104/// // Scroll down 400px and right 50px at a fixed speed.
105/// tab.scroll_with(ScrollOptions {
106///     dx: 50.0,
107///     dy: -400.0,
108///     speed: Some(800),
109/// }).await?;
110/// # Ok(()) }
111/// ```
112#[derive(Debug, Clone, Default)]
113pub struct ScrollOptions {
114    /// Horizontal scroll distance in pixels (`synthesizeScrollGesture.xDistance`).
115    pub dx: f64,
116    /// Vertical scroll distance in pixels (`synthesizeScrollGesture.yDistance`).
117    /// Negative scrolls the page down (CDP convention); positive scrolls up.
118    pub dy: f64,
119    /// Optional gesture speed in pixels/second (`synthesizeScrollGesture.speed`).
120    /// Omitted from the dispatch when `None`.
121    pub speed: Option<i64>,
122}

124/// Runtime user-agent override for [`Tab::set_user_agent_with`].
125///
126/// `accept_language` / `platform` are optional refinements — leave them
127/// `None` to override only the UA string. Each `None` field is omitted from
128/// the `Emulation.setUserAgentOverride` dispatch entirely (not sent as
129/// `null`).
130///
131/// # Stealth interaction (read before using under a stealth profile)
132///
133/// The active [`StealthProfile`](zendriver_stealth::StealthProfile) observer
134/// already issues `Emulation.setUserAgentOverride` carrying a coherent
135/// `userAgentMetadata` block (UA Client-Hints). This override is
136/// **last-write-wins and sends NO `userAgentMetadata`**, so applying it under
137/// the Spoofed profile *clobbers* that Client-Hints coherence and can
138/// *increase* fingerprint detectability (the UA string and the UA-CH high
139/// entropy values would disagree). For stealth, prefer setting the UA through
140/// the stealth profile instead. Use this for non-stealth tabs or a deliberate
141/// per-tab UA change where you accept the coherence trade-off.
142///
143/// # Examples
144///
145/// ```no_run
146/// # async fn ex() -> zendriver::Result<()> {
147/// use zendriver::UserAgentOverride;
148/// # let browser = zendriver::Browser::builder().launch().await?;
149/// # let tab = browser.main_tab();
150/// tab.set_user_agent_with(UserAgentOverride {
151///     user_agent: "Mozilla/5.0 (custom) Gecko/20100101 Firefox/123.0".into(),
152///     accept_language: Some("en-US,en;q=0.9".into()),
153///     platform: Some("Linux x86_64".into()),
154/// }).await?;
155/// # Ok(()) }
156/// ```
157#[derive(Debug, Clone, Default)]
158pub struct UserAgentOverride {
159    /// Full `User-Agent` request-header / `navigator.userAgent` string
160    /// (`Emulation.setUserAgentOverride.userAgent`). Required.
161    pub user_agent: String,
162    /// `Accept-Language` header + `navigator.language(s)` override
163    /// (`Emulation.setUserAgentOverride.acceptLanguage`). Omitted when `None`.
164    pub accept_language: Option<String>,
165    /// `navigator.platform` override
166    /// (`Emulation.setUserAgentOverride.platform`). Omitted when `None`.
167    pub platform: Option<String>,
168}

170/// Document load milestone for [`Tab::wait_for_ready_state`].
171///
172/// Maps to the three values of the DOM `document.readyState` property and
173/// orders them by progress: `Loading` < `Interactive` < `Complete`. Passing a
174/// target to `wait_for_ready_state` polls until the document has reached *at
175/// least* that milestone — e.g. waiting for `Interactive` also returns once
176/// the page is fully `Complete`.
177///
178/// # Examples
179///
180/// ```no_run
181/// # async fn ex() -> zendriver::Result<()> {
182/// use zendriver::ReadyState;
183/// # let browser = zendriver::Browser::builder().launch().await?;
184/// # let tab = browser.main_tab();
185/// tab.goto("https://example.com").await?;
186/// tab.wait_for_ready_state(ReadyState::Complete).await?;
187/// # Ok(()) }
188/// ```
189#[derive(Debug, Clone, Copy, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
190pub enum ReadyState {
191    /// `document.readyState === "loading"` — the document is still parsing.
192    #[serde(rename = "loading")]
193    Loading,
194    /// `document.readyState === "interactive"` — parsed, but sub-resources
195    /// (images, stylesheets, frames) may still be loading.
196    #[serde(rename = "interactive")]
197    Interactive,
198    /// `document.readyState === "complete"` — the document and all
199    /// sub-resources have finished loading (the `load` event has fired).
200    #[serde(rename = "complete")]
201    Complete,
202}
203
204impl ReadyState {
205    /// Monotonic progress rank: `Loading` = 0 < `Interactive` = 1 <
206    /// `Complete` = 2. Used by [`Tab::wait_for_ready_state`] to decide whether
207    /// the observed state has reached the requested milestone.
208    fn rank(self) -> u8 {
209        match self {
210            ReadyState::Loading => 0,
211            ReadyState::Interactive => 1,
212            ReadyState::Complete => 2,
213        }
214    }
215
216    /// Parse a raw `document.readyState` string into a [`ReadyState`].
217    /// Returns `None` for any value other than the three documented states.
218    fn from_dom_str(s: &str) -> Option<Self> {
219        match s {
220            "loading" => Some(ReadyState::Loading),
221            "interactive" => Some(ReadyState::Interactive),
222            "complete" => Some(ReadyState::Complete),
223            _ => None,
224        }
225    }
226}

255/// Handle to a single CDP target session — one open page in Chrome.
256///
257/// `Tab` is `Clone` (cheap — wraps an `Arc`) and `Send + Sync`, so the same
258/// handle can be passed across `tokio::spawn` boundaries freely. Dropping
259/// the last clone tears down the per-Tab background tasks (network tracker,
260/// frame lifecycle subscriber) but does NOT close the page in Chrome — call
261/// [`Tab::close`] for an explicit teardown.
262///
263/// Obtain a `Tab` from [`crate::Browser::main_tab`], [`crate::Browser::new_tab`],
264/// or [`crate::Browser::tabs`].

360impl Drop for TabInner {
361    fn drop(&mut self) {
362        // Signal the spawned `InFlightTracker::run` task to exit. The task
363        // selects on this token alongside the four `Network.*` subscriber
364        // streams; cancellation unblocks the select even if no events are
365        // arriving. Without this the task would leak per Tab on shutdown.
366        self.network_cancel.cancel();
367        // Signal the spawned `frame::lifecycle::run` task to exit. Same
368        // posture as `network_cancel` above — the task selects on this
369        // token alongside the three `Page.frame*` subscriber streams.
370        self.frame_lifecycle_cancel.cancel();
371    }
372}

374/// Tunables for [`Tab::wait_for_idle_opts`].
375///
376/// Construct from [`IdleOptions::default`] and override the fields you care
377/// about:
378///
379/// ```no_run
380/// # use std::time::Duration;
381/// # use zendriver::IdleOptions;
382/// let opts = IdleOptions {
383///     max_inflight_age: Some(Duration::from_secs(5)),
384///     ..Default::default()
385/// };
386/// ```
387#[derive(Debug, Clone)]
388pub struct IdleOptions {
389    /// Outer bound on the whole wait. [`Tab::wait_for_idle_opts`] returns
390    /// [`ZendriverError::Timeout`] once it elapses. Default: 30 s.
391    pub timeout: Duration,
392    /// The in-flight set must stay empty for this long to count as idle.
393    /// Default: 500 ms.
394    pub quiet_window: Duration,
395    /// Requests that have been in flight *longer than* this are ignored when
396    /// judging idleness — they are treated as stuck/background (a hung beacon,
397    /// long-poll, SSE stream, …) rather than active page loading. This lets
398    /// idle resolve even while such a request is still technically open.
399    ///
400    /// `None` (the default) waits for **every** request to terminate, which is
401    /// the historical behavior: a single never-completing request keeps the
402    /// tab non-idle until `timeout`.
403    pub max_inflight_age: Option<Duration>,
404    /// How the wait reacts to a CDP event-stream delivery gap (a lagging
405    /// broadcast subscriber, a reconnect, or the WebSocket dying) observed
406    /// during the wait. Default: [`IdleLossPolicy::Lenient`] — a gap is
407    /// tolerated and the call still resolves best-effort, matching every
408    /// prior release's behavior. Opt into [`IdleLossPolicy::Strict`] to fail
409    /// loudly with [`ZendriverError::EventStreamIncomplete`] instead of
410    /// silently trusting a possibly-incomplete observation.
411    pub loss_policy: IdleLossPolicy,
412}
413
414impl Default for IdleOptions {
415    fn default() -> Self {
416        Self {
417            timeout: Duration::from_secs(30),
418            quiet_window: Duration::from_millis(500),
419            max_inflight_age: None,
420            loss_policy: IdleLossPolicy::Lenient,
421        }
422    }
423}

425impl Tab {
426    pub(crate) fn new(
427        session: SessionHandle,
428        browser: std::sync::Weak<crate::browser::BrowserInner>,
429        input: Arc<InputController>,
430        target_id: String,
431    ) -> Self {
432        // Build the per-Tab network tracker + spawn its background subscriber
433        // task. The task calls `Network.enable` once, then maintains the
434        // in-flight set in response to `Network.requestWillBeSent` (insert)
435        // and `loadingFailed` / `loadingFinished` (remove) events arriving on
436        // this tab's session — `responseReceived` (headers only) is
437        // deliberately not terminal. `wait_for_idle` reads from the same
438        // `network_tracker` Arc.
439        let network_tracker = crate::network_idle::InFlightTracker::new();
440        let network_cancel = tokio_util::sync::CancellationToken::new();
441        tokio::spawn({
442            let tracker = network_tracker.clone();
443            let session_for_task = session.clone();
444            let cancel_for_task = network_cancel.clone();
445            async move {
446                tracker.run(session_for_task, cancel_for_task).await;
447            }
448        });
449
450        // Build the per-Tab frames registry + spawn the lifecycle
451        // subscriber. The task calls `Page.enable` once, then mutates the
452        // registry in response to `Page.frameAttached` (insert),
453        // `Page.frameNavigated` (update url / insert if unseen) and
454        // `Page.frameDetached` (remove). The `Arc<RwLock<_>>` lives on
455        // `TabInner::frames` so `Tab::frames` / `frame_by_url` /
456        // `frame_by_name` can take snapshots without going through the
457        // tracker task. The `Weak<TabInner>` is wired in via
458        // `Arc::new_cyclic` below so every `Frame` constructed by the
459        // subscriber can upgrade back to the owning Tab.
460        let frames: Arc<tokio::sync::RwLock<HashMap<String, Frame>>> =
461            Arc::new(tokio::sync::RwLock::new(HashMap::new()));
462        let frame_lifecycle_cancel = tokio_util::sync::CancellationToken::new();
463
464        let inner = Arc::new_cyclic(|weak: &std::sync::Weak<TabInner>| {
465            tokio::spawn({
466                let session_for_task = session.clone();
467                let frames_for_task = frames.clone();
468                let weak_for_task = weak.clone();
469                let cancel_for_task = frame_lifecycle_cancel.clone();
470                async move {
471                    crate::frame::lifecycle::run(
472                        session_for_task,
473                        frames_for_task,
474                        weak_for_task,
475                        cancel_for_task,
476                    )
477                    .await;
478                }
479            });
480            TabInner {
481                session,
482                isolated_world: tokio::sync::Mutex::new(IsolatedWorldCache::default()),
483                browser,
484                input,
485                target_id,
486                network_tracker,
487                network_cancel,
488                main_frame: tokio::sync::OnceCell::new(),
489                #[cfg(feature = "expect")]
490                download_setup: tokio::sync::OnceCell::new(),
491                frames,
492                frame_lifecycle_cancel,
493                pending_load: tokio::sync::Mutex::new(None),
494                download_behavior_set: std::sync::atomic::AtomicBool::new(false),
495            }
496        });
497
498        Self { inner }
499    }

654    /// The top-level [`Frame`] for this tab.
655    ///
656    /// First call dispatches `Page.getFrameTree` on the tab's session,
657    /// extracts the top-level frame's `id` / `url` / `name`, and constructs
658    /// a [`Frame`] whose session is this tab's session (the main frame is
659    /// always same-process). The result is cached internally so subsequent
660    /// calls return the same `Frame` clone without a round-trip.
661    ///
662    /// # Errors
663    ///
664    /// Returns [`ZendriverError::Navigation`] if Chrome's response is
665    /// missing the top-level frame id.
666    ///
667    /// # Examples
668    ///
669    /// ```no_run
670    /// # async fn ex() -> zendriver::Result<()> {
671    /// # let browser = zendriver::Browser::builder().launch().await?;
672    /// # let tab = browser.main_tab();
673    /// tab.goto("https://example.com").await?;
674    /// let main = tab.main_frame().await?;
675    /// assert!(main.url().await.contains("example.com"));
676    /// # Ok(()) }
677    /// ```
678    pub async fn main_frame(&self) -> Result<Frame> {
679        let frame = self
680            .inner
681            .main_frame
682            .get_or_try_init(|| async {
683                let tree = self.call("Page.getFrameTree", json!({})).await?;
684                let frame_node = &tree["frameTree"]["frame"];
685                let frame_id = frame_node["id"]
686                    .as_str()
687                    .ok_or_else(|| {
688                        ZendriverError::Navigation(
689                            "Page.getFrameTree missing frameTree.frame.id".into(),
690                        )
691                    })?
692                    .to_string();
693                let url = frame_node["url"].as_str().unwrap_or("").to_string();
694                let name = frame_node["name"].as_str().map(str::to_string);
695                Ok::<_, ZendriverError>(Frame::new(
696                    frame_id,
697                    None,
698                    url,
699                    name,
700                    self.inner.session.clone(),
701                    Arc::downgrade(&self.inner),
702                ))
703            })
704            .await?;
705        Ok(frame.clone())
706    }

709    ///
710    /// The registry is maintained by an internal lifecycle subscriber spawned
711    /// when the Tab is constructed (see the [`crate::frame::lifecycle`]
712    /// module). Includes the top-level frame (once Chrome has emitted at
713    /// least one `Page.frameAttached` or `Page.frameNavigated` for it) plus

892    /// Navigate the tab to `url`.
893    ///
894    /// Does NOT wait for the load to complete — call [`Tab::wait_for_load`]
895    /// (or [`Tab::wait_for_idle`]) afterward to block on the navigation.
896    ///
897    /// # Errors
898    ///
899    /// Returns [`ZendriverError::Navigation`] when Chrome reports
900    /// `errorText` on the `Page.navigate` response (e.g. DNS failure,
901    /// connection refused, invalid URL).
902    ///
903    /// # Examples
904    ///
905    /// ```no_run
906    /// # async fn ex() -> zendriver::Result<()> {
907    /// # let browser = zendriver::Browser::builder().launch().await?;
908    /// # let tab = browser.main_tab();
909    /// tab.goto("https://example.com").await?;
910    /// tab.wait_for_load().await?;
911    /// # Ok(()) }
912    /// ```
913    pub async fn goto(&self, url: impl AsRef<str>) -> Result<()> {
914        // Enable Page domain so we get FrameStoppedLoading events.
915        self.call("Page.enable", json!({})).await?;
916        // Subscribe to Page.frameStoppedLoading BEFORE issuing Page.navigate.
917        // The transport's event bus is a tokio broadcast channel, so a
918        // subscriber created after the event has already been published can
919        // never observe it. By subscribing first, then handing a oneshot to
920        // `wait_for_load`, we guarantee the next load event lands in our
921        // receiver regardless of how fast the page loads (e.g. localhost
922        // wiremock fixtures in CI).
923        let mut stream = self
924            .inner
925            .session
926            .subscribe::<Value>("Page.frameStoppedLoading");
927        let (tx, rx) = tokio::sync::oneshot::channel();
928        tokio::spawn(async move {
929            if stream.next().await.is_some() {
930                let _ = tx.send(());
931            }
932        });
933        *self.inner.pending_load.lock().await = Some(rx);
934
935        let url_s = url.as_ref().to_string();
936        let res = self.call("Page.navigate", json!({ "url": url_s })).await?;
937        if let Some(err) = res.get("errorText").and_then(|v| v.as_str()) {
938            if !err.is_empty() {
939                return Err(ZendriverError::Navigation(err.to_string()));
940            }
941        }
942        Ok(())
943    }

945    /// Wait until the main frame's load event fires.
946    ///
947    /// Subscribes to `Page.frameStoppedLoading` and waits for the first
948    /// event. Bounded by a 30s timeout.
949    ///
950    /// # Errors
951    ///
952    /// Returns [`ZendriverError::Timeout`] when no load event arrives
953    /// within 30s; [`ZendriverError::Navigation`] if the event stream
954    /// closes (transport teardown).
955    ///
956    /// # Examples
957    ///
958    /// ```no_run
959    /// # async fn ex() -> zendriver::Result<()> {
960    /// # let browser = zendriver::Browser::builder().launch().await?;
961    /// # let tab = browser.main_tab();
962    /// tab.goto("https://example.com").await?;
963    /// tab.wait_for_load().await?;
964    /// # Ok(()) }
965    /// ```
966    pub async fn wait_for_load(&self) -> Result<()> {
967        // Preferred path: consume the oneshot stashed by `goto`, which
968        // subscribed to `Page.frameStoppedLoading` BEFORE the navigation
969        // request — guaranteed delivery.
970        if let Some(rx) = self.inner.pending_load.lock().await.take() {
971            timeout(DEFAULT_LOAD_TIMEOUT, rx)
972                .await
973                .map_err(|_| ZendriverError::Timeout(DEFAULT_LOAD_TIMEOUT))?
974                .map_err(|_| ZendriverError::Navigation("page event stream closed".into()))?;
975            return Ok(());
976        }
977        // Fallback: no pending navigation (e.g. caller invoked
978        // `wait_for_load` without a preceding `goto`, or after the page
979        // navigated itself). Subscribe + short-circuit on the current
980        // `document.readyState` so an already-loaded page returns
981        // immediately rather than blocking on a missed event.
982        let mut stream = self
983            .inner
984            .session
985            .subscribe::<Value>("Page.frameStoppedLoading");
986        let ready: Option<String> = self
987            .call(
988                "Runtime.evaluate",
989                json!({
990                    "expression": "document.readyState",
991                    "returnByValue": true,
992                }),
993            )
994            .await
995            .ok()
996            .and_then(|v| v.get("result")?.get("value")?.as_str().map(str::to_owned));
997        if ready.as_deref() == Some("complete") {
998            return Ok(());
999        }
1000        timeout(DEFAULT_LOAD_TIMEOUT, stream.next())
1001            .await
1002            .map_err(|_| ZendriverError::Timeout(DEFAULT_LOAD_TIMEOUT))?
1003            .ok_or_else(|| ZendriverError::Navigation("page event stream closed".into()))?;
1004        Ok(())
1005    }

1007    /// Block until the document reaches at least the `until` load milestone.
1008    ///
1009    /// Polls `document.readyState` (via a main-world `Runtime.evaluate`) every
1010    /// [`READY_STATE_POLL_INTERVAL`] and returns as soon as the observed state
1011    /// ranks at or above `until` (see [`ReadyState`] for the
1012    /// `Loading < Interactive < Complete` ordering). Bounded by a 30s outer
1013    /// timeout, matching [`Tab::wait_for_load`].
1014    ///
1015    /// This is the finer-grained sibling of [`Tab::wait_for_load`]: use it when
1016    /// you want to proceed at `Interactive` (DOM parsed, scripts runnable)
1017    /// without waiting for every image / stylesheet that `Complete` implies.
1018    ///
1019    /// # Errors
1020    ///
1021    /// Returns [`ZendriverError::Timeout`] if the requested state is not
1022    /// reached within 30s; propagates [`ZendriverError::JsException`] if the
1023    /// `document.readyState` read raises.
1024    ///
1025    /// # Examples
1026    ///
1027    /// ```no_run
1028    /// # async fn ex() -> zendriver::Result<()> {
1029    /// use zendriver::ReadyState;
1030    /// # let browser = zendriver::Browser::builder().launch().await?;
1031    /// # let tab = browser.main_tab();
1032    /// tab.goto("https://example.com").await?;
1033    /// // Proceed as soon as the DOM is parsed, don't wait for sub-resources.
1034    /// tab.wait_for_ready_state(ReadyState::Interactive).await?;
1035    /// # Ok(()) }
1036    /// ```
1037    pub async fn wait_for_ready_state(&self, until: ReadyState) -> Result<()> {
1038        let deadline = tokio::time::Instant::now() + DEFAULT_LOAD_TIMEOUT;
1039        loop {
1040            let state: String = self.evaluate_main("document.readyState").await?;
1041            if let Some(observed) = ReadyState::from_dom_str(&state) {
1042                if observed.rank() >= until.rank() {
1043                    return Ok(());
1044                }
1045            }
1046            if tokio::time::Instant::now() >= deadline {
1047                return Err(ZendriverError::Timeout(DEFAULT_LOAD_TIMEOUT));
1048            }
1049            tokio::time::sleep(READY_STATE_POLL_INTERVAL).await;
1050        }
1051    }

1053    /// Evaluate a JavaScript expression in an isolated world.
1054    ///
1055    /// Runs in a sandbox where page globals are NOT visible — the default for
1056    /// stealth-safe execution. The result is deserialized into `T`.
1057    ///
1058    /// If the cached isolated-world execution context was destroyed (e.g. by
1059    /// a page navigation), the cache is invalidated and the evaluation is
1060    /// retried once. One retry is enough: the failure mode this guards
1061    /// against is "navigation happened between cache-fetch and `Runtime.evaluate`",
1062    /// which is a one-shot race — recreating the world for the new context
1063    /// and re-issuing the call clears it. If the same call fails the
1064    /// second attempt the page has a real problem (target gone, isolated
1065    /// world refused to recreate) and further retries would only mask it.
1066    ///
1067    /// # Errors
1068    ///
1069    /// Returns [`ZendriverError::JsException`] when the expression raises;
1070    /// [`ZendriverError::Serde`] when the result cannot be decoded into `T`;
1071    /// [`ZendriverError::Navigation`] when the execution context is missing.
1072    ///
1073    /// # Examples
1074    ///
1075    /// ```no_run
1076    /// # async fn ex() -> zendriver::Result<()> {
1077    /// # let browser = zendriver::Browser::builder().launch().await?;
1078    /// # let tab = browser.main_tab();
1079    /// tab.goto("https://example.com").await?;
1080    /// let n: i32 = tab.evaluate("1 + 2").await?;
1081    /// assert_eq!(n, 3);
1082    /// # Ok(()) }
1083    /// ```
1084    pub async fn evaluate<T: DeserializeOwned>(&self, js: impl AsRef<str>) -> Result<T> {
1085        let js = js.as_ref();
1086        for attempt in 0..2 {
1087            let ctx_id = self.ensure_isolated_world().await?;
1088            let res = self
1089                .call(
1090                    "Runtime.evaluate",
1091                    json!({
1092                        "expression": js,
1093                        "contextId": ctx_id,
1094                        "returnByValue": true,
1095                        "awaitPromise": true,
1096                    }),
1097                )
1098                .await;
1099            match res {
1100                Ok(v) => {
1101                    if let Some(details) = v.get("exceptionDetails") {
1102                        let msg = details
1103                            .get("exception")
1104                            .and_then(|e| e.get("description"))
1105                            .and_then(|d| d.as_str())
1106                            .unwrap_or("unknown")
1107                            .to_string();
1108                        return Err(ZendriverError::JsException(msg));
1109                    }
1110                    let value = v
1111                        .get("result")
1112                        .and_then(|r| r.get("value"))
1113                        .cloned()
1114                        .unwrap_or(Value::Null);
1115                    return serde_json::from_value(value).map_err(ZendriverError::Serde);
1116                }
1117                // Chrome returns -32000 "Cannot find context with specified
1118                // id" when the execution context we cached was destroyed
1119                // (typically by a navigation). `From<CallError>` maps that
1120                // to `Navigation` (see `error.rs`), so we match on that
1121                // variant here — not on `Cdp` as the original P2 plan
1122                // suggested.

1687    pub async fn scroll_with(&self, opts: ScrollOptions) -> Result<()> {
1688        // Fixed in-viewport anchor for the gesture. Picking a constant point
1689        // keeps the call deterministic + single-dispatch (no

1776    /// Wait for network idle with full control over the policy via
1777    /// [`IdleOptions`].
1778    ///
1779    /// Algorithm: poll the in-flight set with a `Notify`-driven wake (or a
1780    /// 50ms fallback tick). Each iteration computes the number of *active*
1781    /// requests — every in-flight request when
1782    /// [`IdleOptions::max_inflight_age`] is `None`, otherwise only those in
1783    /// flight for less than that age (older ones are treated as stuck /
1784    /// background and ignored). Track `quiet_start = Some(now)` on the first
1785    /// observation of zero active requests; reset to `None` whenever the active
1786    /// count is non-zero or a membership change fires. Return once
1787    /// `now - quiet_start >= quiet_window`.
1788    ///
1789    /// The 50ms tick bounds latency both for the already-idle case (no further
1790    /// events fire) and for an age-out crossing (which emits no CDP event), so
1791    /// worst-case latency to detect "stayed idle long enough" is
1792    /// `quiet_window + 50ms`.
1793    ///
1794    /// Under [`IdleOptions::loss_policy`] = [`IdleLossPolicy::Strict`], the
1795    /// wait additionally races against this tab's connection's accounted
1796    /// event stream; the first `Lagged` / `Reconnected` / `Disconnected`
1797    /// boundary observed while the wait is in progress aborts it early with
1798    /// [`ZendriverError::EventStreamIncomplete`]. Under the default
1799    /// [`IdleLossPolicy::Lenient`] this extra stream is never subscribed to,
1800    /// so a Lenient wait costs nothing beyond what the tracker's own
1801    /// best-effort subscription already pays.
1802    ///
1803    /// # Errors
1804    ///
1805    /// Returns [`ZendriverError::Timeout`] (carrying [`IdleOptions::timeout`])
1806    /// once the outer deadline elapses. Returns
1807    /// [`ZendriverError::EventStreamIncomplete`] under
1808    /// [`IdleLossPolicy::Strict`] if a delivery gap is observed before the
1809    /// network settles; [`IdleLossPolicy::Lenient`] (the default) never
1810    /// returns this variant.
1811    ///
1812    /// # Examples
1813    ///
1814    /// ```no_run
1815    /// # use std::time::Duration;
1816    /// # use zendriver::IdleOptions;
1817    /// # async fn ex() -> zendriver::Result<()> {
1818    /// # let browser = zendriver::Browser::builder().launch().await?;
1819    /// # let tab = browser.main_tab();
1820    /// tab.goto("https://example.com").await?;
1821    /// // Resolve even if a beacon / long-poll stays open past 5s.
1822    /// tab.wait_for_idle_opts(IdleOptions {
1823    ///     max_inflight_age: Some(Duration::from_secs(5)),
1824    ///     ..Default::default()
1825    /// }).await?;
1826    /// # Ok(()) }
1827    /// ```
1828    pub async fn wait_for_idle_opts(&self, opts: IdleOptions) -> Result<()> {
1829        let IdleOptions {
1830            timeout,
1831            quiet_window,
1832            max_inflight_age,
1833            loss_policy,
1834        } = opts;
1835        let tracker = self.inner.network_tracker.clone();
1836        let deadline = tokio::time::Instant::now() + timeout;
1837        let mut quiet_start: Option<tokio::time::Instant> = None;
1838
1839        // Under `Strict`, subscribe to this tab's connection's accounted
1840        // event stream for the lifetime of this call so a delivery gap
1841        // during the wait aborts it instead of silently reporting a
1842        // possibly-wrong idle. Under `Lenient` we deliberately never call
1843        // `subscribe_raw_accounted` — the accounted bus only pays its
1844        // per-event clone/send cost when it has a live subscriber, so a
1845        // Lenient wait stays as cheap as it always was.
1846        let mut loss_boundary: std::pin::Pin<
1847            Box<dyn futures::Stream<Item = AccountedRawEvent> + Send>,
1848        > = match loss_policy {
1849            IdleLossPolicy::Strict => {
1850                Box::pin(self.session().connection().subscribe_raw_accounted())
1851            }
1852            IdleLossPolicy::Lenient => Box::pin(futures::stream::pending()),
1853        };
1854
1855        loop {
1856            // Arm the notification interest BEFORE reading the in-flight set
1857            // so a notification fired between the read and the `select!`
1858            // below is still delivered. `Notify::notified()` only catches
1859            // notifications fired after the future has been `enable()`d, so
1860            // doing it the other way around would let a request that started
1861            // *and finished* inside the quiet window slip past us with a
1862            // sustained count of 0 — `wait_for_idle` would return early.
1863            let notif = tracker.notifier.notified();
1864            tokio::pin!(notif);
1865            notif.as_mut().enable();
1866
1867            // "Active" = requests that still count toward busy-ness. With
1868            // `max_inflight_age` set, a request in flight longer than that age
1869            // is treated as stuck/background and excluded, so a never-
1870            // terminating request can no longer pin the tab non-idle forever.
1871            let active_count = {
1872                let set = tracker.in_flight.lock().await;
1873                match max_inflight_age {
1874                    None => set.len(),
1875                    Some(age) => {
1876                        let now = tokio::time::Instant::now();
1877                        set.values()
1878                            .filter(|inserted| now.duration_since(**inserted) < age)
1879                            .count()
1880                    }
1881                }
1882            };
1883            if active_count == 0 {
1884                let now = tokio::time::Instant::now();
1885                match quiet_start {
1886                    None => quiet_start = Some(now),
1887                    Some(start) if now.duration_since(start) >= quiet_window => {
1888                        return Ok(());
1889                    }
1890                    _ => {}
1891                }
1892            } else {
1893                quiet_start = None;
1894            }
1895            if tokio::time::Instant::now() >= deadline {
1896                return Err(ZendriverError::Timeout(timeout));
1897            }
1898            tokio::select! {
1899                () = tokio::time::sleep(Duration::from_millis(50)) => {}
1900                () = notif => {
1901                    // A membership change fired since we armed `notif`. Reset
1902                    // the quiet window — even if the set is back to zero by
1903                    // the next iteration, real activity occurred during this
1904                    // window so it doesn't count as "idle".
1905                    quiet_start = None;
1906                }
1907                Some(boundary) = loss_boundary.next() => {
1908                    match boundary {
1909                        AccountedRawEvent::Lagged { .. }
1910                        | AccountedRawEvent::Reconnected { .. }
1911                        | AccountedRawEvent::Disconnected { .. } => {
1912                            // Under `Lenient` this branch is unreachable —
1913                            // `loss_boundary` is `stream::pending()`, which
1914                            // never yields. Under `Strict` we can no longer
1915                            // prove nothing relevant to idleness was missed,
1916                            // so refuse to report a possibly-wrong idle.
1917                            return Err(ZendriverError::EventStreamIncomplete);
1918                        }
1919                        AccountedRawEvent::Event { .. } => {
1920                            // Delivered in order — no loss, keep waiting.
1921                        }
1922                    }
1923                }
1924            }
1925        }
1926    }

2254    /// [`FrameResourceMatch`] records (resource URL + owning frame id).
2255    ///
2256    /// `query` is treated as a plain substring by Chrome (the underlying
2257    /// `searchInResource` defaults to non-regex, case-sensitive matching).
2258    /// Resources that error on search (e.g. a body Chrome no longer retains)
2259    /// are skipped rather than failing the whole call.
2260    ///
2261    /// # Errors
2262    ///
2263    /// Returns [`ZendriverError::Navigation`] if `Page.getResourceTree`'s
2264    /// response is missing the frame tree; transport errors from
2265    /// `getResourceTree` itself propagate.
2266    ///
2267    /// # Examples
2268    ///
2269    /// ```no_run
2270    /// # async fn ex() -> zendriver::Result<()> {
2271    /// # let browser = zendriver::Browser::builder().launch().await?;
2272    /// # let tab = browser.main_tab();
2273    /// tab.goto("https://example.com").await?;
2274    /// let hits = tab.search_frame_resources("__INITIAL_STATE__").await?;
2275    /// println!("{} resources matched", hits.len());
2276    /// # Ok(()) }
2277    /// ```
2278    pub async fn search_frame_resources(&self, query: &str) -> Result<Vec<FrameResourceMatch>> {
2279        let tree = self.call("Page.getResourceTree", json!({})).await?;
2280        let root = tree.get("frameTree").ok_or_else(|| {
2281            ZendriverError::Navigation("Page.getResourceTree missing frameTree".into())
2282        })?;
2283
2284        // Flatten the tree into (frame_id, resource_url) pairs. The frame node
2285        // carries its id under `frame.id`; resources live in `resources[]`
2286        // (each with a `url`), and nested frames in `childFrames[]` (each a
2287        // FrameResourceTree of the same shape).
2288        let mut pairs: Vec<(String, String)> = Vec::new();
2289        let mut stack = vec![root];
2290        while let Some(node) = stack.pop() {
2291            let frame_id = node["frame"]["id"].as_str().unwrap_or("").to_string();
2292            if let Some(resources) = node.get("resources").and_then(Value::as_array) {
2293                for res in resources {
2294                    if let Some(url) = res.get("url").and_then(Value::as_str) {
2295                        pairs.push((frame_id.clone(), url.to_string()));
2296                    }
2297                }
2298            }
2299            if let Some(children) = node.get("childFrames").and_then(Value::as_array) {
2300                stack.extend(children.iter());
2301            }
2302        }
2303
2304        // Search each resource; collect the ones with a non-empty match list.
2305        let mut matches = Vec::new();
2306        for (frame_id, url) in pairs {
2307            let res = self
2308                .call(
2309                    "Page.searchInResource",
2310                    json!({ "frameId": frame_id, "url": url, "query": query }),
2311                )
2312                .await;
2313            // Skip resources Chrome can't search (stale body, unsupported
2314            // type) rather than aborting the whole sweep.
2315            let Ok(res) = res else { continue };
2316            let has_match = res
2317                .get("result")
2318                .and_then(Value::as_array)
2319                .is_some_and(|m| !m.is_empty());
2320            if has_match {
2321                matches.push(FrameResourceMatch { url, frame_id });
2322            }
2323        }
2324        Ok(matches)
2325    }

2645    /// Download the resource at `url` into the tab's download directory,
2646    /// driven entirely from the page itself.
2647    ///
2648    /// Ports nodriver's `download_file` mechanism: it injects a main-world
2649    /// script that `fetch`es `url`, wraps the response body in a `Blob`,
2650    /// creates an object URL, and clicks a synthetic `<a download>` anchor —
2651    /// so the bytes flow through the page's own network context (cookies,

2653    /// configured download behavior. The anchor is removed and the object URL
2654    /// revoked shortly after the click.
2655    ///

2657    /// [`Tab::set_download_path`] or an earlier `download_file`), a default of
2658    /// `<cwd>/downloads` is created and installed first — matching nodriver.
2659    /// When `filename` is `None` the saved name is derived from the URL's last
2660    /// path segment (query string stripped).
2661    ///
2662    /// Returns once the injection script has been dispatched — it does **not**
2663    /// wait for the download to finish. For await/inspect semantics use
2664    /// [`Tab::expect_download`] (gated by the `expect` feature).
2665    ///
2666    /// # Errors
2667    ///
2668    /// Returns [`ZendriverError::Io`] if the default download directory cannot
2669    /// be created; propagates [`ZendriverError::JsException`] if the injected
2670    /// fetch/anchor script raises (e.g. a CORS-blocked `fetch`).
2671    ///
2672    /// # Examples
2673    ///
2674    /// ```no_run
2675    /// # async fn ex() -> zendriver::Result<()> {
2676    /// # let browser = zendriver::Browser::builder().launch().await?;
2677    /// # let tab = browser.main_tab();
2678    /// tab.set_download_path("/tmp/dl").await?;
2679    /// tab.download_file("https://example.com/file.pdf", None).await?;
2680    /// # Ok(()) }
2681    /// ```
2682    pub async fn download_file(
2683        &self,
2684        url: impl Into<String>,
2685        filename: Option<PathBuf>,
2686    ) -> Result<()> {
2687        let url = url.into();
2688
2689        // Establish a default download directory if the caller never set one
2690        // (mirrors nodriver's `_download_behavior` guard so we don't clobber
2691        // an explicitly-chosen directory).
2692        if !self
2693            .inner
2694            .download_behavior_set
2695            .load(std::sync::atomic::Ordering::Relaxed)
2696        {
2697            let dir = std::env::current_dir()?.join("downloads");
2698            std::fs::create_dir_all(&dir)?;
2699            self.set_download_path(dir).await?;
2700        }
2701
2702        // Derive the saved filename from the URL tail (query stripped) when
2703        // not supplied, matching nodriver.
2704        let filename = filename
2705            .as_ref()
2706            .map(|p| p.to_string_lossy().to_string())
2707            .unwrap_or_else(|| {
2708                url.rsplit('/')
2709                    .next()
2710                    .unwrap_or("")
2711                    .split('?')
2712                    .next()
2713                    .unwrap_or("")
2714                    .to_string()
2715            });
2716
2717        // Inject the fetch→blob→anchor[download]→click sequence into the page
2718        // main world (the document's own network context). Values are JSON-
2719        // encoded so quotes/specials in the URL or filename can't break out.
2720        let url_lit = serde_json::to_string(&url)?;
2721        let name_lit = serde_json::to_string(&filename)?;
2722        let js = format!(
2723            "(async () => {{ \
2724                const response = await fetch({url_lit}); \
2725                const blob = await response.blob(); \
2726                const href = URL.createObjectURL(blob); \
2727                const a = document.createElement('a'); \
2728                a.href = href; \
2729                a.download = {name_lit}; \
2730                document.body.appendChild(a); \
2731                a.click(); \
2732                setTimeout(() => {{ document.body.removeChild(a); URL.revokeObjectURL(href); }}, 500); \
2733            }})()"
2734        );
2735        let _: Value = self.evaluate_main(js).await?;
2736        Ok(())
2737    }

2865    /// Register a one-shot expectation for the first `Page.fileChooserOpened`
2866    /// on this tab, and answer it with `paths` — bypassing the OS file
2867    /// dialog the same way [`Element::upload_files`](crate::Element::upload_files)
2868    /// does, but for **button/label-triggered** pickers (a hidden
2869    /// `<input type="file">` clicked from a JS handler, or any custom
2870    /// widget that ultimately opens one) that `upload_files` can't reach
2871    /// because it only knows how to target a direct `<input type="file">`'s
2872    /// backend node.
2873    ///
2874    /// `paths` is captured up front (converted via `to_string_lossy()`,
2875    /// matching CDP's UTF-8 string contract) and applied automatically the
2876    /// instant the chooser opens.
2877    ///
2878    /// `async` because `Page.setInterceptFileChooserDialog { enabled: true
2879    /// }` must reach Chrome before the caller's next line — usually the
2880    /// click that opens the picker — or that click could race Chrome into
2881    /// showing the real OS dialog instead of firing the intercept event.
2882    /// Same reasoning as [`Tab::expect_download`]'s own setup call.
2883    ///
2884    /// The returned
2885    /// [`FileChooserExpectation`](crate::expect::file_chooser::FileChooserExpectation)
2886    /// is awaitable directly (`expectation.await`) or via the
2887    /// Playwright-style `expectation.matched().await`; configure the
2888    /// timeout via
2889    /// [`timeout`](crate::expect::file_chooser::FileChooserExpectation::timeout)
2890    /// before awaiting. If the expectation is dropped before a chooser
2891    /// opens (timeout, early return, panic unwind), the intercept is
2892    /// disabled best-effort so a later real dialog isn't silently
2893    /// swallowed.
2894    ///
2895    /// # Examples
2896    ///
2897    /// ```no_run
2898    /// # async fn ex() -> zendriver::Result<()> {
2899    /// # let browser = zendriver::Browser::builder().launch().await?;
2900    /// # let tab = browser.main_tab();
2901    /// let fc = tab.expect_file_chooser(&["/tmp/photo.jpg"]).await?;
2902    /// let button = tab.find().css("#upload-btn").one().await?;
2903    /// button.click().await?; // opens the picker via a hidden input
2904    /// fc.await?; // intercepts fileChooserOpened + sets the files
2905    /// # Ok(()) }
2906    /// ```
2907    ///
2908    /// Gated by the `expect` cargo feature.
2909    pub async fn expect_file_chooser<P: AsRef<std::path::Path>>(
2910        &self,
2911        paths: &[P],
2912    ) -> Result<crate::expect::file_chooser::FileChooserExpectation> {
2913        let files: Vec<String> = paths
2914            .iter()
2915            .map(|p| p.as_ref().to_string_lossy().into_owned())
2916            .collect();
2917        crate::expect::file_chooser::register(self.session(), files).await
2918    }
