1//! Per-page handle to a single CDP target session.
2//!
3//! [`Tab`] is the primary interaction surface in zendriver — most workflows
4//! are some sequence of `goto`, `find().css(...).one()`, `evaluate`,
5//! `screenshot`, and `wait_for_idle`. Each [`Tab`] owns its own
6//! [`InputController`] (cursor + held-modifier state), its own per-tab
7//! frame registry, and its own in-flight network tracker, so multiple tabs
8//! in the same [`crate::Browser`] don't interfere with one another.
9//!
10//! ```no_run
11//! # async fn ex() -> zendriver::Result<()> {
12//! let browser = zendriver::Browser::builder().launch().await?;
13//! let tab = browser.main_tab();
14//! tab.goto("https://example.com").await?;
15//! tab.wait_for_load().await?;
16//! let title: String = tab.evaluate_main("document.title").await?;
17//! assert_eq!(title, "Example Domain");
18//! # Ok(()) }
19//! ```

40/// Handle to a single CDP target session — one open page in Chrome.
41///
42/// `Tab` is `Clone` (cheap — wraps an `Arc`) and `Send + Sync`, so the same
43/// handle can be passed across `tokio::spawn` boundaries freely. Dropping
44/// the last clone tears down the per-Tab background tasks (network tracker,
45/// frame lifecycle subscriber) but does NOT close the page in Chrome — call
46/// [`Tab::close`] for an explicit teardown.
47///
48/// Obtain a `Tab` from [`crate::Browser::main_tab`], [`crate::Browser::new_tab`],
49/// or [`crate::Browser::tabs`].
50#[derive(Clone, Debug)]
51pub struct Tab {
52    pub(crate) inner: Arc<TabInner>,
53}
54
55#[derive(Debug)]
56pub(crate) struct TabInner {
57    pub(crate) session: SessionHandle,
58    pub(crate) isolated_world: tokio::sync::Mutex<IsolatedWorldCache>,
59    /// Weak ref to the owning `BrowserInner`. Used by [`Tab::cookies`] to
60    /// hand back a [`crate::CookieJar`] bound to the browser's root
61    /// connection (Chrome's cookie store is browser-scoped, so per-tab jars
62    /// would all dispatch the same way). Reserved for future P4 tasks
63    /// (tabs registry walks, storage). `Weak` breaks the Browser→Tab→Browser
64    /// cycle.
65    pub(crate) browser: std::sync::Weak<crate::browser::BrowserInner>,
66    /// Per-Tab input controller. Each tab owns its own cursor + held-modifier
67    /// state — distinct tabs in the same Browser have independent pointers.
68    /// `Element` actions clone this `Arc` to drive `mouse::*` / `keyboard::*`
69    /// dispatch helpers; the shared mutex inside `InputController` serializes
70    /// per-tab writes without crossing tab boundaries.
71    pub(crate) input: Arc<InputController>,
72    /// CDP `targetId` for the page target this tab wraps. Cached at Tab
73    /// construction time (from `Target.attachedToTarget`'s `target_info`)
74    /// so multi-tab orchestration (`Browser::new_tab` correlation,
75    /// `Tab::activate`, `Tab::close`'s `Target.closeTarget` upgrade) can
76    /// dispatch by `targetId` without re-querying `Target.getTargetInfo`
77    /// per call.
78    pub(crate) target_id: String,
79    /// Per-Tab in-flight network request tracker. Constructed in
80    /// [`Tab::new`] alongside a background task (spawned via
81    /// [`crate::network_idle::InFlightTracker::run`]) that subscribes to
82    /// `Network.*` events and maintains the set. Consulted by
83    /// [`Tab::wait_for_idle`] / [`Tab::wait_for_idle_with`] for Playwright
84    /// `networkidle` semantics.
85    pub(crate) network_tracker: Arc<crate::network_idle::InFlightTracker>,
86    /// Cancellation token for the background tracker task. Fires on
87    /// [`Drop`] so the spawned task exits cleanly when the last clone of
88    /// this Tab goes away. Cloned by the spawned task at construction
89    /// time; cancelling here propagates to the task's `tokio::select!`
90    /// loop within one event tick.
91    pub(crate) network_cancel: tokio_util::sync::CancellationToken,
92    /// Lazily-discovered main [`Frame`] for this tab. First call to
93    /// [`Tab::main_frame`] sends `Page.getFrameTree`, extracts the top-level
94    /// frame id/url/name, constructs a `Frame` (sharing this tab's
95    /// session — the main frame is always same-process), and stores it
96    /// here. Subsequent calls return the cached `Frame` clone without
97    /// another round-trip.
98    pub(crate) main_frame: tokio::sync::OnceCell<Frame>,
99    /// Per-Tab download coordinator. Lazily initialized on the first
100    /// [`Tab::expect_download`] call (gated `expect`) — the constructor
101    /// allocates a tempdir, dispatches `Browser.setDownloadBehavior` once,
102    /// and spawns a long-running `Page.downloadProgress` subscriber. Held
103    /// behind a [`tokio::sync::OnceCell`] so the wiring happens exactly
104    /// once per Tab; subsequent `expect_download` calls reuse the same
105    /// coordinator (and therefore the same tempdir + subscriber).
106    ///
107    /// `Arc` because both the [`Tab`] (via this cell) and the spawned
108    /// progress subscriber task hold references to the same coordinator
109    /// state for the Tab's entire lifetime.
110    #[cfg(feature = "expect")]
111    pub(crate) download_setup:
112        tokio::sync::OnceCell<Arc<crate::expect::download::DownloadCoordinator>>,
113    /// Per-Tab frames registry keyed by CDP `frameId`. Populated by the
114    /// background subscriber spawned in [`Tab::new`] via
115    /// [`crate::frame::lifecycle::run`] which mutates the map in response
116    /// to `Page.frameAttached` / `Page.frameDetached` /
117    /// `Page.frameNavigated` events on this tab's session. Read by
118    /// [`Tab::frames`] / [`Tab::frame_by_url`] / [`Tab::frame_by_name`].
119    ///
120    /// Same-origin sub-frames go in this map directly; out-of-process
121    /// iframes (OOPIFs) take the `Target.attachedToTarget` path wired in
122    /// T16 and land here only after that observer registers them.
123    pub(crate) frames: Arc<tokio::sync::RwLock<HashMap<String, Frame>>>,
124    /// Cancellation token for the frame lifecycle subscriber task. Mirror
125    /// of [`TabInner::network_cancel`]: fires on [`Drop`] so the spawned
126    /// task exits cleanly when the last clone of this Tab goes away. The
127    /// task selects on this token alongside the three `Page.frame*`
128    /// subscriber streams so cancellation unblocks the select even if no
129    /// events are arriving.
130    pub(crate) frame_lifecycle_cancel: tokio_util::sync::CancellationToken,
131    /// Oneshot receiver published by [`Tab::goto`] (subscribes to
132    /// `Page.frameStoppedLoading` BEFORE issuing `Page.navigate` so the
133    /// event can't race past) and consumed by [`Tab::wait_for_load`].
134    /// `None` when there is no pending navigation — `wait_for_load` falls
135    /// back to a fresh subscribe + `document.readyState` short-circuit.
136    pub(crate) pending_load: tokio::sync::Mutex<Option<tokio::sync::oneshot::Receiver<()>>>,
137}

139impl Drop for TabInner {
140    fn drop(&mut self) {
141        // Signal the spawned `InFlightTracker::run` task to exit. The task
142        // selects on this token alongside the four `Network.*` subscriber
143        // streams; cancellation unblocks the select even if no events are
144        // arriving. Without this the task would leak per Tab on shutdown.
145        self.network_cancel.cancel();
146        // Signal the spawned `frame::lifecycle::run` task to exit. Same
147        // posture as `network_cancel` above — the task selects on this
148        // token alongside the three `Page.frame*` subscriber streams.
149        self.frame_lifecycle_cancel.cancel();
150    }
151}

153impl Tab {
154    pub(crate) fn new(
155        session: SessionHandle,
156        browser: std::sync::Weak<crate::browser::BrowserInner>,
157        input: Arc<InputController>,
158        target_id: String,
159    ) -> Self {
160        // Build the per-Tab network tracker + spawn its background subscriber
161        // task. The task calls `Network.enable` once, then maintains the
162        // in-flight set in response to `Network.requestWillBeSent` /
163        // `responseReceived` / `loadingFailed` / `loadingFinished` events
164        // arriving on this tab's session. `wait_for_idle` reads from the
165        // same `network_tracker` Arc.
166        let network_tracker = crate::network_idle::InFlightTracker::new();
167        let network_cancel = tokio_util::sync::CancellationToken::new();
168        tokio::spawn({
169            let tracker = network_tracker.clone();
170            let session_for_task = session.clone();
171            let cancel_for_task = network_cancel.clone();
172            async move {
173                tracker.run(session_for_task, cancel_for_task).await;
174            }
175        });
176
177        // Build the per-Tab frames registry + spawn the lifecycle
178        // subscriber. The task calls `Page.enable` once, then mutates the
179        // registry in response to `Page.frameAttached` (insert),
180        // `Page.frameNavigated` (update url / insert if unseen) and
181        // `Page.frameDetached` (remove). The `Arc<RwLock<_>>` lives on
182        // `TabInner::frames` so `Tab::frames` / `frame_by_url` /
183        // `frame_by_name` can take snapshots without going through the
184        // tracker task. The `Weak<TabInner>` is wired in via
185        // `Arc::new_cyclic` below so every `Frame` constructed by the
186        // subscriber can upgrade back to the owning Tab.
187        let frames: Arc<tokio::sync::RwLock<HashMap<String, Frame>>> =
188            Arc::new(tokio::sync::RwLock::new(HashMap::new()));
189        let frame_lifecycle_cancel = tokio_util::sync::CancellationToken::new();
190
191        let inner = Arc::new_cyclic(|weak: &std::sync::Weak<TabInner>| {
192            tokio::spawn({
193                let session_for_task = session.clone();
194                let frames_for_task = frames.clone();
195                let weak_for_task = weak.clone();
196                let cancel_for_task = frame_lifecycle_cancel.clone();
197                async move {
198                    crate::frame::lifecycle::run(
199                        session_for_task,
200                        frames_for_task,
201                        weak_for_task,
202                        cancel_for_task,
203                    )
204                    .await;
205                }
206            });
207            TabInner {
208                session,
209                isolated_world: tokio::sync::Mutex::new(IsolatedWorldCache::default()),
210                browser,
211                input,
212                target_id,
213                network_tracker,
214                network_cancel,
215                main_frame: tokio::sync::OnceCell::new(),
216                #[cfg(feature = "expect")]
217                download_setup: tokio::sync::OnceCell::new(),
218                frames,
219                frame_lifecycle_cancel,
220                pending_load: tokio::sync::Mutex::new(None),
221            }
222        });
223
224        Self { inner }
225    }

305    /// The top-level [`Frame`] for this tab.
306    ///
307    /// First call dispatches `Page.getFrameTree` on the tab's session,
308    /// extracts the top-level frame's `id` / `url` / `name`, and constructs
309    /// a [`Frame`] whose session is this tab's session (the main frame is
310    /// always same-process). The result is cached internally so subsequent
311    /// calls return the same `Frame` clone without a round-trip.
312    ///
313    /// # Errors
314    ///
315    /// Returns [`ZendriverError::Navigation`] if Chrome's response is
316    /// missing the top-level frame id.
317    ///
318    /// # Examples
319    ///
320    /// ```no_run
321    /// # async fn ex() -> zendriver::Result<()> {
322    /// # let browser = zendriver::Browser::builder().launch().await?;
323    /// # let tab = browser.main_tab();
324    /// tab.goto("https://example.com").await?;
325    /// let main = tab.main_frame().await?;
326    /// assert!(main.url().await.contains("example.com"));
327    /// # Ok(()) }
328    /// ```
329    pub async fn main_frame(&self) -> Result<Frame> {
330        let frame = self
331            .inner
332            .main_frame
333            .get_or_try_init(|| async {
334                let tree = self.call("Page.getFrameTree", json!({})).await?;
335                let frame_node = &tree["frameTree"]["frame"];
336                let frame_id = frame_node["id"]
337                    .as_str()
338                    .ok_or_else(|| {
339                        ZendriverError::Navigation(
340                            "Page.getFrameTree missing frameTree.frame.id".into(),
341                        )
342                    })?
343                    .to_string();
344                let url = frame_node["url"].as_str().unwrap_or("").to_string();
345                let name = frame_node["name"].as_str().map(str::to_string);
346                Ok::<_, ZendriverError>(Frame::new(
347                    frame_id,
348                    None,
349                    url,
350                    name,
351                    self.inner.session.clone(),
352                    Arc::downgrade(&self.inner),
353                ))
354            })
355            .await?;
356        Ok(frame.clone())
357    }

359    /// Snapshot of all currently-registered frames for this tab.
360    ///
361    /// The registry is maintained by an internal lifecycle subscriber spawned
362    /// when the Tab is constructed (see the [`crate::frame::lifecycle`]
363    /// module). Includes the top-level frame (once Chrome has emitted at
364    /// least one `Page.frameAttached` or `Page.frameNavigated` for it) plus

366    /// this map via the [`crate::frame::oopif`] observer path.
367    ///
368    /// Order is unspecified ([`HashMap`] iteration); callers that need a
369    /// stable order should sort by [`Frame::id`] or [`Frame::url`].

371    /// # Examples
372    ///

374    /// # async fn ex() -> zendriver::Result<()> {
375    /// # let browser = zendriver::Browser::builder().launch().await?;
376    /// # let tab = browser.main_tab();
377    /// tab.goto("https://example.com").await?;

384        Ok(self.inner.frames.read().await.values().cloned().collect())

387    /// First frame in [`Tab::frames`] whose URL contains `url_substr`.
388    ///
389    /// Linear scan over the registry. Useful for picking a frame by its
390    /// origin (e.g. `tab.frame_by_url("docs.google.com")`) without knowing
391    /// the exact path. Returns `Ok(None)` if no frame matches; the registry
392    /// lock is released before returning so concurrent updates can land.
393    ///
394    /// # Examples
395    ///
396    /// ```no_run
397    /// # async fn ex() -> zendriver::Result<()> {
398    /// # let browser = zendriver::Browser::builder().launch().await?;
399    /// # let tab = browser.main_tab();
400    /// tab.goto("https://example.com").await?;
401    /// if let Some(iframe) = tab.frame_by_url("youtube.com").await? {
402    ///     println!("found iframe: {}", iframe.url().await);
403    /// }
404    /// # Ok(()) }
405    /// ```
406    pub async fn frame_by_url(&self, url_substr: &str) -> Result<Option<Frame>> {
407        let map = self.inner.frames.read().await;
408        for frame in map.values() {
409            if frame.url().await.contains(url_substr) {
410                return Ok(Some(frame.clone()));
411            }
412        }
413        Ok(None)
414    }

538    /// Navigate the tab to `url`.
539    ///
540    /// Does NOT wait for the load to complete — call [`Tab::wait_for_load`]
541    /// (or [`Tab::wait_for_idle`]) afterward to block on the navigation.
542    ///
543    /// # Errors
544    ///
545    /// Returns [`ZendriverError::Navigation`] when Chrome reports
546    /// `errorText` on the `Page.navigate` response (e.g. DNS failure,
547    /// connection refused, invalid URL).
548    ///
549    /// # Examples
550    ///
551    /// ```no_run
552    /// # async fn ex() -> zendriver::Result<()> {
553    /// # let browser = zendriver::Browser::builder().launch().await?;
554    /// # let tab = browser.main_tab();
555    /// tab.goto("https://example.com").await?;
556    /// tab.wait_for_load().await?;
557    /// # Ok(()) }
558    /// ```
559    pub async fn goto(&self, url: impl AsRef<str>) -> Result<()> {
560        // Enable Page domain so we get FrameStoppedLoading events.
561        self.call("Page.enable", json!({})).await?;
562        // Subscribe to Page.frameStoppedLoading BEFORE issuing Page.navigate.
563        // The transport's event bus is a tokio broadcast channel, so a
564        // subscriber created after the event has already been published can
565        // never observe it. By subscribing first, then handing a oneshot to
566        // `wait_for_load`, we guarantee the next load event lands in our
567        // receiver regardless of how fast the page loads (e.g. localhost
568        // wiremock fixtures in CI).
569        let mut stream = self
570            .inner
571            .session
572            .subscribe::<Value>("Page.frameStoppedLoading");
573        let (tx, rx) = tokio::sync::oneshot::channel();
574        tokio::spawn(async move {
575            if stream.next().await.is_some() {
576                let _ = tx.send(());
577            }
578        });
579        *self.inner.pending_load.lock().await = Some(rx);
580
581        let url_s = url.as_ref().to_string();
582        let res = self.call("Page.navigate", json!({ "url": url_s })).await?;
583        if let Some(err) = res.get("errorText").and_then(|v| v.as_str()) {
584            if !err.is_empty() {
585                return Err(ZendriverError::Navigation(err.to_string()));
586            }
587        }
588        Ok(())
589    }

591    /// Wait until the main frame's load event fires.
592    ///
593    /// Subscribes to `Page.frameStoppedLoading` and waits for the first
594    /// event. Bounded by a 30s timeout.
595    ///
596    /// # Errors
597    ///
598    /// Returns [`ZendriverError::Timeout`] when no load event arrives
599    /// within 30s; [`ZendriverError::Navigation`] if the event stream
600    /// closes (transport teardown).
601    ///
602    /// # Examples
603    ///
604    /// ```no_run
605    /// # async fn ex() -> zendriver::Result<()> {
606    /// # let browser = zendriver::Browser::builder().launch().await?;
607    /// # let tab = browser.main_tab();
608    /// tab.goto("https://example.com").await?;
609    /// tab.wait_for_load().await?;
610    /// # Ok(()) }
611    /// ```
612    pub async fn wait_for_load(&self) -> Result<()> {
613        // Preferred path: consume the oneshot stashed by `goto`, which
614        // subscribed to `Page.frameStoppedLoading` BEFORE the navigation
615        // request — guaranteed delivery.
616        if let Some(rx) = self.inner.pending_load.lock().await.take() {
617            timeout(DEFAULT_LOAD_TIMEOUT, rx)
618                .await
619                .map_err(|_| ZendriverError::Timeout(DEFAULT_LOAD_TIMEOUT))?
620                .map_err(|_| ZendriverError::Navigation("page event stream closed".into()))?;
621            return Ok(());
622        }
623        // Fallback: no pending navigation (e.g. caller invoked
624        // `wait_for_load` without a preceding `goto`, or after the page
625        // navigated itself). Subscribe + short-circuit on the current
626        // `document.readyState` so an already-loaded page returns
627        // immediately rather than blocking on a missed event.
628        let mut stream = self
629            .inner
630            .session
631            .subscribe::<Value>("Page.frameStoppedLoading");
632        let ready: Option<String> = self
633            .call(
634                "Runtime.evaluate",
635                json!({
636                    "expression": "document.readyState",
637                    "returnByValue": true,
638                }),
639            )
640            .await
641            .ok()
642            .and_then(|v| v.get("result")?.get("value")?.as_str().map(str::to_owned));
643        if ready.as_deref() == Some("complete") {
644            return Ok(());
645        }
646        timeout(DEFAULT_LOAD_TIMEOUT, stream.next())
647            .await
648            .map_err(|_| ZendriverError::Timeout(DEFAULT_LOAD_TIMEOUT))?
649            .ok_or_else(|| ZendriverError::Navigation("page event stream closed".into()))?;
650        Ok(())
651    }

653    /// Evaluate a JavaScript expression in an isolated world.
654    ///
655    /// Runs in a sandbox where page globals are NOT visible — the default for
656    /// stealth-safe execution. The result is deserialized into `T`.
657    ///
658    /// If the cached isolated-world execution context was destroyed (e.g. by
659    /// a page navigation), the cache is invalidated and the evaluation is
660    /// retried once. One retry is enough: the failure mode this guards
661    /// against is "navigation happened between cache-fetch and `Runtime.evaluate`",
662    /// which is a one-shot race — recreating the world for the new context
663    /// and re-issuing the call clears it. If the same call fails the
664    /// second attempt the page has a real problem (target gone, isolated
665    /// world refused to recreate) and further retries would only mask it.
666    ///
667    /// # Errors
668    ///
669    /// Returns [`ZendriverError::JsException`] when the expression raises;
670    /// [`ZendriverError::Serde`] when the result cannot be decoded into `T`;
671    /// [`ZendriverError::Navigation`] when the execution context is missing.
672    ///
673    /// # Examples
674    ///
675    /// ```no_run
676    /// # async fn ex() -> zendriver::Result<()> {
677    /// # let browser = zendriver::Browser::builder().launch().await?;
678    /// # let tab = browser.main_tab();
679    /// tab.goto("https://example.com").await?;
680    /// let n: i32 = tab.evaluate("1 + 2").await?;
681    /// assert_eq!(n, 3);
682    /// # Ok(()) }
683    /// ```
684    pub async fn evaluate<T: DeserializeOwned>(&self, js: impl AsRef<str>) -> Result<T> {
685        let js = js.as_ref();
686        for attempt in 0..2 {
687            let ctx_id = self.ensure_isolated_world().await?;
688            let res = self
689                .call(
690                    "Runtime.evaluate",
691                    json!({
692                        "expression": js,
693                        "contextId": ctx_id,
694                        "returnByValue": true,
695                        "awaitPromise": true,
696                    }),
697                )
698                .await;
699            match res {
700                Ok(v) => {
701                    if let Some(details) = v.get("exceptionDetails") {
702                        let msg = details
703                            .get("exception")
704                            .and_then(|e| e.get("description"))
705                            .and_then(|d| d.as_str())
706                            .unwrap_or("unknown")
707                            .to_string();
708                        return Err(ZendriverError::JsException(msg));
709                    }
710                    let value = v
711                        .get("result")
712                        .and_then(|r| r.get("value"))
713                        .cloned()
714                        .unwrap_or(Value::Null);
715                    return serde_json::from_value(value).map_err(ZendriverError::Serde);
716                }
717                // Chrome returns -32000 "Cannot find context with specified
718                // id" when the execution context we cached was destroyed
719                // (typically by a navigation). `From<CallError>` maps that
720                // to `Navigation` (see `error.rs`), so we match on that
721                // variant here — not on `Cdp` as the original P2 plan
722                // suggested.
723                Err(ZendriverError::Navigation(ref m))
724                    if attempt == 0 && m.contains("Cannot find context") =>
725                {
726                    self.inner.isolated_world.lock().await.context_id = None;
727                    continue;
728                }
729                Err(e) => return Err(e),
730            }
731        }
732        unreachable!()
733    }

1124    /// Wait until the tab's network has been idle for `quiet_window`,
1125    /// bounded by `timeout`.
1126    ///
1127    /// Algorithm: poll the in-flight set with a `Notify`-driven wake (or a
1128    /// 50ms fallback tick). Track `quiet_start = Some(now)` on the first
1129    /// observation of an empty set; reset to `None` on any observation
1130    /// where the set is non-empty. Return once `now - quiet_start
1131    /// >= quiet_window`.
1132    ///
1133    /// The 50ms tick is a safety net for the case where the tracker is
1134    /// already at 0 in-flight requests and no further events fire to wake
1135    /// the notifier. Worst-case latency to detect "stayed idle long enough"
1136    /// is `quiet_window + 50ms`.
1137    ///
1138    /// # Errors
1139    ///
1140    /// Returns [`ZendriverError::Timeout`] (carrying the supplied `timeout`)
1141    /// once the outer deadline elapses.
1142    ///
1143    /// # Examples
1144    ///
1145    /// ```no_run
1146    /// # use std::time::Duration;
1147    /// # async fn ex() -> zendriver::Result<()> {
1148    /// # let browser = zendriver::Browser::builder().launch().await?;
1149    /// # let tab = browser.main_tab();
1150    /// tab.goto("https://example.com").await?;
1151    /// tab.wait_for_idle_with(
1152    ///     Duration::from_secs(60),
1153    ///     Duration::from_secs(1),
1154    /// ).await?;
1155    /// # Ok(()) }
1156    /// ```
1157    pub async fn wait_for_idle_with(
1158        &self,
1159        timeout: Duration,
1160        quiet_window: Duration,
1161    ) -> Result<()> {
1162        let tracker = self.inner.network_tracker.clone();
1163        let deadline = tokio::time::Instant::now() + timeout;
1164        let mut quiet_start: Option<tokio::time::Instant> = None;
1165        loop {
1166            // Arm the notification interest BEFORE reading the in-flight set
1167            // so a notification fired between the read and the `select!`
1168            // below is still delivered. `Notify::notified()` only catches
1169            // notifications fired after the future has been `enable()`d, so
1170            // doing it the other way around would let a request that started
1171            // *and finished* inside the quiet window slip past us with a
1172            // sustained count of 0 — `wait_for_idle` would return early.
1173            let notif = tracker.notifier.notified();
1174            tokio::pin!(notif);
1175            notif.as_mut().enable();
1176
1177            let in_flight_count = tracker.in_flight.lock().await.len();
1178            if in_flight_count == 0 {
1179                let now = tokio::time::Instant::now();
1180                match quiet_start {
1181                    None => quiet_start = Some(now),
1182                    Some(start) if now.duration_since(start) >= quiet_window => {
1183                        return Ok(());
1184                    }
1185                    _ => {}
1186                }
1187            } else {
1188                quiet_start = None;
1189            }
1190            if tokio::time::Instant::now() >= deadline {
1191                return Err(ZendriverError::Timeout(timeout));
1192            }
1193            tokio::select! {
1194                () = tokio::time::sleep(Duration::from_millis(50)) => {}
1195                () = notif => {
1196                    // A membership change fired since we armed `notif`. Reset
1197                    // the quiet window — even if the set is back to zero by
1198                    // the next iteration, real activity occurred during this
1199                    // window so it doesn't count as "idle".
1200                    quiet_start = None;
1201                }
1202            }
1203        }
1204    }
1205}

2356    /// Regression for the "burst within tick" race: a request that fires
2357    /// *and finishes* inside one 50ms poll tick takes the in-flight set
2358    /// 0 → 1 → 0 without ever being observed at a >0 read. A naive
2359    /// implementation that only checks the set len would see sustained
2360    /// 0 and resolve early. The fix arms a `Notify::notified()` future
2361    /// before each count read so the two membership events from the burst
2362    /// both hit an armed waker; on the next iteration we wake via the
2363    /// notifier arm and reset `quiet_start`.
2364    #[tokio::test]
2365    async fn wait_for_idle_burst_inside_tick_resets_quiet_window() {
2366        let (mut mock, conn) = MockConnection::pair();
2367        let sess = SessionHandle::new(conn.clone(), "S1");
2368        let tab = Tab::new_for_test(sess);
2369
2370        let id_enable =
2371            tokio::time::timeout(Duration::from_secs(2), mock.expect_cmd("Network.enable"))
2372                .await
2373                .expect("tracker did not send Network.enable within 2s");
2374        mock.reply(id_enable, json!({})).await;
2375
2376        // Start with the set already at 0 — wait_for_idle should accumulate
2377        // a quiet window from T0. The inner `wait_for_idle_with` timeout is
2378        // deliberately generous (30s) so a slow / loaded CI runner doesn't
2379        // flake the test; the correctness assertion below uses a strict
2380        // lower bound on `elapsed` and an outer 10s tokio::time::timeout to
2381        // catch a too-early resolve or a hang.
2382        let fut = tokio::spawn({
2383            let t = tab.clone();
2384            async move {
2385                t.wait_for_idle_with(Duration::from_secs(30), Duration::from_millis(300))
2386                    .await
2387            }
2388        });
2389        let started_at = tokio::time::Instant::now();
2390
2391        // Let the wait_for_idle loop tick once on an empty set so
2392        // `quiet_start` is firmly armed.
2393        tokio::time::sleep(Duration::from_millis(75)).await;
2394
2395        // Burst: fire R-burst's requestWillBeSent + responseReceived
2396        // back-to-back. The tracker should observe both transitions and
2397        // notify on each; the wait_for_idle loop should wake on the notif
2398        // arm and reset quiet_start even though the set's instantaneous
2399        // value returns to 0.
2400        mock.emit_event_for_session(
2401            "Network.requestWillBeSent",
2402            json!({ "requestId": "Rburst" }),
2403            "S1",
2404        )
2405        .await;
2406        mock.emit_event_for_session(
2407            "Network.responseReceived",
2408            json!({ "requestId": "Rburst" }),
2409            "S1",
2410        )
2411        .await;
2412
2413        // Outer timeout deliberately generous (10s) so a slow / loaded
2414        // CI runner doesn't flake the test. The correctness assertion
2415        // below uses a strict lower bound on `elapsed` to catch a
2416        // too-early resolve regardless of how long the slack window is.
2417        let res = tokio::time::timeout(Duration::from_secs(10), fut)
2418            .await
2419            .expect("wait_for_idle did not resolve within 10s");
2420        res.unwrap().unwrap();
2421
2422        // Lower bound: 75ms initial sleep + 300ms quiet window after the
2423        // burst's last notification = 375ms. A bug that ignored the burst
2424        // would resolve at started_at + 300ms = 300ms.
2425        let elapsed = started_at.elapsed();
2426        assert!(
2427            elapsed >= Duration::from_millis(355),
2428            "wait_for_idle resolved too early ({elapsed:?}); 0→1→0 burst inside \
2429             quiet window must reset quiet_start",
2430        );
2431
2432        conn.shutdown();
2433    }
