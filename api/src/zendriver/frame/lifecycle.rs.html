1//! Frame lifecycle event subscriber — internal.
2//!
3//! Spawned per [`crate::Tab`] at construction time. Subscribes to
4//! `Page.frameAttached`, `Page.frameDetached`, and `Page.frameNavigated`
5//! events on the owning tab's session and maintains the tab's frames
6//! registry (read via [`crate::Tab::frames`]).
7//!
8//! ## Wiring
9//!
10//! Each [`crate::Tab`] owns one background task. At construction time the
11//! task:
12//! 1. Subscribes to the three `Page.frame*` event streams BEFORE awaiting
13//!    `Page.enable` — avoids the race where Chrome fires events between
14//!    the enable reply and our subscription registration.
15//! 2. Fires `Page.enable` once on the tab's session so Chrome starts
16//!    emitting `Page.frame*` events.
17//! 3. Loops over the merged event stream, mutating the registry:
18//!    - `Page.frameAttached` — construct a new [`crate::Frame`] sharing
19//!      the tab's session (same-origin sub-frame; OOPIF frames arrive
20//!      through the [`crate::frame::oopif`] observer path on their own
21//!      child session, NOT this stream) and insert it under `frameId`.
22//!    - `Page.frameNavigated` — update the existing entry's URL in
23//!      place; insert a fresh [`crate::Frame`] if no entry exists.
24//!    - `Page.frameDetached` — remove the entry from the registry.
25//!
26//! The task runs until its [`tokio_util::sync::CancellationToken`] fires —
27//! typically when the owning Tab is dropped.
28
29use std::collections::HashMap;
30use std::sync::{Arc, Weak};
31
32use futures::StreamExt;
33use serde::Deserialize;
34use tokio::sync::RwLock;
35use tokio_util::sync::CancellationToken;
36use tracing::{trace, warn};
37use zendriver_transport::SessionHandle;
38
39use crate::frame::Frame;
40use crate::tab::TabInner;
41
42/// Minimal projection of `Page.frameAttached` — only the fields we need
43/// to construct a [`Frame`]. URL/name are not on this payload; they arrive
44/// on a subsequent `Page.frameNavigated`.
45#[derive(Debug, Deserialize)]
46struct FrameAttachedEvent {
47    #[serde(rename = "frameId")]
48    frame_id: String,
49    #[serde(rename = "parentFrameId")]
50    parent_frame_id: Option<String>,
51}
52
53/// Minimal projection of `Page.frameDetached` — only `frameId` matters for
54/// registry removal.
55#[derive(Debug, Deserialize)]
56struct FrameDetachedEvent {
57    #[serde(rename = "frameId")]
58    frame_id: String,
59}
60
61/// Minimal projection of `Page.frameNavigated`. Chrome nests the frame
62/// metadata under `frame: {...}`.
63#[derive(Debug, Deserialize)]
64struct FrameNavigatedEvent {
65    frame: NavigatedFrameInner,
66}
67
68#[derive(Debug, Deserialize)]
69struct NavigatedFrameInner {
70    id: String,
71    #[serde(rename = "parentId", default)]
72    parent_id: Option<String>,
73    #[serde(default)]
74    url: String,
75    #[serde(default)]
76    name: Option<String>,
77}
78
79/// Drive the lifecycle subscriber until `cancel` fires.
80///
81/// `frames` is the [`RwLock`]-protected registry shared with the owning
82/// [`TabInner`]. `tab_weak` is the [`Weak<TabInner>`] handed to each
83/// constructed [`Frame`] so its `tab_for_synthesize` upgrade can return
84/// the owning `Tab`.
85///
86/// Same fire-and-forget posture as
87/// [`crate::network_idle::InFlightTracker::run`] for the `Page.enable`
88/// call: failure is logged + ignored so the subscriber keeps running on a
89/// previously-enabled Page domain.
90pub(crate) async fn run(
91    session: SessionHandle,
92    frames: Arc<RwLock<HashMap<String, Frame>>>,
93    tab_weak: Weak<TabInner>,
94    cancel: CancellationToken,
95) {
96    let mut attached = session.subscribe::<FrameAttachedEvent>("Page.frameAttached");
97    let mut detached = session.subscribe::<FrameDetachedEvent>("Page.frameDetached");
98    let mut navigated = session.subscribe::<FrameNavigatedEvent>("Page.frameNavigated");
99
100    // Fire-and-forget `Page.enable`. Same rationale as
101    // `InFlightTracker::run`: the mock harness never replies to this
102    // call, and in production the subscribe streams above are already
103    // registered, so awaiting the response would only serialize the
104    // first arriving event behind the enable round-trip.
105    let enable_session = session.clone();
106    tokio::spawn(async move {
107        if let Err(e) = enable_session
108            .call("Page.enable", serde_json::json!({}))
109            .await
110        {
111            warn!(error = %e, "frame::lifecycle: Page.enable failed; frame events may be inactive");
112        }
113    });
114
115    loop {
116        tokio::select! {
117            () = cancel.cancelled() => {
118                trace!("frame::lifecycle: cancellation received, exiting");
119                return;
120            }
121            Some(ev) = attached.next() => {
122                let frame = Frame::new(
123                    ev.frame_id.clone(),
124                    ev.parent_frame_id.clone(),
125                    String::new(),
126                    None,
127                    session.clone(),
128                    tab_weak.clone(),
129                );
130                // Chrome occasionally fires `frameAttached` twice for a
131                // single iframe — once with a provisional `frameId` and
132                // again with the committed one — without an intervening
133                // `frameDetached`. Observed on `srcdoc` iframes under
134                // `--headless=new`. The provisional entry always has an
135                // empty URL (it never received a `frameNavigated`); the
136                // newer attach carries the real id. Sweep any
137                // same-parent provisional siblings before inserting so
138                // the registry doesn't accumulate dead frames that
139                // `Page.createIsolatedWorld` would reject with
140                // "No frame for given id found".
141                let mut map = frames.write().await;
142                if let Some(parent) = ev.parent_frame_id.as_deref() {
143                    let stale: Vec<String> = map
144                        .iter()
145                        .filter_map(|(id, f)| {
146                            if id == &ev.frame_id {
147                                return None;
148                            }
149                            if f.inner.parent_frame_id.as_deref() != Some(parent) {
150                                return None;
151                            }
152                            let url_slot = f.inner.url.try_read().ok()?;
153                            if url_slot.is_empty() {
154                                Some(id.clone())
155                            } else {
156                                None
157                            }
158                        })
159                        .collect();
160                    for id in stale {
161                        map.remove(&id);
162                    }
163                }
164                map.insert(ev.frame_id, frame);
165            }
166            Some(ev) = navigated.next() => {
167                let frame_id = ev.frame.id.clone();
168                let new_url = ev.frame.url.clone();
169                // Update in place if known; otherwise treat the navigation
170                // as the implicit attach (Chrome may emit `frameNavigated`
171                // for the main frame before any subscriber sees an explicit
172                // `frameAttached`).
173                {
174                    let map = frames.read().await;
175                    if let Some(existing) = map.get(&frame_id) {
176                        let mut url_slot = existing.inner.url.write().await;
177                        *url_slot = new_url;
178                        continue;
179                    }
180                }
181                let frame = Frame::new(
182                    frame_id.clone(),
183                    ev.frame.parent_id,
184                    ev.frame.url,
185                    ev.frame.name,
186                    session.clone(),
187                    tab_weak.clone(),
188                );
189                frames.write().await.insert(frame_id, frame);
190            }
191            Some(ev) = detached.next() => {
192                frames.write().await.remove(&ev.frame_id);
193            }
194            else => {
195                trace!("frame::lifecycle: all event streams closed, exiting");
196                return;
197            }
198        }
199    }
200}
