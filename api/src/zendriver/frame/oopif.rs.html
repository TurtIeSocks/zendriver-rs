1//! Out-of-process iframe (OOPIF) Frame attach — internal.
2//!
3//! Extends the internal tab registrar observer to handle
4//! `Target.attachedToTarget` events where `target_info.kind == "iframe"`,
5//! constructing a [`crate::Frame`] with the new child session and
6//! registering it in the parent tab's frames map.
7//!
8//! ## Parent-tab discovery
9//!
10//! When Chrome attaches an OOPIF target the payload identifies the iframe
11//! by `target_id` (always present) and — on Chromium 90+ — by
12//! `opener_frame_id` (the host iframe's CDP `frameId` inside the parent
13//! document). To find the owning [`crate::tab::Tab`] we walk the browser-
14//! wide tabs registry and look for the first tab whose frames map already
15//! contains an entry under one of these ids:
16//!
17//! 1. `opener_frame_id` if Chrome supplied one — the canonical match.
18//! 2. Otherwise `target_id` — for OOPIFs Chrome typically uses the
19//!    `targetId` as the `frameId` of the iframe inside the parent
20//!    document (so the parent's `Page.frameAttached` handler has already
21//!    registered a same-id entry by the time the OOPIF target attaches).
22//!
23//! If neither id matches anything in the registry the OOPIF attaches
24//! before any parent observed `Page.frameAttached` for the host iframe
25//! — a rare race. We log a warning and skip registration; the OOPIF's
26//! session is still alive (the actor will release the debugger after the
27//! observer returns) but no `Frame` handle is exposed for it. The next
28//! `Page.frameAttached` from the parent tab will create a same-origin
29//! placeholder entry that callers can use for queries on the parent side,
30//! and the OOPIF target can be discovered manually via `Browser::cdp()`
31//! if needed.
32
33use std::sync::Arc;
34
35use tracing::{trace, warn};
36use zendriver_transport::{SessionHandle, TargetInfo};
37
38use crate::browser::BrowserInner;
39use crate::frame::Frame;

41/// Construct a [`Frame`] for the attached OOPIF target and register it in
42/// the parent tab's frames map. Returns `Some(())` on success, `None` if no
43/// parent tab could be located.
44///
45/// Called by [`crate::browser::TabRegistrar::on_target_attached`] for events
46/// with `target_info.kind == "iframe"`. The returned [`Frame`] shares the
47/// passed-in [`SessionHandle`] (the OOPIF's distinct child session) — so
48/// any subsequent `Runtime.evaluate` / `Page.*` dispatch lands in the
49/// out-of-process renderer, not the parent tab's renderer.
50pub(crate) async fn register_oopif_frame(
51    browser: &Arc<BrowserInner>,
52    target_info: &TargetInfo,
53    session: SessionHandle,
54) -> Option<()> {
55    // Candidate frame_ids to look up in each tab's frames map. Prefer the
56    // explicit opener_frame_id (Chrome 90+); fall back to the target_id
57    // (Chrome uses targetId == frameId for OOPIF hosts in many cases).
58    let mut candidates: Vec<&str> = Vec::with_capacity(2);
59    if let Some(opener) = target_info.opener_frame_id.as_deref() {
60        candidates.push(opener);
61    }
62    candidates.push(&target_info.target_id);
63
64    let tabs = browser.tabs.read().await;
65    for tab in tabs.values() {
66        let frames = tab.inner.frames.read().await;
67        for cand in &candidates {
68            if frames.contains_key(*cand) {
69                drop(frames);
70
71                // The OOPIF's own frame_id is ALWAYS its target_id — the
72                // unique CDP identifier for this child renderer's frame.
73                // The matched `cand` only tells us how we found the parent
74                // (either via Chrome's opener_frame_id hint or via the
75                // common targetId == frameId convention); it must NOT be
76                // reused as the new Frame's id, or two OOPIFs sharing a
77                // parent_frame_id would collide and overwrite each other
78                // in the frames map. The matched candidate becomes the
79                // new frame's `parent_frame_id` linkage.
80                //
81                // url/name unknown at attach time (a subsequent
82                // Page.frameNavigated on the child session will populate
83                // them via the lifecycle subscriber spawned by the parent
84                // Tab... actually the parent's subscriber listens on the
85                // parent's session, not the OOPIF session, so URL stays
86                // empty until a future per-OOPIF lifecycle is wired).
87                let frame_id = target_info.target_id.clone();
88                let parent_frame_id = (*cand).to_string();
89                let oopif = Frame::new(
90                    frame_id.clone(),
91                    Some(parent_frame_id),
92                    String::new(),
93                    None,
94                    session,
95                    Arc::downgrade(&tab.inner),
96                );
97                tab.inner.frames.write().await.insert(frame_id, oopif);
98                trace!(
99                    target_id = %target_info.target_id,
100                    "OOPIF frame registered under parent tab",
101                );
102                return Some(());
103            }
104        }
105    }
106    warn!(
107        target_id = %target_info.target_id,
108        opener_frame_id = ?target_info.opener_frame_id,
109        "OOPIF attached with no matching parent frame; skipping registration",
110    );
111    None
112}

114/// Walk every tab in the browser registry and remove any [`Frame`] whose
115/// underlying session matches `session_id`. Returns `true` if any frame
116/// was removed.
117///
118/// Called by [`crate::browser::TabRegistrar::on_target_detached`] as the
119/// counterpart to [`register_oopif_frame`]: when an OOPIF's child session
120/// goes away, drop its placeholder from the hosting tab's frames map.
121pub(crate) async fn deregister_oopif_frame(browser: &Arc<BrowserInner>, session_id: &str) -> bool {
122    let tabs = browser.tabs.read().await;
123    for tab in tabs.values() {
124        let to_remove: Vec<String> = {
125            let frames = tab.inner.frames.read().await;
126            frames
127                .iter()
128                .filter(|(_, frame)| frame.session().session_id() == session_id)
129                .map(|(frame_id, _)| frame_id.clone())
130                .collect()
131        };
132        if !to_remove.is_empty() {
133            let mut frames = tab.inner.frames.write().await;
134            for frame_id in &to_remove {
135                frames.remove(frame_id);
136            }
137            return true;
138        }
139    }
140    false
141}
