1//! Browser lifecycle: executable discovery, subprocess spawn, WS attach,
2//! graceful teardown.
3//!
4//! Entry point is [`Browser::builder`] — start there for any zendriver
5//! workflow. The launched [`Browser`] owns the Chrome subprocess and the
6//! transport actor; dropping it terminates Chrome. Spawn additional pages
7//! via [`Browser::new_tab`] and reach the initial page via

42/// Which Chromium-family browser channel to discover at launch.
43///
44/// Passed via [`BrowserBuilder::channel`]; consumed by
45/// [`find_chrome_executable_for_channel`] to pick the per-OS candidate path
46/// table. [`BrowserBuilder::executable`] still overrides channel discovery
47/// entirely.
48///
49/// # Examples
50///
51/// ```no_run
52/// use zendriver::Channel;
53/// let builder = zendriver::Browser::builder().channel(Channel::Brave);
54/// ```
55#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Default)]
56pub enum Channel {
57    /// Google Chrome (stable). Same discovery as [`Channel::Auto`] minus the
58    /// Chromium fallbacks.
59    Chrome,
60    /// Open-source Chromium build.
61    Chromium,
62    /// Brave Browser.
63    Brave,
64    /// Microsoft Edge.
65    Edge,
66    /// First Chromium-family browser found — the historical default (Chrome,
67    /// then Chromium). Use a specific channel to force Brave / Edge.
68    #[default]
69    Auto,
70}

72/// Look for a Chromium-family binary on PATH and in conventional locations.
73///
74/// Returns the first path that exists, scanning the [`Channel::Auto`]
75/// candidate table (Chrome, then Chromium). For a specific browser channel
76/// (Brave / Edge / …) use [`find_chrome_executable_for_channel`].
77///
78/// # Errors
79///
80/// Returns [`BrowserError::ExecutableNotFound`] with the full list of
81/// searched paths when no installation is found.
82///
83/// # Examples
84///
85/// ```no_run
86/// match zendriver::browser::find_chrome_executable() {
87///     Ok(p) => println!("found chrome at {}", p.display()),
88///     Err(e) => eprintln!("no chrome installed: {e}"),
89/// }
90/// ```
91pub fn find_chrome_executable() -> Result<PathBuf, BrowserError> {
92    find_chrome_executable_for_channel(Channel::Auto)
93}

95/// Look for the binary of a specific [`Channel`] on PATH and in conventional
96/// per-OS install locations.
97///
98/// Returns the first candidate that exists. [`Channel::Auto`] reproduces the
99/// historical first-found behavior (Chrome, then Chromium).
100///
101/// # Errors
102///
103/// Returns [`BrowserError::ExecutableNotFound`] with the searched candidate
104/// list when none of the channel's paths exist.
105pub fn find_chrome_executable_for_channel(channel: Channel) -> Result<PathBuf, BrowserError> {
106    let candidates = candidate_paths_for_channel(channel);
107    for c in &candidates {
108        if c.exists() {
109            return Ok(c.clone());
110        }
111    }
112    Err(BrowserError::ExecutableNotFound {
113        searched: candidates,
114    })
115}

117/// Build the ordered candidate-path list for `channel`.
118///
119/// PATH lookups come first (so a PATH-resolved binary wins over a fixed
120/// install dir), followed by the per-OS conventional install locations for
121/// the requested channel. Factored out (and `pub(crate)`) so unit tests can
122/// assert the table without requiring the browser installed.
123///
124/// On Windows the machine-wide `Program Files` locations are listed ahead of
125/// the per-user `%LOCALAPPDATA%` ones: callers take the first candidate that
126/// *exists*, so a machine with a system-wide install resolves exactly as it
127/// always has, and the per-user path is reached only where discovery would
128/// otherwise have found nothing.
129pub(crate) fn candidate_paths_for_channel(channel: Channel) -> Vec<PathBuf> {
130    let mut v = Vec::new();
131
132    // PATH lookups — names vary by channel.
133    let path_names: &[&str] = match channel {
134        Channel::Chrome => &["google-chrome", "google-chrome-stable", "chrome"],
135        Channel::Chromium => &["chromium", "chromium-browser"],
136        Channel::Brave => &["brave-browser", "brave"],
137        Channel::Edge => &["microsoft-edge", "microsoft-edge-stable"],
138        Channel::Auto => &[
139            "google-chrome",
140            "google-chrome-stable",
141            "chromium",
142            "chromium-browser",
143            "chrome",
144        ],
145    };
146    for name in path_names {
147        if let Some(p) = which_on_path(name) {
148            v.push(p);
149        }
150    }
151
152    // Platform-specific known locations, per channel.
153    #[cfg(target_os = "macos")]
154    {
155        let chrome = "/Applications/Google Chrome.app/Contents/MacOS/Google Chrome";
156        let chromium = "/Applications/Chromium.app/Contents/MacOS/Chromium";
157        let brave = "/Applications/Brave Browser.app/Contents/MacOS/Brave Browser";
158        let edge = "/Applications/Microsoft Edge.app/Contents/MacOS/Microsoft Edge";
159        match channel {
160            Channel::Chrome => v.push(PathBuf::from(chrome)),
161            Channel::Chromium => v.push(PathBuf::from(chromium)),
162            Channel::Brave => v.push(PathBuf::from(brave)),
163            Channel::Edge => v.push(PathBuf::from(edge)),
164            Channel::Auto => {
165                v.push(PathBuf::from(chrome));
166                v.push(PathBuf::from(chromium));
167            }
168        }
169    }
170    #[cfg(target_os = "linux")]
171    {
172        match channel {
173            Channel::Chrome => {
174                v.push(PathBuf::from("/usr/bin/google-chrome"));
175                v.push(PathBuf::from("/usr/bin/google-chrome-stable"));
176            }
177            Channel::Chromium => {
178                v.push(PathBuf::from("/usr/bin/chromium"));
179                v.push(PathBuf::from("/usr/bin/chromium-browser"));
180                v.push(PathBuf::from("/snap/bin/chromium"));
181            }
182            Channel::Brave => {
183                v.push(PathBuf::from("/usr/bin/brave-browser"));
184                v.push(PathBuf::from("/usr/bin/brave"));
185                v.push(PathBuf::from("/opt/brave.com/brave/brave-browser"));
186            }
187            Channel::Edge => {
188                v.push(PathBuf::from("/usr/bin/microsoft-edge"));
189                v.push(PathBuf::from("/usr/bin/microsoft-edge-stable"));
190                v.push(PathBuf::from("/opt/microsoft/msedge/microsoft-edge"));
191            }
192            Channel::Auto => {
193                v.push(PathBuf::from("/usr/bin/google-chrome"));
194                v.push(PathBuf::from("/usr/bin/chromium"));
195                v.push(PathBuf::from("/usr/bin/chromium-browser"));
196                v.push(PathBuf::from("/snap/bin/chromium"));
197            }
198        }
199    }
200    #[cfg(target_os = "windows")]
201    {
202        let chrome = [
203            r"C:\Program Files\Google\Chrome\Application\chrome.exe",
204            r"C:\Program Files (x86)\Google\Chrome\Application\chrome.exe",
205        ];
206        let brave = [
207            r"C:\Program Files\BraveSoftware\Brave-Browser\Application\brave.exe",
208            r"C:\Program Files (x86)\BraveSoftware\Brave-Browser\Application\brave.exe",
209        ];
210        let edge = [
211            r"C:\Program Files\Microsoft\Edge\Application\msedge.exe",
212            r"C:\Program Files (x86)\Microsoft\Edge\Application\msedge.exe",
213        ];
214
215        // Per-user installs live under `%LOCALAPPDATA%` — where Chrome's
216        // installer puts it whenever it runs without admin rights, which is the
217        // common outcome on a locked-down or personal desktop. Only the two
218        // `Program Files` (machine-wide) locations were ever checked, so on such
219        // a machine discovery found nothing and `launch()` failed unless the
220        // caller happened to set an explicit `chrome_path` / `CHROME_BIN`.
221        //
222        // `None` when the variable is unset (a service account, a stripped
223        // environment), in which case the table is simply what it was before.
224        let local_app_data = std::env::var_os("LOCALAPPDATA")
225            .filter(|v| !v.is_empty())
226            .map(PathBuf::from);
227
228        match channel {
229            Channel::Chrome | Channel::Chromium | Channel::Auto => {
230                for p in chrome {
231                    v.push(PathBuf::from(p));
232                }
233                if let Some(local) = &local_app_data {
234                    v.push(local.join(r"Google\Chrome\Application\chrome.exe"));
235                }
236            }
237            Channel::Brave => {
238                for p in brave {
239                    v.push(PathBuf::from(p));
240                }
241                if let Some(local) = &local_app_data {
242                    v.push(local.join(r"BraveSoftware\Brave-Browser\Application\brave.exe"));
243                }
244            }
245            Channel::Edge => {
246                for p in edge {
247                    v.push(PathBuf::from(p));
248                }
249                if let Some(local) = &local_app_data {
250                    v.push(local.join(r"Microsoft\Edge\Application\msedge.exe"));
251                }
252            }
253        }
254    }
255
256    v
257}

259fn which_on_path(name: &str) -> Option<PathBuf> {
260    let path = std::env::var_os("PATH")?;
261    for dir in std::env::split_paths(&path) {
262        let full = dir.join(name);
263        if full.is_file() {
264            return Some(full);
265        }
266        #[cfg(target_os = "windows")]
267        {
268            let with_exe = dir.join(format!("{name}.exe"));
269            if with_exe.is_file() {
270                return Some(with_exe);
271            }
272        }
273    }
274    None
275}

277/// Extract the `host:port` authority from a `ws://HOST:PORT/...` (or `wss://`)
278/// DevTools endpoint URL.
279///
280/// Returns `None` for inputs that don't carry a recognizable
281/// `scheme://authority` shape. Used to compose [`Tab::inspector_url`] from the
282/// endpoint the browser connected to.
283pub(crate) fn debug_host_port_from_ws(ws_url: &str) -> Option<String> {
284    // Strip the scheme, then take everything up to the first `/` (the path).
285    let after_scheme = ws_url
286        .strip_prefix("ws://")
287        .or_else(|| ws_url.strip_prefix("wss://"))?;
288    let authority = after_scheme
289        .split('/')
290        .next()
291        .unwrap_or(after_scheme)
292        .trim();
293    if authority.is_empty() {
294        None
295    } else {
296        Some(authority.to_string())
297    }
298}

300/// Parse a `DevTools listening on ws://...` line from Chrome's stderr.
301pub(crate) fn parse_devtools_line(line: &str) -> Option<String> {
302    // Format: `DevTools listening on ws://127.0.0.1:NNNN/devtools/browser/UUID`
303    let needle = "DevTools listening on ";
304    let idx = line.find(needle)?;
305    let rest = &line[idx + needle.len()..];
306    let end = rest.find(char::is_whitespace).unwrap_or(rest.len());
307    let url = rest[..end].trim();
308    if url.starts_with("ws://") || url.starts_with("wss://") {
309        Some(url.to_string())
310    } else {
311        None
312    }
313}

315/// Chrome's own record of the debug port it bound, written into the
316/// `user_data_dir`. A second, independent source for the WS endpoint.
317const DEVTOOLS_ACTIVE_PORT_FILE: &str = "DevToolsActivePort";
318
319/// How often [`poll_devtools_active_port`] re-reads the file while waiting for
320/// Chrome to create it.
321const DEVTOOLS_PORT_POLL_INTERVAL: Duration = Duration::from_millis(25);
322
323/// Parse Chrome's `DevToolsActivePort` file into a `ws://` endpoint.
324///
325/// The file is two lines: the port Chrome actually bound (which is the only way
326/// to learn it when launched with `--remote-debugging-port=0`), then the
327/// browser target's path (`/devtools/browser/<uuid>`).
328///
329/// Both lines are required. The file is polled while Chrome may still be
330/// creating it, so a partial read must yield `None` and let the caller keep
331/// waiting rather than resolve an endpoint that will not dial.
332pub(crate) fn parse_devtools_active_port(contents: &str) -> Option<String> {
333    let mut lines = contents.lines();
334    let port: u16 = lines.next()?.trim().parse().ok()?;
335    let path = lines.next()?.trim();
336    // Guard a torn read: the path line must at least look like a path.
337    if !path.starts_with('/') {
338        return None;
339    }
340    Some(format!("ws://127.0.0.1:{port}{path}"))
341}
342
343/// Wait for Chrome to write `<user_data_dir>/DevToolsActivePort`, then return
344/// the `ws://` endpoint it describes.
345///
346/// Never returns until the file exists and parses — callers bound it with a
347/// timeout (and race it against the stderr read).
348pub(crate) async fn poll_devtools_active_port(user_data_dir: &Path) -> String {
349    let path = user_data_dir.join(DEVTOOLS_ACTIVE_PORT_FILE);
350    loop {
351        if let Ok(contents) = tokio::fs::read_to_string(&path).await {
352            if let Some(url) = parse_devtools_active_port(&contents) {
353                return url;
354            }
355        }
356        tokio::time::sleep(DEVTOOLS_PORT_POLL_INTERVAL).await;
357    }
358}

360/// Load (or first-time generate + persist) the fingerprint [`Seed`] bound to a
361/// `user_data_dir`.
362///
363/// Stored as a single base-10 `u64` in `<dir>/.zd_persona_seed`. On reuse the
364/// file is read and parsed; on first use (or any read/parse failure) a fresh
365/// [`Seed::random`] is generated and written, giving a stable per-profile
366/// fingerprint across runs. All filesystem errors are swallowed (best-effort):
367/// a dir that can't be written simply yields a fresh random seed each launch
368/// rather than failing the build.
369fn persisted_seed(dir: &Path) -> Seed {
370    let path = dir.join(".zd_persona_seed");
371    if let Ok(contents) = std::fs::read_to_string(&path) {
372        if let Ok(value) = contents.trim().parse::<u64>() {
373            return Seed::from_u64(value);
374        }
375    }
376    let seed = Seed::random();
377    let _ = std::fs::create_dir_all(dir);
378    let _ = std::fs::write(&path, seed.value().to_string());
379    seed
380}

579    /// Auto-respond to proxy / HTTP basic-auth challenges with

1044    /// Compute the effective [`Persona`] this builder will hand to the stealth
1045    /// observer.
1046    ///
1047    /// Resolution order: base persona ([`Persona::system`] when unset) →
1048    /// [`persona_overlay`](BrowserBuilder::persona_overlay) (field-wise merge) →
1049    /// each [`surface`](BrowserBuilder::surface) override → a seed persisted
1050    /// alongside [`user_data_dir`](BrowserBuilder::user_data_dir) when the
1051    /// resolved persona has no explicit seed.
1052    ///
1053    /// Callable before launch — it does not require a live browser.
1054    #[must_use]
1055    pub fn resolved_persona(&self) -> Persona {
1056        // Did the caller pin a seed explicitly (via `.persona` or
1057        // `.persona_overlay`)? Capture this BEFORE the `system()` fallback,
1058        // which injects a fresh random seed of its own — an explicit seed must
1059        // win over the `user_data_dir`-persisted one below.
1060        let explicit_seed = self
1061            .persona
1062            .as_ref()
1063            .and_then(|p| p.seed)
1064            .or_else(|| self.persona_overlay.as_ref().and_then(|p| p.seed));
1065
1066        let mut persona = self.persona.clone().unwrap_or_else(Persona::system);
1067        if let Some(overlay) = &self.persona_overlay {
1068            persona = persona.overlay(overlay.clone());
1069        }
1070        for (surface, strategy) in &self.surface_overrides {
1071            persona.apply_surface_override(*surface, *strategy);
1072        }
1073
1074        // Seed persistence: when the caller did not pin a seed but a
1075        // `user_data_dir` is set, the seed is read from (or written to) a file
1076        // inside that dir so the same profile yields a stable fingerprint
1077        // across runs. This overrides the ephemeral random seed that
1078        // `Persona::system()` injects. Without a `user_data_dir`, the random
1079        // seed stands (fresh identity per launch).
1080        if explicit_seed.is_none() {
1081            if let Some(dir) = &self.user_data_dir {
1082                persona.seed = Some(persisted_seed(dir));
1083            }
1084        }
1085
1086        persona
1087    }

1110    /// Compute the full argv that would be passed to Chrome. Exposed to
1111    /// tests + snapshots; called internally by `launch`.
1112    pub(crate) fn build_flags(&self, user_data_dir: &Path) -> Vec<String> {
1113        let mut v = Vec::with_capacity(10 + self.extra_args.len());
1114        v.push("--remote-debugging-port=0".to_string());
1115        v.push(format!("--user-data-dir={}", user_data_dir.display()));
1116        v.push("--no-first-run".to_string());
1117        v.push("--no-default-browser-check".to_string());
1118        // Suppress the Chrome "Save password?" / autofill bubbles + onboarding
1119        // popups that otherwise hijack focus inside automated runs.
1120        // See cdpdriver/zendriver#13.
1121        v.push("--password-store=basic".to_string());
1122        v.push("--disable-save-password-bubble".to_string());
1123        // Base disable-features set. When extensions are requested we MUST also
1124        // turn off `DisableLoadExtensionCommandLineSwitch`, or Chrome 136+
1125        // silently ignores `--load-extension`. The stealth profiles carry that
1126        // feature in their own `--disable-features=IsolateOrigins,…` line, but
1127        // an Off profile would otherwise miss it, so merge it into the base
1128        // line here — Chrome unions every `--disable-features` occurrence, so
1129        // the redundancy under a stealth profile is harmless.
1130        if self.extensions.is_empty() {
1131            v.push(
1132                "--disable-features=PasswordManagerOnboarding,AutofillServerCommunication"
1133                    .to_string(),
1134            );
1135        } else {
1136            v.push(
1137                "--disable-features=PasswordManagerOnboarding,AutofillServerCommunication,DisableLoadExtensionCommandLineSwitch"
1138                    .to_string(),
1139            );
1140        }
1141        if self.headless.unwrap_or(true) {
1142            v.push("--headless=new".to_string());
1143            v.push("--disable-gpu".to_string());
1144        }
1145        // Expert mode: relax web-security + site isolation. Emitted only when
1146        // `expert(true)` so the default flag set / snapshots are unchanged.
1147        if self.expert {
1148            v.push("--disable-web-security".to_string());
1149            v.push("--disable-site-isolation-trials".to_string());
1150        }
1151        // Extension side-loading flags. `self.extensions` holds already-resolved
1152        // directories at this point (`launch` unzips any `.crx` into tempdirs
1153        // and rewrites the list before calling `build_flags`). Skipped entirely
1154        // when no extensions are configured so the default argv is untouched.
1155        if !self.extensions.is_empty() {
1156            let joined = self
1157                .extensions
1158                .iter()
1159                .map(|p| p.display().to_string())
1160                .collect::<Vec<_>>()
1161                .join(",");
1162            v.push(format!("--load-extension={joined}"));
1163            v.push(format!("--disable-extensions-except={joined}"));
1164            v.push("--enable-unsafe-extension-debugging".to_string());
1165        }
1166        // C4 dedicated flags. Emitted only when explicitly configured so the
1167        // default-builder flag set (and its snapshots) is unchanged. Placed
1168        // before user `extra_args` so caller-supplied flags still come last.
1169        if let Some(lang) = self.lang.as_ref() {
1170            v.push(format!("--lang={lang}"));
1171        }
1172        if let Some(ua) = self.user_agent.as_ref() {
1173            v.push(format!("--user-agent={ua}"));
1174        }
1175        // Sandbox off → --no-sandbox. Default (None / Some(true)) emits
1176        // nothing; the CI auto-disable is handled separately in `launch`.
1177        // NOTE: root-uid (euid 0) auto-disable is intentionally NOT done here
1178        // — that needs a `geteuid` syscall (no std API) and we decline to add
1179        // a `rustix` / `unsafe libc` dependency for it. Callers running as
1180        // root should opt in explicitly with `.sandbox(false)`.
1181        if self.sandbox == Some(false) {
1182            v.push("--no-sandbox".to_string());
1183        }
1184        v.extend(self.extra_args.iter().cloned());
1185        // Start the initial tab on `about:blank` instead of Chrome's default
1186        // New Tab Page. The NTP (`chrome://new-tab-page`) issues its own
1187        // network requests — realbox icons plus, on a fresh user-data-dir,
1188        // external Google fetches (logo/doodle, OneGoogle, suggest, background
1189        // collections). Those land on the *main tab's* CDP session, so they
1190        // pollute the in-flight set that [`Tab::wait_for_idle`] watches; on a
1191        // network-restricted host (e.g. CI with no outbound egress) they never
1192        // complete, and the first `wait_for_idle` after launch hangs until its
1193        // timeout. A blank start page issues zero requests, so idle detection
1194        // sees only the caller's own navigation traffic. Every mainstream
1195        // driver (Puppeteer, Playwright, nodriver) starts on `about:blank` for
1196        // the same reason.
1197        //
1198        // Emitted as the final positional argument (a bare URL, not a flag),
1199        // and skipped when the caller already supplied their own positional
1200        // start URL via `.arg`/`.args` — an explicit start page still wins.
1201        if !self.extra_args.iter().any(|a| !a.starts_with('-')) {
1202            v.push("about:blank".to_string());
1203        }
1204        v
1205    }

1207    /// Build the combined [`HostMatcher`] from all configured sources, or
1208    /// `None` if nothing was requested. Called once at launch.
1209    #[cfg(feature = "tracker-blocking")]
1210    async fn build_tracker_matcher(
1211        &self,
1212    ) -> Result<Option<std::sync::Arc<crate::HostMatcher>>, ZendriverError> {
1213        let mut domains: Vec<String> = Vec::new();
1214        if self.block_trackers {
1215            domains.extend(crate::tracker::bundled_hosts());
1216        }
1217        domains.extend(self.tracker_blocklist_domains.iter().cloned());
1218        for path in &self.tracker_blocklist_files {
1219            let text = std::fs::read_to_string(path)?; // -> ZendriverError::Io
1220            domains.extend(crate::tracker::parse_blocklist(&text));
1221        }
1222        for url in &self.tracker_blocklist_urls {
1223            let hosts = crate::tracker::load_or_download_blocklist(url).await?; // -> Io
1224            domains.extend(hosts);
1225        }
1226        if domains.is_empty() {
1227            return Ok(None);
1228        }
1229        Ok(Some(std::sync::Arc::new(crate::HostMatcher::new(domains))))
1230    }

1450#[async_trait::async_trait]
1451impl TargetObserver for TabRegistrar {
1452    fn name(&self) -> &'static str {
1453        "tab-registrar"
1454    }
1455
1456    async fn on_target_attached(&self, session: PausedSession<'_>) -> Result<(), ObserverError> {
1457        let Some(weak) = self.browser.get() else {
1458            // EXPECTED on the launch path, exactly once per browser: the
1459            // initial target is attached during `finish_connect`, which is
1460            // upstream of the `Arc::new_cyclic` that produces the
1461            // `Weak<BrowserInner>` this observer needs. So the first attach
1462            // always lands here and bails — and `finish_connect` compensates
1463            // by inserting the main tab into the registry by hand right after
1464            // it calls `set_browser` (see the call site).
1465            //
1466            // This was a `warn!` claiming it "should not happen in practice".
1467            // It happens on every single launch, which made real registrar
1468            // warnings (the chrome-scheme page below) impossible to spot in a
1469            // log — so it is `debug!`. If it ever fires for a target OTHER
1470            // than the initial one, that target is genuinely unregistered:
1471            // the tab is then invisible to `Browser::tabs()` and closeable by
1472            // nothing, which is worth the noise it costs to find.
1473            debug!(
1474                target_id = %session.target_info.target_id,
1475                url = %session.target_info.url,
1476                "tab-registrar: target attached before the browser weak ref was wired; skipping \
1477                 registration (expected exactly once, for the initial target)"
1478            );
1479            return Ok(());
1480        };
1481        let Some(browser) = weak.upgrade() else {
1482            // Browser dropped between event arrival and observer body —
1483            // nothing to register against.
1484            return Ok(());
1485        };
1486
1487        match session.target_info.kind.as_str() {
1488            "iframe" => {
1489                // Out-of-process iframe: register a Frame under the parent
1490                // tab's frames map. The OOPIF carries a distinct child
1491                // session; the helper resolves the parent tab by walking
1492                // the tabs registry for a matching frame_id (preferring
1493                // `opener_frame_id` when present, falling back to
1494                // `target_id`).
1495                let conn = session.connection().clone();
1496                let new_session = SessionHandle::new(conn, session.session_id.to_string());
1497                crate::frame::oopif::register_oopif_frame(
1498                    &browser,
1499                    session.target_info,
1500                    new_session,
1501                )
1502                .await;
1503                Ok(())
1504            }
1505            "page" => {
1506                // Skip Chrome-internal pages (chrome://newtab/ that Chrome
1507                // auto-opens when the last user tab closes, devtools://, etc).
1508                // These aren't pages the caller drove zendriver to create and
1509                // would inflate `tab_count` / `tabs()` with unwanted entries.
1510                //
1511                // Skipping the *registry* is right; skipping it **silently**
1512                // was not. A chrome-scheme page Chrome opened on its own is a
1513                // real window: it is invisible to `Browser::tabs()`, so no
1514                // caller can close it, and `close()` only kills the tracked
1515                // PID — on Windows that leaves it orphaned on screen. This is
1516                // one of the two live candidates for the double-window
1517                // symptom, so make it observable rather than inferring it from
1518                // a screenshot.
1519                let url = &session.target_info.url;
1520                if url.starts_with("chrome://")
1521                    || url.starts_with("devtools://")
1522                    || url.starts_with("chrome-extension://")
1523                    || url.starts_with("chrome-untrusted://")
1524                {
1525                    warn!(
1526                        url = %url,
1527                        target_id = %session.target_info.target_id,
1528                        "chrome opened an internal page on its own; not tracked as a tab \
1529                         (if a stray window is visible, this is it)",
1530                    );
1531                    return Ok(());
1532                }
1533                let conn = session.connection().clone();
1534                let new_session = SessionHandle::new(conn, session.session_id.to_string());
1535                // Clone the session before it is moved into Tab::new so the
1536                // tracker-blocking install can use it afterwards.
1537                #[cfg(feature = "tracker-blocking")]
1538                let new_session_for_tracker = new_session.clone();
1539                let input = InputController::new(self.input_profile.clone());
1540                let weak_inner = Arc::downgrade(&browser);
1541                let tab = Tab::new(
1542                    new_session,
1543                    weak_inner,
1544                    input,
1545                    session.target_info.target_id.clone(),
1546                );
1547
1548                // Dedupe by `target_id`: if `flatten: true` re-fires
1549                // `attachedToTarget` for the same target under a different
1550                // sessionId (observed on `--headless=new`), the old entry
1551                // would otherwise linger and inflate `tab_count`.
1552                let target_id_str = session.target_info.target_id.clone();
1553                let mut tabs = browser.tabs.write().await;
1554                tabs.retain(|_, t| t.target_id() != target_id_str);
1555                tabs.insert(session.session_id.to_string(), tab);
1556                drop(tabs);
1557                // Wake any `new_tab_at` callers waiting on this insert.
1558                browser.tabs_changed.notify_waiters();
1559
1560                // Tracker blocking: if configured, install a BlockHosts
1561                // interception on this new page tab's session and park the
1562                // handle (keyed by sessionId) so it lives with the browser.
1563                #[cfg(feature = "tracker-blocking")]
1564                if let Some(matcher) = browser.tracker_matcher.clone() {
1565                    let handle =
1566                        zendriver_interception::InterceptBuilder::new(&new_session_for_tracker)
1567                            .block_hosts(matcher)
1568                            .start();
1569                    browser
1570                        .tracker_handles
1571                        .lock()
1572                        .await
1573                        .insert(new_session_for_tracker.session_id().to_string(), handle);
1574                }
1575
1576                Ok(())
1577            }
1578            _ => {
1579                // Workers / service workers / etc — out of scope for the
1580                // current registrar; ignored silently. Future P4 tasks
1581                // may add explicit handling.
1582                Ok(())
1583            }
1584        }
1585    }
1586
1587    async fn on_target_detached(&self, session_id: &str) {
1588        let Some(weak) = self.browser.get() else {
1589            return;
1590        };
1591        let Some(browser) = weak.upgrade() else {
1592            return;
1593        };
1594        // Tab path first — if the detached session backs a Tab, remove it.
1595        // If it was an OOPIF (no matching tab), fall through to the OOPIF
1596        // sweep which walks each tab's frames map.
1597        let removed_tab = browser.tabs.write().await.remove(session_id).is_some();
1598        if removed_tab {
1599            // Mirror the insert-side notify so any future watchers (e.g.
1600            // a `wait_for_tab_count` helper) can listen on the same channel.
1601            browser.tabs_changed.notify_waiters();
1602        } else {
1603            let _ = crate::frame::oopif::deregister_oopif_frame(&browser, session_id).await;
1604        }
1605        // Drop any tracker-blocking handle for this session (stops its actor).
1606        #[cfg(feature = "tracker-blocking")]
1607        {
1608            browser.tracker_handles.lock().await.remove(session_id);
1609        }
1610    }

1626/// Budget for everything *after* Chrome advertises its WS endpoint: the
1627/// WebSocket dial plus the initial `Target.setAutoAttach` / `Target.getTargets`
1628/// / `Target.attachToTarget` round-trips.
1629///
1630/// Before this existed, every step past the `WS_ENDPOINT_TIMEOUT`-guarded
1631/// stderr read was unbounded, so a Chrome whose CDP responder was slow or
1632/// wedged hung `launch()` **forever** — no error, no retry. Bounding it turns
1633/// that invisible hang into a retryable [`BrowserError::HandshakeTimeout`].
1634///
1635/// **PROVISIONAL.** 30s is a deliberately generous placeholder chosen to sit
1636/// well clear of a warm handshake (single-digit milliseconds against a local
1637/// socket) while still bounding the pathological case. It is **not** yet
1638/// calibrated against a real measurement: the motivating failure is a Windows
1639/// *cold* start (first launch of an OS session, before GPU/DXGI/DWM bring-up
1640/// and any Defender first-run scan of `chrome.exe` are paid), which cannot be
1641/// measured from the macOS/Linux dev boxes this was written on. Re-tune once a
1642/// cold Windows 11 launch has been timed; prefer raising it over lowering it,
1643/// since a false timeout costs a whole retried launch while a slow success
1644/// costs only latency.
1645const HANDSHAKE_TIMEOUT: Duration = Duration::from_secs(30);

1647/// [`CREATE_NEW_PROCESS_GROUP`][1] — spawn Chrome as the root of its own
1648/// process group so a console `Ctrl+C` is not delivered to it.
1649///
1650/// Without this, `Ctrl+C` in a terminal running a zendriver program goes to
1651/// every process in the console's group, Chrome included: Chrome starts tearing
1652/// itself down at the same moment our own handler starts an orchestrated
1653/// shutdown, and the two race. With it, the `Ctrl+C` reaches only our process
1654/// and Chrome goes down the one way we control — CDP quit, then the job object.

1656/// Declared locally rather than pulled from a `windows`-crate dependency: it is

1665/// A Windows [job object][1] holding the launched Chrome **and every process it
1666/// spawns**, so the whole tree can be terminated as one unit.
1667///
1668/// # Why this exists
1669///
1670/// Windows has no parent→child reaping. `close()` tracks exactly one `Child` —
1671/// the `chrome.exe` that `Command::spawn` returned — but Chrome's renderer, GPU,
1672/// utility and crashpad-handler processes are spawned *by that chrome.exe*, and
1673/// `TerminateProcess` against the parent does nothing to them. They survive as
1674/// orphans that no code path in this crate can ever reach. Unix is spared only
1675/// because SIGTERM triggers Chrome's own handler, which cascades the shutdown
1676/// itself.
1677///
1678/// A job object is the OS-level fix: processes assigned to a job are joined by
1679/// every process *they* subsequently spawn, and
1680/// `JOB_OBJECT_LIMIT_KILL_ON_JOB_CLOSE` terminates all of them the moment the
1681/// last handle to the job closes. That "last handle closes" wording is doing
1682/// more work than it looks: it holds even when *this* process dies abruptly —
1683/// a panic, a `SIGKILL`, an end-task from Task Manager — because Windows closes
1684/// our handles as part of tearing us down. The Chrome tree cannot outlive its
1685/// owner even when the owner never gets to run cleanup code. That self-healing
1686/// property is most of the value here, so [`kill_tree`](Self::kill_tree) is an
1687/// optimization for the orderly path rather than the mechanism itself.
1688///
1689/// # Non-Windows
1690///
1691/// A zero-sized no-op. Kept as a real (non-`cfg`-gated) field on
1692/// [`BrowserInner`] rather than a `#[cfg(windows)]` one so that the struct has
1693/// the same shape on every platform: the ~10 construction sites, the tests
1694/// among them, then compile identically everywhere, and a macOS `cargo check`
1695/// still type-checks the plumbing. Only the internals are conditional.
1696///
1697/// [1]: https://learn.microsoft.com/en-us/windows/win32/procthread/job-objects
1698pub(crate) struct ProcessJob {
1699    /// `Some` only for a Windows `launch()` whose assignment succeeded.
1700    /// `Mutex` because the kill happens through `&self` (via `Arc<BrowserInner>`)
1701    /// and has to *take* the job to drop it; `std::sync` because the only
1702    /// operation is a non-blocking `take()`, never held across an `.await`.
1703    #[cfg(windows)]
1704    job: std::sync::Mutex<Option<win32job::Job>>,
1705}
1706
1707impl ProcessJob {
1708    /// A handle that confines nothing: `connect()` (we do not own the process),
1709    /// every non-Windows platform, and the fallback when confinement fails.
1710    pub(crate) fn none() -> Self {
1711        Self {
1712            #[cfg(windows)]
1713            job: std::sync::Mutex::new(None),
1714        }
1715    }
1716
1717    /// Confine `child` — and everything it goes on to spawn — to a fresh job
1718    /// object that kills the lot when its last handle closes.
1719    ///
1720    /// **Never fails.** Assignment is legitimately refused in environments we do
1721    /// not control (an outer job that forbids nesting, some sandboxes and CI
1722    /// containers), and a browser that cannot be confined is still a perfectly
1723    /// usable browser. Any error is logged and degrades to [`Self::none`],
1724    /// which leaves `close()` on exactly the single-PID kill it performed
1725    /// before this existed — the pre-existing behavior, not a new failure mode.
1726    /// It must never panic and must never fail `launch()`.
1727    #[cfg(windows)]
1728    pub(crate) fn confine(child: &Child) -> Self {
1729        // `raw_handle` is `None` only once the child has been reaped, which
1730        // cannot have happened yet — but a launch is not worth panicking over.
1731        let Some(handle) = child.raw_handle() else {
1732            warn!(
1733                "chrome exited before it could be confined to a job object; \
1734                 falling back to single-process termination"
1735            );
1736            return Self::none();
1737        };
1738
1739        match Self::try_confine(handle) {
1740            Ok(job) => {
1741                debug!("confined chrome to a job object; its process tree will die with it");
1742                Self {
1743                    job: std::sync::Mutex::new(Some(job)),
1744                }
1745            }
1746            Err(e) => {
1747                warn!(
1748                    error = %e,
1749                    "could not confine chrome to a job object; falling back to \
1750                     single-process termination — chrome's renderer/GPU/utility \
1751                     processes may outlive close()"
1752                );
1753                Self::none()
1754            }
1755        }
1756    }
1757
1758    /// The fallible half of [`Self::confine`], split out so every failure path
1759    /// funnels through one `warn!` instead of being handled three times.
1760    ///
1761    /// Note the ordering: the limit is set *at creation*, before any process is
1762    /// assigned. Assigning first and setting the limit second would leave a
1763    /// window in which the job holds Chrome but would not kill it.
1764    #[cfg(windows)]
1765    fn try_confine(
1766        handle: std::os::windows::io::RawHandle,
1767    ) -> Result<win32job::Job, win32job::JobError> {
1768        let mut info = win32job::ExtendedLimitInfo::new();
1769        info.limit_kill_on_job_close();
1770        let job = win32job::Job::create_with_limit_info(&info)?;
1771        // `assign_process` takes the raw `HANDLE` as an `isize`. A pointer →
1772        // integer cast is safe; no `unsafe` block is needed anywhere here,
1773        // which is the whole reason `win32job` is preferred over raw FFI in a
1774        // crate that denies `unsafe_code`.
1775        job.assign_process(handle as isize)?;
1776        Ok(job)
1777    }
1778
1779    /// No-op stand-in on platforms without job objects, where Chrome's own
1780    /// SIGTERM handler already cascades the shutdown.
1781    #[cfg(not(windows))]
1782    pub(crate) fn confine(_child: &Child) -> Self {
1783        Self::none()
1784    }
1785
1786    /// Terminate every process still in the job, immediately.
1787    ///
1788    /// Returns `true` if a job was actually closed, so the caller can tell
1789    /// "the tree is being killed" from "there is no job; fall back to the
1790    /// single-PID kill".
1791    ///
1792    /// Dropping the job *is* the kill: closing the last handle to a
1793    /// `KILL_ON_JOB_CLOSE` job is what terminates its members, and there is no
1794    /// separate "terminate" call to make. Idempotent — a second call finds the
1795    /// slot empty and reports `false`.
1796    #[cfg(windows)]
1797    pub(crate) fn kill_tree(&self) -> bool {
1798        let taken = self
1799            .job
1800            .lock()
1801            .unwrap_or_else(std::sync::PoisonError::into_inner)
1802            .take();
1803
1804        match taken {
1805            Some(job) => {
1806                // Explicit, because this drop is load-bearing rather than
1807                // incidental: it is the syscall that kills the tree.
1808                drop(job);
1809                true
1810            }
1811            None => false,
1812        }
1813    }
1814
1815    /// No-op stand-in; there is never a job to close off Windows.
1816    ///
1817    /// Called only from `close()`'s `cfg(not(unix))` branch, so a Unix build
1818    /// never reaches it. Scoped to `unix` rather than a blanket allow so the
1819    /// Windows build — where this is load-bearing — still reports it if the
1820    /// call site is ever lost.
1821    #[cfg(not(windows))]
1822    #[cfg_attr(unix, allow(dead_code))]
1823    pub(crate) fn kill_tree(&self) -> bool {
1824        false
1825    }
1826}

1843/// The freshly-spawned Chrome process, shared between [`guard_handshake`] and
1844/// [`finish_connect`].
1845///
1846/// Shared rather than moved because ownership has to survive a *dropped*
1847/// future: [`finish_connect`] installs the child on [`BrowserInner`] only after
1848/// its CDP round-trips succeed, so when the handshake budget expires mid-flight
1849/// the guard must still be able to reach the child and kill it. A plain
1850/// `Option<Child>` moved into the handshake future would vanish with it,
1851/// leaving the orphan `chrome.exe` this type exists to prevent.
1852///
1853/// `std::sync::Mutex` (not tokio's) is deliberate: the only operation is a
1854/// non-blocking `take()`, never held across an `.await`.
1855pub(crate) type ChildSlot = Arc<std::sync::Mutex<Option<Child>>>;
1856
1857/// Run the post-endpoint handshake under `budget`, guaranteeing that a launch
1858/// which times out leaves no orphan Chrome behind.
1859///
1860/// `handshake` is the WS dial + [`finish_connect`] sequence. On expiry the
1861/// future is dropped (cancelling the in-flight CDP call) and any child still
1862/// parked in `child_slot` is killed **and reaped** before returning, so the
1863/// process is gone by the time the caller sees the error rather than whenever
1864/// tokio's orphan queue next runs.
1865///
1866/// The `kill_on_drop(true)` set at spawn stays as a backstop for the narrow
1867/// window after [`finish_connect`] has moved the child onto `BrowserInner` (the
1868/// `Arc<BrowserInner>` then drops with the future, and
1869/// [`Drop for BrowserInner`] relies on `kill_on_drop`). The explicit kill here
1870/// covers the whole span before that — which is where every CDP round-trip,
1871/// and therefore every realistic stall, actually lives.
1872///
1873/// # Errors
1874///
1875/// Returns [`BrowserError::HandshakeTimeout`] when `budget` expires; otherwise
1876/// passes the handshake's own result through untouched.
1877pub(crate) async fn guard_handshake<F>(
1878    budget: Duration,
1879    child_slot: &ChildSlot,
1880    handshake: F,
1881) -> Result<Arc<BrowserInner>, ZendriverError>
1882where
1883    F: std::future::Future<Output = Result<Arc<BrowserInner>, ZendriverError>>,
1884{
1885    match timeout(budget, handshake).await {
1886        Ok(res) => res,
1887        Err(_elapsed) => {
1888            let orphan = child_slot
1889                .lock()
1890                .unwrap_or_else(std::sync::PoisonError::into_inner)
1891                .take();
1892            if let Some(mut child) = orphan {
1893                warn!(
1894                    budget = ?budget,
1895                    "chrome CDP handshake exceeded its budget; killing the spawned child",
1896                );
1897                // `kill()` is start_kill + wait, so the process is reaped
1898                // before we return — no zombie, no orphan.
1899                let _ = child.kill().await;
1900            }
1901            Err(BrowserError::HandshakeTimeout.into())
1902        }
1903    }
1904}

1954/// Run the post-connect CDP handshake and assemble [`BrowserInner`].
1955///
1956/// Shared by [`BrowserBuilder::launch`] and [`BrowserBuilder::connect`]: both
1957/// arrive here with a live [`Connection`] (observer chain already attached)
1958/// and need the same sequence — enable browser-scoped auto-attach, discover +
1959/// attach the main page target, build the self-referential `Arc<BrowserInner>`,
1960/// then backfill the registrar's weak ref and the main-tab registry entry.
1961///
1962/// The only spawn-vs-attach differences are carried in [`FinishConnect`]: the
1963/// owned `Child` / `TempDir` (present only for `launch`) and the
1964/// `owns_process` flag.
1965pub(crate) async fn finish_connect(
1966    args: FinishConnect,
1967) -> Result<Arc<BrowserInner>, ZendriverError> {
1968    let FinishConnect {
1969        conn,
1970        registrar,
1971        input_profile,
1972        child,
1973        job,
1974        owned_tmp,
1975        extension_dirs,
1976        debug_host_port,
1977        ws_url,
1978        owns_process,
1979        #[cfg(feature = "tracker-blocking")]
1980        tracker_matcher,
1981    } = args;
1982
1983    // Enable auto-attach with debugger-pause BEFORE attaching to the initial
1984    // target. Sent at browser scope (no session_id) so it covers both the
1985    // initial target and any subsequently-opened pages/iframes.
1986    conn.call_raw(
1987        "Target.setAutoAttach",
1988        json!({
1989            "autoAttach": true,
1990            "waitForDebuggerOnStart": true,
1991            "flatten": true,
1992        }),
1993        None,
1994    )
1995    .await?;
1996
1997    // Discover the initial target via Target.getTargets (prefer a page).
1998    let list = conn.call_raw("Target.getTargets", json!({}), None).await?;
1999    let no_targets = Vec::new();
2000    let targets = list["targetInfos"].as_array().unwrap_or(&no_targets);
2001    // Preference rule (unchanged): the first page target, else the first
2002    // target of any type.
2003    let target_id = targets
2004        .iter()
2005        .find(|t| t["type"] == "page")
2006        .or_else(|| targets.first())
2007        .and_then(|t| t["targetId"].as_str())
2008        .ok_or_else(|| ZendriverError::Navigation("no initial target found".into()))?
2009        .to_string();
2010
2011    // Every *other* page target is a window we are not driving. Chrome opens
2012    // some on its own (a Windows first-run window, a `chrome://newtab/`), and
2013    // `.find()`-ing one target used to silently discard the rest: invisible to
2014    // `Browser::tabs()`, and closed by nothing — `close()` only ever killed the
2015    // one tracked PID. Collect them now, sweep them once we are attached.
2016    //
2017    // Only for a Chrome we spawned. On the `connect` path those extra pages are
2018    // the user's own tabs in their own browser; closing them would be
2019    // destructive and is emphatically not ours to do. Non-page targets (workers,
2020    // etc.) are never swept — they are not windows.
2021    let strays: Vec<String> = if owns_process {
2022        targets
2023            .iter()
2024            .filter(|t| t["type"] == "page")
2025            .filter_map(|t| t["targetId"].as_str())
2026            .filter(|id| *id != target_id)
2027            .map(str::to_string)
2028            .collect()
2029    } else {
2030        Vec::new()
2031    };
2032
2033    // Attach to the initial target. This triggers `Target.attachedToTarget`
2034    // which the actor routes through observers (`on_target_attached`) and
2035    // then releases via `Runtime.runIfWaitingForDebugger`.
2036    //
2037    // The `TabRegistrar` observer (in the chain) will try to insert into
2038    // `BrowserInner.tabs` for the main tab too. That insertion is a no-op
2039    // because the weak ref isn't wired yet (`OnceLock` empty → observer logs
2040    // at debug and skips). We re-insert the main tab manually below so the
2041    // registry is consistent post-connect.
2042    let attach = conn
2043        .call_raw(
2044            "Target.attachToTarget",
2045            json!({ "targetId": target_id, "flatten": true }),
2046            None,
2047        )
2048        .await?;
2049    let session_id = attach["sessionId"]
2050        .as_str()
2051        .ok_or_else(|| ZendriverError::Navigation("attach returned no sessionId".into()))?
2052        .to_string();
2053
2054    // Sweep the stray windows now that our own tab is safely attached.
2055    // Best-effort: a stray we cannot close is worth a warning, not a failed
2056    // launch — the caller has a perfectly usable browser either way.
2057    for stray in strays {
2058        match conn
2059            .call_raw("Target.closeTarget", json!({ "targetId": stray }), None)
2060            .await
2061        {
2062            Ok(_) => debug!(target_id = %stray, "closed stray page target at connect"),
2063            Err(e) => {
2064                warn!(target_id = %stray, error = %e, "could not close stray page target");
2065            }
2066        }
2067    }
2068
2069    // Wrap session in Tab; build BrowserInner via the canonical
2070    // `Arc::new_cyclic` self-referential pattern.
2071    let session_id_for_registry = session_id.clone();
2072    let target_id_for_main_tab = target_id.clone();
2073    let inner = Arc::new_cyclic(|weak: &std::sync::Weak<BrowserInner>| {
2074        let session = SessionHandle::new(conn.clone(), session_id);
2075        let main_tab_input = InputController::new(input_profile.clone());
2076        let main_tab = Tab::new(
2077            session,
2078            weak.clone(),
2079            main_tab_input,
2080            target_id_for_main_tab,
2081        );
2082        BrowserInner {
2083            conn,
2084            main_tab,
2085            // Take ownership of the child only now, once every CDP round-trip
2086            // has succeeded. Until this point `guard_handshake` owns the
2087            // cleanup: if the budget expires mid-handshake the child is still
2088            // in the slot and gets killed there.
2089            child: tokio::sync::Mutex::new(
2090                child
2091                    .lock()
2092                    .unwrap_or_else(std::sync::PoisonError::into_inner)
2093                    .take(),
2094            ),
2095            // Moves with the child, for the same reason: until this point the
2096            // job belongs to the handshake future, so a timed-out launch drops
2097            // it and takes the whole Chrome tree down with it.
2098            job,
2099            _user_data: owned_tmp,
2100            _extension_dirs: extension_dirs,
2101            owns_process,
2102            stealth_input_profile: input_profile,
2103            tabs: tokio::sync::RwLock::new(HashMap::new()),
2104            debug_host_port,
2105            ws_url,
2106            tabs_changed: tokio::sync::Notify::new(),
2107            #[cfg(feature = "interception")]
2108            proxy_auth_handle: std::sync::OnceLock::new(),
2109            #[cfg(feature = "tracker-blocking")]
2110            tracker_matcher,
2111            #[cfg(feature = "tracker-blocking")]
2112            tracker_handles: tokio::sync::Mutex::new(HashMap::new()),
2113        }
2114    });
2115
2116    // Wire the registrar's weak ref + manually insert the main tab (it was
2117    // attached before the weak ref existed, so the observer bailed early).
2118    registrar.set_browser(Arc::downgrade(&inner));
2119    inner
2120        .tabs
2121        .write()
2122        .await
2123        .insert(session_id_for_registry, inner.main_tab.clone());
2124
2125    Ok(inner)
2126}

2128/// Resolve a `webSocketDebuggerUrl` from a DevTools HTTP base by issuing a
2129/// minimal `GET {endpoint}/json/version` over a raw [`tokio::net::TcpStream`].
2130///
2131/// Hand-rolled (HTTP/1.0, read-until-close, split headers from body, parse the
2132/// JSON body) so the always-on dependency set does not grow an HTTP client —
2133/// `connect`'s `ws://` path needs no HTTP at all, and the `fetcher`-gated
2134/// `reqwest` must not leak into the default build. Mirrors nodriver /
2135/// zendriver-py, which read the same `webSocketDebuggerUrl` field.
2136///
2137/// # Errors
2138///
2139/// Returns [`BrowserError::DevtoolsParse`] when the endpoint is malformed, the
2140/// TCP round-trip fails / times out, the response has no body, or the JSON
2141/// lacks a string `webSocketDebuggerUrl`.
2142pub(crate) async fn resolve_ws_from_http(endpoint: &str) -> Result<String, ZendriverError> {
2143    use tokio::io::{AsyncReadExt, AsyncWriteExt};
2144    use tokio::net::TcpStream;
2145
2146    // Parse `scheme://host[:port]` → (host, port). Strip any trailing path.
2147    let after_scheme = endpoint
2148        .strip_prefix("http://")
2149        .or_else(|| endpoint.strip_prefix("https://"))
2150        .ok_or(BrowserError::DevtoolsParse)?;
2151    let authority = after_scheme
2152        .split('/')
2153        .next()
2154        .unwrap_or(after_scheme)
2155        .trim();
2156    if authority.is_empty() {
2157        return Err(BrowserError::DevtoolsParse.into());
2158    }
2159    // host:port split — default to 80 when no explicit port (DevTools is
2160    // typically explicit, e.g. 9222, but be lenient).
2161    let (host, port) = match authority.rsplit_once(':') {
2162        Some((h, p)) => (
2163            h,
2164            p.parse::<u16>().map_err(|_| BrowserError::DevtoolsParse)?,
2165        ),
2166        None => (authority, 80u16),
2167    };
2168
2169    let body = timeout(JSON_VERSION_TIMEOUT, async {
2170        let mut stream = TcpStream::connect((host, port))
2171            .await
2172            .map_err(BrowserError::SpawnFailed)?;
2173        // HTTP/1.0 + Connection: close → server closes the socket after the
2174        // response, so a read-to-end cleanly delimits the message.
2175        let req = format!(
2176            "GET /json/version HTTP/1.0\r\nHost: {authority}\r\nAccept: application/json\r\nConnection: close\r\n\r\n"
2177        );
2178        stream
2179            .write_all(req.as_bytes())
2180            .await
2181            .map_err(BrowserError::SpawnFailed)?;
2182        stream.flush().await.map_err(BrowserError::SpawnFailed)?;
2183        let mut buf = Vec::new();
2184        stream
2185            .read_to_end(&mut buf)
2186            .await
2187            .map_err(BrowserError::SpawnFailed)?;
2188        Ok::<Vec<u8>, ZendriverError>(buf)
2189    })
2190    .await
2191    .map_err(|_| BrowserError::WsTimeout)??;
2192
2193    parse_ws_from_json_version(&body)
2194}

2196/// Split an HTTP/1.x response into its body and parse the
2197/// `webSocketDebuggerUrl` field from the JSON.
2198///
2199/// Factored out of [`resolve_ws_from_http`] so the parse — the part most worth
2200/// asserting — is unit-testable without a socket. Splits on the first
2201/// `\r\n\r\n` (header/body boundary); if absent, treats the whole buffer as the
2202/// JSON body (lenient toward bodies returned without standard headers in
2203/// tests).

2205    let text = String::from_utf8_lossy(raw);
2206    // Header/body split on the blank line; fall back to the whole buffer.
2207    let body = text
2208        .split_once("\r\n\r\n")
2209        .map(|(_, b)| b)
2210        .unwrap_or(&text)
2211        .trim();
2212    if body.is_empty() {
2213        return Err(BrowserError::DevtoolsParse.into());
2214    }
2215    let parsed: serde_json::Value =
2216        serde_json::from_str(body).map_err(|_| BrowserError::DevtoolsParse)?;
2217    parsed["webSocketDebuggerUrl"]
2218        .as_str()
2219        .map(|s| s.to_string())
2220        .ok_or_else(|| BrowserError::DevtoolsParse.into())
2221}

2223/// Resolve every entry in `extensions` to an unpacked-extension *directory*,
2224/// in place, returning the tempdirs that back any unzipped `.crx` archives.
2225///
2226/// `--load-extension` only accepts directories, so a `.crx` file is unzipped
2227/// into a fresh [`TempDir`] and the slot is rewritten to that directory.
2228/// Entries that are already directories pass through unchanged. The returned
2229/// tempdirs MUST be kept alive for as long as Chrome runs (they are parked on
2230/// [`BrowserInner`]); dropping one deletes the extracted extension out from
2231/// under the running browser.
2232///
2233/// A `.crx` is a ZIP with a short binary header (magic `Cr24` + version +
2234/// signature lengths) prepended; we locate the embedded ZIP by scanning for
2235/// the local-file-header magic (`PK\x03\x04`) and unzip from there, so both
2236/// CRX2 and CRX3 layouts work without parsing the header fields.
2237///
2238/// # Errors
2239///
2240/// Returns [`BrowserError::ExtensionLoad`] when a configured path does not
2241/// exist, when a `.crx` cannot be read / contains no ZIP payload / fails to
2242/// unzip, or when an entry is neither a directory nor a `.crx` file.
2243async fn resolve_extension_dirs(
2244    extensions: &mut [PathBuf],
2245) -> Result<Vec<TempDir>, ZendriverError> {
2246    let mut tempdirs = Vec::new();
2247    for slot in extensions.iter_mut() {
2248        if slot.is_dir() {
2249            continue;
2250        }
2251        let is_crx = slot
2252            .extension()
2253            .is_some_and(|e| e.eq_ignore_ascii_case("crx"));
2254        if !is_crx {
2255            return Err(BrowserError::ExtensionLoad {
2256                path: slot.clone(),
2257                reason: if slot.exists() {
2258                    "extension path is neither a directory nor a .crx file".into()
2259                } else {
2260                    "extension path does not exist".into()
2261                },
2262            }
2263            .into());
2264        }
2265        let (dir, td) = unzip_crx(slot).await?;
2266        *slot = dir;
2267        tempdirs.push(td);
2268    }
2269    Ok(tempdirs)
2270}

2272/// Unzip a single `.crx` at `crx_path` into a fresh [`TempDir`], returning the
2273/// extracted directory path alongside the owning tempdir handle.
2274///
2275/// The blocking ZIP walk runs on [`tokio::task::spawn_blocking`]. Path safety
2276/// mirrors the fetcher's extractor: entries with absolute / parent-traversal
2277/// paths and (on Unix) symlink entries are rejected so a hostile `.crx` can't
2278/// escape the tempdir.
2279async fn unzip_crx(crx_path: &Path) -> Result<(PathBuf, TempDir), ZendriverError> {
2280    let crx_path = crx_path.to_path_buf();
2281    let td = tempfile::Builder::new()
2282        .prefix("zendriver-ext-")
2283        .tempdir()
2284        .map_err(BrowserError::SpawnFailed)?;
2285    let dest = td.path().to_path_buf();
2286    let dest_for_blocking = dest.clone();
2287    let crx_for_blocking = crx_path.clone();
2288
2289    tokio::task::spawn_blocking(move || unzip_crx_blocking(&crx_for_blocking, &dest_for_blocking))
2290        .await
2291        .map_err(|e| BrowserError::ExtensionLoad {
2292            path: crx_path.clone(),
2293            reason: format!("unzip task join error: {e}"),
2294        })??;
2295
2296    Ok((dest, td))
2297}

2299/// Synchronous `.crx` → directory unzip body (runs on a blocking thread).
2300fn unzip_crx_blocking(crx_path: &Path, dest_dir: &Path) -> Result<(), ZendriverError> {
2301    use std::io::Cursor;
2302
2303    let bytes = std::fs::read(crx_path).map_err(|e| BrowserError::ExtensionLoad {
2304        path: crx_path.to_path_buf(),
2305        reason: format!("read failed: {e}"),
2306    })?;
2307    // A `.crx` prepends a binary header before the ZIP. Find the first ZIP
2308    // local-file-header signature (`PK\x03\x04`) and treat everything from
2309    // there as the archive. A bare `.zip` (no CRX header) also works since the
2310    // signature is at offset 0.
2311    let zip_start = bytes
2312        .windows(4)
2313        .position(|w| w == [0x50, 0x4B, 0x03, 0x04])
2314        .ok_or_else(|| BrowserError::ExtensionLoad {
2315            path: crx_path.to_path_buf(),
2316            reason: "no ZIP payload found in .crx".into(),
2317        })?;
2318    let mut archive = zip::ZipArchive::new(Cursor::new(&bytes[zip_start..])).map_err(|e| {
2319        BrowserError::ExtensionLoad {
2320            path: crx_path.to_path_buf(),
2321            reason: format!("not a valid ZIP: {e}"),
2322        }
2323    })?;
2324
2325    for i in 0..archive.len() {
2326        let mut entry = archive
2327            .by_index(i)
2328            .map_err(|e| BrowserError::ExtensionLoad {
2329                path: crx_path.to_path_buf(),
2330                reason: format!("zip entry {i}: {e}"),
2331            })?;
2332        // Reject unsafe paths (absolute / parent-traversal).
2333        let Some(rel_path) = entry.enclosed_name() else {
2334            return Err(BrowserError::ExtensionLoad {
2335                path: crx_path.to_path_buf(),
2336                reason: format!("zip entry has unsafe path: {}", entry.name()),
2337            }
2338            .into());
2339        };
2340        // Refuse symlink entries — the primary zip-slip follow-on vector.
2341        #[cfg(unix)]
2342        if let Some(mode) = entry.unix_mode() {
2343            const S_IFMT: u32 = 0o170_000;
2344            const S_IFLNK: u32 = 0o120_000;
2345            if mode & S_IFMT == S_IFLNK {
2346                return Err(BrowserError::ExtensionLoad {
2347                    path: crx_path.to_path_buf(),
2348                    reason: format!("zip entry {rel_path:?} is a symlink; refusing"),
2349                }
2350                .into());
2351            }
2352        }
2353        let out_path = dest_dir.join(&rel_path);
2354        if entry.is_dir() {
2355            std::fs::create_dir_all(&out_path).map_err(|e| BrowserError::ExtensionLoad {
2356                path: crx_path.to_path_buf(),
2357                reason: format!("mkdir {out_path:?}: {e}"),
2358            })?;
2359            continue;
2360        }
2361        if let Some(parent) = out_path.parent() {
2362            std::fs::create_dir_all(parent).map_err(|e| BrowserError::ExtensionLoad {
2363                path: crx_path.to_path_buf(),
2364                reason: format!("mkdir {parent:?}: {e}"),
2365            })?;
2366        }
2367        let mut out =
2368            std::fs::File::create(&out_path).map_err(|e| BrowserError::ExtensionLoad {
2369                path: crx_path.to_path_buf(),
2370                reason: format!("create {out_path:?}: {e}"),
2371            })?;
2372        std::io::copy(&mut entry, &mut out).map_err(|e| BrowserError::ExtensionLoad {
2373            path: crx_path.to_path_buf(),
2374            reason: format!("write {out_path:?}: {e}"),
2375        })?;
2376    }
2377    Ok(())
2378}

2380impl BrowserBuilder {
2381    /// Spawn Chrome and attach. Returns once the main tab is bound.
2382    ///
2383    /// When a [`StealthProfile`] is set (the default), this:
2384    /// 1. Resolves a [`zendriver_stealth::Fingerprint`] from the resolved
2385    ///    Chrome executable.
2386    /// 2. Prepends the profile's [`StealthObserver`] to the observer chain.
2387    /// 3. Appends the profile's stealth flags to the launch argv.
2388    /// 4. Sends `Target.setAutoAttach { waitForDebuggerOnStart: true }` at
2389    ///    browser scope so the actor can route pauses through observers
2390    ///    before any page script runs.
2391    ///
2392    /// # Errors
2393    ///
2394    /// Returns [`ZendriverError::Browser`] when Chrome can't be discovered
2395    /// or spawned, [`ZendriverError::Stealth`] when fingerprint resolution
2396    /// fails, [`ZendriverError::Transport`] when the WebSocket attach times
2397    /// out.
2398    ///
2399    /// # Examples
2400    ///
2401    /// ```no_run
2402    /// # async fn ex() -> zendriver::Result<()> {
2403    /// let browser = zendriver::Browser::builder().launch().await?;
2404    /// # browser.close().await?;
2405    /// # Ok(()) }
2406    /// ```
2407    pub async fn launch(mut self) -> Result<Browser, ZendriverError> {
2408        // 1. Resolve Chrome executable.
2409        // Precedence: explicit `.executable(...)` > `CHROME_BIN` env var >
2410        // per-channel platform discovery. The env-var hop lets CI (and local
2411        // devs pointing at Canary / a downloaded Chrome-for-Testing build)
2412        // override the discovery path without code changes. The configured
2413        // `channel` (default `Auto`) only steers the final discovery step.
2414        let exe = match self.executable.clone() {
2415            Some(p) => p,
2416            None => match std::env::var("CHROME_BIN").ok().filter(|s| !s.is_empty()) {
2417                Some(p) => PathBuf::from(p),
2418                None => find_chrome_executable_for_channel(self.channel)?,
2419            },
2420        };
2421
2422        // 1b. Resolve extensions: unzip any `.crx` into a tempdir and rewrite
2423        // `self.extensions` to the resolved unpacked-directory paths so
2424        // `build_flags` emits `--load-extension=<dir,…>`. Directory entries
2425        // pass through untouched. The returned tempdirs are handed to
2426        // `BrowserInner` below so the extracted dirs outlive Chrome.
2427        let extension_dirs = resolve_extension_dirs(&mut self.extensions).await?;
2428
2429        // 2. Resolve the per-tab InputProfile from the active StealthProfile
2430        // (or zero-overhead `native` when stealth is off). Cached on
2431        // `BrowserInner` so `Browser::new_tab` + the `TabRegistrar` observer
2432        // can build fresh per-Tab controllers without re-resolving the
2433        // profile each time.
2434        let input_profile = self
2435            .stealth
2436            .as_ref()
2437            .map_or_else(zendriver_stealth::InputProfile::native, |sp| {
2438                sp.input_profile()
2439            });
2440
2441        // 3. Build the `TabRegistrar` observer. Holds a `OnceLock` for the
2442        // `Weak<BrowserInner>` that gets wired in step 10 — observers must
2443        // be passed to `connect_with_observers` before the cyclic `Arc` is
2444        // resolved, so the weak ref is filled in later. Retained here so
2445        // we can `set_browser` after construction.
2446        let registrar = Arc::new(TabRegistrar::new(input_profile.clone()));
2447
2448        // 4. Resolve fingerprint + build observer chain + profile flags.
2449        // Observer order: stealth (patches each new target) → tab registrar
2450        // (records the resulting Tab handle) → user-supplied observers →
2451        // force-open-shadow-roots (independent of the stealth bundle).
2452        let (mut observers, extra_flags): (Vec<Arc<dyn TargetObserver>>, Vec<String>) =
2453            if let Some(ref profile) = self.stealth {
2454                let fp = profile.resolve_fingerprint(&exe)?;
2455                let stealth_obs: Arc<dyn TargetObserver> = Arc::new(StealthObserver::with_persona(
2456                    profile.clone(),
2457                    fp,
2458                    self.resolved_persona(),
2459                ));
2460                let mut obs_vec = Vec::with_capacity(3 + self.extra_observers.len());
2461                obs_vec.push(stealth_obs);
2462                obs_vec.push(registrar.clone() as Arc<dyn TargetObserver>);
2463                obs_vec.extend(self.extra_observers.iter().cloned());
2464                (obs_vec, profile.build_flags())
2465            } else {
2466                let mut obs_vec = Vec::with_capacity(2 + self.extra_observers.len());
2467                obs_vec.push(registrar.clone() as Arc<dyn TargetObserver>);
2468                obs_vec.extend(self.extra_observers.iter().cloned());
2469                (obs_vec, Vec::new())
2470            };
2471        // Append the force-open-shadow-roots observer last so its injected
2472        // attachShadow override runs independently of (and after) the stealth
2473        // bundle. Only added when the caller opted in — keeps the default
2474        // observer chain untouched.
2475        if self.force_open_shadow_roots {
2476            observers.push(Arc::new(crate::expert::ShadowRootObserver) as Arc<dyn TargetObserver>);
2477        }
2478
2479        // 5. Allocate user_data_dir (or use a TempDir we keep alive until shutdown).
2480        let (user_data_path, owned_tmp) = match self.user_data_dir.clone() {
2481            Some(p) => (p, None),
2482            None => {
2483                let td = tempfile::Builder::new()
2484                    .prefix("zendriver-")
2485                    .tempdir()
2486                    .map_err(BrowserError::SpawnFailed)?;
2487                (td.path().to_path_buf(), Some(td))
2488            }
2489        };

2491        // Write Chrome profile preferences (popup suppression for owned temp
2492        // profiles; explicit user prefs always). Best-effort — see preferences.rs.
2493        crate::preferences::write_preferences(
2494            &user_data_path,
2495            owned_tmp.is_some(),
2496            &self.preferences,
2497        );
2498
2499        let mut flags = self.build_flags(&user_data_path);
2500        flags.extend(extra_flags);
2501        // CI-friendly defaults: when running under CI (the runner sets
2502        // `CI=true`), Chrome's user-namespace sandbox refuses to start
2503        // because the GitHub-Actions / Docker container runs as root,
2504        // and the small /dev/shm in the container OOMs the renderer on
2505        // real workloads. Auto-add `--no-sandbox` and
2506        // `--disable-dev-shm-usage` unless the caller already supplied
2507        // them (so explicit user opt-in still wins).
2508        if std::env::var("CI").is_ok() {

2510                if !flags.iter().any(|f| f == needed) {
2511                    flags.push(needed.into());
2512                }

2518        // spawn. The default `user_data_dir` is a fresh tempdir so there is
2519        // nothing to find, but a caller-supplied dir can carry a stale file
2520        // naming a long-dead port — which the poll below would otherwise
2521        // resolve instantly and hand us an endpoint that dials nothing.
2522        let _ = std::fs::remove_file(user_data_path.join(DEVTOOLS_ACTIVE_PORT_FILE));

2527            .stdin(Stdio::null())
2528            .stdout(Stdio::null())
2529            .stderr(Stdio::piped())

2533        // Ctrl+C cannot signal it behind our back and race the orchestrated
2534        // shutdown in `close()`. See [`CREATE_NEW_PROCESS_GROUP`].

2544        // Confine Chrome's whole process tree to a job object, immediately —
2545        // before its first `.await`, so nothing can be scheduled between the
2546        // spawn and the assignment.
2547        //
2548        // Job membership is inherited by processes spawned *after* assignment,
2549        // which is every Chrome helper that matters: chrome.exe has to complete
2550        // a good deal of startup before it forks its first crashpad/GPU child,
2551        // and this runs within microseconds of the spawn returning.

2553        // `CREATE_SUSPENDED` + assign-before-resume would close that window

2558        // covers, in a crate that denies `unsafe_code`, to close a race whose

2563        // Never fails: see [`ProcessJob::confine`].

2579        //
2580        // Deliberately additive: an stderr EOF still fails fast with

2584        // poll can only ever win *earlier*; it never delays a failure.

2588                    while let Ok(Some(line)) = lines.next_line().await {
2589                        debug!(line = %line, "chrome stderr");
2590                        if let Some(url) = parse_devtools_line(&line) {
2591                            return Ok::<String, ZendriverError>(url);
2592                        }
2593                    }
2594                    Err(BrowserError::DevtoolsParse.into())
2595                } => res,
2596                url = poll_devtools_active_port(&user_data_path) => {
2597                    debug!(ws_url = %url, "resolved endpoint from DevToolsActivePort");

2599                }
2600            }
2601        })
2602        .await
2603        .map_err(|_| BrowserError::WsTimeout)??;

2607        // Identical for spawn (`launch`) and attach (`connect`); the only
2608        // differences are the owned `Child` / `TempDir` we hand it and the
2609        // `owns_process` flag (true here — we spawned Chrome).

2612
2613        // Everything from here to a live `BrowserInner` runs under a single

2615        // its endpoint; the dial and the CDP round-trips that follow used to be
2616        // unbounded, so a Chrome that came up but never answered CDP hung
2617        // `launch()` forever. `guard_handshake` bounds them and kills the

2624            finish_connect(FinishConnect {
2625                conn,
2626                registrar,
2627                input_profile,
2628                child: child_slot.clone(),
2629                job,
2630                owned_tmp,
2631                extension_dirs,
2632                debug_host_port: debug_host_port_from_ws(&ws_url),
2633                ws_url: Some(ws_url.clone()),
2634                owns_process: true,
2635                #[cfg(feature = "tracker-blocking")]
2636                tracker_matcher,
2637            })
2638            .await
2639        })
2640        .await?;
2641
2642        // 13. If a custom downloads_dir was set, configure browser-scoped
2643        // download behavior so all tabs (current + future) save into it.
2644        // See cdpdriver/zendriver#88.
2645        if let Some(dir) = self.downloads_dir.as_ref() {
2646            inner
2647                .conn
2648                .call_raw(
2649                    "Browser.setDownloadBehavior",
2650                    json!({
2651                        "behavior": "allow",
2652                        "downloadPath": dir.display().to_string(),
2653                        "eventsEnabled": true,
2654                    }),
2655                    None,
2656                )
2657                .await?;
2658        }
2659
2660        // 14. Spawn a single main-session interception actor carrying BOTH
2661        // proxy-auth (handle_auth) and tracker blocking (block_hosts), so they
2662        // don't double-resolve the same Fetch.requestPaused. New tabs get their

2664        // main-tab-only). The InterceptHandle is parked on BrowserInner so the
2665        // actor lives as long as the Browser does; dropping BrowserInner drops
2666        // the handle which cancels the actor. See cdpdriver/zendriver#208.
2667        #[cfg(feature = "interception")]
2668        {
2669            let main_session = inner.main_tab.session().clone();
2670            let mut builder = zendriver_interception::InterceptBuilder::new(&main_session);
2671            let mut needs_actor = false;
2672            if let Some((user, pass)) = self.proxy_auth.clone() {
2673                builder = builder.handle_auth(user, pass);
2674                needs_actor = true;
2675            }
2676            #[cfg(feature = "tracker-blocking")]
2677            if let Some(matcher) = inner.tracker_matcher.clone() {
2678                builder = builder.block_hosts(matcher);
2679                needs_actor = true;
2680            }
2681            if needs_actor {
2682                let handle = builder.start();
2683                let _ = inner.proxy_auth_handle.set(handle);
2684            }
2685        }
2686
2687        Ok(Browser { inner })
2688    }

2691    /// a new process.
2692    ///
2693    /// `endpoint` is detected by scheme:
2694    /// - `ws://…` / `wss://…` — used directly as the DevTools browser
2695    ///   WebSocket URL (the `webSocketDebuggerUrl` Chrome prints to stderr as
2696    ///   `DevTools listening on …`).
2697    /// - `http://host:port` / `https://host:port` — a DevTools HTTP base;
2698    ///   `connect` performs `GET {endpoint}/json/version` and reads
2699    ///   `webSocketDebuggerUrl` from the JSON body (matching nodriver /
2700    ///   zendriver-py).
2701    ///
2702    /// The connected [`Browser`] does **not** own the Chrome process: its
2703    /// [`Browser::close`] shuts down only the transport, and dropping it does
2704    /// **not** terminate Chrome. Use this to drive a long-lived browser you
2705    /// started elsewhere.
2706    ///
2707    /// `.stealth(profile)` and `.observer(..)` still apply, but only to
2708    /// targets attached **after** this call — the stealth observer is wired
2709    /// into the same browser-wide `Target.setAutoAttach { flatten: true }`
2710    /// that fires on newly-attached targets. Tabs already open when you
2711    /// connect predate the observer chain and are **not** patched.
2712    ///
2713    /// The spawn-only builder fields — [`BrowserBuilder::executable`],
2714    /// [`BrowserBuilder::user_data_dir`], [`BrowserBuilder::downloads_dir`],
2715    /// and any launch flags ([`BrowserBuilder::arg`] / headless / sandbox /
2716    /// channel / lang / user-agent) — are **ignored** on the connect path,
2717    /// since no process is launched.
2718    ///
2719    /// # Errors
2720    ///
2721    /// Returns [`ZendriverError::Browser`] when an `http(s)://` endpoint
2722    /// cannot be resolved to a `webSocketDebuggerUrl`, and
2723    /// [`ZendriverError::Transport`] when the WebSocket attach fails.
2724    ///
2725    /// # Examples
2726    ///
2727    /// ```no_run
2728    /// # async fn ex() -> zendriver::Result<()> {
2729    /// // Chrome already started with `--remote-debugging-port=9222`.
2730    /// let browser = zendriver::Browser::builder()
2731    ///     .connect("http://127.0.0.1:9222").await?;
2732    /// let tab = browser.main_tab();
2733    /// tab.goto("https://example.com").await?;
2734    /// // Does NOT kill the Chrome we attached to:
2735    /// browser.close().await?;
2736    /// # Ok(()) }
2737    /// ```
2738    pub async fn connect(self, endpoint: impl Into<String>) -> Result<Browser, ZendriverError> {
2739        let endpoint = endpoint.into();
2740
2741        // Resolve the browser WebSocket URL from the endpoint scheme.
2742        let ws_url = if endpoint.starts_with("ws://") || endpoint.starts_with("wss://") {
2743            endpoint
2744        } else if endpoint.starts_with("http://") || endpoint.starts_with("https://") {
2745            resolve_ws_from_http(&endpoint).await?
2746        } else {
2747            return Err(BrowserError::DevtoolsParse.into());
2748        };
2749
2750        // Resolve the per-tab InputProfile + build the observer chain exactly
2751        // like `launch` does (stealth observer → tab registrar → user
2752        // observers). The spawn-only branches (`executable`, flags, TempDir)
2753        // are intentionally skipped — `connect` never launches a process.
2754        let input_profile = self
2755            .stealth
2756            .as_ref()
2757            .map_or_else(zendriver_stealth::InputProfile::native, |sp| {
2758                sp.input_profile()
2759            });
2760        let registrar = Arc::new(TabRegistrar::new(input_profile.clone()));
2761        let mut observers: Vec<Arc<dyn TargetObserver>> = if let Some(ref profile) = self.stealth {
2762            // No executable to resolve a fingerprint from on the connect path;
2763            // fall back to the profile's default fingerprint.
2764            let fp = profile.resolve_fingerprint(Path::new(""))?;
2765            let stealth_obs: Arc<dyn TargetObserver> = Arc::new(StealthObserver::with_persona(
2766                profile.clone(),
2767                fp,
2768                self.resolved_persona(),
2769            ));
2770            let mut obs_vec = Vec::with_capacity(3 + self.extra_observers.len());
2771            obs_vec.push(stealth_obs);
2772            obs_vec.push(registrar.clone() as Arc<dyn TargetObserver>);
2773            obs_vec.extend(self.extra_observers.iter().cloned());
2774            obs_vec
2775        } else {
2776            let mut obs_vec = Vec::with_capacity(2 + self.extra_observers.len());
2777            obs_vec.push(registrar.clone() as Arc<dyn TargetObserver>);
2778            obs_vec.extend(self.extra_observers.iter().cloned());
2779            obs_vec
2780        };
2781        // Same opt-in force-open-shadow-roots observer as the launch path.
2782        if self.force_open_shadow_roots {
2783            observers.push(Arc::new(crate::expert::ShadowRootObserver) as Arc<dyn TargetObserver>);
2784        }
2785
2786        debug!(ws_url = %ws_url, "attaching to running chrome");
2787        let conn = zendriver_transport::connect_with_observers(&ws_url, observers).await?;
2788
2789        // Same post-connect handshake as `launch`, but with no owned process:
2790        // no `Child`, no `TempDir`, no job, `owns_process = false`.
2791        let inner = finish_connect(FinishConnect {
2792            conn,
2793            registrar,
2794            input_profile,
2795            child: ChildSlot::default(),
2796            // We attached to a Chrome we did not spawn. Confining someone
2797            // else's browser to our job would kill it — and every tab they had
2798            // open — when this process exits.
2799            job: ProcessJob::none(),
2800            owned_tmp: None,
2801            extension_dirs: Vec::new(),
2802            debug_host_port: debug_host_port_from_ws(&ws_url),
2803            ws_url: Some(ws_url),
2804            owns_process: false,
2805            #[cfg(feature = "tracker-blocking")]
2806            tracker_matcher: None,
2807        })
2808        .await?;
2809
2810        Ok(Browser { inner })
2811    }

3404    /// Re-establish a dropped connection to the **same** still-running Chrome
3405    /// process (scoped reconnect — see the invalidation caveat below).
3406    ///
3407    /// When a CDP call returns [`ZendriverError::Disconnected`] the WebSocket
3408    /// died but the Chrome process is typically still alive — its browser-level
3409    /// `/devtools/browser/<id>` endpoint survives. `reconnect` re-dials that
3410    /// endpoint on the existing [`Connection`] (so raw event subscribers stay
3411    /// attached to the same broadcast bus), re-applies the WebSocket size
3412    /// config, re-arms `Target.setAutoAttach { flatten: true }` — which
3413    /// re-fires the observer chain so the stealth patches re-inject on every
3414    /// target — and refreshes the tab registry.
3415    ///
3416    /// # Handle invalidation — IMPORTANT
3417    ///
3418    /// Re-attaching to Chrome's targets yields **new `sessionId`s**. Every
3419    /// [`Tab`], `Frame`, and `Element` handle obtained before the reconnect
3420    /// caches a now-stale session id and **must not be reused** — calls on
3421    /// them will fail. After `reconnect` returns, **re-acquire** handles:
3422    ///
3423    /// - live page handles via [`Browser::tabs`] (the registry is rebuilt from
3424    ///   the re-attached targets);
3425    /// - **not** via [`Browser::main_tab`], which still returns the
3426    ///   pre-reconnect handle (its cached session id is stale). Swapping the
3427    ///   cached main-tab handle in place is part of the deferred
3428    ///   transparent-reconnect work; for now treat `main_tab()` as invalid
3429    ///   after a reconnect and use `tabs()`.
3430    ///
3431    /// # What is NOT restored (deferred)
3432    ///
3433    /// This is the scoped v1. It does **not** replay per-feature domain arming
3434    /// (`Network.enable`, active `Fetch` interception rules, `DOMStorage.enable`,
3435    /// download behavior, …) that individual features turned on before the
3436    /// drop — re-arm those yourself after reconnecting. Transparent
3437    /// handle-preserving reconnect (session-id remap so existing `Tab`s keep
3438    /// working) is tracked as follow-up work.
3439    ///
3440    /// # Errors
3441    ///
3442    /// Returns [`ZendriverError::Disconnected`] if this browser was constructed
3443    /// without a known ws endpoint (e.g. a test harness) so there is nothing to
3444    /// re-dial, and [`ZendriverError::Transport`] / [`ZendriverError::Cdp`] if
3445    /// the re-dial or the post-reconnect handshake fails. On a failed re-dial
3446    /// the existing (dead) connection is left untouched.
3447    ///
3448    /// # Examples
3449    ///
3450    /// ```no_run
3451    /// # async fn ex() -> zendriver::Result<()> {
3452    /// # let browser = zendriver::Browser::builder().launch().await?;
3453    /// # async fn do_work(_t: &zendriver::Tab) -> zendriver::Result<()> { Ok(()) }
3454    /// // A long-running scraper that survives a dropped socket:
3455    /// let tab = browser.main_tab();
3456    /// if let Err(zendriver::ZendriverError::Disconnected) = do_work(&tab).await {
3457    ///     browser.reconnect().await?; // re-dial the same Chrome process
3458    ///     // Old handles are stale now — re-acquire from the rebuilt registry:
3459    ///     let tab = browser.tabs().await.into_iter().next().unwrap();
3460    ///     do_work(&tab).await?;
3461    /// }
3462    /// # Ok(()) }
3463    /// ```
3464    pub async fn reconnect(&self) -> Result<(), ZendriverError> {
3465        let ws_url = self
3466            .inner
3467            .ws_url
3468            .clone()
3469            .ok_or(ZendriverError::Disconnected)?;
3470
3471        // Re-dial onto the SAME Connection (same event bus, same observer
3472        // chain). Dials first; only swaps the live socket on success, so a
3473        // failed reconnect doesn't tear down a connection that might recover.
3474        self.inner.conn.redial(&ws_url).await?;
3475
3476        // Drop the stale tab registry — every entry's session id is invalid
3477        // now. The re-armed auto-attach below re-populates it from the
3478        // re-attached targets via the `TabRegistrar` observer.
3479        self.inner.tabs.write().await.clear();
3480
3481        // Re-arm browser-scoped auto-attach. Chrome re-fires
3482        // `Target.attachedToTarget` for every existing target, which runs the
3483        // observer chain again — stealth re-injects on each target, and the
3484        // registrar re-inserts a fresh `Tab` (with a fresh session id) for each
3485        // page. `flatten: true` re-applies the single-socket flat session model.
3486        self.inner
3487            .conn
3488            .call_raw(
3489                "Target.setAutoAttach",
3490                json!({
3491                    "autoAttach": true,
3492                    "waitForDebuggerOnStart": true,
3493                    "flatten": true,
3494                }),
3495                None,
3496            )
3497            .await?;
3498
3499        // Wake any `new_tab_at` waiters now that the registry has been reset
3500        // and is being repopulated by the observer chain.
3501        self.inner.tabs_changed.notify_waiters();
3502        Ok(())
3503    }

3505    /// Graceful shutdown of the Chrome subprocess.
3506    ///
3507    /// Asks Chrome to quit over CDP (`Browser.close`) first, then falls back to
3508    /// SIGTERM / wait 5s / SIGKILL if it refuses, never answers, or answers but
3509    /// does not exit. Cleans up the `user_data_dir` tempdir if one was
3510    /// allocated at launch time.
3511    ///
3512    /// The CDP quit matters because the signal path only ever targets the
3513    /// single `chrome.exe` PID we tracked at spawn. Any *other* window Chrome
3514    /// opened (a first-run window, a `chrome://newtab/`) is not that PID's
3515    /// business on Windows, where there is no SIGTERM to cascade — so it
3516    /// survives `close()` and orphans forever. `Browser.close` is Chrome's
3517    /// protocol-native quit: it closes every window and exits the whole tree.
3518    ///
3519    /// For a browser produced by [`BrowserBuilder::connect`] (we did not spawn
3520    /// Chrome) this shuts down only the transport and leaves the attached
3521    /// process running — `Browser.close` is never sent, since quitting a
3522    /// browser we merely attached to would take the user's windows with it.
3523    ///
3524    /// # Examples
3525    ///
3526    /// ```no_run
3527    /// # async fn ex() -> zendriver::Result<()> {
3528    /// let browser = zendriver::Browser::builder().launch().await?;
3529    /// // ... drive the browser ...
3530    /// browser.close().await?;
3531    /// # Ok(()) }
3532    /// ```
3533    pub async fn close(self) -> Result<(), ZendriverError> {
3534        self.close_within(BROWSER_CLOSE_TIMEOUT).await
3535    }

3537    /// [`Browser::close`] with an injectable budget for the `Browser.close`
3538    /// round-trip, so tests can exercise the timeout→hard-kill fallback
3539    /// without waiting out the real [`BROWSER_CLOSE_TIMEOUT`].
3540    pub(crate) async fn close_within(
3541        self,
3542        browser_close_budget: Duration,
3543    ) -> Result<(), ZendriverError> {
3544        // Attached (non-owning) sessions must never terminate the process we
3545        // connected to — only the transport is torn down. Spawn-only below.
3546        if !self.inner.owns_process {
3547            self.inner.conn.shutdown();
3548            return Ok(());
3549        }
3550
3551        // Ask Chrome to quit over the protocol first. This must precede
3552        // `conn.shutdown()` — that tears down the very transport the command
3553        // travels over.
3554        let quit = timeout(
3555            browser_close_budget,
3556            self.inner.conn.call_raw("Browser.close", json!({}), None),
3557        )
3558        .await;
3559        let acked = match quit {
3560            // Chrome acknowledged the quit.
3561            Ok(Ok(_)) => true,
3562            // Chrome commonly drops the socket on quit instead of replying;
3563            // the transport dying here means the quit landed.
3564            Ok(Err(zendriver_transport::CallError::Transport(_))) => true,
3565            // Chrome never answered. Unreachable while `browser_close_budget`
3566            // (3s) stays far tighter than the transport's per-call default
3567            // (180s) — the outer timeout below wins — but matched explicitly
3568            // so a retuned budget can never make this report "refused", which
3569            // it categorically is not: nothing was heard, let alone declined.
3570            Ok(Err(e @ zendriver_transport::CallError::Timeout { .. })) => {
3571                warn!(error = %e, "Browser.close went unanswered; falling back to signal shutdown");
3572                false
3573            }
3574            // An RPC refusal means Chrome heard us and declined — nothing to
3575            // wait for, go straight to signals.
3576            Ok(Err(e)) => {
3577                warn!(error = %e, "Browser.close was refused; falling back to signal shutdown");
3578                false
3579            }
3580            Err(_elapsed) => {
3581                warn!(
3582                    budget = ?browser_close_budget,
3583                    "Browser.close went unanswered; falling back to signal shutdown",
3584                );
3585                false
3586            }
3587        };
3588
3589        self.inner.conn.shutdown();
3590
3591        let mut child_guard = self.inner.child.lock().await;
3592        if let Some(mut child) = child_guard.take() {
3593            if acked {
3594                // Chrome took the quit — give it the same grace to actually
3595                // exit. If it does, no signal is ever sent and the whole
3596                // process tree went down with it.
3597                if let Ok(Ok(_status)) = timeout(SHUTDOWN_GRACE, child.wait()).await {
3598                    return Ok(());
3599                }
3600                warn!("chrome accepted Browser.close but did not exit; hard-killing");
3601            }
3602            // Try graceful exit first. On Unix, tokio's `start_kill` is
3603            // `kill(pid, SIGKILL)` — too aggressive for graceful shutdown.
3604            // We send SIGTERM ourselves and fall back to SIGKILL on timeout.
3605            #[cfg(unix)]
3606            {
3607                if let Some(pid) = child.id() {
3608                    // SIGTERM gives Chrome a chance to flush + clean up; SIGKILL fallback below.
3609                    // Safety: just sending a signal to a known pid; no shared state.
3610                    #[allow(unsafe_code)]
3611                    unsafe {
3612                        libc::kill(pid as i32, libc::SIGTERM);
3613                    }
3614                }
3615            }
3616            #[cfg(not(unix))]
3617            {
3618                // Close the job first: on Windows this is the only step that
3619                // reaches Chrome's renderer/GPU/utility/crashpad children.
3620                // `start_kill` is a `TerminateProcess` against the one tracked
3621                // `chrome.exe`, and nothing on Windows cascades that to the
3622                // tree, so on its own it orphans every helper.
3623                if self.inner.job.kill_tree() {
3624                    debug!("closed chrome's job object; its whole process tree is terminating");
3625                }
3626                // Still sent unconditionally. When a job exists this is
3627                // redundant (the OS has already terminated this pid); when
3628                // confinement was refused it is the entire kill, exactly as
3629                // before this fix. The `wait`/`kill` fallback below covers both.
3630                let _ = child.start_kill();
3631            }
3632            match timeout(SHUTDOWN_GRACE, child.wait()).await {
3633                Ok(Ok(_status)) => {}
3634                _ => {
3635                    let _ = child.kill().await;
3636                }
3637            }
3638        }
3639        Ok(())
3640    }

3754/// Hard-shutdown fallback. `Drop` cannot be async, so it cannot perform the
3755/// SIGTERM-then-wait-then-SIGKILL dance [`Browser::close`] runs. Instead:
3756///
3757/// 1. [`Connection::shutdown`] signals the transport actor to stop reading;
3758///    pending CDP calls fail with a transport error.
3759/// 2. The child [`std::process::Child`] is dropped via tokio's
3760///    `kill_on_drop(true)` (set at spawn time), which sends `SIGKILL`
3761///    immediately — Chrome gets no grace period to flush state.
3762/// 3. The optional `user_data_dir` [`TempDir`] is dropped, deleting the
3763///    profile.
3764///
3765/// In short: dropping the [`Browser`] is the panic-safety / scope-exit path.
3766/// For a graceful shutdown that flushes Chrome state cleanly, call
3767/// [`Browser::close`] explicitly before the [`Browser`] goes out of scope.
3768///
3769/// For a browser produced by [`BrowserBuilder::connect`] (`owns_process ==
3770/// false`) there is no owned `Child`, so step 2 is a no-op: dropping detaches
3771/// the transport but never signals the attached Chrome process.
3772impl Drop for BrowserInner {
3773    fn drop(&mut self) {
3774        self.conn.shutdown();
3775        // We can't `.await` in Drop. If `close()` was not called explicitly,
3776        // we rely on `kill_on_drop(true)` set on the spawned Command, which
3777        // causes tokio to SIGKILL the child when the Child is dropped.
3778        // The TempDir for user_data_dir is dropped here too.
3779        //
3780        // On Windows `kill_on_drop` has the same blind spot as every other
3781        // single-pid kill — it does not reach Chrome's helper processes. The
3782        // `job` field covers them without needing a line of code here: dropping
3783        // it closes the last handle to a `KILL_ON_JOB_CLOSE` job, which is
3784        // itself the syscall that terminates the tree. That is also why it
3785        // survives paths that never run Drop at all (a `SIGKILL`, an end-task
3786        // from Task Manager): Windows closes our handles as it tears us down,
3787        // and the tree dies with them.
3788        //
3789        // Attached (`connect`) browsers hold no `Child` — `owns_process` is
3790        // false and the field is `None` — so nothing is killed here.
3791        debug_assert!(
3792            self.owns_process || self.child.try_lock().map(|g| g.is_none()).unwrap_or(true),
3793            "a non-owning Browser must not hold a Child handle",
3794        );
3795    }
3796}
