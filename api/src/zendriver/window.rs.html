1//! OS-window bounds + state control for a [`Tab`].
2//!
3//! Chrome models each top-level page inside an OS window with a position,
4//! size, and a window state (normal / minimized / maximized / fullscreen).
5//! These live in the `Browser` CDP domain, addressed by `windowId` — which is
6//! itself resolved from the tab's `targetId` via
7//! `Browser.getWindowForTarget`. Because they are `Browser`-domain commands,
8//! they dispatch at **browser scope** (no `sessionId`), exactly like
9//! [`Tab::activate`].
10//!
11//! ```no_run
12//! # async fn ex() -> zendriver::Result<()> {
13//! # let browser = zendriver::Browser::builder().launch().await?;
14//! # let tab = browser.main_tab();
15//! tab.set_window_size(1280, 800).await?;
16//! tab.maximize().await?;
17//! # Ok(()) }
18//! ```
19//!
20//! ## CDP state-change rule
21//!
22//! `Browser.setWindowBounds` rejects a payload that mixes a non-`normal`
23//! `windowState` with explicit geometry. So [`Tab::maximize`] /
24//! [`Tab::minimize`] / [`Tab::fullscreen`] send `bounds: { windowState }`
25//! **alone**, while [`Tab::set_window_size`] sends `bounds: { width, height }`
26//! with the state left implicit (normal). [`Tab::set_window_bounds`] honors
27//! whatever [`WindowBounds`] you hand it, so it is your responsibility there
28//! not to combine a state change with geometry.
29

36/// State of the OS window hosting a tab.
37///
38/// Serializes to the CDP `Browser.WindowState` wire strings
39/// (`"normal"` / `"minimized"` / `"maximized"` / `"fullscreen"`).
40#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
41#[serde(rename_all = "lowercase")]
42pub enum WindowState {
43    /// Normal (windowed) — the only state that may carry explicit geometry.
44    Normal,
45    /// Minimized to the OS taskbar / dock.
46    Minimized,
47    /// Maximized to fill the screen work area.
48    Maximized,
49    /// Fullscreen (chrome / OS decorations hidden).
50    Fullscreen,
51}
52
53impl WindowState {
54    /// CDP wire string for this state.
55    ///
56    /// Cheap (`&'static str`); mirrors [`crate::Format::as_cdp`].
57    ///
58    /// # Examples
59    ///
60    /// ```
61    /// use zendriver::WindowState;
62    /// assert_eq!(WindowState::Maximized.as_cdp(), "maximized");
63    /// assert_eq!(WindowState::Normal.as_cdp(), "normal");
64    /// ```
65    #[must_use]
66    pub fn as_cdp(&self) -> &'static str {
67        match self {
68            Self::Normal => "normal",
69            Self::Minimized => "minimized",
70            Self::Maximized => "maximized",
71            Self::Fullscreen => "fullscreen",
72        }
73    }
74}

76/// Position, size, and state of the OS window hosting a tab.
77///
78/// All geometry fields are optional: `None` means "leave unchanged" on a
79/// [`Tab::set_window_bounds`] write, and "not reported" on a
80/// [`Tab::window_bounds`] read (Chrome omits geometry for minimized windows).
81/// Coordinates and dimensions are device-independent pixels.
82#[derive(Debug, Clone, Default, PartialEq, Eq)]
83pub struct WindowBounds {
84    /// Window left edge (screen X), in DIP.
85    pub left: Option<i64>,
86    /// Window top edge (screen Y), in DIP.
87    pub top: Option<i64>,
88    /// Window width, in DIP.
89    pub width: Option<i64>,
90    /// Window height, in DIP.
91    pub height: Option<i64>,
92    /// Window state. `None` leaves it unchanged on a write.
93    pub state: Option<WindowState>,
94}
95
96impl WindowBounds {
97    /// Serialize to a CDP `Browser.Bounds` JSON object.
98    ///
99    /// Only the set fields are emitted; `state` becomes `windowState`.
100    fn to_cdp(&self) -> Value {
101        let mut bounds = Map::new();
102        if let Some(v) = self.left {
103            bounds.insert("left".to_string(), json!(v));
104        }
105        if let Some(v) = self.top {
106            bounds.insert("top".to_string(), json!(v));
107        }
108        if let Some(v) = self.width {
109            bounds.insert("width".to_string(), json!(v));
110        }
111        if let Some(v) = self.height {
112            bounds.insert("height".to_string(), json!(v));
113        }
114        if let Some(state) = self.state {
115            bounds.insert(
116                "windowState".to_string(),
117                Value::String(state.as_cdp().into()),
118            );
119        }
120        Value::Object(bounds)
121    }
122
123    /// Parse a CDP `Browser.Bounds` JSON object into a [`WindowBounds`].
124    fn from_cdp(v: &Value) -> Self {
125        Self {
126            left: v.get("left").and_then(Value::as_i64),
127            top: v.get("top").and_then(Value::as_i64),
128            width: v.get("width").and_then(Value::as_i64),
129            height: v.get("height").and_then(Value::as_i64),
130            state: v
131                .get("windowState")
132                .and_then(Value::as_str)
133                .and_then(|s| match s {
134                    "normal" => Some(WindowState::Normal),
135                    "minimized" => Some(WindowState::Minimized),
136                    "maximized" => Some(WindowState::Maximized),
137                    "fullscreen" => Some(WindowState::Fullscreen),
138                    _ => None,
139                }),
140        }
141    }
142}

144impl Tab {
145    /// Resolve the CDP `windowId` for this tab's target.
146    ///
147    /// Browser-scope `Browser.getWindowForTarget { targetId }` (no
148    /// `sessionId`), mirroring [`Tab::activate`]'s dispatch.
149    async fn window_id(&self) -> Result<i64> {
150        let target_id = self.target_id().to_string();
151        let res = self
152            .inner
153            .session
154            .connection()
155            .call_raw(
156                "Browser.getWindowForTarget",
157                json!({ "targetId": target_id }),
158                None,
159            )
160            .await?;
161        res.get("windowId").and_then(Value::as_i64).ok_or_else(|| {
162            ZendriverError::Navigation("Browser.getWindowForTarget returned no windowId".into())
163        })
164    }
165
166    /// Read this tab's current OS-window bounds + state.
167    ///
168    /// Sends `Browser.getWindowForTarget { targetId }` at browser scope and
169    /// parses the response's `bounds`. Minimized windows may omit geometry
170    /// (those fields come back as `None`).
171    ///
172    /// # Errors
173    ///
174    /// Returns [`ZendriverError::Navigation`] when Chrome's response is
175    /// missing `windowId` or `bounds`.
176    ///
177    /// # Examples
178    ///
179    /// ```no_run
180    /// # async fn ex() -> zendriver::Result<()> {
181    /// # let browser = zendriver::Browser::builder().launch().await?;
182    /// # let tab = browser.main_tab();
183    /// let bounds = tab.window_bounds().await?;
184    /// println!("{:?}", bounds.state);
185    /// # Ok(()) }
186    /// ```
187    pub async fn window_bounds(&self) -> Result<WindowBounds> {
188        let target_id = self.target_id().to_string();
189        let res = self
190            .inner
191            .session
192            .connection()
193            .call_raw(
194                "Browser.getWindowForTarget",
195                json!({ "targetId": target_id }),
196                None,
197            )
198            .await?;
199        let bounds = res.get("bounds").ok_or_else(|| {
200            ZendriverError::Navigation("Browser.getWindowForTarget returned no bounds".into())
201        })?;
202        Ok(WindowBounds::from_cdp(bounds))
203    }
204
205    /// Set this tab's OS-window bounds + state.
206    ///
207    /// Resolves the `windowId` (via `Browser.getWindowForTarget`), then sends
208    /// `Browser.setWindowBounds { windowId, bounds }` at browser scope. The
209    /// `bounds` object carries only the [`WindowBounds`] fields you set.
210    ///
211    /// Per the CDP contract, do **not** combine a non-`normal`
212    /// [`WindowState`] with explicit geometry in a single call — use the
213    /// dedicated [`Tab::maximize`] / [`Tab::minimize`] / [`Tab::fullscreen`]
214    /// helpers for state changes and [`Tab::set_window_size`] for sizing.
215    ///
216    /// # Errors
217    ///
218    /// Returns [`ZendriverError::Navigation`] when the `windowId` lookup
219    /// fails.
220    ///
221    /// # Examples
222    ///
223    /// ```no_run
224    /// # use zendriver::WindowBounds;
225    /// # async fn ex() -> zendriver::Result<()> {
226    /// # let browser = zendriver::Browser::builder().launch().await?;
227    /// # let tab = browser.main_tab();
228    /// tab.set_window_bounds(WindowBounds {
229    ///     left: Some(0),
230    ///     top: Some(0),
231    ///     width: Some(1024),
232    ///     height: Some(768),
233    ///     state: None,
234    /// }).await?;
235    /// # Ok(()) }
236    /// ```
237    pub async fn set_window_bounds(&self, bounds: WindowBounds) -> Result<()> {
238        let window_id = self.window_id().await?;
239        self.inner
240            .session
241            .connection()
242            .call_raw(
243                "Browser.setWindowBounds",
244                json!({ "windowId": window_id, "bounds": bounds.to_cdp() }),
245                None,
246            )
247            .await?;
248        Ok(())
249    }
250
251    /// Resize this tab's OS window to `width` × `height` DIP.
252    ///
253    /// Sends `bounds: { width, height }` with the window state implicitly
254    /// `normal` — no `windowState` key — so it never trips the CDP
255    /// state-vs-geometry rule.
256    ///
257    /// # Examples
258    ///
259    /// ```no_run
260    /// # async fn ex() -> zendriver::Result<()> {
261    /// # let browser = zendriver::Browser::builder().launch().await?;
262    /// # let tab = browser.main_tab();
263    /// tab.set_window_size(1280, 800).await?;
264    /// # Ok(()) }
265    /// ```
266    pub async fn set_window_size(&self, width: i64, height: i64) -> Result<()> {
267        self.set_window_bounds(WindowBounds {
268            width: Some(width),
269            height: Some(height),
270            ..Default::default()
271        })
272        .await
273    }
274
275    /// Maximize this tab's OS window.
276    ///
277    /// Sends `bounds: { windowState: "maximized" }` alone (no geometry), per
278    /// the CDP state-change rule.
279    ///
280    /// # Examples
281    ///
282    /// ```no_run
283    /// # async fn ex() -> zendriver::Result<()> {
284    /// # let browser = zendriver::Browser::builder().launch().await?;
285    /// # let tab = browser.main_tab();
286    /// tab.maximize().await?;
287    /// # Ok(()) }
288    /// ```
289    pub async fn maximize(&self) -> Result<()> {
290        self.set_window_state(WindowState::Maximized).await
291    }
292
293    /// Minimize this tab's OS window.
294    ///
295    /// Sends `bounds: { windowState: "minimized" }` alone (no geometry), per
296    /// the CDP state-change rule.
297    ///
298    /// # Examples
299    ///
300    /// ```no_run
301    /// # async fn ex() -> zendriver::Result<()> {
302    /// # let browser = zendriver::Browser::builder().launch().await?;
303    /// # let tab = browser.main_tab();
304    /// tab.minimize().await?;
305    /// # Ok(()) }
306    /// ```
307    pub async fn minimize(&self) -> Result<()> {
308        self.set_window_state(WindowState::Minimized).await
309    }
310
311    /// Put this tab's OS window into fullscreen.
312    ///
313    /// Sends `bounds: { windowState: "fullscreen" }` alone (no geometry), per
314    /// the CDP state-change rule.
315    ///
316    /// # Examples
317    ///
318    /// ```no_run
319    /// # async fn ex() -> zendriver::Result<()> {
320    /// # let browser = zendriver::Browser::builder().launch().await?;
321    /// # let tab = browser.main_tab();
322    /// tab.fullscreen().await?;
323    /// # Ok(()) }
324    /// ```
325    pub async fn fullscreen(&self) -> Result<()> {
326        self.set_window_state(WindowState::Fullscreen).await
327    }
328
329    /// Send a bare `windowState`-only bounds change.
330    ///
331    /// Shared by [`Tab::maximize`] / [`Tab::minimize`] / [`Tab::fullscreen`]:
332    /// the `bounds` object carries exactly one key, `windowState`, satisfying
333    /// the CDP rule that a state transition must not be mixed with geometry.
334    async fn set_window_state(&self, state: WindowState) -> Result<()> {
335        self.set_window_bounds(WindowBounds {
336            state: Some(state),
337            ..Default::default()
338        })
339        .await
340    }
341}

351    /// `window_bounds()` dispatches `Browser.getWindowForTarget { targetId }`
352    /// at browser scope (no sessionId) and parses the mocked windowId +
353    /// bounds into a [`WindowBounds`].
354    #[tokio::test]
355    async fn window_bounds_dispatches_get_window_for_target() {
356        let (mut mock, conn) = MockConnection::pair();
357        let sess = SessionHandle::new(conn.clone(), "S42");
358        let tab = Tab::new_for_test(sess);
359
360        let fut = tokio::spawn({
361            let t = tab.clone();
362            async move { t.window_bounds().await }
363        });
364
365        let id = mock.expect_cmd("Browser.getWindowForTarget").await;
366        let sent = mock.last_sent();
367        assert_eq!(sent["params"]["targetId"], "test-target-S42");
368        // Browser-scope command — no session_id.
369        assert!(sent.get("sessionId").is_none());
370        mock.reply(
371            id,
372            json!({
373                "windowId": 7,
374                "bounds": { "left": 0, "top": 0, "width": 1280, "height": 800, "windowState": "normal" },
375            }),
376        )
377        .await;
378
379        let bounds = fut.await.unwrap().unwrap();
380        assert_eq!(bounds.width, Some(1280));
381        assert_eq!(bounds.height, Some(800));
382        assert_eq!(bounds.state, Some(WindowState::Normal));
383        conn.shutdown();
384    }

386    /// `maximize()` first resolves the windowId, then sends
387    /// `Browser.setWindowBounds` whose `bounds` has `windowState: "maximized"`
388    /// and NO geometry keys.
389    #[tokio::test]
390    async fn maximize_sends_only_window_state() {
391        let (mut mock, conn) = MockConnection::pair();
392        let sess = SessionHandle::new(conn.clone(), "S1");
393        let tab = Tab::new_for_test(sess);
394
395        let fut = tokio::spawn({
396            let t = tab.clone();
397            async move { t.maximize().await }
398        });
399
400        // Step 1: windowId lookup.
401        let lookup_id = mock.expect_cmd("Browser.getWindowForTarget").await;
402        mock.reply(lookup_id, json!({ "windowId": 3, "bounds": {} }))
403            .await;
404
405        // Step 2: setWindowBounds with windowState-only bounds.
406        let set_id = mock.expect_cmd("Browser.setWindowBounds").await;
407        let sent = mock.last_sent();
408        assert_eq!(sent["params"]["windowId"], 3);
409        let bounds = &sent["params"]["bounds"];
410        assert_eq!(bounds["windowState"], "maximized");
411        // CDP rule: a state change must carry no geometry.
412        assert!(bounds.get("width").is_none());
413        assert!(bounds.get("height").is_none());
414        assert!(bounds.get("left").is_none());
415        assert!(bounds.get("top").is_none());
416        // Browser-scope command — no session_id.
417        assert!(sent.get("sessionId").is_none());
418        mock.reply(set_id, json!({})).await;
419
420        fut.await.unwrap().unwrap();
421        conn.shutdown();
422    }

424    /// `set_window_size(w, h)` sends `bounds: { width, height }` with NO

428        let (mut mock, conn) = MockConnection::pair();

430        let tab = Tab::new_for_test(sess);

432        let fut = tokio::spawn({

437        let lookup_id = mock.expect_cmd("Browser.getWindowForTarget").await;

446        assert_eq!(bounds["height"], 768);
447        // Sizing must not carry a windowState key.
448        assert!(bounds.get("windowState").is_none());
449        mock.reply(set_id, json!({})).await;
450
451        fut.await.unwrap().unwrap();
