1//! Per-Tab DOM storage handle (localStorage / sessionStorage) backed by CDP
2//! `DOMStorage.*` methods.
3//!
4//! Each [`Storage`] handle is configured at construction with an `is_local`
5//! flag selecting either localStorage (true) or sessionStorage (false) for
6//! the owning [`crate::Tab`]'s current origin. Obtain handles via
7//! [`crate::Tab::local_storage`] / [`crate::Tab::session_storage`].
8//!
9//! ```no_run
10//! # async fn ex() -> zendriver::Result<()> {
11//! # let browser = zendriver::Browser::builder().launch().await?;
12//! # let tab = browser.main_tab();
13//! tab.goto("https://example.com").await?;
14//! let ls = tab.local_storage();
15//! ls.set("theme", "dark").await?;
16//! assert_eq!(ls.get("theme").await?.as_deref(), Some("dark"));
17//! # Ok(()) }
18//! ```
19//!
20//! ## Why fetch the URL on every call
21//!
22//! CDP's `DOMStorage` domain identifies a storage area by `StorageId {
23//! securityOrigin, isLocalStorage }`. The origin must reflect the tab's
24//! *current* document — a navigation between calls would otherwise read
25//! storage from the wrong origin. Rather than cache + invalidate on
26//! `Page.frameNavigated`, the handle re-fetches the URL via
27//! [`crate::tab::Tab::url`] (one cheap `Target.getTargetInfo` round-trip)
28//! on each operation. Storage ops are infrequent enough that the extra
29//! round-trip is preferable to the cache-coherency footgun.
30//!
31//! ## `DOMStorage.enable` once-per-handle
32//!
33//! Chrome rejects `DOMStorage` reads/writes until the domain is enabled on
34//! the session. Each [`Storage`] handle gates the first call through a
35//! [`tokio::sync::OnceCell`] so the enable round-trip fires exactly once
36//! per handle — subsequent calls skip straight to the data op. The same
37//! storage handle re-used across many calls thus pays the enable cost
38//! exactly once.
39

50/// Cheap-to-clone handle to a single DOM storage area.
51///
52/// localStorage or sessionStorage for the owning tab's current origin.
53/// Construct via [`crate::Tab::local_storage`] /
54/// [`crate::Tab::session_storage`]. Wrapping is `Arc`-cheap; pass the
55/// handle around freely. All methods are async — each call performs at
56/// least one CDP round-trip (origin discovery), plus the data op.
57#[derive(Clone, Debug)]
58pub struct Storage {
59    inner: Arc<StorageInner>,
60}
61
62#[derive(Debug)]
63struct StorageInner {
64    /// Tab session used for `DOMStorage.*` calls. For both local and session
65    /// storage this is the tab's primary session — DOMStorage is page-scoped,
66    /// not per-frame.
67    session: SessionHandle,
68    /// `true` selects localStorage; `false` selects sessionStorage. Encoded
69    /// into every CDP `storageId` payload.
70    is_local: bool,
71    /// Weak ref to the owning tab so we can call [`crate::tab::Tab::url`]
72    /// on each op without forcing the tab to outlive the handle. If the
73    /// tab has been dropped, storage ops error with
74    /// [`ZendriverError::Storage`] rather than panic.
75    tab: Weak<TabInner>,
76    /// Gates the once-per-handle `DOMStorage.enable` call. Chrome rejects
77    /// `getDOMStorageItems` / `setDOMStorageItem` / etc. until the domain
78    /// is enabled on the session; we lazily enable on first use and cache
79    /// the success so subsequent calls skip the round-trip.
80    enable_once: OnceCell<()>,
81}
82
83impl Storage {
84    /// Construct a storage handle. Typically called by
85    /// [`crate::tab::Tab::local_storage`] / `session_storage` rather than
86    /// user code — those carry the right [`SessionHandle`] +
87    /// [`Weak<TabInner>`] for the owning tab.
88    #[must_use]
89    pub(crate) fn new(session: SessionHandle, is_local: bool, tab: Weak<TabInner>) -> Self {
90        Self {
91            inner: Arc::new(StorageInner {
92                session,
93                is_local,
94                tab,
95                enable_once: OnceCell::new(),
96            }),
97        }
98    }

100    /// Whether this handle targets `localStorage` (true) or
101    /// `sessionStorage` (false).
102    ///
103    /// # Examples
104    ///
105    /// ```no_run
106    /// # async fn ex() -> zendriver::Result<()> {
107    /// # let browser = zendriver::Browser::builder().launch().await?;
108    /// # let tab = browser.main_tab();
109    /// assert!(tab.local_storage().is_local());
110    /// assert!(!tab.session_storage().is_local());
111    /// # Ok(()) }
112    /// ```
113    #[must_use]
114    pub fn is_local(&self) -> bool {
115        self.inner.is_local
116    }

118    /// Look up a single value by key. Returns `None` if the key is absent.
119    ///
120    /// Fetches the full storage area (CDP has no single-key getter) and
121    /// scans for the match.
122    ///
123    /// # Examples
124    ///
125    /// ```no_run
126    /// # async fn ex() -> zendriver::Result<()> {
127    /// # let browser = zendriver::Browser::builder().launch().await?;
128    /// # let tab = browser.main_tab();
129    /// if let Some(v) = tab.local_storage().get("theme").await? {
130    ///     println!("theme: {v}");
131    /// }
132    /// # Ok(()) }
133    /// ```
134    pub async fn get(&self, key: &str) -> Result<Option<String>> {
135        let all = self.get_all().await?;
136        Ok(all.get(key).cloned())
137    }

139    /// Snapshot the entire storage area as a `HashMap`.
140    ///
141    /// Maps to `DOMStorage.getDOMStorageItems`. Order is unspecified.
142    ///
143    /// # Examples
144    ///
145    /// ```no_run
146    /// # async fn ex() -> zendriver::Result<()> {
147    /// # let browser = zendriver::Browser::builder().launch().await?;
148    /// # let tab = browser.main_tab();
149    /// for (k, v) in tab.local_storage().get_all().await? {
150    ///     println!("{k}={v}");
151    /// }
152    /// # Ok(()) }
153    /// ```
154    pub async fn get_all(&self) -> Result<HashMap<String, String>> {
155        self.ensure_enabled().await?;
156        // Hold a strong ref to the Tab across the full op so the session
157        // can't be torn down between the `storage_id` URL fetch and the
158        // `DOMStorage.*` call.
159        let tab = self.upgraded_tab()?;
160        let storage_id = self.build_storage_id(&tab).await?;
161        let resp = self
162            .inner
163            .session
164            .call(
165                "DOMStorage.getDOMStorageItems",
166                json!({ "storageId": storage_id }),
167            )
168            .await?;
169        drop(tab);
170        parse_entries(&resp)
171    }

173    /// Insert or replace a single key-value pair.
174    ///
175    /// Maps to `DOMStorage.setDOMStorageItem` — Chrome treats the value as
176    /// opaque text (per the Storage API spec; non-string values must be
177    /// stringified by the caller).
178    ///
179    /// # Examples
180    ///
181    /// ```no_run
182    /// # async fn ex() -> zendriver::Result<()> {
183    /// # let browser = zendriver::Browser::builder().launch().await?;
184    /// # let tab = browser.main_tab();
185    /// tab.local_storage().set("theme", "dark").await?;
186    /// # Ok(()) }
187    /// ```
188    pub async fn set(&self, key: &str, value: &str) -> Result<()> {
189        self.ensure_enabled().await?;
190        let tab = self.upgraded_tab()?;
191        let storage_id = self.build_storage_id(&tab).await?;
192        self.inner
193            .session
194            .call(
195                "DOMStorage.setDOMStorageItem",
196                json!({
197                    "storageId": storage_id,
198                    "key": key,
199                    "value": value,
200                }),
201            )
202            .await?;
203        drop(tab);
204        Ok(())
205    }

207    /// Set many entries.
208    ///
209    /// Convenience wrapper over [`Self::set`] — there is no CDP bulk-set for
210    /// DOMStorage, so this issues N round-trips. Order of dispatch follows
211    /// the `HashMap`'s iteration order (unspecified).
212    ///
213    /// # Examples
214    ///
215    /// ```no_run
216    /// # use std::collections::HashMap;
217    /// # async fn ex() -> zendriver::Result<()> {
218    /// # let browser = zendriver::Browser::builder().launch().await?;
219    /// # let tab = browser.main_tab();
220    /// let mut items = HashMap::new();
221    /// items.insert("theme".to_string(), "dark".to_string());
222    /// items.insert("lang".to_string(), "en".to_string());
223    /// tab.local_storage().set_many(&items).await?;
224    /// # Ok(()) }
225    /// ```
226    pub async fn set_many(&self, items: &HashMap<String, String>) -> Result<()> {
227        for (k, v) in items {
228            self.set(k, v).await?;
229        }
230        Ok(())
231    }

233    /// Remove a single key.
234    ///
235    /// Maps to `DOMStorage.removeDOMStorageItem`. Missing keys are silently
236    /// ignored (matches the Storage API `removeItem` semantics).
237    ///
238    /// # Examples
239    ///
240    /// ```no_run
241    /// # async fn ex() -> zendriver::Result<()> {
242    /// # let browser = zendriver::Browser::builder().launch().await?;
243    /// # let tab = browser.main_tab();
244    /// tab.local_storage().remove("theme").await?;
245    /// # Ok(()) }
246    /// ```
247    pub async fn remove(&self, key: &str) -> Result<()> {
248        self.ensure_enabled().await?;
249        let tab = self.upgraded_tab()?;
250        let storage_id = self.build_storage_id(&tab).await?;
251        self.inner
252            .session
253            .call(
254                "DOMStorage.removeDOMStorageItem",
255                json!({ "storageId": storage_id, "key": key }),
256            )
257            .await?;
258        drop(tab);
259        Ok(())
260    }

262    /// Empty the entire storage area for this origin.
263    ///
264    /// Maps to `DOMStorage.clear`. Equivalent to calling
265    /// `localStorage.clear()` / `sessionStorage.clear()` from page JS.
266    ///
267    /// # Examples
268    ///
269    /// ```no_run
270    /// # async fn ex() -> zendriver::Result<()> {
271    /// # let browser = zendriver::Browser::builder().launch().await?;
272    /// # let tab = browser.main_tab();
273    /// tab.local_storage().clear().await?;
274    /// # Ok(()) }
275    /// ```
276    pub async fn clear(&self) -> Result<()> {
277        self.ensure_enabled().await?;
278        let tab = self.upgraded_tab()?;
279        let storage_id = self.build_storage_id(&tab).await?;
280        self.inner
281            .session
282            .call("DOMStorage.clear", json!({ "storageId": storage_id }))
283            .await?;
284        drop(tab);
285        Ok(())
286    }

288    /// Number of entries in this storage area.
289    ///
290    /// Implemented as [`Self::get_all`] + `.len()` — CDP has no dedicated
291    /// length op.
292    ///
293    /// # Examples
294    ///
295    /// ```no_run
296    /// # async fn ex() -> zendriver::Result<()> {
297    /// # let browser = zendriver::Browser::builder().launch().await?;
298    /// # let tab = browser.main_tab();
299    /// let n = tab.local_storage().len().await?;
300    /// println!("{n} entries");
301    /// # Ok(()) }
302    /// ```
303    pub async fn len(&self) -> Result<usize> {
304        Ok(self.get_all().await?.len())
305    }

307    /// Lazily dispatch `DOMStorage.enable` exactly once per handle. The
308    /// gate is a [`tokio::sync::OnceCell`] — concurrent callers race to
309    /// the same single dispatch and all see the same result.
310    async fn ensure_enabled(&self) -> Result<()> {
311        self.inner
312            .enable_once
313            .get_or_try_init(|| async {
314                self.inner
315                    .session
316                    .call("DOMStorage.enable", json!({}))
317                    .await?;
318                Ok::<(), ZendriverError>(())
319            })
320            .await?;
321        Ok(())
322    }
323
324    /// Upgrade the cached [`Weak<TabInner>`] into a strong [`crate::Tab`]
325    /// handle. The caller is expected to hold the returned `Tab` across
326    /// the full storage op so the underlying session can't be torn down
327    /// between the origin lookup and the `DOMStorage.*` dispatch.
328    fn upgraded_tab(&self) -> Result<crate::tab::Tab> {
329        let tab_inner = self
330            .inner
331            .tab
332            .upgrade()
333            .ok_or_else(|| ZendriverError::Storage("owning tab has been dropped".into()))?;
334        Ok(crate::tab::Tab { inner: tab_inner })
335    }
336
337    /// Build the CDP `StorageId` payload — `{ securityOrigin, isLocalStorage }`
338    /// — from the supplied tab's current URL. The caller passes a live
339    /// `&Tab` (typically the one held by [`Self::upgraded_tab`]) so the
340    /// URL read and any follow-on CDP call see the same session.
341    async fn build_storage_id(&self, tab: &crate::tab::Tab) -> Result<Value> {
342        let url = tab.url().await?;
343        let origin = url.origin().ascii_serialization();
344        Ok(json!({
345            "securityOrigin": origin,
346            "isLocalStorage": self.inner.is_local,
347        }))
348    }
349}

351/// Parse `DOMStorage.getDOMStorageItems` response. The `entries` field is
352/// an array of `[key, value]` two-element string arrays.
353#[allow(clippy::result_large_err)] // ZendriverError variance is the project-wide return type
354fn parse_entries(resp: &Value) -> Result<HashMap<String, String>> {
355    let arr = resp
356        .get("entries")
357        .and_then(|v| v.as_array())
358        .ok_or_else(|| ZendriverError::Storage("response missing `entries` array".into()))?;
359    let mut out = HashMap::with_capacity(arr.len());
360    for entry in arr {
361        let pair = entry.as_array().ok_or_else(|| {
362            ZendriverError::Storage("`entries` element is not a [key, value] array".into())
363        })?;
364        if pair.len() < 2 {
365            return Err(ZendriverError::Storage(
366                "`entries` element has fewer than 2 fields".into(),
367            ));
368        }
369        let k = pair[0]
370            .as_str()
371            .ok_or_else(|| ZendriverError::Storage("entry key is not a string".into()))?
372            .to_string();
373        let v = pair[1]
374            .as_str()
375            .ok_or_else(|| ZendriverError::Storage("entry value is not a string".into()))?
376            .to_string();
377        out.insert(k, v);
378    }
379    Ok(out)
380}

391    /// [`Storage::get`] dispatches `DOMStorage.enable` exactly once (on
392    /// first use), then resolves the storage origin via
393    /// `Target.getTargetInfo` and reads via `DOMStorage.getDOMStorageItems`
394    /// with `isLocalStorage: true`.
395    #[tokio::test]
396    async fn get_enables_domstorage_then_reads_with_local_flag() {
397        let (mut mock, conn) = MockConnection::pair();
398        let sess = SessionHandle::new(conn.clone(), "S1");
399        let tab = Tab::new_for_test(sess.clone());
400        let storage = Storage::new(sess, true, Arc::downgrade(&tab.inner));
401
402        let fut = tokio::spawn({
403            let s = storage.clone();
404            async move { s.get("theme").await }
405        });
406
407        // 1. DOMStorage.enable (once-per-handle gate).
408        let id_enable = mock.expect_cmd("DOMStorage.enable").await;
409        mock.reply(id_enable, json!({})).await;
410
411        // 2. Target.getTargetInfo for origin discovery (Tab::url).
412        let id_info = mock.expect_cmd("Target.getTargetInfo").await;
413        mock.reply(
414            id_info,
415            json!({ "targetInfo": { "url": "https://x.test/page" } }),
416        )
417        .await;
418
419        // 3. DOMStorage.getDOMStorageItems with the derived storageId.
420        let id_get = mock.expect_cmd("DOMStorage.getDOMStorageItems").await;
421        let sent = mock.last_sent();
422        assert_eq!(
423            sent["params"]["storageId"]["securityOrigin"],
424            "https://x.test"
425        );
426        assert_eq!(sent["params"]["storageId"]["isLocalStorage"], true);
427        mock.reply(
428            id_get,
429            json!({ "entries": [["theme", "dark"], ["lang", "en"]] }),
430        )
431        .await;
432
433        let got = fut.await.unwrap().unwrap();
434        assert_eq!(got, Some("dark".to_string()));
435
436        conn.shutdown();
437    }

439    /// [`Storage::set`] dispatches `DOMStorage.setDOMStorageItem` with the

450        let fut = tokio::spawn({

458        let id_info = mock.expect_cmd("Target.getTargetInfo").await;
