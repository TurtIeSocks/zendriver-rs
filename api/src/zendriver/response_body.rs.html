1//! Bounded response-body capture.
2//!
3//! Unbounded body capture ([`crate::monitor::NetworkExchange::body`] and
4//! friends) risks OOM on a large response, or blowing an agent's context
5//! window when the body is forwarded verbatim. [`BoundedBody`] gives callers
6//! an explicit cap: capture at most `max_bytes` of a body and get back a
7//! `truncated` flag that says whether anything was cut, so a short body is
8//! never silently mistaken for a truncated one (and vice versa).
9//!
10//! This module is a standalone primitive — it does not itself fetch or
11//! decode bodies. A caller (the network monitor, in a later change) is
12//! expected to decode the raw body bytes (e.g. base64-decode a CDP
13//! `getResponseBody` result) and pass the decoded bytes to
14//! [`BoundedBody::capture`].
15
16/// A body captured up to a byte cap, with explicit truncation bookkeeping.
17///
18/// Construct via [`BoundedBody::capture`]. All bounding happens against the
19/// **raw decoded byte length** of the input — never a base64 or other
20/// encoded representation — so a fully-captured small body is never
21/// misreported as truncated.
22#[derive(Debug, Clone, PartialEq, Eq)]
23pub struct BoundedBody {
24    /// The captured bytes: either the full body, or the first `max_bytes` of
25    /// it if the body exceeded the cap.
26    pub bytes: Vec<u8>,
27    /// `true` if `bytes` is a prefix of a larger body (i.e. the body's raw
28    /// length exceeded `max_bytes`); `false` if `bytes` holds the entire
29    /// body.
30    pub truncated: bool,
31    /// The full (pre-truncation) raw byte length of the body, regardless of
32    /// how much of it was kept in `bytes`. Always equal to `bytes.len()` as
33    /// a `u64` when `truncated` is `false`.
34    pub full_len: u64,
35}
36
37impl BoundedBody {
38    /// Capture up to `max_bytes` of `full`, a raw (already-decoded) byte
39    /// slice.
40    ///
41    /// - If `full.len() <= max_bytes`, every byte is kept and `truncated` is
42    ///   `false`.
43    /// - Otherwise only the first `max_bytes` bytes are kept and `truncated`
44    ///   is `true`.
45    /// - `full_len` is always `full.len()` as a `u64`, i.e. the full
46    ///   pre-truncation length — it does not shrink when the body is
47    ///   truncated, so callers can report "captured N of M bytes".
48    ///
49    /// `max_bytes == 0` means **unbounded**: the entire body is kept and
50    /// `truncated` is always `false`. This lets a caller opt out of bounding
51    /// (e.g. a config default of `0`) without a separate code path — there
52    /// is no "reject" behavior to special-case.
53    ///
54    /// `full` must already be the raw decoded bytes of the body (e.g. after
55    /// base64-decoding a CDP `getResponseBody` result). Bounding is always
56    /// computed against this raw length, never against an encoded
57    /// representation's length — a body whose base64 form is longer than
58    /// `max_bytes` but whose raw form is not must never be reported as
59    /// truncated.
60    ///
61    /// ```
62    /// use zendriver::BoundedBody;
63    ///
64    /// let body = BoundedBody::capture(b"hello world", 5);
65    /// assert!(body.truncated);
66    /// assert_eq!(body.bytes, b"hello");
67    /// assert_eq!(body.full_len, 11);
68    /// ```
69    #[must_use]
70    pub fn capture(full: &[u8], max_bytes: usize) -> Self {
71        let full_len = full.len() as u64;
72
73        if max_bytes == 0 || full.len() <= max_bytes {
74            return Self {
75                bytes: full.to_vec(),
76                truncated: false,
77                full_len,
78            };
79        }
80
81        Self {
82            bytes: full[..max_bytes].to_vec(),
83            truncated: true,
84            full_len,
85        }
86    }
87}

89#[cfg(test)]
90mod tests {
91    use super::*;
92
93    #[test]
94    fn body_under_bound_is_not_truncated() {
95        let full = b"hello";
96        let result = BoundedBody::capture(full, 10);
97
98        assert!(!result.truncated);
99        assert_eq!(result.bytes, full.to_vec());
100        assert_eq!(result.full_len, full.len() as u64);
101    }
102
103    #[test]
104    fn body_over_bound_is_truncated_to_prefix() {
105        let full = b"hello world";
106        let max_bytes = 5;
107        let result = BoundedBody::capture(full, max_bytes);
108
109        assert!(result.truncated);
110        assert_eq!(result.bytes.len(), max_bytes);
111        assert_eq!(result.bytes, full[..max_bytes].to_vec());
112        assert_eq!(result.full_len, full.len() as u64);
113    }
114
115    #[test]
116    fn body_exactly_at_bound_is_not_truncated() {
117        let full = b"hello";
118        let result = BoundedBody::capture(full, full.len());
119
120        assert!(!result.truncated);
121        assert_eq!(result.bytes, full.to_vec());
122        assert_eq!(result.full_len, full.len() as u64);
123    }
124
125    #[test]
126    fn max_bytes_zero_is_unbounded() {
127        let full = vec![0xAB; 10_000];
128        let result = BoundedBody::capture(&full, 0);
129
130        assert!(!result.truncated);
131        assert_eq!(result.bytes, full);
132        assert_eq!(result.full_len, full.len() as u64);
133    }
134
135    /// Bounding must be computed against the RAW decoded byte length, never
136    /// a base64 (or other encoded) length. Construct a raw body small enough
137    /// to fit under `max_bytes`, but whose base64 encoding (~4/3 larger,
138    /// plus padding) would exceed `max_bytes` — and prove `capture` still
139    /// reports it as fully captured.
140    #[test]
141    fn bounding_uses_raw_length_not_base64_encoded_length() {
142        use base64::Engine as _;
143        use base64::engine::general_purpose::STANDARD as BASE64;
144
145        // 40 raw bytes.
146        let full = vec![0x42; 40];
147        let encoded = BASE64.encode(&full);
148
149        // The base64 form is longer than the raw form (encoding overhead),
150        // and specifically longer than a bound that comfortably fits the
151        // raw bytes.
152        let max_bytes = 40;
153        assert!(
154            encoded.len() > max_bytes,
155            "test setup: base64 form ({}) must exceed max_bytes ({max_bytes}) \
156             to exercise the raw-vs-encoded distinction",
157            encoded.len()
158        );
159
160        // capture() is called with the RAW bytes (as a caller would after
161        // base64-decoding), so it must NOT report truncation even though the
162        // as-yet-undecoded encoded form would have exceeded the bound.
163        let result = BoundedBody::capture(&full, max_bytes);
164
165        assert!(
166            !result.truncated,
167            "raw body fits under max_bytes; must not be truncated even though \
168             its base64 encoding would have exceeded max_bytes"
169        );
170        assert_eq!(result.bytes, full);
171        assert_eq!(result.full_len, full.len() as u64);
172    }
173
174    #[test]
175    fn empty_body_is_not_truncated() {
176        let result = BoundedBody::capture(b"", 10);
177
178        assert!(!result.truncated);
179        assert!(result.bytes.is_empty());
180        assert_eq!(result.full_len, 0);
181    }
