1//! Browser-wide cookie store backed by CDP `Network.*Cookies*` methods.
2//!
3//! [`CookieJar`] wraps a [`Connection`] and exposes ergonomic CRUD over
4//! Chrome's cookie store. The jar is cheap to clone — internally an `Arc`
5//! over a thin inner struct — so it can be passed around freely.
6//!
7//! A jar is either **browser-scoped** ([`CookieJar::new`], via
8//! [`crate::Browser::cookies`]) — reading/writing the default
9//! `BrowserContext` — or **session-scoped** ([`CookieJar::for_session`], via
10//! [`crate::Tab::cookies`]) — routing each command with the tab's `sessionId`
11//! so it resolves against that tab's own `BrowserContext`. The two are
12//! equivalent when every tab shares the default context, and differ only
13//! when a caller opens tabs under per-target isolated `BrowserContext`s.
14//!
15//! ```no_run
16//! # async fn ex() -> zendriver::Result<()> {
17//! # let browser = zendriver::Browser::builder().launch().await?;
18//! let jar = browser.cookies();
19//! jar.set(zendriver::Cookie {
20//!     name: "sid".into(),
21//!     value: "abc123".into(),
22//!     domain: ".example.com".into(),
23//!     path: "/".into(),
24//!     http_only: true,
25//!     secure: true,
26//!     same_site: Some(zendriver::SameSite::Lax),
27//!     ..Default::default()
28//! }).await?;
29//! # Ok(()) }
30//! ```
31//!
32//! ## Serialization — snake_case on disk, camelCase on the wire
33//!
34//! The public [`Cookie`] struct uses idiomatic snake_case for JSON output
35//! (so users' on-disk JSON looks Rust-natural), while a private internal
36//! mirror handles the CDP camelCase rename. Lossless conversion in both
37//! directions; users see clean snake_case JSON, CDP sees camelCase, neither
38//! side has to know the other exists.

50/// SameSite policy as defined by RFC 6265bis.
51///
52/// Mirrors the CDP `Network.CookieSameSite` enum. Serializes as
53/// `"Strict"` / `"Lax"` / `"None"` to match CDP exactly (and the
54/// standard's capitalization).
55#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
56pub enum SameSite {
57    /// First-party only — never sent on cross-site requests.
58    Strict,
59    /// Sent on cross-site top-level navigations only (the default
60    /// modern-browser behavior for unspecified SameSite).
61    Lax,
62    /// Always sent, including in third-party contexts. Requires `Secure`.
63    None,
64}

66/// Cookie priority hint as defined by the (non-standard) `Priority`
67/// attribute Chrome honors during eviction.
68///
69/// Mirrors the CDP `Network.CookiePriority` enum. Serializes as
70/// `"Low"` / `"Medium"` / `"High"` to match CDP exactly.
71#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
72pub enum CookiePriority {
73    /// Lowest retention priority — evicted first under storage pressure.
74    Low,
75    /// Default priority when the attribute is unspecified.
76    Medium,
77    /// Highest retention priority — evicted last.
78    High,
79}

81/// Scheme of the request that set the cookie.
82///
83/// Mirrors the CDP `Network.CookieSourceScheme` enum. Serializes as
84/// `"Unset"` / `"NonSecure"` / `"Secure"` to match CDP exactly. CDP uses
85/// this together with `secure` to implement the "schemeful same-site"
86/// model; `Unset` is the default for cookies set without a known scheme.
87#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
88pub enum CookieSourceScheme {
89    /// Scheme unknown / not recorded.
90    Unset,
91    /// Cookie was set over a non-secure (`http`) origin.
92    NonSecure,
93    /// Cookie was set over a secure (`https`) origin.
94    Secure,
95}

97/// CHIPS partition key — the top-level site a partitioned cookie is scoped
98/// to, plus whether it was set from a cross-site context. Mirrors CDP's
99/// `Network.CookiePartitionKey` (M119+).

133/// A single HTTP cookie.
134///
135/// Field shape matches the public Rust/JSON contract (snake_case). An
136/// internal mirror handles the CDP camelCase rename — see the module-level
137/// docs.
138///
139/// - `expires` is a Unix-epoch timestamp in **seconds** (with sub-second
140///   precision), matching CDP's `Network.TimeSinceEpoch`. `None` means
141///   "session cookie" (deleted when the browser closes).
142/// - `url` is a constructor-time convenience for [`CookieJar::set`]: when
143///   present, CDP infers `domain` + `path` + `secure` from it. CDP never
144///   emits this field on reads, so it's always `None` on cookies returned
145///   by [`CookieJar::all`] / [`CookieJar::for_url`].
146#[derive(Debug, Clone, Default, PartialEq, Serialize, Deserialize)]
147pub struct Cookie {
148    /// Cookie name.
149    pub name: String,
150    /// Cookie value.
151    pub value: String,
152    /// Domain the cookie applies to. Leading dot (e.g. `.example.com`)
153    /// matches the domain and all subdomains.
154    pub domain: String,
155    /// URL path the cookie applies to (typically `"/"`).
156    pub path: String,
157    /// Expiration timestamp in seconds since Unix epoch. `None` for a
158    /// session cookie (deleted when the browser closes).
159    #[serde(default, skip_serializing_if = "Option::is_none")]
160    pub expires: Option<f64>,
161    /// `true` if the cookie has the `HttpOnly` flag (not accessible via
162    /// JavaScript).
163    #[serde(default)]
164    pub http_only: bool,
165    /// `true` if the cookie is only sent over HTTPS.
166    #[serde(default)]
167    pub secure: bool,
168    /// SameSite policy, if specified.
169    #[serde(default, skip_serializing_if = "Option::is_none")]
170    pub same_site: Option<SameSite>,
171    /// Origin URL for [`CookieJar::set`] convenience — CDP infers
172    /// `domain` + `path` + `secure` from it. Always `None` on cookies
173    /// returned by reads.
174    #[serde(default, skip_serializing_if = "Option::is_none")]
175    pub url: Option<String>,
176    /// Eviction-priority hint (`Network.CookiePriority`). `None` lets
177    /// Chrome apply its default (`Medium`).
178    #[serde(default, skip_serializing_if = "Option::is_none")]
179    pub priority: Option<CookiePriority>,
180    /// First-Party Sets membership flag (`sameParty`). Rarely set by
181    /// hand; populated by CDP on reads when known.
182    #[serde(default, skip_serializing_if = "Option::is_none")]
183    pub same_party: Option<bool>,
184    /// Scheme of the origin that set the cookie
185    /// (`Network.CookieSourceScheme`). Drives schemeful same-site.
186    #[serde(default, skip_serializing_if = "Option::is_none")]
187    pub source_scheme: Option<CookieSourceScheme>,
188    /// Source port of the origin that set the cookie. `-1` in CDP means
189    /// "unspecified"; modeled as a plain `Option<i32>` here.
190    #[serde(default, skip_serializing_if = "Option::is_none")]
191    pub source_port: Option<i32>,
192    /// CHIPS partition key — the top-level site the partitioned cookie is
193    /// scoped to, plus whether it was set from a cross-site context.
194    /// Mirrors CDP's `Network.CookiePartitionKey` (M119+).
195    #[serde(default, skip_serializing_if = "Option::is_none")]
196    pub partition_key: Option<CookiePartitionKey>,
197}

273/// Cheap-to-clone handle to a Chrome cookie store.
274///
275/// Wraps a [`Connection`] in an [`Arc`]; cloning is reference-bump cheap.
276///
277/// ## Scoping — default vs. per-target `BrowserContext`
278///
279/// A jar dispatches its `Storage.*`/`Network.*` cookie commands either at
280/// **browser scope** (no `sessionId`) or scoped to a **specific target's
281/// session** (`sessionId` attached):
282///
283/// - [`CookieJar::new`] — browser scope, no `sessionId`. Reads and writes
284///   the **default** `BrowserContext`. This is what [`crate::Browser::cookies`]
285///   returns.
286/// - [`CookieJar::for_session`] — carries a target's `sessionId`, so the
287///   command resolves against **that target's own `BrowserContext`**. This is
288///   what [`crate::Tab::cookies`] returns.
289///
290/// The distinction matters for callers that open pages under **per-target
291/// isolated `BrowserContext`s** (`Target.createBrowserContext`): the default
292/// context and an isolated context have **separate** cookie stores, so a
293/// browser-scope read returns the wrong context's cookies for such a tab.
294/// Routing the command through the tab's own session fixes this — Chrome
295/// resolves the cookie store from the session's target. For the common case
296/// (a single default context shared by every tab), the two are equivalent.
297#[derive(Clone, Debug)]
298pub struct CookieJar {
299    inner: Arc<CookieJarInner>,
300}
301
302#[derive(Debug)]
303struct CookieJarInner {
304    conn: Connection,
305    /// When `Some`, every cookie command is routed with this CDP `sessionId`
306    /// so it resolves against the session's target `BrowserContext`. When
307    /// `None`, commands go at browser scope (the default `BrowserContext`).
308    session_id: Option<String>,
309}
310
311impl CookieJar {
312    /// Construct a browser-scope jar around a [`Connection`].
313    ///
314    /// Cookie commands dispatch with **no** `sessionId`, so they read and
315    /// write the **default** `BrowserContext`. Typically called by
316    /// [`crate::Browser::cookies`] rather than user code.
317    #[must_use]
318    pub fn new(conn: Connection) -> Self {
319        Self {
320            inner: Arc::new(CookieJarInner {
321                conn,
322                session_id: None,
323            }),
324        }
325    }
326
327    /// Construct a jar scoped to a specific target's CDP `session_id`.
328    ///
329    /// Cookie commands dispatch **with** that `sessionId`, so Chrome resolves
330    /// them against the session's target `BrowserContext` — the correct store
331    /// for a tab opened under a per-target isolated `BrowserContext`.
332    /// Typically called by [`crate::Tab::cookies`] rather than user code.
333    #[must_use]
334    pub fn for_session(conn: Connection, session_id: impl Into<String>) -> Self {
335        Self {
336            inner: Arc::new(CookieJarInner {
337                conn,
338                session_id: Some(session_id.into()),
339            }),
340        }
341    }
342
343    /// The `sessionId` (if any) every cookie command is routed with. Cloned
344    /// per call because [`Connection::call_raw`] takes an owned `Option`.
345    fn session(&self) -> Option<String> {
346        self.inner.session_id.clone()
347    }
348
349    /// Return every cookie in the browser's store.
350    ///
351    /// Maps to `Storage.getCookies`.
352    ///
353    /// # Examples
354    ///
355    /// ```no_run
356    /// # async fn ex() -> zendriver::Result<()> {
357    /// # let browser = zendriver::Browser::builder().launch().await?;
358    /// let jar = browser.cookies();
359    /// for c in jar.all().await? {
360    ///     println!("{}={} ({})", c.name, c.value, c.domain);
361    /// }
362    /// # Ok(()) }
363    /// ```
364    pub async fn all(&self) -> Result<Vec<Cookie>> {
365        let resp = self
366            .inner
367            .conn
368            .call_raw("Storage.getCookies", json!({}), self.session())
369            .await?;
370        parse_cookies(&resp)
371    }
372
373    /// Return cookies that would be sent for `url`.
374    ///
375    /// Maps to `Network.getCookies` with `urls: [url]`. Accepting a parsed
376    /// [`url::Url`] surfaces malformed-URL errors at construction time
377    /// instead of as a confusing empty result.
378    ///
379    /// # Examples
380    ///
381    /// ```no_run
382    /// # async fn ex() -> zendriver::Result<()> {
383    /// # let browser = zendriver::Browser::builder().launch().await?;
384    /// let jar = browser.cookies();
385    /// let url = url::Url::parse("https://example.com/").unwrap();
386    /// let cookies = jar.for_url(&url).await?;
387    /// # let _ = cookies;
388    /// # Ok(()) }
389    /// ```
390    pub async fn for_url(&self, url: &url::Url) -> Result<Vec<Cookie>> {
391        let resp = self
392            .inner
393            .conn
394            .call_raw(
395                "Network.getCookies",
396                json!({ "urls": [url.as_str()] }),
397                self.session(),
398            )
399            .await?;
400        parse_cookies(&resp)
401    }
402
403    /// Set a single cookie.
404    ///
405    /// Maps to `Storage.setCookies` with a one-element batch. The singular
406    /// `Network.setCookie` is removed from newer Chrome / Chromium builds
407    /// (the CDP method is reported as not found), so the bulk endpoint is
408    /// the portable choice and is what the parent zendriver-python project
409    /// uses too.
410    ///
411    /// # Errors
412    ///
413    /// Returns [`ZendriverError::Cookie`] if CDP rejects the cookie.
414    ///
415    /// # Examples
416    ///
417    /// ```no_run
418    /// # use zendriver::{Cookie, SameSite};
419    /// # async fn ex() -> zendriver::Result<()> {
420    /// # let browser = zendriver::Browser::builder().launch().await?;
421    /// browser.cookies().set(Cookie {
422    ///     name: "sid".into(),
423    ///     value: "abc".into(),
424    ///     domain: ".example.com".into(),
425    ///     path: "/".into(),
426    ///     http_only: true,
427    ///     secure: true,
428    ///     same_site: Some(SameSite::Lax),
429    ///     ..Default::default()
430    /// }).await?;
431    /// # Ok(()) }
432    /// ```
433    pub async fn set(&self, cookie: Cookie) -> Result<()> {
434        let cdp: CdpCookie = cookie.into();
435        let cdp_json = serde_json::to_value(&cdp).map_err(ZendriverError::Serde)?;
436        self.inner
437            .conn
438            .call_raw(
439                "Storage.setCookies",
440                json!({ "cookies": [cdp_json] }),
441                self.session(),
442            )
443            .await?;
444        Ok(())
445    }
446
447    /// Set many cookies in a single CDP call.
448    ///
449    /// Maps to `Storage.setCookies`. Faster than looping over [`Self::set`]
450    /// for bulk loads (one round-trip instead of N).
451    ///
452    /// # Examples
453    ///
454    /// ```no_run
455    /// # use zendriver::Cookie;
456    /// # async fn ex() -> zendriver::Result<()> {
457    /// # let browser = zendriver::Browser::builder().launch().await?;
458    /// let cookies = vec![
459    ///     Cookie { name: "a".into(), value: "1".into(), domain: ".x.com".into(),
460    ///              path: "/".into(), ..Default::default() },
461    /// ];
462    /// browser.cookies().set_many(cookies).await?;
463    /// # Ok(()) }
464    /// ```
465    pub async fn set_many(&self, cookies: Vec<Cookie>) -> Result<()> {
466        let cdp: Vec<CdpCookie> = cookies.into_iter().map(CdpCookie::from).collect();
467        self.inner
468            .conn
469            .call_raw(
470                "Storage.setCookies",
471                json!({ "cookies": cdp }),
472                self.session(),
473            )
474            .await?;
475        Ok(())
476    }
477
478    /// Delete cookies matching `name` and optional `domain` / `path`.
479    ///
480    /// Maps to `Network.deleteCookies`. CDP requires `name`; `domain` and
481    /// `path` are optional narrowers (when omitted, all cookies with the
482    /// given name across all domains/paths are deleted).
483    ///
484    /// # Examples
485    ///
486    /// ```no_run
487    /// # async fn ex() -> zendriver::Result<()> {
488    /// # let browser = zendriver::Browser::builder().launch().await?;
489    /// browser.cookies().delete("sid", Some(".example.com"), Some("/")).await?;
490    /// # Ok(()) }
491    /// ```
492    pub async fn delete(&self, name: &str, domain: Option<&str>, path: Option<&str>) -> Result<()> {
493        let mut params = json!({ "name": name });
494        if let Some(d) = domain {
495            params["domain"] = json!(d);
496        }
497        if let Some(p) = path {
498            params["path"] = json!(p);
499        }
500        self.inner
501            .conn
502            .call_raw("Network.deleteCookies", params, self.session())
503            .await?;
504        Ok(())
505    }
506
507    /// Clear the entire browser cookie store.
508    ///
509    /// Maps to `Storage.clearCookies` (the modern CDP method; newer
510    /// Chrome / Chromium builds dropped `Storage.clearCookies`
511    /// from the wire). No params, no response payload.
512    ///
513    /// # Examples
514    ///
515    /// ```no_run
516    /// # async fn ex() -> zendriver::Result<()> {
517    /// # let browser = zendriver::Browser::builder().launch().await?;
518    /// browser.cookies().clear().await?;
519    /// # Ok(()) }
520    /// ```
521    pub async fn clear(&self) -> Result<()> {
522        self.inner
523            .conn
524            .call_raw("Storage.clearCookies", json!({}), self.session())
525            .await?;
526        Ok(())
527    }
528}

530/// Shared parser for `Storage.getCookies` and `Network.getCookies` —
531/// both responses use `{ cookies: CdpCookie[] }`.
532#[allow(clippy::result_large_err)] // ZendriverError variance is the project-wide return type
533fn parse_cookies(resp: &serde_json::Value) -> Result<Vec<Cookie>> {
534    let arr = resp
535        .get("cookies")
536        .and_then(|v| v.as_array())
537        .ok_or_else(|| ZendriverError::Cookie("response missing `cookies` array".into()))?;
538    let mut out = Vec::with_capacity(arr.len());
539    for v in arr {
540        let cdp: CdpCookie = serde_json::from_value(v.clone()).map_err(ZendriverError::Serde)?;
541        out.push(cdp.into());
542    }
543    Ok(out)
544}
