1//! Per-frame isolated-world execution-context cache.
2//!
3//! Shared between [`crate::tab::Tab`] (main-frame eval) and
4//! [`crate::frame::Frame`] (per-frame eval). Holds the discovered main frame
5//! id (cached after the first `Page.getFrameTree` round-trip) and the
6//! `executionContextId` returned by `Page.createIsolatedWorld` (cached
7//! after first use; invalidated when Chrome reports
8//! "Cannot find context with specified id", typically after a navigation
9//! destroys the previous context).
10//!
11//! Pre-P4 this lived as a private type inside `tab.rs`. P4 promotes it to
12//! a shared module so [`crate::frame::Frame`] can carry its own per-frame
13//! cache without duplicating the struct definition. The visibility stays
14//! `pub(crate)` — neither the cache nor its fields are part of the public
15//! API.
16
17/// Cache of the discovered main-frame id + the executionContextId for that
18/// frame's `zendriver-eval` isolated world.
19///
20/// Both fields start `None`; the first call to
21/// [`crate::tab::Tab::ensure_isolated_world`] populates them in one
22/// round-trip pair (`Page.getFrameTree` then `Page.createIsolatedWorld`).
23/// Subsequent calls short-circuit on the cached `context_id`.
24///
25/// When [`crate::tab::Tab::evaluate`] catches Chrome's `-32000 Cannot find
26/// context with specified id` error, it sets `context_id = None` (leaving
27/// `main_frame_id` intact since the frame is still around) and retries —
28/// the next `ensure_isolated_world` call re-runs `Page.createIsolatedWorld`
29/// only, skipping the `Page.getFrameTree` round-trip.
30#[derive(Default, Debug)]
31pub(crate) struct IsolatedWorldCache {
32    pub(crate) main_frame_id: Option<String>,
33    pub(crate) context_id: Option<i64>,
34}
