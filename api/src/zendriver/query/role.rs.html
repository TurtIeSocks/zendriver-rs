7/// An ARIA role for use with [`crate::query::FindBuilder::role`].
8///
9/// Covers the common WAI-ARIA roles. For roles not enumerated here, use
10/// [`AriaRole::Other`] with a `&'static str` of the role name.
11///
12/// # Examples
13///
14/// ```
15/// use zendriver::AriaRole;
16/// assert_eq!(AriaRole::Button.to_css(), r#"[role="button"]"#);
17/// ```
18#[derive(Debug, Clone, Copy, PartialEq, Eq)]
19pub enum AriaRole {
20    /// `role="button"`.
21    Button,
22    /// `role="link"`.
23    Link,
24    /// `role="textbox"`.
25    Textbox,
26    /// `role="combobox"`.
27    Combobox,
28    /// `role="checkbox"`.
29    Checkbox,
30    /// `role="radio"`.
31    Radio,
32    /// `role="tab"`.
33    Tab,
34    /// `role="menu"`.
35    Menu,
36    /// `role="menuitem"`.
37    Menuitem,
38    /// `role="dialog"`.
39    Dialog,
40    /// `role="heading"`.
41    Heading,
42    /// `role="banner"`.
43    Banner,
44    /// `role="navigation"`.
45    Navigation,
46    /// `role="main"`.
47    Main,
48    /// `role="article"`.
49    Article,
50    /// `role="list"`.
51    List,
52    /// `role="listitem"`.
53    Listitem,
54    /// `role="row"`.
55    Row,
56    /// `role="cell"`.
57    Cell,
58    /// `role="columnheader"`.
59    Columnheader,
60    /// `role="rowheader"`.
61    Rowheader,
62    /// Escape hatch for ARIA roles not enumerated above.
63    Other(&'static str),
64}
65
66impl AriaRole {
67    /// Compile to a CSS attribute selector.
68    ///
69    /// Tag-implicit roles (e.g. `<button>` implies role=button) are NOT
70    /// auto-included; users querying by role get only explicit
71    /// `[role="..."]` matches. This avoids surprising matches against
72    /// tag-implicit elements that may have different accessibility behavior.
73    ///
74    /// # Examples
75    ///
76    /// ```
77    /// use zendriver::AriaRole;
78    /// assert_eq!(AriaRole::Button.to_css(), r#"[role="button"]"#);
79    /// assert_eq!(AriaRole::Other("tooltip").to_css(), r#"[role="tooltip"]"#);
80    /// ```
81    #[must_use]
82    pub fn to_css(self) -> String {
83        let name = match self {
84            AriaRole::Button => "button",
85            AriaRole::Link => "link",
86            AriaRole::Textbox => "textbox",
87            AriaRole::Combobox => "combobox",
88            AriaRole::Checkbox => "checkbox",
89            AriaRole::Radio => "radio",
90            AriaRole::Tab => "tab",
91            AriaRole::Menu => "menu",
92            AriaRole::Menuitem => "menuitem",
93            AriaRole::Dialog => "dialog",
94            AriaRole::Heading => "heading",
95            AriaRole::Banner => "banner",
96            AriaRole::Navigation => "navigation",
97            AriaRole::Main => "main",
98            AriaRole::Article => "article",
99            AriaRole::List => "list",
100            AriaRole::Listitem => "listitem",
101            AriaRole::Row => "row",
102            AriaRole::Cell => "cell",
103            AriaRole::Columnheader => "columnheader",
104            AriaRole::Rowheader => "rowheader",
105            AriaRole::Other(s) => s,
106        };
107        format!("[role=\"{name}\"]")
108    }
109}
