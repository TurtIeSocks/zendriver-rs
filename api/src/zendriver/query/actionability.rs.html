1//! Playwright-style actionability predicates: visible / stable / enabled /
2//! receives_pointer. Each runs a small JS function on the element's remote
3//! handle via `Element::call_on_main` and returns a `bool`.
4//!
5//! The aggregate gate (`wait_actionable`) and `NotActionable` emission land
6//! in T15; T14 only ships the four predicate primitives + the
7//! `ActionabilityCheck` requirements struct that downstream actions use to
8//! describe which checks they need.
9
10use std::time::Duration;
11
12use serde_json::json;
13use tokio::time::Instant;
14
15use crate::element::Element;
16use crate::error::{Result, ZendriverError};

18/// Set of actionability checks an action wants the element to satisfy
19/// before its CDP dispatch. Per-field booleans gate the corresponding
20/// `check_*` predicate in `wait_actionable` (T15). Three named presets
21/// cover the common combinations (`FULL`, `VISIBLE_ONLY`, `TEXT_INPUT`);
22/// callers may also construct ad-hoc sets directly.
23#[allow(dead_code)] // First caller (`wait_actionable`) lands in T15.
24#[derive(Clone, Copy, Debug, PartialEq, Eq)]
25pub(crate) struct ActionabilityCheck {
26    pub visible: bool,
27    pub stable: bool,
28    pub enabled: bool,
29    pub receives_pointer: bool,
30}
31
32impl ActionabilityCheck {
33    /// All four predicates — used by `click` and similar pointer-driven
34    /// actions where layout stability + an unobstructed hit point matter.
35    #[allow(dead_code)] // Consumed by `click_with` in T20.
36    pub(crate) const FULL: Self = Self {
37        visible: true,
38        stable: true,
39        enabled: true,
40        receives_pointer: true,
41    };
42
43    /// Visibility only — used by `screenshot` (we just need pixels to
44    /// capture; we don't care if a sibling overlay covers part of the
45    /// element).
46    #[allow(dead_code)] // Consumed by `Element::screenshot` in T26.
47    pub(crate) const VISIBLE_ONLY: Self = Self {
48        visible: true,
49        stable: false,
50        enabled: false,
51        receives_pointer: false,
52    };
53
54    /// Text-input combo — used by `type_text` / `focus` where the element
55    /// must be visible + enabled but doesn't need a hit-tested pointer
56    /// path (keystrokes route through the focused element, not the
57    /// cursor's position).
58    #[allow(dead_code)] // Consumed by `type_text` / `focus` in T19/T22.
59    pub(crate) const TEXT_INPUT: Self = Self {
60        visible: true,
61        stable: false,
62        enabled: true,
63        receives_pointer: false,
64    };
65}

67/// `true` iff the element is rendered: attached to the document, has a
68/// non-`null` `offsetParent` (catches `display: none` ancestors), a
69/// positive bbox, computed `visibility !== 'hidden'`, and computed
70/// `opacity !== '0'`.
71#[allow(dead_code)] // First callers (`is_visible` + `wait_actionable`) land in T15/T18.
72pub(crate) async fn check_visible(el: &Element) -> Result<bool> {
73    let js = r#"
74        function(el) {
75            if (!el || !el.isConnected) return false;
76            // offsetParent === null catches `display: none` on the element or any ancestor
77            // (except for fixed-position elements, which are handled by the bbox check below).
78            if (el.offsetParent === null && getComputedStyle(el).position !== 'fixed') return false;
79            const rect = el.getBoundingClientRect();
80            if (rect.width <= 0 || rect.height <= 0) return false;
81            const style = getComputedStyle(el);
82            if (style.visibility === 'hidden') return false;
83            if (style.opacity === '0') return false;
84            return true;
85        }
86    "#;
87    let res = el.call_on_main(js, json!([])).await?;
88    Ok(res.get("value").and_then(|v| v.as_bool()).unwrap_or(false))
89}

91/// `true` iff the element's bounding box is unchanged across two
92/// `requestAnimationFrame` ticks (within 0.5px on each of x/y/w/h). This
93/// catches mid-transition layout shifts that would race a synthesized
94/// click.
95#[allow(dead_code)] // First caller (`wait_actionable`) lands in T15.
96pub(crate) async fn check_stable(el: &Element) -> Result<bool> {
97    let js = r#"
98        function(el) {
99            return new Promise(resolve => {
100                if (!el || !el.isConnected) { resolve(false); return; }
101                const first = el.getBoundingClientRect();
102                requestAnimationFrame(() => {
103                    requestAnimationFrame(() => {
104                        const second = el.getBoundingClientRect();
105                        const stable =
106                            Math.abs(first.x - second.x) < 0.5 &&
107                            Math.abs(first.y - second.y) < 0.5 &&
108                            Math.abs(first.width - second.width) < 0.5 &&
109                            Math.abs(first.height - second.height) < 0.5;
110                        resolve(stable);
111                    });
112                });
113            });
114        }
115    "#;
116    let res = el.call_on_main(js, json!([])).await?;
117    Ok(res.get("value").and_then(|v| v.as_bool()).unwrap_or(false))
118}

120/// `true` iff the element is not disabled: native `el.disabled` is
121/// false-ish AND `aria-disabled` is not `'true'`. Non-form elements
122/// (which have no `disabled` property) are considered enabled.
123#[allow(dead_code)] // First callers (`is_enabled` + `wait_actionable`) land in T15/T18.
124pub(crate) async fn check_enabled(el: &Element) -> Result<bool> {
125    let js = r#"
126        function(el) {
127            if (!el) return false;
128            // `disabled === false` for form controls; `undefined` for non-form elements
129            // (which we treat as enabled).
130            if (el.disabled === true) return false;
131            const ariaDisabled = el.getAttribute && el.getAttribute('aria-disabled');
132            if (ariaDisabled === 'true') return false;
133            return true;
134        }
135    "#;
136    let res = el.call_on_main(js, json!([])).await?;
137    Ok(res.get("value").and_then(|v| v.as_bool()).unwrap_or(false))
138}

140/// `true` iff a synthesized click at the element's bbox center would land
141/// on the element (or one of its descendants). Walks the ancestor chain
142/// of `document.elementFromPoint(cx, cy)`; if our element appears in that
143/// chain, pointer events reach it. Returns `false` when a sibling overlay
144/// covers the hit point.
145#[allow(dead_code)] // First caller (`wait_actionable`) lands in T15.
146pub(crate) async fn check_receives_pointer(el: &Element) -> Result<bool> {
147    let js = r#"
148        function(el) {
149            if (!el || !el.isConnected) return false;
150            const rect = el.getBoundingClientRect();
151            if (rect.width <= 0 || rect.height <= 0) return false;
152            const cx = rect.left + rect.width / 2;
153            const cy = rect.top + rect.height / 2;
154            let hit = document.elementFromPoint(cx, cy);
155            while (hit) {
156                if (hit === el) return true;
157                hit = hit.parentElement;
158            }
159            return false;
160        }
161    "#;
162    let res = el.call_on_main(js, json!([])).await?;
163    Ok(res.get("value").and_then(|v| v.as_bool()).unwrap_or(false))
164}

166/// Poll each predicate in `require` at 50 ms intervals until all enabled
167/// checks pass, or `timeout` elapses. On deadline, returns
168/// [`ZendriverError::NotActionable`] with the first-failing check's
169/// human-readable reason ("not visible", "not enabled", "not stable (still
170/// animating)", or "occluded by overlay").
171///
172/// Predicates are evaluated in fixed order: visible → enabled → stable →
173/// receives_pointer. This matches Playwright's gate ordering and avoids
174/// running the more-expensive stability + hit-testing checks while the
175/// element is still hidden or disabled.
176#[allow(dead_code)] // First callers (`click_with`, `hover`, `focus`, …) land in T19/T20/T22/T26.
177pub(crate) async fn wait_actionable(
178    el: &Element,
179    require: ActionabilityCheck,
180    timeout: Duration,
181) -> Result<()> {
182    let deadline = Instant::now() + timeout;
183    let poll_interval = Duration::from_millis(50);
184    loop {
185        let mut failed_reason: Option<&'static str> = None;
186        if require.visible && !check_visible(el).await? {
187            failed_reason = Some("not visible");
188        } else if require.enabled && !check_enabled(el).await? {
189            failed_reason = Some("not enabled");
190        } else if require.stable && !check_stable(el).await? {
191            failed_reason = Some("not stable (still animating)");
192        } else if require.receives_pointer && !check_receives_pointer(el).await? {
193            failed_reason = Some("occluded by overlay");
194        }
195        match failed_reason {
196            None => return Ok(()),
197            Some(reason) => {
198                if Instant::now() >= deadline {
199                    return Err(ZendriverError::NotActionable(timeout, reason.to_owned()));
200                }
201                tokio::time::sleep(poll_interval).await;
202            }
203        }
204    }
