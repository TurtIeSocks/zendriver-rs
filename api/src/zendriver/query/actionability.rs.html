1//! Playwright-style actionability predicates: visible / stable / enabled /
2//! receives_pointer. Each runs a small JS function on the element's remote
3//! handle via `Element::call_on_main` and returns a `bool`.
4//!
5//! The aggregate gate ([`wait_actionable`]) polls the four predicates in
6//! order and raises `NotActionable` on deadline; `Element::click_with`,
7//! `hover`, `focus`, `type_text`, and `screenshot` (in `element::actions`
8//! / `element::screenshot`) each pick the [`ActionabilityCheck`] preset
9//! matching what they need before dispatching.
10
11use std::time::Duration;

13use serde_json::json;
14use tokio::time::Instant;
15
16use crate::element::Element;
17use crate::error::{Result, ZendriverError};

19/// Set of actionability checks an action wants the element to satisfy
20/// before its CDP dispatch. Per-field booleans gate the corresponding
21/// `check_*` predicate in `wait_actionable`. Three named presets
22/// cover the common combinations (`FULL`, `VISIBLE_ONLY`, `TEXT_INPUT`);
23/// callers may also construct ad-hoc sets directly.
24#[derive(Clone, Copy, Debug, PartialEq, Eq)]
25pub(crate) struct ActionabilityCheck {
26    pub visible: bool,
27    pub stable: bool,
28    pub enabled: bool,
29    pub receives_pointer: bool,
30}
31
32impl ActionabilityCheck {
33    /// All four predicates — used by `click` and similar pointer-driven
34    /// actions where layout stability + an unobstructed hit point matter.
35    pub(crate) const FULL: Self = Self {
36        visible: true,
37        stable: true,
38        enabled: true,
39        receives_pointer: true,
40    };
41
42    /// Visibility only — used by `screenshot` (we just need pixels to
43    /// capture; we don't care if a sibling overlay covers part of the
44    /// element).
45    pub(crate) const VISIBLE_ONLY: Self = Self {
46        visible: true,
47        stable: false,
48        enabled: false,
49        receives_pointer: false,
50    };
51
52    /// Text-input combo — used by `type_text` / `focus` where the element
53    /// must be visible + enabled but doesn't need a hit-tested pointer
54    /// path (keystrokes route through the focused element, not the
55    /// cursor's position).
56    pub(crate) const TEXT_INPUT: Self = Self {
57        visible: true,
58        stable: false,
59        enabled: true,
60        receives_pointer: false,
61    };
62}

64/// `true` iff the element is rendered: attached to the document, has a
65/// non-`null` `offsetParent` (catches `display: none` ancestors), a
66/// positive bbox, computed `visibility !== 'hidden'`, and computed
67/// `opacity !== '0'`.
68///
69/// Live callers: `Element::is_visible` and the `visible_only` filter in
70/// `FindBuilder`/`FindAllBuilder` (`query::mod`).
71pub(crate) async fn check_visible(el: &Element) -> Result<bool> {
72    let js = r#"
73        function(el) {
74            if (!el || !el.isConnected) return false;
75            // offsetParent === null catches `display: none` on the element or any ancestor
76            // (except for fixed-position elements, which are handled by the bbox check below).
77            if (el.offsetParent === null && getComputedStyle(el).position !== 'fixed') return false;
78            const rect = el.getBoundingClientRect();
79            if (rect.width <= 0 || rect.height <= 0) return false;
80            const style = getComputedStyle(el);
81            if (style.visibility === 'hidden') return false;
82            if (style.opacity === '0') return false;
83            return true;
84        }
85    "#;
86    let res = el.call_on_main(js, json!([])).await?;
87    Ok(res.get("value").and_then(|v| v.as_bool()).unwrap_or(false))
88}

90/// `true` iff the element's bounding box is unchanged across two
91/// `requestAnimationFrame` ticks (within 0.5px on each of x/y/w/h). This
92/// catches mid-transition layout shifts that would race a synthesized
93/// click.
94pub(crate) async fn check_stable(el: &Element) -> Result<bool> {
95    let js = r#"
96        function(el) {
97            return new Promise(resolve => {
98                if (!el || !el.isConnected) { resolve(false); return; }
99                const first = el.getBoundingClientRect();
100                requestAnimationFrame(() => {
101                    requestAnimationFrame(() => {
102                        const second = el.getBoundingClientRect();
103                        const stable =
104                            Math.abs(first.x - second.x) < 0.5 &&
105                            Math.abs(first.y - second.y) < 0.5 &&
106                            Math.abs(first.width - second.width) < 0.5 &&
107                            Math.abs(first.height - second.height) < 0.5;
108                        resolve(stable);
109                    });
110                });
111            });
112        }
113    "#;
114    let res = el.call_on_main(js, json!([])).await?;
115    Ok(res.get("value").and_then(|v| v.as_bool()).unwrap_or(false))
116}

118/// `true` iff the element is not disabled: native `el.disabled` is
119/// false-ish AND `aria-disabled` is not `'true'`. Non-form elements
120/// (which have no `disabled` property) are considered enabled.
121pub(crate) async fn check_enabled(el: &Element) -> Result<bool> {
122    let js = r#"
123        function(el) {
124            if (!el) return false;
125            // `disabled === false` for form controls; `undefined` for non-form elements
126            // (which we treat as enabled).
127            if (el.disabled === true) return false;
128            const ariaDisabled = el.getAttribute && el.getAttribute('aria-disabled');
129            if (ariaDisabled === 'true') return false;
130            return true;
131        }
132    "#;
133    let res = el.call_on_main(js, json!([])).await?;
134    Ok(res.get("value").and_then(|v| v.as_bool()).unwrap_or(false))
135}

137/// `true` iff a synthesized click at the element's bbox center would land
138/// on the element (or one of its descendants). Walks the ancestor chain
139/// of `document.elementFromPoint(cx, cy)`; if our element appears in that
140/// chain, pointer events reach it. Returns `false` when a sibling overlay
141/// covers the hit point.
142pub(crate) async fn check_receives_pointer(el: &Element) -> Result<bool> {
143    let js = r#"
144        function(el) {
145            if (!el || !el.isConnected) return false;
146            const rect = el.getBoundingClientRect();
147            if (rect.width <= 0 || rect.height <= 0) return false;
148            const cx = rect.left + rect.width / 2;
149            const cy = rect.top + rect.height / 2;
150            let hit = document.elementFromPoint(cx, cy);
151            while (hit) {
152                if (hit === el) return true;
153                hit = hit.parentElement;
154            }
155            return false;
156        }
157    "#;
158    let res = el.call_on_main(js, json!([])).await?;
159    Ok(res.get("value").and_then(|v| v.as_bool()).unwrap_or(false))
160}

162/// Poll each predicate in `require` at 50 ms intervals until all enabled
163/// checks pass, or `timeout` elapses. On deadline, returns
164/// [`ZendriverError::NotActionable`] with the first-failing check's
165/// human-readable reason ("not visible", "not enabled", "not stable (still
166/// animating)", or "occluded by overlay").
167///
168/// Predicates are evaluated in fixed order: visible → enabled → stable →
169/// receives_pointer. This matches Playwright's gate ordering and avoids
170/// running the more-expensive stability + hit-testing checks while the
171/// element is still hidden or disabled.
172pub(crate) async fn wait_actionable(
173    el: &Element,
174    require: ActionabilityCheck,
175    timeout: Duration,
176) -> Result<()> {
177    let deadline = Instant::now() + timeout;
178    let poll_interval = Duration::from_millis(50);
179    loop {
180        let mut failed_reason: Option<&'static str> = None;
181        if require.visible && !check_visible(el).await? {
182            failed_reason = Some("not visible");
183        } else if require.enabled && !check_enabled(el).await? {
184            failed_reason = Some("not enabled");
185        } else if require.stable && !check_stable(el).await? {
186            failed_reason = Some("not stable (still animating)");
187        } else if require.receives_pointer && !check_receives_pointer(el).await? {
188            failed_reason = Some("occluded by overlay");
189        }
190        match failed_reason {
191            None => return Ok(()),
192            Some(reason) => {
193                if Instant::now() >= deadline {
194                    return Err(ZendriverError::NotActionable(timeout, reason.to_owned()));
195                }
196                tokio::time::sleep(poll_interval).await;
197            }
198        }
199    }
