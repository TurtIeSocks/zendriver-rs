1//! `IpApiResolver`: derive the exit-IP country (and exact timezone, when
2//! available) via a proxied HTTP probe.
3
4use std::time::Duration;
5
6use async_trait::async_trait;
7use zendriver_stealth::geo::{Country, GeoResolver, ResolvedGeo};
8
9/// Resolves the apparent country — and, when present, the exact IANA
10/// timezone — by querying an IP-geolocation service (default
11/// `http://ip-api.com/json` — **plaintext**; the proxy operator can tamper
12/// with the response in transit, so override [`Self::endpoint`] to an HTTPS
13/// service if response integrity matters for your threat model) through the
14/// browser's proxy. Opt-in via `BrowserBuilder::geo_auto`; endpoint
15/// overridable; swap the whole thing out with a custom [`GeoResolver`].
16pub struct IpApiResolver {
17    endpoint: String,
18    proxy: Option<String>,
19    proxy_auth: Option<(String, String)>,
20    timeout: Duration,
21}
22
23impl Default for IpApiResolver {
24    fn default() -> Self {
25        Self::new()
26    }
27}
28
29impl IpApiResolver {
30    /// A resolver hitting `http://ip-api.com/json` directly (no proxy), with
31    /// a 5-second timeout. Chain [`Self::endpoint`] / [`Self::timeout`] to
32    /// customize; `BrowserBuilder::geo_auto` wires the proxy (and its
33    /// credentials, if any) via the crate-private [`Self::with_proxy`].
34    #[must_use]
35    pub fn new() -> Self {
36        Self {
37            endpoint: "http://ip-api.com/json".into(),
38            proxy: None,
39            proxy_auth: None,
40            timeout: Duration::from_secs(5),
41        }
42    }
43
44    /// Override the probe endpoint (default `http://ip-api.com/json`, which
45    /// is plaintext HTTP — a malicious or compromised proxy operator can
46    /// observe or tamper with the response since it's routed through
47    /// `with_proxy`; override to an HTTPS endpoint if you need integrity).
48    /// Must return a JSON body with a top-level `countryCode` string field;
49    /// an optional top-level `timezone` string field (ip-api's exact IANA
50    /// zone for the exit IP) is used when present, falling back to the
51    /// country-representative zone otherwise.
52    #[must_use]
53    pub fn endpoint(mut self, url: impl Into<String>) -> Self {
54        self.endpoint = url.into();
55        self
56    }
57
58    /// Override the request timeout (default 5s).
59    #[must_use]
60    pub fn timeout(mut self, d: Duration) -> Self {
61        self.timeout = d;
62        self
63    }
64
65    /// Route the probe through `server` (mirrors the browser's own proxy so
66    /// the resolved country matches the exit IP Chrome will actually use),
67    /// authenticating with `auth` (`user`, `pass`) when the proxy requires
68    /// it — passed to reqwest via [`reqwest::Proxy::basic_auth`], never
69    /// embedded in the proxy URL string, so it can't leak into an error
70    /// `Display`.
71    ///
72    /// Called by `BrowserBuilder::geo_auto`, wiring `self.proxy` from
73    /// [`crate::browser::BrowserBuilder::proxy`] through here.
74    #[must_use]
75    pub(crate) fn with_proxy(
76        mut self,
77        server: Option<String>,
78        auth: Option<(String, String)>,
79    ) -> Self {
80        self.proxy = server;
81        self.proxy_auth = auth;
82        self
83    }
84}

86#[async_trait]
87impl GeoResolver for IpApiResolver {
88    async fn resolve(&self) -> Option<ResolvedGeo> {
89        let mut builder = reqwest::Client::builder().timeout(self.timeout);
90        if let Some(p) = &self.proxy {
91            match reqwest::Proxy::all(p) {
92                Ok(mut px) => {
93                    if let Some((user, pass)) = &self.proxy_auth {
94                        px = px.basic_auth(user, pass);
95                    }
96                    builder = builder.proxy(px);
97                }
98                Err(e) => {
99                    // `reqwest::Error`'s `Display` only ever includes the
100                    // failed proxy/target URL text, never proxy credentials
101                    // (those are sent via `basic_auth`, not embedded in the
102                    // URL) — safe to log as-is.
103                    tracing::warn!(error = %e, "geo probe: bad proxy; skipping");
104                    return None;
105                }
106            }
107        }
108        let client = builder.build().ok()?;
109        let resp = match client.get(&self.endpoint).send().await {
110            Ok(r) => r,
111            Err(e) => {
112                tracing::warn!(error = %e, "geo probe request failed");
113                return None;
114            }
115        };
116        let body: serde_json::Value = resp.json().await.ok()?;
117        let cc = body.get("countryCode").and_then(|v| v.as_str())?;
118        let country = match Country::try_from(cc) {
119            Ok(c) => c,
120            Err(_) => {
121                tracing::warn!(country = %cc, "geo probe: unrecognized country code");
122                return None;
123            }
124        };
125        let timezone = body
126            .get("timezone")
127            .and_then(|v| v.as_str())
128            .filter(|s| !s.is_empty())
129            .map(str::to_string);
130        Some(ResolvedGeo { country, timezone })
131    }
132}

138    #[tokio::test]
139    async fn resolves_country_from_ipapi_json() {
140        let server = wiremock::MockServer::start().await;
141        wiremock::Mock::given(wiremock::matchers::method("GET"))
142            .respond_with(
143                wiremock::ResponseTemplate::new(200).set_body_string(r#"{"countryCode":"DE"}"#),
144            )
145            .mount(&server)
146            .await;
147        let r = IpApiResolver::new().endpoint(server.uri());
148        assert_eq!(
149            r.resolve().await,
150            Some(ResolvedGeo {
151                country: Country::try_from("DE").unwrap(),
152                timezone: None,
153            })
154        );
155    }

157    /// A body carrying ip-api's `timezone` field must thread it through as
158    /// the EXACT probe timezone, not just the country.
159    #[tokio::test]
160    async fn resolves_exact_timezone_from_ipapi_json() {
161        let server = wiremock::MockServer::start().await;
162        wiremock::Mock::given(wiremock::matchers::method("GET"))
163            .respond_with(
164                wiremock::ResponseTemplate::new(200)
165                    .set_body_string(r#"{"countryCode":"US","timezone":"America/Los_Angeles"}"#),

167            .mount(&server)
168            .await;
169        let r = IpApiResolver::new().endpoint(server.uri());
170        assert_eq!(
171            r.resolve().await,
172            Some(ResolvedGeo {
173                country: Country::try_from("US").unwrap(),
174                timezone: Some("America/Los_Angeles".to_string()),
175            })
176        );
177    }

210    /// C1: `with_proxy`'s `auth` must actually reach the underlying
211    /// `reqwest::Proxy` (via `basic_auth`), not just be stored and ignored.
212    /// A real authenticated-proxy wiremock (mocking the CONNECT/tunnel
213    /// handshake a real forward proxy performs) is impractical with
214    /// `wiremock` (it's an HTTP server, not a proxy), so this instead stands
215    /// a plain wiremock server in for the proxy and drives a plain-HTTP
216    /// target through it — reqwest relays plain-HTTP-through-HTTP-proxy
217    /// requests as absolute-form requests directly to the proxy's socket
218    /// with a `Proxy-Authorization` header when `basic_auth` was set, so the
219    /// mock server IS the thing that receives (and can assert on) that
220    /// header. If `with_proxy`'s `auth` were dropped (the C1 bug), the mock
221    /// (which requires the header) would never match and `resolve()` would
222    /// return `None`.
223    #[tokio::test]
224    async fn threads_proxy_credentials_into_reqwest_proxy() {
225        let proxy = wiremock::MockServer::start().await;
226        wiremock::Mock::given(wiremock::matchers::method("GET"))
227            .and(wiremock::matchers::header_exists("Proxy-Authorization"))
228            .respond_with(
229                wiremock::ResponseTemplate::new(200).set_body_string(r#"{"countryCode":"DE"}"#),
230            )
231            .mount(&proxy)
232            .await;
233
234        let r = IpApiResolver::new()
235            .endpoint("http://geo-probe.invalid/json")
236            .with_proxy(Some(proxy.uri()), Some(("bob".into(), "s3cret".into())));
237        assert_eq!(
238            r.resolve().await.map(|g| g.country),
239            Some(Country::try_from("DE").unwrap())
240        );
241    }
242
243    /// Without credentials, the mock (which requires `Proxy-Authorization`)
244    /// must NOT match — a control proving the above test isn't a false
245    /// positive from some other matcher laxity.
246    #[tokio::test]
247    async fn no_credentials_means_no_proxy_auth_header() {
248        let proxy = wiremock::MockServer::start().await;

250            .and(wiremock::matchers::header_exists("Proxy-Authorization"))
