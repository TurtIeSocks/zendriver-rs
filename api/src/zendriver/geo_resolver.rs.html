5use async_trait::async_trait;

8/// Resolves the apparent country by querying an IP-geolocation service
9/// (default `http://ip-api.com/json` — **plaintext**; the proxy operator can
10/// tamper with the response in transit, so override [`Self::endpoint`] to an
11/// HTTPS service if response integrity matters for your threat model)
12/// through the browser's proxy. Opt-in via `BrowserBuilder::geo_auto`;
13/// endpoint overridable; swap the whole thing out with a custom
14/// [`GeoResolver`].
15pub struct IpApiResolver {
16    endpoint: String,
17    proxy: Option<String>,
18    proxy_auth: Option<(String, String)>,
19    timeout: Duration,
20}
21
22impl Default for IpApiResolver {
23    fn default() -> Self {
24        Self::new()
25    }
26}
27
28impl IpApiResolver {
29    /// A resolver hitting `http://ip-api.com/json` directly (no proxy), with
30    /// a 5-second timeout. Chain [`Self::endpoint`] / [`Self::timeout`] to
31    /// customize; `BrowserBuilder::geo_auto` wires the proxy (and its
32    /// credentials, if any) via the crate-private [`Self::with_proxy`].
33    #[must_use]
34    pub fn new() -> Self {
35        Self {
36            endpoint: "http://ip-api.com/json".into(),
37            proxy: None,
38            proxy_auth: None,
39            timeout: Duration::from_secs(5),
40        }
41    }
42
43    /// Override the probe endpoint (default `http://ip-api.com/json`, which
44    /// is plaintext HTTP — a malicious or compromised proxy operator can
45    /// observe or tamper with the response since it's routed through
46    /// `with_proxy`; override to an HTTPS endpoint if you need integrity).
47    /// Must return a JSON body with a top-level `countryCode` string field.
48    #[must_use]
49    pub fn endpoint(mut self, url: impl Into<String>) -> Self {
50        self.endpoint = url.into();
51        self
52    }
53
54    /// Override the request timeout (default 5s).
55    #[must_use]
56    pub fn timeout(mut self, d: Duration) -> Self {
57        self.timeout = d;
58        self
59    }
60
61    /// Route the probe through `server` (mirrors the browser's own proxy so
62    /// the resolved country matches the exit IP Chrome will actually use),
63    /// authenticating with `auth` (`user`, `pass`) when the proxy requires
64    /// it — passed to reqwest via [`reqwest::Proxy::basic_auth`], never
65    /// embedded in the proxy URL string, so it can't leak into an error
66    /// `Display`.
67    ///
68    /// Called by `BrowserBuilder::geo_auto`, wiring `self.proxy` from
69    /// [`crate::browser::BrowserBuilder::proxy`] through here.
70    #[must_use]
71    pub(crate) fn with_proxy(
72        mut self,
73        server: Option<String>,
74        auth: Option<(String, String)>,
75    ) -> Self {
76        self.proxy = server;
77        self.proxy_auth = auth;
78        self
79    }
80}

82#[async_trait]
83impl GeoResolver for IpApiResolver {
84    async fn country(&self) -> Option<Country> {
85        let mut builder = reqwest::Client::builder().timeout(self.timeout);
86        if let Some(p) = &self.proxy {
87            match reqwest::Proxy::all(p) {
88                Ok(mut px) => {
89                    if let Some((user, pass)) = &self.proxy_auth {
90                        px = px.basic_auth(user, pass);
91                    }
92                    builder = builder.proxy(px);
93                }
94                Err(e) => {
95                    // `reqwest::Error`'s `Display` only ever includes the
96                    // failed proxy/target URL text, never proxy credentials
97                    // (those are sent via `basic_auth`, not embedded in the
98                    // URL) — safe to log as-is.
99                    tracing::warn!(error = %e, "geo probe: bad proxy; skipping");
100                    return None;
101                }
102            }
103        }
104        let client = builder.build().ok()?;
105        let resp = match client.get(&self.endpoint).send().await {
106            Ok(r) => r,
107            Err(e) => {
108                tracing::warn!(error = %e, "geo probe request failed");
109                return None;
110            }
111        };
112        let body: serde_json::Value = resp.json().await.ok()?;
113        let cc = body.get("countryCode").and_then(|v| v.as_str())?;
114        match Country::try_from(cc) {
115            Ok(c) => Some(c),
116            Err(_) => {
117                tracing::warn!(country = %cc, "geo probe: unrecognized country code");
118                None
119            }
120        }
121    }
122}

124#[cfg(test)]
125mod tests {
126    use super::*;

128    #[tokio::test]
129    async fn resolves_country_from_ipapi_json() {
130        let server = wiremock::MockServer::start().await;
131        wiremock::Mock::given(wiremock::matchers::method("GET"))
132            .respond_with(
133                wiremock::ResponseTemplate::new(200).set_body_string(r#"{"countryCode":"DE"}"#),
134            )
135            .mount(&server)
136            .await;
137        let r = IpApiResolver::new().endpoint(server.uri());
138        assert_eq!(r.country().await, Some(Country::try_from("DE").unwrap()));
139    }
140
141    #[tokio::test]
142    async fn bad_body_yields_none() {
143        let server = wiremock::MockServer::start().await;
144        wiremock::Mock::given(wiremock::matchers::method("GET"))
145            .respond_with(wiremock::ResponseTemplate::new(200).set_body_string("nope"))
146            .mount(&server)
147            .await;
148        assert_eq!(
149            IpApiResolver::new().endpoint(server.uri()).country().await,
150            None
151        );
152    }

154    /// C1: `with_proxy`'s `auth` must actually reach the underlying
155    /// `reqwest::Proxy` (via `basic_auth`), not just be stored and ignored.
156    /// A real authenticated-proxy wiremock (mocking the CONNECT/tunnel
157    /// handshake a real forward proxy performs) is impractical with
158    /// `wiremock` (it's an HTTP server, not a proxy), so this instead stands
159    /// a plain wiremock server in for the proxy and drives a plain-HTTP
160    /// target through it — reqwest relays plain-HTTP-through-HTTP-proxy
161    /// requests as absolute-form requests directly to the proxy's socket
162    /// with a `Proxy-Authorization` header when `basic_auth` was set, so the
163    /// mock server IS the thing that receives (and can assert on) that
164    /// header. If `with_proxy`'s `auth` were dropped (the C1 bug), the mock
165    /// (which requires the header) would never match and `country()` would
166    /// return `None`.
167    #[tokio::test]
168    async fn threads_proxy_credentials_into_reqwest_proxy() {
169        let proxy = wiremock::MockServer::start().await;
170        wiremock::Mock::given(wiremock::matchers::method("GET"))
171            .and(wiremock::matchers::header_exists("Proxy-Authorization"))
172            .respond_with(
173                wiremock::ResponseTemplate::new(200).set_body_string(r#"{"countryCode":"DE"}"#),
174            )
175            .mount(&proxy)
176            .await;
177
178        let r = IpApiResolver::new()
179            .endpoint("http://geo-probe.invalid/json")
180            .with_proxy(Some(proxy.uri()), Some(("bob".into(), "s3cret".into())));
181        assert_eq!(r.country().await, Some(Country::try_from("DE").unwrap()));
182    }
183
184    /// Without credentials, the mock (which requires `Proxy-Authorization`)
185    /// must NOT match — a control proving the above test isn't a false
186    /// positive from some other matcher laxity.
187    #[tokio::test]
188    async fn no_credentials_means_no_proxy_auth_header() {
189        let proxy = wiremock::MockServer::start().await;
190        wiremock::Mock::given(wiremock::matchers::method("GET"))
191            .and(wiremock::matchers::header_exists("Proxy-Authorization"))
192            .respond_with(
193                wiremock::ResponseTemplate::new(200).set_body_string(r#"{"countryCode":"DE"}"#),
194            )
195            .mount(&proxy)
196            .await;
197
198        let r = IpApiResolver::new()
199            .endpoint("http://geo-probe.invalid/json")
200            .with_proxy(Some(proxy.uri()), None);
201        // No `Proxy-Authorization` header sent -> mock doesn't match -> 404
202        // from wiremock -> `.json()` fails -> `country()` yields `None`.
203        assert_eq!(r.country().await, None);
204    }
