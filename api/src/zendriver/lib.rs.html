1//! Async, undetectable Chrome automation over the Chrome DevTools Protocol.
2//!
3//! `zendriver` is the high-level browser-automation entry point built on a
4//! CDP-over-WebSocket transport. The crate aims to feel like Playwright /
5//! Puppeteer for Rust while staying explicit about its CDP underpinnings —
6//! every public type maps cleanly to a CDP surface, every action has a
7//! single-call escape hatch, and stealth is on by default.
8//!
9//! See the [user guide / mdBook](https://turtiesocks.github.io/zendriver-rs/)
10//! for end-to-end walkthroughs covering [installation][book-install],
11//! [quickstart][book-quickstart], [stealth][book-stealth],
12//! [multi-tab][book-multi-tab], [frames][book-frames],
13//! [input][book-input], and migration guides from
14//! [Playwright][book-mig-playwright] /
15//! [zendriver (Python)][book-mig-zendriver] /
16//! [nodriver (Python)][book-mig-nodriver].
17//!
18//! [book-install]: https://turtiesocks.github.io/zendriver-rs/install.html
19//! [book-quickstart]: https://turtiesocks.github.io/zendriver-rs/quickstart.html
20//! [book-stealth]: https://turtiesocks.github.io/zendriver-rs/stealth.html
21//! [book-multi-tab]: https://turtiesocks.github.io/zendriver-rs/multi-tab.html
22//! [book-frames]: https://turtiesocks.github.io/zendriver-rs/frames.html
23//! [book-input]: https://turtiesocks.github.io/zendriver-rs/input.html
24//! [book-mig-playwright]: https://turtiesocks.github.io/zendriver-rs/migration-playwright.html
25//! [book-mig-zendriver]: https://turtiesocks.github.io/zendriver-rs/migration-zendriver-python.html
26//! [book-mig-nodriver]: https://turtiesocks.github.io/zendriver-rs/migration-nodriver-python.html
27//!
28//! # Quickstart
29//!
30//! ```no_run
31//! # async fn ex() -> zendriver::Result<()> {
32//! let browser = zendriver::Browser::builder().launch().await?;
33//! let tab = browser.main_tab();
34//! tab.goto("https://example.com").await?;
35//! let title: String = tab.evaluate_main("document.title").await?;
36//! println!("{title}");
37//! browser.close().await?;
38//! # Ok(()) }
39//! ```
40//!
41//! # Module layout
42//!
43//! - [`browser`] — process lifecycle ([`Browser`], [`BrowserBuilder`]).
44//! - [`tab`] — per-page handle ([`Tab`]); the main interaction surface.
45//! - [`frame`] — same-process + out-of-process iframes ([`Frame`]).
46//! - [`element`] — DOM handle ([`Element`]) returned by [`query`] helpers.
47//! - [`query`] — query builders, ARIA roles, actionability checks.
48//! - [`cookies`] / [`storage`] — browser-scope cookie jar + per-tab storage.
49//! - [`screenshot`] — screenshot builder (PNG / JPEG / WebP / full-page).
50//! - [`input`] — keyboard / mouse / modifier state.
51//! - [`error`] — [`ZendriverError`] + [`Result`] alias.
52//! - [`traits`] — [`Queryable`] / [`Evaluable`] for code generic over
53//!   Tab + Frame + Element.
54//! - [`expect`] (feature `expect`) — Playwright-style `expect_*` helpers.
55//!
56//! # Feature flags
57//!
58//! | Flag | Crate | Purpose |
59//! |------|-------|---------|
60//! | `expect` | in-tree | `expect_request` / `expect_response` / `expect_dialog` / `expect_download` |
61//! | `interception` | `zendriver-interception` | `Fetch.*`-backed request rewriting / abort |
62//! | `cloudflare` | `zendriver-cloudflare` | Cloudflare Turnstile bypass |
63//! | `fetcher` | `zendriver-fetcher` | Chrome-for-Testing download cache |
64//!
65//! Each gated module is re-exported here under `#[cfg(feature = "...")]` so
66//! downstream code can `use zendriver::AbortReason` etc. without depending on
67//! the sub-crate directly.

163/// Convenience entry point: launch a Chrome instance with default settings.
164///
165/// Equivalent to `Browser::builder().launch().await`.
166///
167/// ```no_run
168/// # async fn ex() -> zendriver::Result<()> {
169/// let browser = zendriver::start().await?;
170/// let tab = browser.main_tab();
171/// tab.goto("https://example.com").await?;
172/// # Ok(()) }
173/// ```
174pub async fn start() -> Result<Browser> {
175    Browser::builder().launch().await
176}
