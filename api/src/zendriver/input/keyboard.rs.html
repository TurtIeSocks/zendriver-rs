191/// CDP key-event field bundle for a single printable character.
192///
193/// Returned by [`char_descriptor`]. `code` is the physical-key code
194/// (`"KeyA"`, `"Digit1"`, `"Semicolon"`, ...), `windows_vk` the Windows
195/// virtual key-code, and `shift` whether the character requires the Shift
196/// modifier to produce (uppercase letters, `!`, `:`, etc.).
197#[derive(Debug, Clone, Copy, PartialEq, Eq)]
198pub(crate) struct CharDescriptor {
199    /// CDP physical-key `code` (e.g. `"KeyA"`).
200    pub code: &'static str,
201    /// Windows virtual key-code.
202    pub windows_vk: i32,
203    /// Whether Shift is required to type this character.
204    pub shift: bool,
205}

207/// Resolve a printable character to its CDP key descriptor, mirroring
208/// zendriver-py's `core/keys.py` lookup tables.
209///
210/// Returns `None` for anything that has no single physical-key origin —
211/// whitespace (callers route space/tab/newline to the matching
212/// [`SpecialKey`]) and every non-ASCII / multi-codepoint / emoji character
213/// (callers send those via a `char`-type CDP event instead).
214pub(crate) fn char_descriptor(c: char) -> Option<CharDescriptor> {
215    // a–z → KeyA..KeyZ, vk = ASCII of the uppercase letter, no shift.
216    if c.is_ascii_lowercase() {
217        let upper = c.to_ascii_uppercase();
218        return Some(CharDescriptor {
219            code: key_code_for_letter(upper),
220            windows_vk: upper as i32,
221            shift: false,
222        });
223    }
224    // A–Z → KeyA..KeyZ, vk = ASCII of the letter, shift.
225    if c.is_ascii_uppercase() {
226        return Some(CharDescriptor {
227            code: key_code_for_letter(c),
228            windows_vk: c as i32,
229            shift: true,
230        });
231    }
232    // 0–9 → Digit0..Digit9, vk = ASCII of the digit, no shift.
233    if c.is_ascii_digit() {
234        return Some(CharDescriptor {
235            code: digit_code_for(c),
236            windows_vk: c as i32,
237            shift: false,
238        });
239    }
240    // Shifted digits: ")!@#$%^&*(" — index is the base digit, Shift held.
241    if let Some(idx) = NUM_SHIFT.iter().position(|&s| s == c) {
242        let base = b'0' + idx as u8;
243        return Some(CharDescriptor {
244            code: digit_code_for(base as char),
245            windows_vk: i32::from(base),
246            shift: true,
247        });
248    }
249    // Unshifted punctuation table.
250    if let Some(&(code, vk)) = SPECIAL_CHAR_MAP
251        .iter()
252        .find_map(|(k, v)| (*k == c).then_some(v))
253    {
254        return Some(CharDescriptor {
255            code,
256            windows_vk: vk,
257            shift: false,
258        });
259    }
260    // Shifted punctuation: reuse the base char's code/vk, Shift held.
261    if let Some(base) = SPECIAL_CHAR_SHIFT_MAP
262        .iter()
263        .find_map(|(k, base)| (*k == c).then_some(*base))
264    {
265        if let Some(&(code, vk)) = SPECIAL_CHAR_MAP
266            .iter()
267            .find_map(|(k, v)| (*k == base).then_some(v))
268        {
269            return Some(CharDescriptor {
270                code,
271                windows_vk: vk,
272                shift: true,
273            });
274        }
275    }
276    // Whitespace handled by callers as SpecialKey; everything else (non-ASCII,
277    // accents, emoji, multi-codepoint) routes to a `char`-type event.
278    None
279}

362/// How a [`Key`] should be turned into CDP events.
363#[derive(Debug, Clone, Copy, PartialEq, Eq)]
364pub(crate) enum KeyPress {
365    /// One `char`-type event (text insertion). Used for emoji / non-ASCII /
366    /// multi-codepoint clusters that have no physical-key origin.
367    // Constructed by the A2b public-API layer (`type_keys` / grapheme path).
368    #[allow(dead_code)]
369    Char,
370    /// A full keyDown→keyUp pair (wrapped in modifier keyDown/keyUp events
371    /// when modifiers are active).
372    DownAndUp,
373}

375/// A single `Input.dispatchKeyEvent` payload, built by [`key_events`].
376///
377/// Serialized verbatim into the CDP call. `Option` fields are omitted from
378/// the wire payload when `None` (a `char` event carries no `code`/vk; a
379/// special-key event carries no `text`).
380#[derive(Debug, Clone, PartialEq, Eq)]
381pub(crate) struct KeyEventPayload {
382    /// CDP event type: `keyDown` / `keyUp` / `rawKeyDown` / `char`.
383    pub event_type: &'static str,
384    /// Active modifier bitmask at dispatch time.
385    pub modifiers: i32,
386    /// Inserted/typed text, if any.
387    pub text: Option<String>,
388    /// DOM `key` value, if any.
389    pub key: Option<String>,
390    /// Physical-key `code`, if any.
391    pub code: Option<&'static str>,
392    /// Windows virtual key-code, if any.
393    pub windows_vk: Option<i32>,
394}
395
396impl KeyEventPayload {
397    /// Render to the CDP `Input.dispatchKeyEvent` `params` object.
398    pub fn to_cdp_params(&self) -> serde_json::Value {
399        let mut obj = serde_json::Map::new();
400        obj.insert("type".into(), self.event_type.into());
401        obj.insert("modifiers".into(), self.modifiers.into());
402        if let Some(text) = &self.text {
403            obj.insert("text".into(), text.as_str().into());
404        }
405        if let Some(key) = &self.key {
406            obj.insert("key".into(), key.as_str().into());
407        }
408        if let Some(code) = self.code {
409            obj.insert("code".into(), code.into());
410        }
411        if let Some(vk) = self.windows_vk {
412            obj.insert("windowsVirtualKeyCode".into(), vk.into());
413            obj.insert("nativeVirtualKeyCode".into(), vk.into());
414        }
415        serde_json::Value::Object(obj)
416    }
417}

419/// CDP descriptor for a modifier key when synthesized as a real keystroke.
420/// Returns `(code, key, windowsVirtualKeyCode)`.
421fn modifier_cdp(m: KeyModifiers) -> (&'static str, &'static str, i32) {
422    match m {
423        KeyModifiers::SHIFT => ("ShiftLeft", "Shift", 16),
424        KeyModifiers::CTRL => ("ControlLeft", "Control", 17),
425        KeyModifiers::ALT => ("AltLeft", "Alt", 18),
426        KeyModifiers::META => ("MetaLeft", "Meta", 91),
427        _ => unreachable!("modifier_cdp called with a non-single modifier"),
428    }
429}

431/// Active modifiers in conventional press order: Alt, Ctrl, Meta, Shift.
432fn ordered_modifiers(mods: KeyModifiers) -> impl Iterator<Item = KeyModifiers> {
433    [
434        KeyModifiers::ALT,
435        KeyModifiers::CTRL,
436        KeyModifiers::META,
437        KeyModifiers::SHIFT,
438    ]
439    .into_iter()
440    .filter(move |m| mods.contains(*m))
441}

443/// Build the ordered CDP key-event payloads for a single [`Key`] press.
444///
445/// - `KeyPress::Char`, or a `Key::Char` whose [`char_descriptor`] is `None`
446///   (emoji / non-ASCII), emits a single `char`-type event carrying the
447///   character as both `text` and `key`.
448/// - `KeyPress::DownAndUp` on a printable `Key::Char` resolves the effective
449///   modifiers (`mods` plus Shift if the descriptor requires it) and emits,
450///   in conventional order: a keyDown for each active modifier (accumulating
451///   the modifier bitmask) → the main keyDown → the main keyUp → modifier
452///   keyUps in reverse.
453/// - `KeyPress::DownAndUp` on a `Key::Special` does the same modifier wrap,
454///   with the main events a `rawKeyDown`/`keyUp` pair built from
455///   [`SpecialKey::to_cdp`] (no `text`).
456pub(crate) fn key_events(key: Key, mods: KeyModifiers, kind: KeyPress) -> Vec<KeyEventPayload> {
457    // char-event path: explicit Char kind, or a Char with no descriptor.
458    let force_char = matches!(kind, KeyPress::Char);
459    if let Key::Char(c) = key {
460        let Some(d) = char_descriptor(c).filter(|_| !force_char) else {
461            // Explicit Char kind, or emoji / non-ASCII with no descriptor.
462            let s = c.to_string();
463            return vec![KeyEventPayload {
464                event_type: "char",
465                modifiers: mods.cdp_bits(),
466                text: Some(s.clone()),
467                key: Some(s),
468                code: None,
469                windows_vk: None,
470            }];
471        };
472        let effective = if d.shift {
473            mods | KeyModifiers::SHIFT
474        } else {
475            mods
476        };
477        let main_down = KeyEventPayload {
478            event_type: "keyDown",
479            modifiers: effective.cdp_bits(),
480            text: Some(c.to_string()),
481            key: Some(c.to_string()),
482            code: Some(d.code),
483            windows_vk: Some(d.windows_vk),
484        };
485        let main_up = KeyEventPayload {
486            event_type: "keyUp",
487            ..main_down.clone()
488        };
489        return wrap_with_modifiers(effective, main_down, main_up);
490    }
491
492    // Special key.
493    let Key::Special(k) = key else {
494        unreachable!("Key::Char handled above");
495    };
496    if force_char {
497        // char-event for a special key: send its key string as text (space,
498        // enter→"\r", tab→"\t" handled by the caller mapping; here we use the
499        // CDP `key`). Rarely used, but keep the contract total.
500        let (_, key_str, _) = k.to_cdp();
501        return vec![KeyEventPayload {
502            event_type: "char",
503            modifiers: mods.cdp_bits(),
504            text: Some(key_str.to_string()),
505            key: Some(key_str.to_string()),
506            code: None,
507            windows_vk: None,
508        }];
509    }
510    let (code, key_str, vk) = k.to_cdp();
511    let main_down = KeyEventPayload {
512        event_type: "rawKeyDown",
513        modifiers: mods.cdp_bits(),
514        text: None,
515        key: Some(key_str.to_string()),
516        code: Some(code),
517        windows_vk: Some(vk),
518    };
519    let main_up = KeyEventPayload {
520        event_type: "keyUp",
521        ..main_down.clone()
522    };
523    wrap_with_modifiers(mods, main_down, main_up)
524}

526/// Wrap a main keyDown/keyUp pair with modifier keyDown (accumulating bits,
527/// conventional order) before and modifier keyUp (reverse, clearing bits)
528/// after. `effective` is the full modifier set to synthesize.
529fn wrap_with_modifiers(
530    effective: KeyModifiers,
531    main_down: KeyEventPayload,
532    main_up: KeyEventPayload,
533) -> Vec<KeyEventPayload> {
534    let active: Vec<KeyModifiers> = ordered_modifiers(effective).collect();
535    let mut events = Vec::with_capacity(active.len() * 2 + 2);
536
537    // Modifier keyDowns, accumulating the bitmask as each is pressed.
538    let mut acc = KeyModifiers::empty();
539    for &m in &active {
540        acc |= m;
541        let (code, key_str, vk) = modifier_cdp(m);
542        events.push(KeyEventPayload {
543            event_type: "keyDown",
544            modifiers: acc.cdp_bits(),
545            text: None,
546            key: Some(key_str.to_string()),
547            code: Some(code),
548            windows_vk: Some(vk),
549        });
550    }
551
552    events.push(main_down);
553    events.push(main_up);
554
555    // Modifier keyUps in reverse, clearing the bitmask as each releases.
556    for &m in active.iter().rev() {
557        acc &= !m;
558        let (code, key_str, vk) = modifier_cdp(m);
559        events.push(KeyEventPayload {
560            event_type: "keyUp",
561            modifiers: acc.cdp_bits(),
562            text: None,
563            key: Some(key_str.to_string()),
564            code: Some(code),
565            windows_vk: Some(vk),
566        });
567    }
568
569    events
570}

572/// Map a whitespace character to the [`SpecialKey`] that produces it.
573///
574/// Space → `Space`, tab → `Tab`, newline / carriage-return → `Enter`. These
575/// are the three whitespace forms `char_descriptor` deliberately returns
576/// `None` for, so typing paths route them here instead of a `char` event.
577pub(crate) fn whitespace_special(c: char) -> Option<SpecialKey> {
578    match c {
579        ' ' => Some(SpecialKey::Space),
580        '\t' => Some(SpecialKey::Tab),
581        '\n' | '\r' => Some(SpecialKey::Enter),
582        _ => None,
583    }
584}

586/// Build the CDP events for one grapheme cluster, with `mods` held.
587///
588/// - A single `char` with a [`char_descriptor`] → `DownAndUp` (shift / code /
589///   vk synthesis).
590/// - A single whitespace `char` → the matching [`SpecialKey`] `DownAndUp`.
591/// - Anything else (emoji with modifiers, combining sequences, CJK, accents)
592///   → one `char`-type event carrying the whole cluster as text.
593pub(crate) fn cluster_events(cluster: &str, mods: KeyModifiers) -> Vec<KeyEventPayload> {
594    let mut chars = cluster.chars();
595    if let (Some(c), None) = (chars.next(), chars.clone().next()) {
596        // Exactly one char in the cluster.
597        if char_descriptor(c).is_some() {
598            return key_events(Key::Char(c), mods, KeyPress::DownAndUp);
599        }
600        if let Some(special) = whitespace_special(c) {
601            return key_events(Key::Special(special), mods, KeyPress::DownAndUp);
602        }
603    }
604    // Multi-codepoint cluster, or a single char with no descriptor → char event.
605    vec![KeyEventPayload {
606        event_type: "char",
607        modifiers: mods.cdp_bits(),
608        text: Some(cluster.to_string()),
609        key: Some(cluster.to_string()),
610        code: None,
611        windows_vk: None,
612    }]
613}

627/// An ordered mixed sequence of typed text, special-key presses, and modifier
628/// chords — parity with zendriver-py's `from_mixed_input`.
629///
630/// Build with the chainable methods, then dispatch via
631/// [`crate::Element::type_keys`]. Steps flatten to CDP key events in the order
632/// they were added.
633///
634/// # Examples
635///
636/// ```
637/// use zendriver::{Key, KeyModifiers, KeySequence, SpecialKey};
638/// let seq = KeySequence::new()
639///     .text("Hello ")
640///     .key(SpecialKey::Enter)
641///     .chord(Key::Char('a'), KeyModifiers::CTRL); // Ctrl+A
642/// # let _ = seq;
643/// ```
644#[derive(Debug, Clone, Default)]
645pub struct KeySequence {
646    steps: Vec<KeyStep>,
647}

649impl KeySequence {
650    /// Start an empty sequence.
651    #[must_use]
652    pub fn new() -> Self {
653        Self::default()
654    }
655
656    /// Append literal text (typed grapheme-by-grapheme, with shift / emoji
657    /// handling identical to [`crate::Element::type_text`]).
658    #[must_use]
659    pub fn text(mut self, text: impl Into<String>) -> Self {
660        self.steps.push(KeyStep::Text(text.into()));
661        self
662    }
663
664    /// Append a single special-key press (down + up).
665    #[must_use]
666    pub fn key(mut self, key: SpecialKey) -> Self {
667        self.steps.push(KeyStep::Key(key));
668        self
669    }
670
671    /// Append a modifier chord — `key` pressed while `mods` are held, with
672    /// real modifier keyDown/keyUp wrapper events.
673    #[must_use]
674    pub fn chord(mut self, key: Key, mods: KeyModifiers) -> Self {
675        self.steps.push(KeyStep::Chord(key, mods));
676        self
677    }
678
679    /// Flatten the whole sequence to ordered CDP key-event payloads.
680    pub(crate) fn to_events(&self) -> Vec<KeyEventPayload> {
681        let mut out = Vec::new();
682        for step in &self.steps {
683            match step {
684                KeyStep::Text(text) => {
685                    for cluster in UnicodeSegmentation::graphemes(text.as_str(), true) {
686                        out.extend(cluster_events(cluster, KeyModifiers::empty()));
687                    }
688                }
689                KeyStep::Key(k) => {
690                    out.extend(key_events(
691                        Key::Special(*k),
692                        KeyModifiers::empty(),
693                        KeyPress::DownAndUp,
694                    ));
695                }
696                KeyStep::Chord(key, mods) => {
697                    out.extend(key_events(*key, *mods, KeyPress::DownAndUp));
698                }
699            }
700        }
701        out
702    }
703}

705/// Returns a plausible nearby QWERTY key for `c`, or None for non-alphanumeric.
706/// Used by realistic typing to inject occasional typos.
707pub(crate) fn neighbor_key(c: char, rng: &mut impl rand::Rng) -> Option<char> {
708    use rand::seq::SliceRandom;
709    let lower = c.to_ascii_lowercase();
710    let neighbors: &[char] = match lower {
711        'q' => &['w', 'a', 's'],
712        'w' => &['q', 'e', 'a', 's', 'd'],
713        'e' => &['w', 'r', 's', 'd', 'f'],
714        'r' => &['e', 't', 'd', 'f', 'g'],
715        't' => &['r', 'y', 'f', 'g', 'h'],
716        'y' => &['t', 'u', 'g', 'h', 'j'],
717        'u' => &['y', 'i', 'h', 'j', 'k'],
718        'i' => &['u', 'o', 'j', 'k', 'l'],
719        'o' => &['i', 'p', 'k', 'l'],
720        'p' => &['o', 'l'],
721        'a' => &['q', 'w', 's', 'z'],
722        's' => &['a', 'd', 'w', 'e', 'z', 'x'],
723        'd' => &['s', 'f', 'e', 'r', 'x', 'c'],
724        'f' => &['d', 'g', 'r', 't', 'c', 'v'],
725        'g' => &['f', 'h', 't', 'y', 'v', 'b'],
726        'h' => &['g', 'j', 'y', 'u', 'b', 'n'],
727        'j' => &['h', 'k', 'u', 'i', 'n', 'm'],
728        'k' => &['j', 'l', 'i', 'o', 'm'],
729        'l' => &['k', 'o', 'p'],
730        'z' => &['a', 's', 'x'],
731        'x' => &['z', 'c', 's', 'd'],
732        'c' => &['x', 'v', 'd', 'f'],
733        'v' => &['c', 'b', 'f', 'g'],
734        'b' => &['v', 'n', 'g', 'h'],
735        'n' => &['b', 'm', 'h', 'j'],
736        'm' => &['n', 'j', 'k'],
737        _ => return None,
738    };
739    let pick = neighbors.choose(rng)?;
740    if c.is_ascii_uppercase() {
741        Some(pick.to_ascii_uppercase())
742    } else {
743        Some(*pick)
744    }
745}

1143/// Dispatch a pre-built sequence of CDP key events in order.
1144pub(crate) async fn dispatch_key_events(tab: &Tab, events: &[KeyEventPayload]) -> Result<()> {
1145    for ev in events {
1146        tab.session()
1147            .call("Input.dispatchKeyEvent", ev.to_cdp_params())
1148            .await?;
1149    }
1150    Ok(())
1151}

1153/// Dispatch raw text as a single `char`-type event.
1154///
1155/// Used for multi-codepoint grapheme clusters (emoji with ZWJ/skin-tone
1156/// modifiers, combining sequences) that have no single physical-key origin.
1157// Consumed by the A2b public-API layer (`type_text` grapheme path).
1158#[allow(dead_code)]
1159pub(crate) async fn dispatch_char_text(tab: &Tab, text: &str) -> Result<()> {
1160    let payload = KeyEventPayload {
1161        event_type: "char",
1162        modifiers: 0,
1163        text: Some(text.to_string()),
1164        key: Some(text.to_string()),
1165        code: None,
1166        windows_vk: None,
1167    };
1168    dispatch_key_events(tab, std::slice::from_ref(&payload)).await
1169}

1171/// Dispatch a single character as a full keyDown/keyUp pair (with shift /
1172/// code / vk synthesis via [`key_events`]).
1173///
1174/// `modifier_bits` are the caller's already-held modifiers; emoji /
1175/// non-ASCII chars fall back to a `char`-type event automatically.
1176pub(crate) async fn dispatch_char(tab: &Tab, c: char, modifier_bits: i32) -> Result<()> {
1177    let mods = KeyModifiers::from_bits_truncate(modifier_bits as u8);
1178    let events = key_events(Key::Char(c), mods, KeyPress::DownAndUp);
1179    dispatch_key_events(tab, &events).await
1180}

1182/// Dispatch a named special key (Enter, Tab, etc) as a rawKeyDown/keyUp pair.
1183#[allow(dead_code)]
1184pub(crate) async fn dispatch_special(tab: &Tab, k: SpecialKey, modifier_bits: i32) -> Result<()> {
1185    let mods = KeyModifiers::from_bits_truncate(modifier_bits as u8);
1186    let events = key_events(Key::Special(k), mods, KeyPress::DownAndUp);
1187    dispatch_key_events(tab, &events).await
1188}

1190/// Type `text` with realistic per-character timing, occasional typos, and
1191/// inter-word "thinking" pauses pulled from the InputProfile.
1192///
1193/// Text is segmented into grapheme clusters; each cluster dispatches via
1194/// [`cluster_events`] (so uppercase / symbols synthesize Shift, emoji / CJK
1195/// route to a `char` event, and space / tab / newline become the matching
1196/// [`SpecialKey`]). The per-cluster delay / typo / thinking-pause wrapper is
1197/// preserved; typos still apply only to ASCII letters (via [`neighbor_key`]).
1198#[allow(dead_code)]
1199pub(crate) async fn type_text_realistic(
1200    input: &InputController,
1201    tab: &Tab,
1202    text: &str,
1203) -> Result<()> {
1204    let profile = input.profile.clone();
1205    for cluster in UnicodeSegmentation::graphemes(text, true) {
1206        // Leading char drives the typo / thinking-pause heuristics; the actual
1207        // keystroke uses the whole cluster (handles multi-codepoint emoji).
1208        let lead = cluster.chars().next().unwrap_or('\0');
1209        let (per_char_delay_ms, held_mods, do_typo, typo_char, thinking_pause_ms) = {
1210            let mut s = input.state.lock().await;
1211            let per_char = if profile.per_char_delay_ms_range.0 == 0
1212                && profile.per_char_delay_ms_range.1 == 0
1213            {
1214                0
1215            } else {
1216                rand::Rng::gen_range(
1217                    &mut s.rng,
1218                    profile.per_char_delay_ms_range.0..=profile.per_char_delay_ms_range.1,
1219                )
1220            };
1221            let do_typo =
1222                profile.typo_rate > 0.0 && rand::Rng::r#gen::<f32>(&mut s.rng) < profile.typo_rate;
1223            let typo_char = if do_typo {
1224                neighbor_key(lead, &mut s.rng)
1225            } else {
1226                None
1227            };
1228            let thinking = if lead == ' '
1229                && profile.thinking_pause_ms_range.0 > 0
1230                && rand::Rng::r#gen::<f32>(&mut s.rng) < 0.05
1231            {
1232                rand::Rng::gen_range(
1233                    &mut s.rng,
1234                    profile.thinking_pause_ms_range.0..=profile.thinking_pause_ms_range.1,
1235                )
1236            } else {
1237                0
1238            };
1239            (per_char, s.modifiers_held, do_typo, typo_char, thinking)
1240        };
1241        if per_char_delay_ms > 0 {
1242            tokio::time::sleep(Duration::from_millis(per_char_delay_ms as u64)).await;
1243        }
1244        if thinking_pause_ms > 0 {
1245            tokio::time::sleep(Duration::from_millis(thinking_pause_ms as u64)).await;
1246        }
1247        if do_typo {
1248            if let Some(wrong) = typo_char {
1249                dispatch_char(tab, wrong, held_mods.cdp_bits()).await?;
1250                tokio::time::sleep(Duration::from_millis(80)).await;
1251                dispatch_special(tab, SpecialKey::Backspace, held_mods.cdp_bits()).await?;
1252            }
1253        }
1254        dispatch_key_events(tab, &cluster_events(cluster, held_mods)).await?;
1255    }
1256    Ok(())
1257}

1259/// Type `text` as fast as possible — no delays, no typos.
1260///
1261/// Segments into grapheme clusters and dispatches each via
1262/// [`cluster_events`], so the full unicode / shift / special-key handling of
1263/// the realistic path applies without any timing jitter.
1264#[allow(dead_code)]
1265pub(crate) async fn type_text_fast(input: &InputController, tab: &Tab, text: &str) -> Result<()> {
1266    let held_mods = input.state.lock().await.modifiers_held;
1267    for cluster in UnicodeSegmentation::graphemes(text, true) {
1268        dispatch_key_events(tab, &cluster_events(cluster, held_mods)).await?;
1269    }
1270    Ok(())
1271}
