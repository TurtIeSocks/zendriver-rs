1//! Offline generator for the vendored CLDR country -> language table.
2//! Run via `cargo run -p locale-gen`. NOT published.
3
4use serde_json::Value;
5
6/// One emitted table row: an uppercase country code and its ranked language
7/// subtags (dominant first).
8#[derive(Debug, PartialEq, Eq)]
9pub struct CountryRow {
10    pub cc: String,
11    pub langs: Vec<String>,
12}
13
14/// Keep a language if it is official-ish OR spoken by at least this percent.
15const POP_THRESHOLD: f64 = 20.0;
16
17/// Transform CLDR `territoryInfo` JSON into sorted, ranked rows.
18pub fn rows_from_cldr(v: &Value) -> Vec<CountryRow> {
19    let mut rows = Vec::new();
20    let Some(map) = v["supplemental"]["territoryInfo"].as_object() else {
21        return rows;
22    };
23    for (cc, info) in map {
24        if cc.len() != 2 || !cc.bytes().all(|b| b.is_ascii_alphabetic()) {
25            continue; // skip "001" etc.
26        }
27        let Some(lp) = info.get("languagePopulation").and_then(Value::as_object) else {
28            continue;
29        };
30        let mut langs: Vec<(String, f64, bool)> = lp
31            .iter()
32            .filter_map(|(lang, d)| {
33                let pop = d
34                    .get("_populationPercent")
35                    .and_then(Value::as_str)
36                    .and_then(|s| s.parse::<f64>().ok())
37                    .unwrap_or(0.0);
38                let official = d
39                    .get("_officialStatus")
40                    .and_then(Value::as_str)
41                    .is_some_and(|s| s.starts_with("official") || s == "de_facto_official");
42                if official || pop >= POP_THRESHOLD {
43                    let subtag = lang.split('_').next().unwrap_or(lang).to_lowercase();
44                    Some((subtag, pop, official))
45                } else {
46                    None
47                }
48            })
49            .collect();
50        langs.sort_by(|a, b| {
51            b.2.cmp(&a.2)
52                .then(b.1.partial_cmp(&a.1).unwrap_or(std::cmp::Ordering::Equal))
53        });
54        let mut seen = std::collections::HashSet::new();
55        let ranked: Vec<String> = langs
56            .into_iter()
57            .map(|(s, _, _)| s)
58            .filter(|s| seen.insert(s.clone()))
59            .collect();
60        if !ranked.is_empty() {
61            rows.push(CountryRow {
62                cc: cc.to_uppercase(),
63                langs: ranked,
64            });
65        }
66    }
67    rows.sort_by(|a, b| a.cc.cmp(&b.cc));
68    rows
69}

71/// Curated primary-zone overrides for large multi-timezone countries whose
72/// `zone1970.tab` first row is not the most-populous zone. Kept minimal;
73/// each entry must be verified against a real `zone1970.tab`, never guessed.
74///
75/// - `RU`: first row is `Europe/Kaliningrad` (~1M); Moscow time covers the
76///   large majority of Russia's population.
77/// - `UA`: first row is the shared `RU,UA` `Europe/Simferopol` (Crimea,
78///   ~2M); mainland Ukraine (`Europe/Kyiv`) is ~40M+.
79/// - `CA`: first row is `America/St_Johns` (Newfoundland, ~500K); Ontario +
80///   Quebec (`America/Toronto`) hold the bulk of Canada's population.
81/// - `AU`: first row is `Australia/Lord_Howe` (~350 residents); New South
82///   Wales (`Australia/Sydney`) is Australia's most populous zone.
83/// - `BR`: first row is `America/Noronha` (~3K residents); southeast Brazil
84///   (`America/Sao_Paulo`) holds the large majority of Brazil's population.
85const PRIMARY_TZ_OVERRIDES: &[(&str, &str)] = &[
86    ("RU", "Europe/Moscow"),
87    ("UA", "Europe/Kyiv"),
88    ("CA", "America/Toronto"),
89    ("AU", "Australia/Sydney"),
90    ("BR", "America/Sao_Paulo"),
91    // Add others ONLY where the parsed first-occurrence is demonstrably not
92    // the most-populous zone; verify against a real zone1970.tab before adding.
93];
94
95/// Parse `zone1970.tab` into a sorted `(country, IANA zone)` list: each
96/// country's FIRST-appearing row wins, then `PRIMARY_TZ_OVERRIDES` are applied.
97pub fn tz_rows_from_zone_tab(raw: &str) -> Vec<(String, String)> {
98    use std::collections::BTreeMap;
99    let mut map: BTreeMap<String, String> = BTreeMap::new();
100    for line in raw.lines() {
101        if line.starts_with('#') || line.trim().is_empty() {
102            continue;
103        }
104        let mut cols = line.split('\t');
105        let codes = cols.next().unwrap_or("");
106        let _coords = cols.next();
107        let zone = cols.next().unwrap_or("").trim();
108        if zone.is_empty() {
109            continue;
110        }
111        for cc in codes.split(',') {
112            map.entry(cc.to_string())
113                .or_insert_with(|| zone.to_string()); // first wins
114        }
115    }
116    for (cc, z) in PRIMARY_TZ_OVERRIDES {
117        map.insert((*cc).to_string(), (*z).to_string());
118    }
119    map.into_iter().collect() // BTreeMap -> already sorted by country
120}

122/// Emit the `table.rs` source for the stealth crate: the CLDR-derived
123/// country->language table and the IANA-derived country->timezone table,
124/// in one generated file body.
125pub fn emit_rust(
126    cldr_version: &str,
127    rows: &[CountryRow],
128    tzdata_version: &str,
129    tz_rows: &[(String, String)],
130) -> String {
131    let mut s = String::new();
132    s.push_str(&format!(
133        "// CLDR {cldr_version}, tzdata {tzdata_version}. Generated by `cargo run -p locale-gen`. DO NOT EDIT.\n"
134    ));
135    s.push_str(&format!(
136        "pub(crate) static CLDR_VERSION: &str = \"{cldr_version}\";\n\n"
137    ));
138    s.push_str("/// Sorted by country code (binary-searchable). Ranked language subtags, dominant first.\n");
139    s.push_str("pub(crate) static COUNTRIES: &[(&str, &[&str])] = &[\n");
140    for r in rows {
141        let langs = r
142            .langs
143            .iter()
144            .map(|l| format!("\"{l}\""))
145            .collect::<Vec<_>>()
146            .join(", ");
147        s.push_str(&format!("    (\"{}\", &[{}]),\n", r.cc, langs));
148    }
149    s.push_str("];\n\n");
150    s.push_str(&format!(
151        "pub(crate) static TZDATA_VERSION: &str = \"{tzdata_version}\";\n\n"
152    ));
153    s.push_str(
154        "/// Sorted by country code (binary-searchable). Representative IANA zone per country.\n",
155    );
156    s.push_str("pub(crate) static TIMEZONES: &[(&str, &str)] = &[\n");
157    for (cc, zone) in tz_rows {
158        s.push_str(&format!("    (\"{cc}\", \"{zone}\"),\n"));
159    }
160    s.push_str("];\n");
161    s
162}

164#[cfg(test)]
165mod tests {
166    use super::*;
167
168    fn fixture() -> Value {
169        serde_json::from_str(include_str!("../tests/fixtures/territoryInfo-mini.json")).unwrap()
170    }
171
172    const TZ_FIXTURE: &str = include_str!("../tests/fixtures/zone1970-mini.tab");
173
174    fn lookup<'a>(rows: &'a [(String, String)], cc: &str) -> Option<&'a str> {
175        rows.iter().find(|(c, _)| c == cc).map(|(_, z)| z.as_str())
176    }
177
178    #[test]
179    fn tz_first_occurrence_per_country() {
180        // fixture rows: "US ... America/New_York", "US ... America/Chicago" -> US = New_York
181        let rows = tz_rows_from_zone_tab(TZ_FIXTURE);
182        assert_eq!(lookup(&rows, "US"), Some("America/New_York"));
183        assert_eq!(lookup(&rows, "FR"), Some("Europe/Paris"));
184    }
185
186    #[test]
187    fn tz_override_wins() {
188        // fixture has RU first row = Europe/Kaliningrad; override forces Moscow
189        let rows = tz_rows_from_zone_tab(TZ_FIXTURE);
190        assert_eq!(lookup(&rows, "RU"), Some("Europe/Moscow"));
191    }
192
193    #[test]
194    fn tz_rows_sorted() {
195        let rows = tz_rows_from_zone_tab(TZ_FIXTURE);
196        assert!(rows.windows(2).all(|w| w[0].0 <= w[1].0));
197    }
198
199    #[test]
200    fn rows_skip_world_region_and_rank() {
201        let rows = rows_from_cldr(&fixture());
202        let ccs: Vec<&str> = rows.iter().map(|r| r.cc.as_str()).collect();
203        assert_eq!(ccs, vec!["CH", "US"]); // sorted, "001" skipped
204        let us = rows.iter().find(|r| r.cc == "US").unwrap();
205        assert_eq!(us.langs, vec!["en"]); // es below threshold (9.6 < 20, not official)
206        let ch = rows.iter().find(|r| r.cc == "CH").unwrap();
207        assert_eq!(ch.langs[0], "de"); // dominant official first
208        assert!(ch.langs.contains(&"fr".to_string()));
209        assert!(ch.langs.contains(&"it".to_string()));
210    }
211
212    #[test]
213    fn emit_is_valid_rust_shape() {
214        let rows = rows_from_cldr(&fixture());
215        let tz_rows = tz_rows_from_zone_tab(TZ_FIXTURE);
216        let out = emit_rust("46.0.0", &rows, "2026c", &tz_rows);
217        assert!(out.contains("CLDR_VERSION: &str = \"46.0.0\""));
218        assert!(out.contains("(\"US\", &[\"en\"]),"));
219        assert!(out.contains("(\"CH\", &["));
220        assert!(out.contains("TZDATA_VERSION: &str = \"2026c\""));
221        assert!(out.contains("(\"US\", \"America/New_York\"),"));
222        assert!(out.contains("(\"RU\", \"Europe/Moscow\"),"));
223    }
