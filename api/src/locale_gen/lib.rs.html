1//! Offline generator for the vendored CLDR country -> language table.
2//! Run via `cargo run -p locale-gen`. NOT published.
3
4use serde_json::Value;
5
6/// One emitted table row: an uppercase country code and its ranked language
7/// subtags (dominant first).
8#[derive(Debug, PartialEq, Eq)]
9pub struct CountryRow {
10    pub cc: String,
11    pub langs: Vec<String>,
12}
13
14/// Keep a language if it is official-ish OR spoken by at least this percent.
15const POP_THRESHOLD: f64 = 20.0;
16
17/// Transform CLDR `territoryInfo` JSON into sorted, ranked rows.
18pub fn rows_from_cldr(v: &Value) -> Vec<CountryRow> {
19    let mut rows = Vec::new();
20    let Some(map) = v["supplemental"]["territoryInfo"].as_object() else {
21        return rows;
22    };
23    for (cc, info) in map {
24        if cc.len() != 2 || !cc.bytes().all(|b| b.is_ascii_alphabetic()) {
25            continue; // skip "001" etc.
26        }
27        let Some(lp) = info.get("languagePopulation").and_then(Value::as_object) else {
28            continue;
29        };
30        let mut langs: Vec<(String, f64, bool)> = lp
31            .iter()
32            .filter_map(|(lang, d)| {
33                let pop = d
34                    .get("_populationPercent")
35                    .and_then(Value::as_str)
36                    .and_then(|s| s.parse::<f64>().ok())
37                    .unwrap_or(0.0);
38                let official = d
39                    .get("_officialStatus")
40                    .and_then(Value::as_str)
41                    .is_some_and(|s| s.starts_with("official") || s == "de_facto_official");
42                if official || pop >= POP_THRESHOLD {
43                    let subtag = lang.split('_').next().unwrap_or(lang).to_lowercase();
44                    Some((subtag, pop, official))
45                } else {
46                    None
47                }
48            })
49            .collect();
50        langs.sort_by(|a, b| {
51            b.2.cmp(&a.2)
52                .then(b.1.partial_cmp(&a.1).unwrap_or(std::cmp::Ordering::Equal))
53        });
54        let mut seen = std::collections::HashSet::new();
55        let ranked: Vec<String> = langs
56            .into_iter()
57            .map(|(s, _, _)| s)
58            .filter(|s| seen.insert(s.clone()))
59            .collect();
60        if !ranked.is_empty() {
61            rows.push(CountryRow {
62                cc: cc.to_uppercase(),
63                langs: ranked,
64            });
65        }
66    }
67    rows.sort_by(|a, b| a.cc.cmp(&b.cc));
68    rows
69}

71/// Emit the `table.rs` source for the stealth crate.
72pub fn emit_rust(version: &str, rows: &[CountryRow]) -> String {
73    let mut s = String::new();
74    s.push_str(&format!(
75        "// CLDR {version}. Generated by `cargo run -p locale-gen`. DO NOT EDIT.\n"
76    ));
77    s.push_str(&format!(
78        "pub(crate) static CLDR_VERSION: &str = \"{version}\";\n\n"
79    ));
80    s.push_str("/// Sorted by country code (binary-searchable). Ranked language subtags, dominant first.\n");
81    s.push_str("pub(crate) static COUNTRIES: &[(&str, &[&str])] = &[\n");
82    for r in rows {
83        let langs = r
84            .langs
85            .iter()
86            .map(|l| format!("\"{l}\""))
87            .collect::<Vec<_>>()
88            .join(", ");
89        s.push_str(&format!("    (\"{}\", &[{}]),\n", r.cc, langs));
90    }
91    s.push_str("];\n");
92    s
93}

95#[cfg(test)]
96mod tests {
97    use super::*;
98
99    fn fixture() -> Value {
100        serde_json::from_str(include_str!("../tests/fixtures/territoryInfo-mini.json")).unwrap()
101    }
102
103    #[test]
104    fn rows_skip_world_region_and_rank() {
105        let rows = rows_from_cldr(&fixture());
106        let ccs: Vec<&str> = rows.iter().map(|r| r.cc.as_str()).collect();
107        assert_eq!(ccs, vec!["CH", "US"]); // sorted, "001" skipped
108        let us = rows.iter().find(|r| r.cc == "US").unwrap();
109        assert_eq!(us.langs, vec!["en"]); // es below threshold (9.6 < 20, not official)
110        let ch = rows.iter().find(|r| r.cc == "CH").unwrap();
111        assert_eq!(ch.langs[0], "de"); // dominant official first
112        assert!(ch.langs.contains(&"fr".to_string()));
113        assert!(ch.langs.contains(&"it".to_string()));
114    }
115
116    #[test]
117    fn emit_is_valid_rust_shape() {
118        let rows = rows_from_cldr(&fixture());
119        let out = emit_rust("46.0.0", &rows);
120        assert!(out.contains("CLDR_VERSION: &str = \"46.0.0\""));
121        assert!(out.contains("(\"US\", &[\"en\"]),"));
122        assert!(out.contains("(\"CH\", &["));
123    }
